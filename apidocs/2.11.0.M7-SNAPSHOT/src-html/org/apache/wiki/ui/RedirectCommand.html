019package org.apache.wiki.ui;
020
021import java.security.Permission;
022
023/**
024 * <p>Defines Commands for redirections to off-site special pages. RedirectCommands do not have associated permissions; the
025 * {@link #requiredPermission()} method will always return <code>null</code>. When combined with a supplied String url,
026 * the {@link #getTarget()} method will return a String, the {@link #getURLPattern()} method will return the supplied target URL,
027 * and {@link #getJSP()} method will return the "cleansed" URL.</p>
028 *
029 * @since 2.4.22
030 */
031public final class RedirectCommand extends AbstractCommand {
032
033    public static final Command REDIRECT = new RedirectCommand( "", "%u%n", null, null );
034
035    /**
036     * Constructs a new Command with a specified wiki context, URL pattern, type, and content template. The WikiPage for this action is
037     * initialized to <code>null</code>.
038     *
039     * @param requestContext the request context
040     * @param urlPattern the URL pattern
041     * @param contentTemplate the content template; may be <code>null</code>
042     * @param target the target of the command
043     * @throws IllegalArgumentException if the request content, URL pattern, or type is <code>null</code>
044     */
045    private RedirectCommand( final String requestContext, final String urlPattern, final String contentTemplate, final String target ) {
046        super( requestContext, urlPattern, contentTemplate, target );
047    }
048    
049    /**
050     * Creates and returns a targeted Command by combining a URL (as String) with this Command. The supplied <code>target</code>
051     * object must be non-<code>null</code> and of type String. The URL passed to the constructor is actually an URL pattern, but it
052     * will be converted to a JSP page if it is a partial URL. If it is a full URL (beginning with <code>http://</code> or
053     * <code>https://</code>), it will be "passed through" without conversion, and the URL pattern will be <code>null</code>.
054     *
055     * @param target the object to combine
056     * @throws IllegalArgumentException if the target is not of the correct type
057     */
058    public Command targetedCommand( final Object target ) {
059        if ( !( target instanceof String ) ) {
060            throw new IllegalArgumentException( "Target must non-null and of type String." );
061        }
062        return new RedirectCommand( getRequestContext(), ( String )target, getContentTemplate(), ( String )target );
063    }
064    
065    /**
066     * @see org.apache.wiki.ui.Command#getName()
067     */
068    public String getName() {
069        final Object target = getTarget();
070        if ( target == null ) {
071            return getJSPFriendlyName();
072        }
073        return target.toString();
074    }
075
076    /**
077     * No-op; always returns <code>null</code>.
078     *
079     * @see org.apache.wiki.ui.Command#requiredPermission()
080     */
081    public Permission requiredPermission() {
082        return null;
083    }
084
085}
