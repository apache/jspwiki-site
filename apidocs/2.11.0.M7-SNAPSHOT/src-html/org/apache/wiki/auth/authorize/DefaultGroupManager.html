051/**
052 * <p>
053 * Facade class for storing, retrieving and managing wiki groups on behalf of AuthorizationManager, JSPs and other presentation-layer
054 * classes. GroupManager works in collaboration with a back-end {@link GroupDatabase}, which persists groups to permanent storage.
055 * </p>
056 * <p>
057 * <em>Note: prior to JSPWiki 2.4.19, GroupManager was an interface; it is now a concrete, final class. The aspects of GroupManager
058 * which previously extracted group information from storage (e.g., wiki pages) have been refactored into the GroupDatabase interface.</em>
059 * </p>
060 * @since 2.4.19
061 */
062public class DefaultGroupManager implements GroupManager, Authorizer, WikiEventListener {
063

072    /** Map with GroupPrincipals as keys, and Groups as values */
073    private final Map< Principal, Group > m_groups = new HashMap<>();

075    /** {@inheritDoc} */
076    @Override
077    public Principal findRole( final String name ) {
078        try {
079            final Group group = getGroup( name );
080            return group.getPrincipal();
081        } catch( final NoSuchPrincipalException e ) {
082            return null;
083        }
084    }

086    /** {@inheritDoc} */
087    @Override
088    public Group getGroup( final String name ) throws NoSuchPrincipalException {
089        final Group group = m_groups.get( new GroupPrincipal( name ) );
090        if( group != null ) {
091            return group;
092        }
093        throw new NoSuchPrincipalException( "Group " + name + " not found." );
094    }

096    /** {@inheritDoc} */
097    @Override
098    public GroupDatabase getGroupDatabase() throws WikiSecurityException {
099        if( m_groupDatabase != null ) {
100            return m_groupDatabase;
101        }
102
103        String dbClassName = "<unknown>";
104        String dbInstantiationError = null;
105        Throwable cause = null;
106        try {
107            final Properties props = m_engine.getWikiProperties();
108            dbClassName = props.getProperty( PROP_GROUPDATABASE );
109            if( dbClassName == null ) {
110                dbClassName = XMLGroupDatabase.class.getName();
111            }
112            log.info( "Attempting to load group database class " + dbClassName );
113            final Class< ? > dbClass = ClassUtil.findClass( "org.apache.wiki.auth.authorize", dbClassName );
114            m_groupDatabase = ( GroupDatabase )dbClass.newInstance();
115            m_groupDatabase.initialize( m_engine, m_engine.getWikiProperties() );
116            log.info( "Group database initialized." );
117        } catch( final ClassNotFoundException e ) {
118            log.error( "GroupDatabase class " + dbClassName + " cannot be found.", e );
119            dbInstantiationError = "Failed to locate GroupDatabase class " + dbClassName;
120            cause = e;
121        } catch( final InstantiationException e ) {
122            log.error( "GroupDatabase class " + dbClassName + " cannot be created.", e );
123            dbInstantiationError = "Failed to create GroupDatabase class " + dbClassName;
124            cause = e;
125        } catch( final IllegalAccessException e ) {
126            log.error( "You are not allowed to access group database class " + dbClassName + ".", e );
127            dbInstantiationError = "Access GroupDatabase class " + dbClassName + " denied";
128            cause = e;
129        } catch( final NoRequiredPropertyException e ) {
130            log.error( "Missing property: " + e.getMessage() + "." );
131            dbInstantiationError = "Missing property: " + e.getMessage();
132            cause = e;
133        }
134
135        if( dbInstantiationError != null ) {
136            throw new WikiSecurityException( dbInstantiationError + " Cause: " + cause.getMessage(), cause );
137        }
138
139        return m_groupDatabase;
140    }

142    /** {@inheritDoc} */
143    @Override
144    public Principal[] getRoles() {
145        return m_groups.keySet().toArray( new Principal[ m_groups.size() ] );
146    }

148    /** {@inheritDoc} */
149    @Override
150    public void initialize( final Engine engine, final Properties props ) throws WikiSecurityException {
151        m_engine = engine;
152
153        try {
154            m_groupDatabase = getGroupDatabase();
155        } catch( final WikiException e ) {
156            throw new WikiSecurityException( e.getMessage(), e );
157        }
158
159        // Load all groups from the database into the cache
160        final Group[] groups = m_groupDatabase.groups();
161        synchronized( m_groups ) {
162            for( final Group group : groups ) {
163                // Add new group to cache; fire GROUP_ADD event
164                m_groups.put( group.getPrincipal(), group );
165                fireEvent( WikiSecurityEvent.GROUP_ADD, group );
166            }
167        }
168
169        // Make the GroupManager listen for WikiEvents (WikiSecurityEvents for changed user profiles)
170        engine.getManager( UserManager.class ).addWikiEventListener( this );
171
172        // Success!
173        log.info( "Authorizer GroupManager initialized successfully; loaded " + groups.length + " group(s)." );
174    }

176    /** {@inheritDoc} */
177    @Override
178    public boolean isUserInRole( final Session session, final Principal role ) {
179        // Always return false if session/role is null, or if role isn't a GroupPrincipal
180        if ( session == null || !( role instanceof GroupPrincipal ) || !session.isAuthenticated() ) {
181            return false;
182        }
183
184        // Get the group we're examining
185        final Group group = m_groups.get( role );
186        if( group == null ) {
187            return false;
188        }
189
190        // Check each user principal to see if it belongs to the group
191        for( final Principal principal : session.getPrincipals() ) {
192            if( AuthenticationManager.isUserPrincipal( principal ) && group.isMember( principal ) ) {
193                return true;
194            }
195        }
196        return false;
197    }

199    /** {@inheritDoc} */
200    @Override
201    public Group parseGroup( String name, String memberLine, final boolean create ) throws WikiSecurityException {
202        // If null name parameter, it's because someone's creating a new group
203        if( name == null ) {
204            if( create ) {
205                name = "MyGroup";
206            } else {
207                throw new WikiSecurityException( "Group name cannot be blank." );
208            }
209        } else if( ArrayUtils.contains( Group.RESTRICTED_GROUPNAMES, name ) ) {
210            // Certain names are forbidden
211            throw new WikiSecurityException( "Illegal group name: " + name );
212        }
213        name = name.trim();
214
215        // Normalize the member line
216        if( InputValidator.isBlank( memberLine ) ) {
217            memberLine = "";
218        }
219        memberLine = memberLine.trim();
220
221        // Create or retrieve the group (may have been previously cached)
222        final Group group = new Group( name, m_engine.getApplicationName() );
223        try {
224            final Group existingGroup = getGroup( name );
225
226            // If existing, clone it
227            group.setCreator( existingGroup.getCreator() );
228            group.setCreated( existingGroup.getCreated() );
229            group.setModifier( existingGroup.getModifier() );
230            group.setLastModified( existingGroup.getLastModified() );
231            for( final Principal existingMember : existingGroup.members() ) {
232                group.add( existingMember );
233            }
234        } catch( final NoSuchPrincipalException e ) {
235            // It's a new group.... throw error if we don't create new ones
236            if( !create ) {
237                throw new NoSuchPrincipalException( "Group '" + name + "' does not exist." );
238            }
239        }
240
241        // If passed members not empty, overwrite
242        final String[] members = extractMembers( memberLine );
243        if( members.length > 0 ) {
244            group.clear();
245            for( final String member : members ) {
246                group.add( new WikiPrincipal( member ) );
247            }
248        }
249
250        return group;
251    }

253    /** {@inheritDoc} */
254    @Override
255    public void removeGroup( final String index ) throws WikiSecurityException {
256        if( index == null ) {
257            throw new IllegalArgumentException( "Group cannot be null." );
258        }
259
260        final Group group = m_groups.get( new GroupPrincipal( index ) );
261        if( group == null ) {
262            throw new NoSuchPrincipalException( "Group " + index + " not found" );
263        }
264
265        // Delete the group
266        // TODO: need rollback procedure
267        synchronized( m_groups ) {
268            m_groups.remove( group.getPrincipal() );
269        }
270        m_groupDatabase.delete( group );
271        fireEvent( WikiSecurityEvent.GROUP_REMOVE, group );
272    }
273
274    /** {@inheritDoc} */
275    @Override
276    public void setGroup( final Session session, final Group group ) throws WikiSecurityException {
277        // TODO: check for appropriate permissions
278
279        // If group already exists, delete it; fire GROUP_REMOVE event
280        final Group oldGroup = m_groups.get( group.getPrincipal() );
281        if( oldGroup != null ) {
282            fireEvent( WikiSecurityEvent.GROUP_REMOVE, oldGroup );
283            synchronized( m_groups ) {
284                m_groups.remove( oldGroup.getPrincipal() );
285            }
286        }
287
288        // Copy existing modifier info & timestamps
289        if( oldGroup != null ) {
290            group.setCreator( oldGroup.getCreator() );
291            group.setCreated( oldGroup.getCreated() );
292            group.setModifier( oldGroup.getModifier() );
293            group.setLastModified( oldGroup.getLastModified() );
294        }
295
296        // Add new group to cache; announce GROUP_ADD event
297        synchronized( m_groups ) {
298            m_groups.put( group.getPrincipal(), group );
299        }
300        fireEvent( WikiSecurityEvent.GROUP_ADD, group );
301
302        // Save the group to back-end database; if it fails, roll back to previous state. Note that the back-end
303        // MUST timestammp the create/modify fields in the Group.
304        try {
305            m_groupDatabase.save( group, session.getUserPrincipal() );
306        }
307
308        // We got an exception! Roll back...
309        catch( final WikiSecurityException e ) {
310            if( oldGroup != null ) {
311                // Restore previous version, re-throw...
312                fireEvent( WikiSecurityEvent.GROUP_REMOVE, group );
313                fireEvent( WikiSecurityEvent.GROUP_ADD, oldGroup );
314                synchronized( m_groups ) {
315                    m_groups.put( oldGroup.getPrincipal(), oldGroup );
316                }
317                throw new WikiSecurityException( e.getMessage() + " (rolled back to previous version).", e );
318            }
319            // Re-throw security exception
320            throw new WikiSecurityException( e.getMessage(), e );
321        }
322    }

324    /**
325     * Extracts carriage-return separated members into a Set of String objects.
326     *
327     * @param memberLine the list of members
328     * @return the list of members
329     */
330    protected String[] extractMembers( final String memberLine ) {
331        final Set< String > members = new HashSet<>();
332        if( memberLine != null ) {
333            final StringTokenizer tok = new StringTokenizer( memberLine, "\n" );
334            while( tok.hasMoreTokens() ) {
335                final String uid = tok.nextToken().trim();
336                if( uid.length() > 0 ) {
337                    members.add( uid );
338                }
339            }
340        }
341        return members.toArray( new String[ members.size() ] );
342    }

358    /** {@inheritDoc} */
359    @Override
360    public void actionPerformed( final WikiEvent event ) {
361        if( !( event instanceof WikiSecurityEvent ) ) {
362            return;
363        }
364
365        final WikiSecurityEvent se = ( WikiSecurityEvent )event;
366        if( se.getType() == WikiSecurityEvent.PROFILE_NAME_CHANGED ) {
367            final Session session = se.getSrc();
368            final UserProfile[] profiles = ( UserProfile[] )se.getTarget();
369            final Principal[] oldPrincipals = new Principal[] { new WikiPrincipal( profiles[ 0 ].getLoginName() ),
370                    new WikiPrincipal( profiles[ 0 ].getFullname() ), new WikiPrincipal( profiles[ 0 ].getWikiName() ) };
371            final Principal newPrincipal = new WikiPrincipal( profiles[ 1 ].getFullname() );
372
373            // Examine each group
374            int groupsChanged = 0;
375            try {
376                for( final Group group : m_groupDatabase.groups() ) {
377                    boolean groupChanged = false;
378                    for( final Principal oldPrincipal : oldPrincipals ) {
379                        if( group.isMember( oldPrincipal ) ) {
380                            group.remove( oldPrincipal );
381                            group.add( newPrincipal );
382                            groupChanged = true;
383                        }
384                    }
385                    if( groupChanged ) {
386                        setGroup( session, group );
387                        groupsChanged++;
388                    }
389                }
390            } catch( final WikiException e ) {
391                // Oooo! This is really bad...
392                log.error( "Could not change user name in Group lists because of GroupDatabase error:" + e.getMessage() );
393            }
394            log.info( "Profile name change for '" + newPrincipal.toString() + "' caused " + groupsChanged + " groups to change also." );
395        }
396    }
