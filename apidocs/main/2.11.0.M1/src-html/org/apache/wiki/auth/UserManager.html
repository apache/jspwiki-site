137    /**
138     * Returns the UserDatabase employed by this WikiEngine. The UserDatabase is
139     * lazily initialized by this method, if it does not exist yet. If the
140     * initialization fails, this method will use the inner class
141     * DummyUserDatabase as a default (which is enough to get JSPWiki running).
142     * @return the dummy user database
143     * @since 2.3
144     */
145    public UserDatabase getUserDatabase()
146    {
147        // FIXME: Must not throw RuntimeException, but something else.
148        if( m_database != null )
149        {
150            return m_database;
151        }
152
153        String dbClassName = UNKNOWN_CLASS;
154
155        try
156        {
157            dbClassName = m_engine.getRequiredProperty( m_engine.getWikiProperties(), PROP_DATABASE );
158
159            log.info("Attempting to load user database class "+dbClassName);
160            final Class<?> dbClass = ClassUtil.findClass( USERDATABASE_PACKAGE, dbClassName );
161            m_database = (UserDatabase) dbClass.newInstance();
162            m_database.initialize( m_engine, m_engine.getWikiProperties() );
163            log.info("UserDatabase initialized.");
164        }
165        catch( final NoRequiredPropertyException e )
166        {
167            log.error( "You have not set the '"+PROP_DATABASE+"'. You need to do this if you want to enable user management by JSPWiki.", e );
168        }
169        catch( final ClassNotFoundException e )
170        {
171            log.error( "UserDatabase class " + dbClassName + " cannot be found", e );
172        }
173        catch( final InstantiationException e )
174        {
175            log.error( "UserDatabase class " + dbClassName + " cannot be created", e );
176        }
177        catch( final IllegalAccessException e )
178        {
179            log.error( "You are not allowed to access this user database class", e );
180        }
181        catch( final WikiSecurityException e )
182        {
183            log.error( "Exception initializing user database: " + e.getMessage(), e );
184        }
185        finally
186        {
187            if( m_database == null )
188            {
189                log.info("I could not create a database object you specified (or didn't specify), so I am falling back to a default.");
190                m_database = new DummyUserDatabase();
191            }
192        }
193
194        return m_database;
195    }

197    /**
198     * <p>Retrieves the {@link org.apache.wiki.auth.user.UserProfile}for the
199     * user in a wiki session. If the user is authenticated, the UserProfile
200     * returned will be the one stored in the user database; if one does not
201     * exist, a new one will be initialized and returned. If the user is
202     * anonymous or asserted, the UserProfile will <i>always</i> be newly
203     * initialized to prevent spoofing of identities. If a UserProfile needs to
204     * be initialized, its
205     * {@link org.apache.wiki.auth.user.UserProfile#isNew()} method will
206     * return <code>true</code>, and its login name will will be set
207     * automatically if the user is authenticated. Note that this method does
208     * not modify the retrieved (or newly created) profile otherwise; other
209     * fields in the user profile may be <code>null</code>.</p>
210     * <p>If a new UserProfile was created, but its
211     * {@link org.apache.wiki.auth.user.UserProfile#isNew()} method returns
212     * <code>false</code>, this method throws an {@link IllegalStateException}.
213     * This is meant as a quality check for UserDatabase providers;
214     * it should only be thrown if the implementation is faulty.</p>
215     * @param session the wiki session, which may not be <code>null</code>
216     * @return the user's profile, which will be newly initialized if the user
217     * is anonymous or asserted, or if the user cannot be found in the user
218     * database
219     */
220    public UserProfile getUserProfile( WikiSession session )
221    {
222        // Look up cached user profile
223        UserProfile profile = m_profiles.get( session );
224        boolean newProfile = profile == null;
225        Principal user = null;
226
227        // If user is authenticated, figure out if this is an existing profile
228        if ( session.isAuthenticated() )
229        {
230            user = session.getUserPrincipal();
231            try
232            {
233                profile = getUserDatabase().find( user.getName() );
234                newProfile = false;
235            }
236            catch( final NoSuchPrincipalException e )
237            {
238            }
239        }
240
241        if ( newProfile )
242        {
243            profile = getUserDatabase().newProfile();
244            if ( user != null )
245            {
246                profile.setLoginName( user.getName() );
247            }
248            if ( !profile.isNew() )
249            {
250                throw new IllegalStateException(
251                        "New profile should be marked 'new'. Check your UserProfile implementation." );
252            }
253        }
254
255        // Stash the profile for next time
256        m_profiles.put( session, profile );
257        return profile;
258    }

260    /**
261     * <p>
262     * Saves the {@link org.apache.wiki.auth.user.UserProfile}for the user in
263     * a wiki session. This method verifies that a user profile to be saved
264     * doesn't collide with existing profiles; that is, the login name
265     * or full name is already used by another profile. If the profile
266     * collides, a <code>DuplicateUserException</code> is thrown. After saving
267     * the profile, the user database changes are committed, and the user's
268     * credential set is refreshed; if custom authentication is used, this means
269     * the user will be automatically be logged in.
270     * </p>
271     * <p>
272     * When the user's profile is saved successfully, this method fires a
273     * {@link WikiSecurityEvent#PROFILE_SAVE} event with the WikiSession as the
274     * source and the UserProfile as target. For existing profiles, if the
275     * user's full name changes, this method also fires a "name changed"
276     * event ({@link WikiSecurityEvent#PROFILE_NAME_CHANGED}) with the
277     * WikiSession as the source and an array containing the old and new
278     * UserProfiles, respectively. The <code>NAME_CHANGED</code> event allows
279     * the GroupManager and PageManager can change group memberships and
280     * ACLs if needed.
281     * </p>
282     * <p>
283     * Note that WikiSessions normally attach event listeners to the
284     * UserManager, so changes to the profile will automatically cause the
285     * correct Principals to be reloaded into the current WikiSession's Subject.
286     * </p>
287     * @param session the wiki session, which may not be <code>null</code>
288     * @param profile the user profile, which may not be <code>null</code>
289     * @throws DuplicateUserException if the proposed profile's login name or full name collides with another
290     * @throws WikiException if the save fails for some reason. If the current user does not have
291     * permission to save the profile, this will be a {@link org.apache.wiki.auth.WikiSecurityException};
292     * if if the user profile must be approved before it can be saved, it will be a
293     * {@link org.apache.wiki.workflow.DecisionRequiredException}. All other WikiException
294     * indicate a condition that is not normal is probably due to mis-configuration
295     */
296    public void setUserProfile( WikiSession session, UserProfile profile ) throws DuplicateUserException, WikiException
297    {
298        // Verify user is allowed to save profile!
299        final Permission p = new WikiPermission( m_engine.getApplicationName(), WikiPermission.EDIT_PROFILE_ACTION );
300        if ( !m_engine.getAuthorizationManager().checkPermission( session, p ) )
301        {
302            throw new WikiSecurityException( "You are not allowed to save wiki profiles." );
303        }
304
305        // Check if profile is new, and see if container allows creation
306        final boolean newProfile = profile.isNew();
307
308        // Check if another user profile already has the fullname or loginname
309        final UserProfile oldProfile = getUserProfile( session );
310        final boolean nameChanged = ( oldProfile == null  || oldProfile.getFullname() == null )
311            ? false
312            : !( oldProfile.getFullname().equals( profile.getFullname() ) &&
313                 oldProfile.getLoginName().equals( profile.getLoginName() ) );
314        UserProfile otherProfile;
315        try
316        {
317            otherProfile = getUserDatabase().findByLoginName( profile.getLoginName() );
318            if ( otherProfile != null && !otherProfile.equals( oldProfile ) )
319            {
320                throw new DuplicateUserException( "security.error.login.taken", profile.getLoginName() );
321            }
322        }
323        catch( final NoSuchPrincipalException e )
324        {
325        }
326        try
327        {
328            otherProfile = getUserDatabase().findByFullName( profile.getFullname() );
329            if ( otherProfile != null && !otherProfile.equals( oldProfile ) )
330            {
331                throw new DuplicateUserException( "security.error.fullname.taken", profile.getFullname() );
332            }
333        }
334        catch( final NoSuchPrincipalException e )
335        {
336        }
337
338        // For new accounts, create approval workflow for user profile save.
339        if ( newProfile && oldProfile != null && oldProfile.isNew() )
340        {
341            final WorkflowBuilder builder = WorkflowBuilder.getBuilder( m_engine );
342            final Principal submitter = session.getUserPrincipal();
343            final Task completionTask = new SaveUserProfileTask( m_engine, session.getLocale() );
344
345            // Add user profile attribute as Facts for the approver (if required)
346            final boolean hasEmail = profile.getEmail() != null;
347            final Fact[] facts = new Fact[ hasEmail ? 4 : 3];
348            facts[0] = new Fact( PREFS_FULL_NAME, profile.getFullname() );
349            facts[1] = new Fact( PREFS_LOGIN_NAME, profile.getLoginName() );
350            facts[2] = new Fact( FACT_SUBMITTER, submitter.getName() );
351            if ( hasEmail )
352            {
353                facts[3] = new Fact( PREFS_EMAIL, profile.getEmail() );
354            }
355            final Workflow workflow = builder.buildApprovalWorkflow( submitter,
356                                                               SAVE_APPROVER,
357                                                               null,
358                                                               SAVE_DECISION_MESSAGE_KEY,
359                                                               facts,
360                                                               completionTask,
361                                                               null );
362
363            workflow.setAttribute( SAVED_PROFILE, profile );
364            m_engine.getWorkflowManager().start(workflow);
365
366            final boolean approvalRequired = workflow.getCurrentStep() instanceof Decision;
367
368            // If the profile requires approval, redirect user to message page
369            if ( approvalRequired )
370            {
371                throw new DecisionRequiredException( "This profile must be approved before it becomes active" );
372            }
373
374            // If the profile doesn't need approval, then just log the user in
375
376            try
377            {
378                final AuthenticationManager mgr = m_engine.getAuthenticationManager();
379                if ( newProfile && !mgr.isContainerAuthenticated() )
380                {
381                    mgr.login( session, null, profile.getLoginName(), profile.getPassword() );
382                }
383            }
384            catch ( final WikiException e )
385            {
386                throw new WikiSecurityException( e.getMessage(), e );
387            }
388
389            // Alert all listeners that the profile changed...
390            // ...this will cause credentials to be reloaded in the wiki session
391            fireEvent( WikiSecurityEvent.PROFILE_SAVE, session, profile );
392        }
393
394        // For existing accounts, just save the profile
395        else
396        {
397            // If login name changed, rename it first
398            if ( nameChanged && oldProfile != null && !oldProfile.getLoginName().equals( profile.getLoginName() ) )
399            {
400                getUserDatabase().rename( oldProfile.getLoginName(), profile.getLoginName() );
401            }
402
403            // Now, save the profile (userdatabase will take care of timestamps for us)
404            getUserDatabase().save( profile );
405
406            if ( nameChanged )
407            {
408                // Fire an event if the login name or full name changed
409                final UserProfile[] profiles = new UserProfile[] { oldProfile, profile };
410                fireEvent( WikiSecurityEvent.PROFILE_NAME_CHANGED, session, profiles );
411            }
412            else
413            {
414                // Fire an event that says we have new a new profile (new principals)
415                fireEvent( WikiSecurityEvent.PROFILE_SAVE, session, profile );
416            }
417        }
418    }

420    /**
421     * <p> Extracts user profile parameters from the HTTP request and populates
422     * a UserProfile with them. The UserProfile will either be a copy of the
423     * user's existing profile (if one can be found), or a new profile (if not).
424     * The rules for populating the profile as as follows: </p> <ul> <li>If the
425     * <code>email</code> or <code>password</code> parameter values differ
426     * from those in the existing profile, the passed parameters override the
427     * old values.</li> <li>For new profiles, the user-supplied
428     * <code>fullname</code> parameter is always
429     * used; for existing profiles the existing value is used, and whatever
430     * value the user supplied is discarded. The wiki name is automatically
431     * computed by taking the full name and extracting all whitespace.</li>
432     * <li>In all cases, the
433     * created/last modified timestamps of the user's existing or new profile
434     * always override whatever values the user supplied.</li> <li>If
435     * container authentication is used, the login name property of the profile
436     * is set to the name of
437     * {@link org.apache.wiki.WikiSession#getLoginPrincipal()}. Otherwise,
438     * the value of the <code>loginname</code> parameter is used.</li> </ul>
439     * @param context the current wiki context
440     * @return a new, populated user profile
441     */
442    public UserProfile parseProfile( WikiContext context )
443    {
444        // Retrieve the user's profile (may have been previously cached)
445        final UserProfile profile = getUserProfile( context.getWikiSession() );
446        final HttpServletRequest request = context.getHttpRequest();
447
448        // Extract values from request stream (cleanse whitespace as needed)
449        String loginName = request.getParameter( PARAM_LOGINNAME );
450        String password = request.getParameter( PARAM_PASSWORD );
451        String fullname = request.getParameter( PARAM_FULLNAME );
452        String email = request.getParameter( PARAM_EMAIL );
453        loginName = InputValidator.isBlank( loginName ) ? null : loginName;
454        password = InputValidator.isBlank( password ) ? null : password;
455        fullname = InputValidator.isBlank( fullname ) ? null : fullname;
456        email = InputValidator.isBlank( email ) ? null : email;
457
458        // A special case if we have container authentication
459        // If authenticated, login name is always taken from container
460        if ( m_engine.getAuthenticationManager().isContainerAuthenticated() &&
461                context.getWikiSession().isAuthenticated() )
462        {
463            loginName = context.getWikiSession().getLoginPrincipal().getName();
464        }
465
466        // Set the profile fields!
467        profile.setLoginName( loginName );
468        profile.setEmail( email );
469        profile.setFullname( fullname );
470        profile.setPassword( password );
471        return profile;
472    }

474    /**
475     * Validates a user profile, and appends any errors to the session errors
476     * list. If the profile is new, the password will be checked to make sure it
477     * isn't null. Otherwise, the password is checked for length and that it
478     * matches the value of the 'password2' HTTP parameter. Note that we have a
479     * special case when container-managed authentication is used and the user
480     * is not authenticated; this will always cause validation to fail. Any
481     * validation errors are added to the wiki session's messages collection
482     * (see {@link WikiSession#getMessages()}.
483     * @param context the current wiki context
484     * @param profile the supplied UserProfile
485     */
486    public void validateProfile( WikiContext context, UserProfile profile )
487    {
488        final boolean isNew = profile.isNew();
489        final WikiSession session = context.getWikiSession();
490        final InputValidator validator = new InputValidator( SESSION_MESSAGES, context );
491        final ResourceBundle rb = Preferences.getBundle( context, InternationalizationManager.CORE_BUNDLE );
492
493        //
494        //  Query the SpamFilter first
495        //
496        final FilterManager fm = m_engine.getFilterManager();
497        final List<PageFilter> ls = fm.getFilterList();
498        for( final PageFilter pf : ls )
499        {
500            if( pf instanceof SpamFilter )
501            {
502                if( ((SpamFilter)pf).isValidUserProfile( context, profile ) == false )
503                {
504                    session.addMessage( SESSION_MESSAGES, "Invalid userprofile" );
505                    return;
506                }
507                break;
508            }
509        }
510
511        // If container-managed auth and user not logged in, throw an error
512        if ( m_engine.getAuthenticationManager().isContainerAuthenticated()
513             && !context.getWikiSession().isAuthenticated() )
514        {
515            session.addMessage( SESSION_MESSAGES, rb.getString("security.error.createprofilebeforelogin") );
516        }
517
518        validator.validateNotNull( profile.getLoginName(), rb.getString("security.user.loginname") );
519        validator.validateNotNull( profile.getFullname(), rb.getString("security.user.fullname") );
520        validator.validate( profile.getEmail(), rb.getString("security.user.email"), InputValidator.EMAIL );
521
522        // If new profile, passwords must match and can't be null
523        if ( !m_engine.getAuthenticationManager().isContainerAuthenticated() )
524        {
525            final String password = profile.getPassword();
526            if ( password == null )
527            {
528                if ( isNew )
529                {
530                    session.addMessage( SESSION_MESSAGES, rb.getString("security.error.blankpassword") );
531                }
532            }
533            else
534            {
535                final HttpServletRequest request = context.getHttpRequest();
536                final String password2 = ( request == null ) ? null : request.getParameter( "password2" );
537                if ( !password.equals( password2 ) )
538                {
539                    session.addMessage( SESSION_MESSAGES, rb.getString("security.error.passwordnomatch") );
540                }
541            }
542        }
543
544        UserProfile otherProfile;
545        final String fullName = profile.getFullname();
546        final String loginName = profile.getLoginName();
547        final String email = profile.getEmail();
548
549        // It's illegal to use as a full name someone else's login name
550        try
551        {
552            otherProfile = getUserDatabase().find( fullName );
553            if ( otherProfile != null && !profile.equals( otherProfile ) && !fullName.equals( otherProfile.getFullname() ) )
554            {
555                final Object[] args = { fullName };
556                session.addMessage( SESSION_MESSAGES, MessageFormat.format( rb.getString("security.error.illegalfullname"), args ) );
557            }
558        }
559        catch ( final NoSuchPrincipalException e)
560        { /* It's clean */ }
561
562        // It's illegal to use as a login name someone else's full name
563        try
564        {
565            otherProfile = getUserDatabase().find( loginName );
566            if ( otherProfile != null && !profile.equals( otherProfile ) && !loginName.equals( otherProfile.getLoginName() ) )
567            {
568                final Object[] args = { loginName };
569                session.addMessage( SESSION_MESSAGES, MessageFormat.format( rb.getString("security.error.illegalloginname"), args ) );
570            }
571        }
572        catch ( final NoSuchPrincipalException e)
573        { /* It's clean */ }
574
575        // It's illegal to use multiple accounts with the same email
576        try
577        {
578            otherProfile = getUserDatabase().findByEmail( email );
579            if ( otherProfile != null
580                && !profile.getUid().equals(otherProfile.getUid()) // Issue JSPWIKI-1042
581                && !profile.equals( otherProfile ) && StringUtils.lowerCase( email ).equals( StringUtils.lowerCase(otherProfile.getEmail() ) ) )
582            {
583                final Object[] args = { email };
584                session.addMessage( SESSION_MESSAGES, MessageFormat.format( rb.getString("security.error.email.taken"), args ) );
585            }
586        }
587        catch ( final NoSuchPrincipalException e)
588        { /* It's clean */ }
589    }

741    /**
742     * Inner class that handles the actual profile save action. Instances
743     * of this class are assumed to have been added to an approval workflow via
744     * {@link org.apache.wiki.workflow.WorkflowBuilder#buildApprovalWorkflow(Principal, String, Task, String, org.apache.wiki.workflow.Fact[], Task, String)};
745     * they will not function correctly otherwise.
746     *
747     */
748    public static class SaveUserProfileTask extends Task
749    {
750        private static final long serialVersionUID = 6994297086560480285L;
751        private final UserDatabase m_db;
752        private final WikiEngine m_engine;
753        private final Locale m_loc;
754
755        /**
756         * Constructs a new Task for saving a user profile.
757         * @param engine the wiki engine
758         * @deprecated will be removed in 2.10 scope. Consider using
759         * {@link #UserManager.SaveUserProfileTask(WikiEngine, Locale)} instead
760         */
761        @Deprecated
762        public SaveUserProfileTask( WikiEngine engine )
763        {
764            super( SAVE_TASK_MESSAGE_KEY );
765            m_engine = engine;
766            m_db = engine.getUserManager().getUserDatabase();
767            m_loc = null;
768        }
769
770        public SaveUserProfileTask( WikiEngine engine, Locale loc )
771        {
772            super( SAVE_TASK_MESSAGE_KEY );
773            m_engine = engine;
774            m_db = engine.getUserManager().getUserDatabase();
775            m_loc = loc;
776        }
777
778        /**
779         * Saves the user profile to the user database.
780         * @return {@link org.apache.wiki.workflow.Outcome#STEP_COMPLETE} if the
781         * task completed successfully
782         * @throws WikiException if the save did not complete for some reason
783         */
784        @Override
785        public Outcome execute() throws WikiException
786        {
787            // Retrieve user profile
788            final UserProfile profile = (UserProfile) getWorkflow().getAttribute( SAVED_PROFILE );
789
790            // Save the profile (userdatabase will take care of timestamps for us)
791            m_db.save( profile );
792
793            // Send e-mail if user supplied an e-mail address
794            if ( profile.getEmail() != null )
795            {
796                try
797                {
798                    final InternationalizationManager i18n = m_engine.getInternationalizationManager();
799                    final String app = m_engine.getApplicationName();
800                    final String to = profile.getEmail();
801                    final String subject = i18n.get( InternationalizationManager.DEF_TEMPLATE, m_loc,
802                                               "notification.createUserProfile.accept.subject", app );
803
804                    final String content = i18n.get( InternationalizationManager.DEF_TEMPLATE, m_loc,
805                                               "notification.createUserProfile.accept.content", app,
806                                               profile.getLoginName(),
807                                               profile.getFullname(),
808                                               profile.getEmail(),
809                                               m_engine.getURL( WikiContext.LOGIN, null, null, true ) );
810                    MailUtil.sendMessage( m_engine.getWikiProperties(), to, subject, content);
811                }
812                catch ( final AddressException e)
813                {
814                }
815                catch ( final MessagingException me )
816                {
817                    log.error( "Could not send registration confirmation e-mail. Is the e-mail server running?", me );
818                }
819            }
820
821            return Outcome.STEP_COMPLETE;
822        }
823    }

864    /**
865     *  Implements the JSON API for usermanager.
866     *  <p>
867     *  Even though this gets serialized whenever container shuts down/restarts,
868     *  this gets reinstalled to the session when JSPWiki starts.  This means
869     *  that it's not actually necessary to save anything.
870     */
871    public static final class JSONUserModule implements WikiAjaxServlet
872    {
873        private volatile UserManager m_manager;
874
875        /**
876         *  Create a new JSONUserModule.
877         *  @param mgr Manager
878         */
879        public JSONUserModule( UserManager mgr )
880        {
881            m_manager = mgr;
882        }
883
884        @Override
885        public String getServletMapping() {
886            return JSON_USERS;
887        }
888
889        @Override
890        public void service(HttpServletRequest req, HttpServletResponse resp, String actionName, List<String> params) throws ServletException, IOException {
891            try {
892                String uid = null;
893                if (params.size()<1) {
894                    return;
895                }
896                uid = params.get(0);
897                log.debug("uid="+uid);
898                if (StringUtils.isNotBlank(uid)) {
899                    final UserProfile prof = getUserInfo(uid);
900                    resp.getWriter().write(AjaxUtil.toJson(prof));
901                }
902            } catch (final NoSuchPrincipalException e) {
903                throw new ServletException(e);
904            }
905        }
906
907        /**
908         *  Directly returns the UserProfile object attached to an uid.
909         *
910         *  @param uid The user id (e.g. WikiName)
911         *  @return A UserProfile object
912         *  @throws NoSuchPrincipalException If such a name does not exist.
913         */
914        public UserProfile getUserInfo( String uid )
915            throws NoSuchPrincipalException
916        {
917            if( m_manager != null )
918            {
919                final UserProfile prof = m_manager.getUserDatabase().find( uid );
920
921                return prof;
922            }
923
924            throw new IllegalStateException("The manager is offline.");
925        }
926    }
