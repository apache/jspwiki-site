019package org.apache.wiki;
020
021import org.apache.log4j.Logger;
022import org.apache.wiki.api.core.Engine;
023import org.apache.wiki.event.WikiEngineEvent;
024import org.apache.wiki.event.WikiEvent;
025import org.apache.wiki.event.WikiEventListener;
026
027
028/**
029 * Abstract Thread subclass that operates in the background; when it detects the {@link WikiEngineEvent#SHUTDOWN} event,
030 * it terminates itself. Subclasses of this method need only implement the method {@link #backgroundTask()}, instead of
031 * the normal {@link Thread#run()}, and provide a constructor that passes the Engine and sleep interval. This class is thread-safe.
032 */
033public abstract class WikiBackgroundThread extends Thread implements WikiEventListener {
034    
035    private static final Logger LOG = Logger.getLogger( WikiBackgroundThread.class );
036    private volatile boolean m_killMe;
037    private final Engine m_engine;
038    private final int m_interval;
039    private static final long POLLING_INTERVAL = 1_000L;
040    
041    /**
042     * Constructs a new instance of this background thread with a specified sleep interval, and adds the new instance 
043     * to the wiki engine's event listeners.
044     * 
045     * @param engine the wiki engine
046     * @param sleepInterval the interval between invocations of
047     * the thread's {@link Thread#run()} method, in seconds
048     */
049    public WikiBackgroundThread( final Engine engine, final int sleepInterval ) {
050        super();
051        m_engine = engine;
052        m_interval = sleepInterval;
053        engine.addWikiEventListener( this );
054        setDaemon( false );
055    }
056    
057    /**
058     * Listens for {@link org.apache.wiki.event.WikiEngineEvent#SHUTDOWN} and, if detected, marks the thread for death.
059     * 
060     * @param event {@inheritDoc}
061     * @see org.apache.wiki.event.WikiEventListener#actionPerformed(org.apache.wiki.event.WikiEvent)
062     */
063    @Override
064    public final void actionPerformed( final WikiEvent event ) {
065        if ( event instanceof WikiEngineEvent ) {
066            if ( event.getType() == WikiEngineEvent.SHUTDOWN ) {
067                LOG.warn( "Detected wiki engine shutdown: killing " + getName() + "." );
068                m_killMe = true;
069            }
070        }
071    }
072    
073    /**
074     * Abstract method that performs the actual work for this background thread; subclasses must implement this method.
075     * 
076     * @throws Exception Any exception can be thrown
077     */
078    public abstract void backgroundTask() throws Exception;
079    
080    /**
081     * Returns the Engine that created this background thread.
082     * 
083     * @return the wiki engine
084     */
085    public Engine getEngine() {
086        return m_engine;
087    }
088    
089    /**
090     *  Requests the shutdown of this background thread.  Note that the shutdown is not immediate.
091     *  
092     *  @since 2.4.92
093     */
094    public void shutdown() {
095        m_killMe = true;
096    }
097    
098    /**
099     * Runs the background thread's {@link #backgroundTask()} method at the interval specified at construction.
100     * The thread will initially pause for a full sleep interval before starting, after which it will execute 
101     * {@link #startupTask()}. This method will cleanly terminate the thread if it has previously been marked as 
102     * dead, before which it will execute {@link #shutdownTask()}. If any of the three methods return an exception, 
103     * it will be re-thrown as a {@link org.apache.wiki.InternalWikiException}.
104     * 
105     * @see java.lang.Thread#run()
106     */
107    @Override
108    public final void run() {
109        try {
110            // Perform the initial startup task
111            final String name = getName();
112            LOG.warn( "Starting up background thread: " + name + ".");
113            startupTask();
114            
115            // Perform the background task; check every second for thread death
116            while( !m_killMe ) {
117                // Perform the background task
118                // log.debug( "Running background task: " + name + "." );
119                backgroundTask();
120                
121                // Sleep for the interval we're supposed to, but wake up every POLLING_INTERVAL to see if thread should die
122                boolean interrupted = false;
123                try {
124                    for( int i = 0; i < m_interval; i++ ) {
125                        Thread.sleep( POLLING_INTERVAL );
126                        if( m_killMe ) {
127                            interrupted = true;
128                            LOG.warn( "Interrupted background thread: " + name + "." );
129                            break;
130                        }
131                    }
132                    if( interrupted ) {
133                        break;
134                    }
135                } catch( final Throwable t ) {
136                    LOG.error( "Background thread error: (stack trace follows)", t );
137                }
138            }
139            
140            // Perform the shutdown task
141            shutdownTask();
142        } catch( final Throwable t ) {
143            LOG.error( "Background thread error: (stack trace follows)", t );
144            throw new InternalWikiException( t.getMessage() ,t );
145        }
146    }
147    
148    /**
149     * Executes a task after shutdown signal was detected. By default, this method does nothing; override it 
150     * to implement custom functionality.
151     * 
152     * @throws Exception Any exception can be thrown.
153     */
154    public void shutdownTask() throws Exception {
155    }
156    
157    /**
158     * Executes a task just after the thread's {@link Thread#run()} method starts, but before the {@link #backgroundTask()} task executes.
159     * By default, this method does nothing; override it to implement custom functionality.
160     * 
161     * @throws Exception Any exception can be thrown.
162     */
163    public void startupTask() throws Exception {
164    }
165    
166}
