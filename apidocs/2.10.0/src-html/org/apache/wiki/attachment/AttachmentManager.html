093        /**
094         *  Creates a new AttachmentManager.  Note that creation will never fail,
095         *  but it's quite likely that attachments do not function.
096         *  <p>
097         *  <b>DO NOT CREATE</b> an AttachmentManager on your own, unless you really
098         *  know what you're doing.  Just use WikiEngine.getAttachmentManager() if
099         *  you're making a module for JSPWiki.
100         *
101         *  @param engine The wikiengine that owns this attachment manager.
102         *  @param props  A list of properties from which the AttachmentManager will seek
103         *  its configuration.  Typically this is the "jspwiki.properties".
104         */
105    
106        // FIXME: Perhaps this should fail somehow.
107        public AttachmentManager( WikiEngine engine, Properties props )
108        {
109            String classname;
110    
111            m_engine = engine;
112    
113    
114            //
115            //  If user wants to use a cache, then we'll use the CachingProvider.
116            //
117            boolean useCache = "true".equals(props.getProperty( PageManager.PROP_USECACHE ));
118    
119            if( useCache )
120            {
121                classname = "org.apache.wiki.providers.CachingAttachmentProvider";
122            }
123            else
124            {
125                classname = props.getProperty( PROP_PROVIDER );
126            }
127    
128            //
129            //  If no class defined, then will just simply fail.
130            //
131            if( classname == null )
132            {
133                log.info( "No attachment provider defined - disabling attachment support." );
134                return;
135            }
136    
137            //
138            //  Create and initialize the provider.
139            //
140            try {
141                if (m_cacheManager.cacheExists(CACHE_NAME)) {
142                    m_dynamicAttachments = m_cacheManager.getCache(CACHE_NAME);
143                } else {
144                    log.info("cache with name " + CACHE_NAME + " not found in ehcache.xml, creating it with defaults.");
145                    m_dynamicAttachments = new Cache(CACHE_NAME, DEFAULT_CACHECAPACITY, false, false, 0, 0);
146                    m_cacheManager.addCache(m_dynamicAttachments);
147                }
148    
149                Class<?> providerclass = ClassUtil.findClass("org.apache.wiki.providers", classname);
150    
151                m_provider = (WikiAttachmentProvider) providerclass.newInstance();
152    
153                m_provider.initialize(m_engine, props);
154            } catch( ClassNotFoundException e )
155            {
156                log.error( "Attachment provider class not found",e);
157            }
158            catch( InstantiationException e )
159            {
160                log.error( "Attachment provider could not be created", e );
161            }
162            catch( IllegalAccessException e )
163            {
164                log.error( "You may not access the attachment provider class", e );
165            }
166            catch( NoRequiredPropertyException e )
167            {
168                log.error( "Attachment provider did not find a property that it needed: "+e.getMessage(), e );
169                m_provider = null; // No, it did not work.
170            }
171            catch( IOException e )
172            {
173                log.error( "Attachment provider reports IO error", e );
174                m_provider = null;
175            }
176        }

238        /**
239         *  Figures out the full attachment name from the context and
240         *  attachment name.
241         *
242         *  @param context The current WikiContext
243         *  @param attachmentname The file name of the attachment.
244         *  @param version A particular version.
245         *  @return Attachment, or null, if no such attachment or version exists.
246         *  @throws ProviderException If something goes wrong.
247         */
248    
249        public Attachment getAttachmentInfo( WikiContext context,
250                                             String attachmentname,
251                                             int version )
252            throws ProviderException
253        {
254            if( m_provider == null )
255            {
256                return null;
257            }
258    
259            WikiPage currentPage = null;
260    
261            if( context != null )
262            {
263                currentPage = context.getPage();
264            }
265    
266            //
267            //  Figure out the parent page of this attachment.  If we can't find it,
268            //  we'll assume this refers directly to the attachment.
269            //
270            int cutpt = attachmentname.lastIndexOf('/');
271    
272            if( cutpt != -1 )
273            {
274                String parentPage = attachmentname.substring(0,cutpt);
275                parentPage = MarkupParser.cleanLink( parentPage );
276                attachmentname = attachmentname.substring(cutpt+1);
277    
278                // If we for some reason have an empty parent page name;
279                // this can't be an attachment
280                if(parentPage.length() == 0) return null;
281    
282                currentPage = m_engine.getPage( parentPage );
283    
284                //
285                // Go check for legacy name
286                //
287                // FIXME: This should be resolved using CommandResolver,
288                //        not this adhoc way.  This also assumes that the
289                //        legacy charset is a subset of the full allowed set.
290                if( currentPage == null )
291                {
292                    currentPage = m_engine.getPage( MarkupParser.wikifyLink( parentPage ) );
293                }
294            }
295    
296            //
297            //  If the page cannot be determined, we cannot possibly find the
298            //  attachments.
299            //
300            if( currentPage == null || currentPage.getName().length() == 0 )
301            {
302                return null;
303            }
304    
305            // System.out.println("Seeking info on "+currentPage+"::"+attachmentname);
306    
307            //
308            //  Finally, figure out whether this is a real attachment or a generated
309            //  attachment.
310            //
311            Attachment att;
312    
313            att = getDynamicAttachment( currentPage.getName()+"/"+attachmentname );
314    
315            if( att == null )
316            {
317                att = m_provider.getAttachmentInfo( currentPage, attachmentname, version );
318            }
319    
320            return att;
321        }

323        /**
324         *  Returns the list of attachments associated with a given wiki page.
325         *  If there are no attachments, returns an empty Collection.
326         *
327         *  @param wikipage The wiki page from which you are seeking attachments for.
328         *  @return a valid collection of attachments.
329         *  @throws ProviderException If there was something wrong in the backend.
330         */
331    
332        // FIXME: This API should be changed to return a List.
333        @SuppressWarnings("unchecked")
334        public Collection listAttachments( WikiPage wikipage )
335            throws ProviderException
336        {
337            if( m_provider == null )
338            {
339                return new ArrayList();
340            }
341    
342            Collection atts = m_provider.listAttachments( wikipage );
343    
344            //
345            //  This is just a sanity check; all of our providers return a Collection.
346            //
347            if( atts instanceof List )
348            {
349                m_engine.getPageSorter().sortPages( (List) atts );
350            }
351    
352            return atts;
353        }

355        /**
356         *  Returns true, if the page has any attachments at all.  This is
357         *  a convinience method.
358         *
359         *
360         *  @param wikipage The wiki page from which you are seeking attachments for.
361         *  @return True, if the page has attachments, else false.
362         */
363        public boolean hasAttachments( WikiPage wikipage )
364        {
365            try
366            {
367                return listAttachments( wikipage ).size() > 0;
368            }
369            catch( Exception e ) {}
370    
371            return false;
372        }

390        /**
391         *  Returns an attachment stream using the particular WikiContext.  This method
392         *  should be used instead of getAttachmentStream(Attachment), since it also allows
393         *  the DynamicAttachments to function.
394         *
395         *  @param ctx The Wiki Context
396         *  @param att The Attachment to find
397         *  @return An InputStream.  May return null, if attachments are disabled.  You must
398         *          take care of closing it.
399         *  @throws ProviderException If the backend fails due to some reason
400         *  @throws IOException If the stream cannot be opened
401         */
402        public InputStream getAttachmentStream( WikiContext ctx, Attachment att )
403            throws ProviderException, IOException
404        {
405            if( m_provider == null )
406            {
407                return null;
408            }
409    
410            if( att instanceof DynamicAttachment )
411            {
412                return ((DynamicAttachment)att).getProvider().getAttachmentData( ctx, att );
413            }
414    
415            return m_provider.getAttachmentData( att );
416        }

432        /**
433         *  Finds a DynamicAttachment.  Normally, you should just use getAttachmentInfo(),
434         *  since that will find also DynamicAttachments.
435         *
436         *  @param name The name of the attachment to look for
437         *  @return An Attachment, or null.
438         *  @see #getAttachmentInfo(String)
439         */
440    
441        public DynamicAttachment getDynamicAttachment(String name) {
442            Element element = m_dynamicAttachments.get(name);
443            if (element != null) {
444                return (DynamicAttachment) element.getObjectValue();
445            } else {
446                //
447                //  Remove from cache, it has expired.
448                //
449                m_dynamicAttachments.put(new Element(name, null));
450    
451                return null;
452            }
453        }

455        /**
456         *  Stores an attachment that lives in the given file.
457         *  If the attachment did not exist previously, this method
458         *  will create it.  If it did exist, it stores a new version.
459         *
460         *  @param att Attachment to store this under.
461         *  @param source A file to read from.
462         *
463         *  @throws IOException If writing the attachment failed.
464         *  @throws ProviderException If something else went wrong.
465         */
466        public void storeAttachment( Attachment att, File source )
467            throws IOException,
468                   ProviderException
469        {
470            FileInputStream in = null;
471    
472            try
473            {
474                in = new FileInputStream( source );
475                storeAttachment( att, in );
476            }
477            finally
478            {
479                if( in != null ) in.close();
480            }
481        }
482    
483        /**
484         *  Stores an attachment directly from a stream.
485         *  If the attachment did not exist previously, this method
486         *  will create it.  If it did exist, it stores a new version.
487         *
488         *  @param att Attachment to store this under.
489         *  @param in  InputStream from which the attachment contents will be read.
490         *
491         *  @throws IOException If writing the attachment failed.
492         *  @throws ProviderException If something else went wrong.
493         */
494        public void storeAttachment( Attachment att, InputStream in )
495            throws IOException,
496                   ProviderException
497        {
498            if( m_provider == null )
499            {
500                return;
501            }
502    
503            //
504            //  Checks if the actual, real page exists without any modifications
505            //  or aliases.  We cannot store an attachment to a non-existent page.
506            //
507            if( !m_engine.getPageManager().pageExists( att.getParentName() ) )
508            {
509                // the caller should catch the exception and use the exception text as an i18n key
510                throw new ProviderException(  "attach.parent.not.exist"  );
511            }
512            
513            m_provider.putAttachmentData( att, in );
514    
515            m_engine.getReferenceManager().updateReferences( att.getName(),
516                                                             new java.util.Vector() );
517    
518            WikiPage parent = new WikiPage( m_engine, att.getParentName() );
519            m_engine.updateReferences( parent );
520    
521            m_engine.getSearchManager().reindexPage( att );
522        }

613        /**
614         *  Validates the filename and makes sure it is legal.  It trims and splits
615         *  and replaces bad characters.
616         *  
617         *  @param filename
618         *  @return A validated name with annoying characters replaced.
619         *  @throws WikiException If the filename is not legal (e.g. empty)
620         */
621        static String validateFileName( String filename )
622            throws WikiException
623        {
624            if( filename == null || filename.trim().length() == 0 )
625            {
626                log.error("Empty file name given.");
627        
628                // the caller should catch the exception and use the exception text as an i18n key
629                throw new WikiException(  "attach.empty.file" );
630            }
631        
632            //
633            //  Should help with IE 5.22 on OSX
634            //
635            filename = filename.trim();
636    
637            // If file name ends with .jsp or .jspf, the user is being naughty!
638            if( filename.toLowerCase().endsWith( ".jsp" ) || filename.toLowerCase().endsWith(".jspf") )
639            {
640                log.info( "Attempt to upload a file with a .jsp/.jspf extension.  In certain cases this " +
641                          "can trigger unwanted security side effects, so we're preventing it." );
642                //
643                // the caller should catch the exception and use the exception text as an i18n key
644                throw new WikiException(  "attach.unwanted.file"  );
645            }
646        
647            //
648            //  Some browser send the full path info with the filename, so we need
649            //  to remove it here by simply splitting along slashes and then taking the path.
650            //
651            
652            String[] splitpath = filename.split( "[/\\\\]" );
653            filename = splitpath[splitpath.length-1];
654            
655            //
656            //  Remove any characters that might be a problem. Most
657            //  importantly - characters that might stop processing
658            //  of the URL.
659            //
660            filename = StringUtils.replaceChars( filename, "#?\"'", "____" );
661        
662            return filename;
663        }
