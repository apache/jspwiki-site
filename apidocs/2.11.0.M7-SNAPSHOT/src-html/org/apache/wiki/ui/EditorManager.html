019package org.apache.wiki.ui;
020
021import org.apache.wiki.WikiContext;
022import org.apache.wiki.modules.WikiModuleInfo;
023
024import javax.servlet.jsp.PageContext;
025import java.util.Collection;
026import java.util.Properties;
027
028
029/**
030 *  Defines an editor manager.  An editor can be added by adding a suitable JSP file under templates/default/editors
031 *  If you want your editor to include any scripts or something, you can simply request it by adding the following in your
032 *  {@code ini/jspwiki_module.xml}:
033 *
034 *  <pre>
035 *  &lt;modules>
036 *   &lt;editor name="myeditor">
037 *       &lt;author>Janne Jalkanen&lt;/author>
038 *       &lt;script>foo.js&lt;/script>
039 *       &lt;stylesheet>foo.css&lt;/stylesheet>
040 *       &lt;path>editors/myeditor.jsp&lt;/path>
041 *   &lt;/editor>
042 *  &lt;/modules>
043 *  </pre>
044 *
045 *  @since 2.4
046 */
047public interface EditorManager {
048
049    /** The property name for setting the editor. Current value is "jspwiki.editor" - not used anymore: replaced by defaultpref.template.editor */
050    String PROP_EDITORTYPE = "jspwiki.editor";
051
052    /** Parameter for changing editors at run-time */
053    String PARA_EDITOR = "editor";
054
055    /** Known name for the plain wikimarkup editor. */
056    String EDITOR_PLAIN = "plain";
057
058    /** Known name for the preview editor component. */
059    String EDITOR_PREVIEW = "preview";
060
061    /** Known attribute name for storing the user edited text inside a HTTP parameter. */
062    String REQ_EDITEDTEXT = "_editedtext";
063
064    /** Known attribute name for storing the user edited text inside a session or a page context */
065    String ATTR_EDITEDTEXT = REQ_EDITEDTEXT;
066
067    /**
068     *  Initializes the EditorManager.  It also registers any editors it can find.
069     *
070     *  @param props  Properties for setup.
071     */
072    void initialize( Properties props );
073
074    /**
075     *  Returns an editor for the current context.  The editor names are matched in a case insensitive manner.  At the moment, the only
076     *  place that this method looks in is the property file, but in the future this will also look at user preferences.
077     *  <p>
078     *  Determines the editor to use by the following order of conditions:
079     *  1. Editor set in User Preferences
080     *  2. Default Editor set in jspwiki.properties
081     *  <p>
082     *  For the PREVIEW context, this method returns the "preview" editor.
083     *
084     * @param context The context that is chosen.
085     * @return The name of the chosen editor. If no match could be found, will revert to the default "plain" editor.
086     */
087    String getEditorName( WikiContext context );
088
089    /**
090     *  Returns a list of editors as Strings of editor names.
091     *
092     *  @return the list of available editors
093     */
094    String[] getEditorList();
095
096    /**
097     *  Convenience method for getting the path to the editor JSP file.
098     *
099     *  @param context WikiContext from where the editor name is retrieved.
100     *  @return e.g. "editors/plain.jsp"
101     */
102    String getEditorPath( WikiContext context );
103
104    /**
105     *  Convenience function which examines the current context and attempts to figure out whether the edited text is in the HTTP
106     *  request parameters or somewhere in the session.
107     *
108     *  @param ctx the JSP page context
109     *  @return the edited text, if present in the session page context or as a parameter
110     */
111    static String getEditedText( final PageContext ctx ) {
112        String usertext = ctx.getRequest().getParameter( REQ_EDITEDTEXT );
113        if( usertext == null ) {
114            usertext = ( String )ctx.findAttribute( ATTR_EDITEDTEXT );
115        }
116
117        return usertext;
118    }
119
120    /**
121     * Returns a collection of modules currently managed by this ModuleManager.  Each
122     * entry is an instance of the WikiModuleInfo class.  This method should return something
123     * which is safe to iterate over, even if the underlying collection changes.
124     *
125     * @return A Collection of WikiModuleInfo instances.
126     */
127    Collection< WikiModuleInfo > modules();
128
129}
