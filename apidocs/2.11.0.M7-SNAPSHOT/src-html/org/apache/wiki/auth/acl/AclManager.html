019package org.apache.wiki.auth.acl;
020
021import org.apache.wiki.WikiPage;
022import org.apache.wiki.api.core.Engine;
023import org.apache.wiki.auth.WikiSecurityException;
024
025import java.util.Properties;
026
027/**
028 *  Specifies how to parse and return ACLs from wiki pages.
029 *
030 *  @since 2.3
031 */
032public interface AclManager {
033
034    /**
035     * Initializes the AclManager with a supplied wiki engine and properties.
036     *
037     * @param engine the wiki engine
038     * @param props the initialization properties
039     */
040    void initialize( Engine engine, Properties props );
041
042    /**
043     * A helper method for parsing textual AccessControlLists. The line is in form
044     * "(ALLOW) <permission><principal>, <principal>, <principal>". This method was moved from Authorizer.
045     *
046     * @param page The current wiki page. If the page already has an ACL, it will be used as a basis for this ACL in order to avoid the
047     *             creation of a new one.
048     * @param ruleLine The rule line, as described above.
049     * @return A valid Access Control List. May be empty.
050     * @throws WikiSecurityException if the ruleLine was faulty somehow.
051     * @since 2.1.121
052     */
053    Acl parseAcl( WikiPage page, String ruleLine ) throws WikiSecurityException;
054
055    /**
056     * Returns the access control list for the page. If the ACL has not been parsed yet, it is done on-the-fly. If the page has a
057     * parent page, then that is tried also. This method was moved from Authorizer; it was consolidated with some code from
058     * AuthorizationManager.
059     *
060     * @param page the wiki page
061     * @since 2.2.121
062     * @return the Acl representing permissions for the page
063     */
064    Acl getPermissions( WikiPage page );
065
066    /**
067     * Sets the access control list for the page and persists it.
068     *
069     * @param page the wiki page
070     * @param acl the access control list
071     * @since 2.5
072     * @throws WikiSecurityException if the ACL cannot be set or persisted
073     */
074    void setPermissions( WikiPage page, Acl acl ) throws WikiSecurityException;
075
076}
