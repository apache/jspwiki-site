019package org.apache.wiki.tags;
020
021import org.apache.wiki.api.core.Context;
022import org.apache.wiki.api.core.Engine;
023import org.apache.wiki.ui.Editor;
024import org.apache.wiki.ui.EditorManager;
025
026import java.util.ArrayList;
027import java.util.Collection;
028
029/**
030 *  Iterates through editors.
031 *
032 *  @since 2.4.12
033 */
034
035public class EditorIteratorTag extends IteratorTag  {
036
037    private static final long serialVersionUID = 0L;
038
039    /** {@inheritDoc} */
040    @Override
041    public final int doStartTag() {
042        m_wikiContext = Context.findContext(pageContext);
043        final Engine engine = m_wikiContext.getEngine();
044        final EditorManager mgr = engine.getManager( EditorManager.class );
045        final String[] editorList = mgr.getEditorList();
046        final Collection< Editor > editors = new ArrayList<>();
047
048        for( final String editor : editorList ) {
049            editors.add( new Editor( m_wikiContext, editor ) );
050        }
051        setList( editors );
052
053        return super.doStartTag();
054    }
055
056}
