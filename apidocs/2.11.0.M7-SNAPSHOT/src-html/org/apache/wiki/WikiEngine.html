098 *  Provides Wiki services to the JSP page.
099 *
100 *  <P>
101 *  This is the main interface through which everything should go.
102 *
103 *  <P>

105 *  by using the WikiEngine.getInstance() method.  Never create a new
106 *  WikiEngine() from scratch, unless you're writing tests.

334    // FIXME: Potential make-things-easier thingy here: no need to fetch the wikiengine anymore
335    //        Wiki.jsp.jspInit() [really old code]; it's probably even faster to fetch it
336    //        using this method every time than go to pageContext.getAttribute().
337
338    public static synchronized WikiEngine getInstance( final ServletContext context, Properties props ) throws InternalWikiException {
339        WikiEngine engine = ( WikiEngine )context.getAttribute( ATTR_WIKIENGINE );
340
341        if( engine == null ) {
342            final String appid = Integer.toString(context.hashCode()); //FIXME: Kludge, use real type.
343            context.log(" Assigning new engine to "+appid);
344            try {
345                if( props == null ) {
346                    props = PropertyReader.loadWebAppProps( context );
347                }
348
349                engine = new WikiEngine( context, appid, props );
350                context.setAttribute( ATTR_WIKIENGINE, engine );
351            } catch( final Exception e ) {
352                context.log( "ERROR: Failed to create a Wiki engine: "+e.getMessage() );
353                log.error( "ERROR: Failed to create a Wiki engine, stacktrace follows " , e);
354                throw new InternalWikiException( "No wiki engine, check logs." , e);
355            }
356
357        }
358
359        return engine;
360    }

374    /**
375     *  Instantiate using this method when you're running as a servlet and
376     *  WikiEngine will figure out where to look for the property
377     *  file.
378     *  Do not use this method - use WikiEngine.getInstance() instead.
379     *
380     *  @param context A ServletContext.
381     *  @param appid   An Application ID.  This application is an unique random string which
382     *                 is used to recognize this WikiEngine.
383     *  @param props   The WikiEngine configuration.
384     *  @throws WikiException If the WikiEngine construction fails.
385     */
386    protected WikiEngine( final ServletContext context, final String appid, final Properties props ) throws WikiException {
387        m_servletContext = context;
388        m_appid          = appid;
389
390        // Stash the WikiEngine in the servlet context
391        if ( context != null ) {
392            context.setAttribute( ATTR_WIKIENGINE,  this );
393            m_rootPath = context.getRealPath("/");
394        }
395
396        try {
397            //  Note: May be null, if JSPWiki has been deployed in a WAR file.
398            initialize( props );
399            log.info( "Root path for this Wiki is: '" + m_rootPath + "'" );
400        } catch( final Exception e ) {
401            final String msg = Release.APPNAME+": Unable to load and setup properties from jspwiki.properties. "+e.getMessage();
402            if ( context != null ) {
403                context.log( msg );
404            }
405            throw new WikiException( msg, e );
406        }
407    }

409    /**
410     *  Does all the real initialization.
411     */
412    private void initialize( final Properties props ) throws WikiException {
413        m_startTime  = new Date();
414        m_properties = props;
415
416        //
417        //  Initialize log4j.  However, make sure that we don't initialize it multiple times.
418        //  By default we load the log4j config statements from jspwiki.properties, unless
419        //  the property jspwiki.use.external.logconfig=true, in that case we let log4j figure out the
420        //  logging configuration.
421        //
422        if( !c_configured ) {
423            final String useExternalLogConfig = TextUtil.getStringProperty( props,"jspwiki.use.external.logconfig","false" );
424            if( useExternalLogConfig == null || useExternalLogConfig.equals( "false" ) ) {
425                PropertyConfigurator.configure( props );
426            }
427            c_configured = true;
428        }
429
430        log.info( "*******************************************" );
431        log.info( Release.APPNAME + " " + Release.getVersionString() + " starting. Whee!" );
432
433        fireEvent( WikiEngineEvent.INITIALIZING ); // begin initialization
434
435        log.debug("Java version: "+System.getProperty("java.runtime.version"));
436        log.debug("Java vendor: "+System.getProperty("java.vm.vendor"));
437        log.debug("OS: "+System.getProperty("os.name")+" "+System.getProperty("os.version")+" "+System.getProperty("os.arch"));
438        log.debug("Default server locale: "+Locale.getDefault());
439        log.debug("Default server timezone: "+TimeZone.getDefault().getDisplayName(true, TimeZone.LONG));
440
441        if( m_servletContext != null )
442        {
443            log.info("Servlet container: "+m_servletContext.getServerInfo() );
444            if( m_servletContext.getMajorVersion() < 2 ||
445                (m_servletContext.getMajorVersion() == 2 && m_servletContext.getMinorVersion() < 4) )
446            {
447                throw new InternalWikiException("I require a container which supports at least version 2.4 of Servlet specification");
448            }
449        }
450
451        log.debug("Configuring WikiEngine...");
452
453        //  Initializes the CommandResolver
454        m_commandResolver  = new CommandResolver( this, props );
455
456        //
457        //  Create and find the default working directory.
458        //
459        m_workDir = TextUtil.getStringProperty( props, PROP_WORKDIR, null );
460
461        if( m_workDir == null ) {
462            m_workDir = System.getProperty("java.io.tmpdir", ".");
463            m_workDir += File.separator+Release.APPNAME+"-"+m_appid;
464        }
465
466        try {
467            final File f = new File( m_workDir );
468            f.mkdirs();
469
470            //
471            //  A bunch of sanity checks
472            //
473            if( !f.exists() ) {
474                throw new WikiException("Work directory does not exist: "+m_workDir);
475            }
476            if( !f.canRead() ) {
477                throw new WikiException("No permission to read work directory: "+m_workDir);
478            }
479            if( !f.canWrite() ) {
480                throw new WikiException("No permission to write to work directory: "+m_workDir);
481            }
482            if( !f.isDirectory() ) {
483                throw new WikiException("jspwiki.workDir does not point to a directory: "+m_workDir);
484            }
485        } catch( final SecurityException e ) {
486            log.fatal( "Unable to find or create the working directory: "+m_workDir, e );
487            throw new IllegalArgumentException( "Unable to find or create the working dir: " + m_workDir, e );
488        }
489
490        log.info( "JSPWiki working directory is '" + m_workDir + "'" );
491
492        m_saveUserInfo   = TextUtil.getBooleanProperty( props, PROP_STOREUSERNAME, m_saveUserInfo );
493        m_useUTF8        = StandardCharsets.UTF_8.name().equals( TextUtil.getStringProperty( props, PROP_ENCODING, StandardCharsets.ISO_8859_1.name() ) );
494        m_beautifyTitle  = TextUtil.getBooleanProperty( props, PROP_BEAUTIFYTITLE, m_beautifyTitle );
495        m_templateDir    = TextUtil.getStringProperty( props, PROP_TEMPLATEDIR, "default" );
496        enforceValidTemplateDirectory();
497        m_frontPage      = TextUtil.getStringProperty( props, PROP_FRONTPAGE,   "Main" );
498
499        //
500        //  Initialize the important modules.  Any exception thrown by the
501        //  managers means that we will not start up.
502        //
503
504        // FIXME: This part of the code is getting unwieldy.  We must think
505        //        of a better way to do the startup-sequence.
506        try {
507            final Class< ? > urlclass = ClassUtil.findClass( "org.apache.wiki.url",
508                                                             TextUtil.getStringProperty( props, PROP_URLCONSTRUCTOR, "DefaultURLConstructor" ) );
509            m_urlConstructor = ( URLConstructor ) urlclass.getDeclaredConstructor().newInstance();
510            m_urlConstructor.initialize( this, props );
511
512            m_pageManager           = ClassUtil.getMappedObject( PageManager.class.getName(), this, props );
513            m_pluginManager         = ClassUtil.getMappedObject( PluginManager.class.getName(), this, props );
514            m_differenceManager     = ClassUtil.getMappedObject( DifferenceManager.class.getName(), this, props );
515            m_attachmentManager     = ClassUtil.getMappedObject( AttachmentManager.class.getName(), this, props );
516            m_variableManager       = ClassUtil.getMappedObject( VariableManager.class.getName(), props );
517            m_renderingManager      = ClassUtil.getMappedObject( RenderingManager.class.getName() );
518            m_searchManager         = ClassUtil.getMappedObject( SearchManager.class.getName(), this, props );
519            m_authenticationManager = ClassUtil.getMappedObject( AuthenticationManager.class.getName() );
520            m_authorizationManager  = ClassUtil.getMappedObject( AuthorizationManager.class.getName() );
521            m_userManager           = ClassUtil.getMappedObject( UserManager.class.getName() );
522            m_groupManager          = ClassUtil.getMappedObject( GroupManager.class.getName() );
523            m_editorManager         = ClassUtil.getMappedObject( EditorManager.class.getName(), this );
524            m_editorManager.initialize( props );
525
526            m_progressManager   = new ProgressManager();
527
528            // Initialize the authentication, authorization, user and acl managers
529            m_authenticationManager.initialize( this, props );
530            m_authorizationManager.initialize( this, props );
531            m_userManager.initialize( this, props );
532            m_groupManager.initialize( this, props );
533            m_aclManager = getAclManager();
534
535            // Start the Workflow manager
536            m_workflowMgr = ClassUtil.getMappedObject(WorkflowManager.class.getName());
537            m_workflowMgr.initialize(this, props);
538            m_tasksManager = ClassUtil.getMappedObject(TasksManager.class.getName());
539
540            m_internationalizationManager = ClassUtil.getMappedObject(InternationalizationManager.class.getName(),this);
541            m_templateManager = ClassUtil.getMappedObject(TemplateManager.class.getName(), this, props );
542
543            // Since we want to use a page filters initilize() method
544            // as a engine startup listener where we can initialize global event listeners,
545            // it must be called lastly, so that all object references in the engine
546            // are availabe to the initialize() method
547            m_filterManager = ClassUtil.getMappedObject(FilterManager.class.getName(), this, props );
548
549            m_adminBeanManager = ClassUtil.getMappedObject(AdminBeanManager.class.getName(),this);
550
551            // RenderingManager depends on FilterManager events.
552            m_renderingManager.initialize( this, props );
553
554            //
555            //  ReferenceManager has the side effect of loading all
556            //  pages.  Therefore after this point, all page attributes
557            //  are available.
558            //
559            //  initReferenceManager is indirectly using m_filterManager, therefore
560            //  it has to be called after it was initialized.
561            //
562            initReferenceManager();
563
564            //
565            //  Hook the different manager routines into the system.
566            //
567            m_filterManager.addPageFilter(m_referenceManager, -1001 );
568            m_filterManager.addPageFilter(m_searchManager, -1002 );
569        } catch( final RuntimeException e ) {
570            // RuntimeExceptions may occur here, even if they shouldn't.
571            log.fatal( "Failed to start managers.", e );
572            throw new WikiException( "Failed to start managers: " + e.getMessage(), e );
573        } catch( final ClassNotFoundException e ) {
574            log.fatal( "JSPWiki could not start, URLConstructor was not found: " + e.getMessage(), e );
575            throw new WikiException( e.getMessage(), e );
576        } catch( final InstantiationException e ) {
577            log.fatal( "JSPWiki could not start, URLConstructor could not be instantiated: " + e.getMessage(), e );
578            throw new WikiException( e.getMessage(), e );
579        } catch( final IllegalAccessException e ) {
580            log.fatal( "JSPWiki could not start, URLConstructor cannot be accessed: " + e.getMessage(), e );
581            throw new WikiException( e.getMessage(), e );
582        } catch( final Exception e ) {
583            // Final catch-all for everything
584            log.fatal( "JSPWiki could not start, due to an unknown exception when starting.",e );
585            throw new WikiException( "Failed to start. Caused by: " + e.getMessage() +
586                                     "; please check log files for better information.", e );
587        }
588
589        //
590        //  Initialize the good-to-have-but-not-fatal modules.
591        //
592        try {
593            if( TextUtil.getBooleanProperty( props, RSSGenerator.PROP_GENERATE_RSS,false ) ) {
594                m_rssGenerator = ClassUtil.getMappedObject( RSSGenerator.class.getName(), this, props );
595            }
596
597            m_pageRenamer = ClassUtil.getMappedObject( PageRenamer.class.getName(), this, props );
598        } catch( final Exception e ) {
599            log.error( "Unable to start RSS generator - JSPWiki will still work, but there will be no RSS feed.", e );
600        }
601
602        // Start the RSS generator & generator thread
603        if( m_rssGenerator != null ) {
604            m_rssFile = TextUtil.getStringProperty( props, RSSGenerator.PROP_RSSFILE, "rss.rdf" );
605            final File rssFile;
606            if( m_rssFile.startsWith( File.separator ) ) { // honor absolute pathnames:
607                rssFile = new File(m_rssFile );
608            } else { // relative path names are anchored from the webapp root path:
609                rssFile = new File( getRootPath(), m_rssFile );
610            }
611            final int rssInterval = TextUtil.getIntegerProperty( props, RSSGenerator.PROP_INTERVAL, 3600 );
612            final RSSThread rssThread = new RSSThread( this, rssFile, rssInterval );
613            rssThread.start();
614        }
615
616        fireEvent( WikiEngineEvent.INITIALIZED ); // initialization complete
617
618        log.info("WikiEngine configured.");
619        m_isConfigured = true;
620    }

622    /**
623     * Checks if the template directory specified in the wiki's properties actually exists. If it doesn't, then {@code m_templateDir} is
624     * set to {@link #DEFAULT_TEMPLATE_NAME}.
625     * <p>
626     * This checks the existence of the <tt>ViewTemplate.jsp</tt> file, which exists in every template using {@code m_servletContext.getRealPath("/")}.
627     * <p>
628     * {@code m_servletContext.getRealPath("/")} can return {@code null} on certain servers/conditions (f.ex, packed wars), an extra check
629     * against {@code m_servletContext.getResource} is made.
630     */
631    void enforceValidTemplateDirectory() {
632        if( m_servletContext != null ) {
633            final String viewTemplate = "templates" + File.separator + getTemplateDir() + File.separator + "ViewTemplate.jsp";
634            boolean exists = new File( m_servletContext.getRealPath("/") + viewTemplate ).exists();
635            if( !exists ) {
636                try {
637                    final URL url = m_servletContext.getResource( viewTemplate );
638                    exists = url != null && StringUtils.isNotEmpty( url.getFile() );
639                } catch( final MalformedURLException e ) {
640                    exists = false;
641                }
642            }
643            if( !exists ) {
644                log.warn( getTemplateDir() + " template not found, updating WikiEngine's default template to " + DEFAULT_TEMPLATE_NAME );
645                m_templateDir = DEFAULT_TEMPLATE_NAME;
646            }
647        }
648    }

1105    /**
1106     *  Helper method for doing the HTML translation.
1107     *
1108     *  @param context The WikiContext in which to do the conversion
1109     *  @param pagedata The data to render
1110     *  @param localLinkHook Is called whenever a wiki link is found
1111     *  @param extLinkHook   Is called whenever an external link is found
1112     *  @param parseAccessRules Parse the access rules if we encounter them
1113     *  @param justParse Just parses the pagedata, does not actually render.  In this case, this methods an empty string.
1114     *  @return HTML-rendered page text.
1115     */
1116    public String textToHTML( final WikiContext context,
1117                              String pagedata,
1118                              final StringTransmutator localLinkHook,
1119                              final StringTransmutator extLinkHook,
1120                              final StringTransmutator attLinkHook,
1121                              final boolean            parseAccessRules,
1122                              final boolean            justParse ) {
1123        String result = "";
1124
1125        if( pagedata == null ) {
1126            log.error("NULL pagedata to textToHTML()");
1127            return null;
1128        }
1129
1130        final boolean runFilters = "true".equals(m_variableManager.getValue(context,PROP_RUNFILTERS,"true"));
1131
1132        try {
1133            final StopWatch sw = new StopWatch();
1134            sw.start();
1135
1136            if( runFilters && m_filterManager != null ) {
1137                pagedata = m_filterManager.doPreTranslateFiltering( context, pagedata );
1138            }
1139
1140            final MarkupParser mp = m_renderingManager.getParser( context, pagedata );
1141            mp.addLocalLinkHook( localLinkHook );
1142            mp.addExternalLinkHook( extLinkHook );
1143            mp.addAttachmentLinkHook( attLinkHook );
1144
1145            if( !parseAccessRules ) {
1146                mp.disableAccessRules();
1147            }
1148
1149            final WikiDocument doc = mp.parse();
1150
1151            //  In some cases it's better just to parse, not to render
1152            if( !justParse ) {
1153                result = m_renderingManager.getHTML( context, doc );
1154
1155                if( runFilters && m_filterManager != null ) {
1156                    result = m_filterManager.doPostTranslateFiltering( context, result );
1157                }
1158            }
1159
1160            sw.stop();
1161
1162            if( log.isDebugEnabled() ) {
1163                log.debug( "Page " + context.getRealPage().getName() + " rendered, took " + sw );
1164            }
1165        } catch( final IOException e ) {
1166            log.error( "Failed to scan page data: ", e );
1167        } catch( final FilterException e ) {
1168            log.error( "page filter threw exception: ", e );
1169            // FIXME: Don't yet know what to do
1170        }
1171
1172        return result;
1173    }

1430     *
1431     * @since 2.3
1432     * @return The current AclManager.
1433     */
1434    public AclManager getAclManager()  {
1435        if( m_aclManager == null ) {
1436            try {
1437                final String s = m_properties.getProperty( PROP_ACL_MANAGER_IMPL, ClassUtil.getMappedClass( AclManager.class.getName() ).getName() );
1438                m_aclManager = ClassUtil.getMappedObject(s); // TODO: I am not sure whether this is the right call
1439                m_aclManager.initialize( this, m_properties );
1440            } catch( final ReflectiveOperationException | IllegalArgumentException e ) {
1441                log.fatal( "unable to instantiate class for AclManager: " + e.getMessage() );
1442                throw new InternalWikiException( "Cannot instantiate AclManager, please check logs.", e );
1443            }
1444        }
1445        return m_aclManager;
