019package org.apache.wiki;
020
021import org.apache.wiki.api.exceptions.NoSuchVariableException;
022import org.apache.wiki.parser.LinkParsingOperations;
023
024/**
025 *  Manages variables.  Variables are case-insensitive.  A list of all available variables is on a Wiki page called "WikiVariables".
026 *
027 *  @since 1.9.20.
028 */
029public interface VariableManager {
030
031    // FIXME: These are probably obsolete.
032    String VAR_ERROR = "error";
033    String VAR_MSG   = "msg";
034
035    /**
036     *  Parses the link and finds a value.  This is essentially used once
037     *  {@link LinkParsingOperations#isVariableLink(String)} has found that
038     *  the link text actually contains a variable.  For example, you could
039     *  pass in "{$username}" and get back "JanneJalkanen".
040     *
041     *  @param  context The WikiContext
042     *  @param  link    The link text containing the variable name.
043     *  @return The variable value.
044     *  @throws IllegalArgumentException If the format is not valid (does not start with "{$", is zero length, etc.)
045     *  @throws NoSuchVariableException If a variable is not known.
046     */
047    String parseAndGetValue( WikiContext context, String link ) throws IllegalArgumentException, NoSuchVariableException;
048
049    /**
050     *  This method does in-place expansion of any variables.  However, the expansion is not done twice, that is,
051     *  a variable containing text $variable will not be expanded.
052     *  <P>
053     *  The variables should be in the same format ({$variablename} as in the web pages.
054     *
055     *  @param context The WikiContext of the current page.
056     *  @param source  The source string.
057     *  @return The source string with variables expanded.
058     */
059    String expandVariables( WikiContext context, String source );
060
061    /**
062     *  Returns the value of a named variable.  See {@link #getValue(WikiContext, String)}. The only difference is that
063     *  this method does not throw an exception, but it returns the given default value instead.
064     *
065     *  @param context WikiContext
066     *  @param varName The name of the variable
067     *  @param defValue A default value.
068     *  @return The variable value, or if not found, the default value.
069     */
070    String getValue( WikiContext context, String varName, String defValue );
071
072    /**
073     *  Shortcut to getValue(). However, this method does not throw a NoSuchVariableException, but returns null
074     *  in case the variable does not exist.
075     *
076     *  @param context WikiContext to look the variable in
077     *  @param name Name of the variable to look for
078     *  @return Variable value, or null, if there is no such variable.
079     *  @since 2.2 on WikiEngine, moved to VariableManager on 2.11.0
080     */
081    String getVariable( WikiContext context, String name );
082
083    /**
084     *  Returns a value of the named variable.  The resolving order is
085     *  <ol>
086     *    <li>Known "constant" name, such as "pagename", etc.  This is so
087     *        that pages could not override certain constants.
088     *    <li>WikiContext local variable.  This allows a programmer to
089     *        set a parameter which cannot be overridden by user.
090     *    <li>HTTP Session
091     *    <li>HTTP Request parameters
092     *    <li>WikiPage variable.  As set by the user with the SET directive.
093     *    <li>jspwiki.properties
094     *  </ol>
095     *
096     *  Use this method only whenever you really need to have a parameter that
097     *  can be overridden by anyone using the wiki.
098     *
099     *  @param context The WikiContext
100     *  @param varName Name of the variable.
101     *
102     *  @return The variable value.
103     *
104     *  @throws IllegalArgumentException If the name is somehow broken.
105     *  @throws NoSuchVariableException If a variable is not known.
106     */
107    String getValue( WikiContext context, String varName ) throws IllegalArgumentException, NoSuchVariableException;
108
109}
