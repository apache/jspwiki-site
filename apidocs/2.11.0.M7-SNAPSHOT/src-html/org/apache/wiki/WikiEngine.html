342    // FIXME: Potential make-things-easier thingy here: no need to fetch the wikiengine anymore
343    //        Wiki.jsp.jspInit() [really old code]; it's probably even faster to fetch it
344    //        using this method every time than go to pageContext.getAttribute().
345
346    public static synchronized WikiEngine getInstance( ServletContext context,
347                                                       Properties props )
348        throws InternalWikiException
349    {
350        WikiEngine engine = (WikiEngine) context.getAttribute( ATTR_WIKIENGINE );
351
352        if( engine == null )
353        {
354            String appid = Integer.toString(context.hashCode()); //FIXME: Kludge, use real type.
355
356            context.log(" Assigning new engine to "+appid);
357            try
358            {
359                if( props == null )
360                {
361                    props = PropertyReader.loadWebAppProps( context );
362                }
363
364                engine = new WikiEngine( context, appid, props );
365                context.setAttribute( ATTR_WIKIENGINE, engine );
366            }
367            catch( Exception e )
368            {
369                context.log( "ERROR: Failed to create a Wiki engine: "+e.getMessage() );
370                log.error( "ERROR: Failed to create a Wiki engine, stacktrace follows " , e);
371                throw new InternalWikiException( "No wiki engine, check logs." , e);
372            }
373
374        }
375
376        return engine;
377    }

412        // Stash the WikiEngine in the servlet context
413        if ( context != null )
414        {
415            context.setAttribute( ATTR_WIKIENGINE,  this );
416            m_rootPath = context.getRealPath("/");
417        }
418
419        try
420        {
421            //
422            //  Note: May be null, if JSPWiki has been deployed in a WAR file.

424            initialize( props );

426        }
427        catch( Exception e )
428        {
429            String msg = Release.APPNAME+": Unable to load and setup properties from jspwiki.properties. "+e.getMessage();
430            if ( context != null )
431            {
432                context.log( msg );
433            }
434            throw new WikiException( msg, e );
435        }
436    }

438    /**
439     *  Does all the real initialization.
440     */
441    private void initialize( Properties props )
442        throws WikiException
443    {
444        m_startTime  = new Date();
445        m_properties = props;
446
447        //
448        //  Initialize log4j.  However, make sure that we don't initialize it multiple times.
449        //  By default we load the log4j config statements from jspwiki.properties, unless
450        //  the property jspwiki.use.external.logconfig=true, in that case we let log4j figure out the
451        //  logging configuration.
452        //
453        if( !c_configured )
454        {
455            String useExternalLogConfig = TextUtil.getStringProperty(props,"jspwiki.use.external.logconfig","false");
456            if( useExternalLogConfig == null || useExternalLogConfig.equals("false"))
457            {
458                PropertyConfigurator.configure( props );
459            }
460            c_configured = true;
461        }
462
463        log.info("*******************************************");
464        log.info(Release.APPNAME+" "+Release.getVersionString()+" starting. Whee!");
465
466        fireEvent( WikiEngineEvent.INITIALIZING ); // begin initialization
467
468        log.debug("Java version: "+System.getProperty("java.runtime.version"));
469        log.debug("Java vendor: "+System.getProperty("java.vm.vendor"));
470        log.debug("OS: "+System.getProperty("os.name")+" "+System.getProperty("os.version")+" "+System.getProperty("os.arch"));
471        log.debug("Default server locale: "+Locale.getDefault());
472        log.debug("Default server timezone: "+TimeZone.getDefault().getDisplayName(true, TimeZone.LONG));
473
474        if( m_servletContext != null )
475        {
476            log.info("Servlet container: "+m_servletContext.getServerInfo() );
477            if( m_servletContext.getMajorVersion() < 2 ||
478                (m_servletContext.getMajorVersion() == 2 && m_servletContext.getMinorVersion() < 4) )
479            {
480                throw new InternalWikiException("I require a container which supports at least version 2.4 of Servlet specification");
481            }
482        }
483
484        log.debug("Configuring WikiEngine...");
485
486        //  Initializes the CommandResolver
487        m_commandResolver  = new CommandResolver( this, props );
488
489        //
490        //  Create and find the default working directory.
491        //
492        m_workDir = TextUtil.getStringProperty( props, PROP_WORKDIR, null );
493
494        if( m_workDir == null ) {
495            m_workDir = System.getProperty("java.io.tmpdir", ".");
496            m_workDir += File.separator+Release.APPNAME+"-"+m_appid;
497        }
498
499        try {
500            File f = new File( m_workDir );
501            f.mkdirs();
502
503            //
504            //  A bunch of sanity checks
505            //
506            if( !f.exists() ) throw new WikiException("Work directory does not exist: "+m_workDir);
507            if( !f.canRead() ) throw new WikiException("No permission to read work directory: "+m_workDir);
508            if( !f.canWrite() ) throw new WikiException("No permission to write to work directory: "+m_workDir);
509            if( !f.isDirectory() ) throw new WikiException("jspwiki.workDir does not point to a directory: "+m_workDir);
510        } catch( SecurityException e ) {
511            log.fatal( "Unable to find or create the working directory: "+m_workDir, e );
512            throw new IllegalArgumentException( "Unable to find or create the working dir: " + m_workDir, e );
513        }
514
515        log.info("JSPWiki working directory is '"+m_workDir+"'");
516
517        m_saveUserInfo   = TextUtil.getBooleanProperty( props, PROP_STOREUSERNAME, m_saveUserInfo );
518        m_useUTF8        = StandardCharsets.UTF_8.name().equals( TextUtil.getStringProperty( props, PROP_ENCODING, StandardCharsets.ISO_8859_1.name() ) );
519        m_beautifyTitle  = TextUtil.getBooleanProperty( props, PROP_BEAUTIFYTITLE, m_beautifyTitle );
520        m_templateDir    = TextUtil.getStringProperty( props, PROP_TEMPLATEDIR, "default" );
521        enforceValidTemplateDirectory();
522        m_frontPage      = TextUtil.getStringProperty( props, PROP_FRONTPAGE,   "Main" );
523
524        //
525        //  Initialize the important modules.  Any exception thrown by the
526        //  managers means that we will not start up.
527        //
528
529        // FIXME: This part of the code is getting unwieldy.  We must think
530        //        of a better way to do the startup-sequence.
531        try
532        {
533            Class< ? > urlclass = ClassUtil.findClass( "org.apache.wiki.url",
534                                                       TextUtil.getStringProperty( props, PROP_URLCONSTRUCTOR, "DefaultURLConstructor" ) );
535            m_urlConstructor = (URLConstructor) urlclass.getDeclaredConstructor().newInstance();
536            m_urlConstructor.initialize( this, props );
537
538            m_pageManager           = ClassUtil.getMappedObject( PageManager.class.getName(), this, props );
539            m_pluginManager         = ClassUtil.getMappedObject( PluginManager.class.getName(), this, props );
540            m_differenceManager     = ClassUtil.getMappedObject( DifferenceManager.class.getName(), this, props );
541            m_attachmentManager     = ClassUtil.getMappedObject( AttachmentManager.class.getName(), this, props );
542            m_variableManager       = ClassUtil.getMappedObject( VariableManager.class.getName(), props );
543            m_renderingManager      = ClassUtil.getMappedObject( RenderingManager.class.getName() );
544            m_searchManager         = ClassUtil.getMappedObject( SearchManager.class.getName(), this, props );
545            m_authenticationManager = ClassUtil.getMappedObject( AuthenticationManager.class.getName() );
546            m_authorizationManager  = ClassUtil.getMappedObject( AuthorizationManager.class.getName() );
547            m_userManager           = ClassUtil.getMappedObject( UserManager.class.getName() );
548            m_groupManager          = ClassUtil.getMappedObject( GroupManager.class.getName() );
549            m_editorManager         = ClassUtil.getMappedObject( EditorManager.class.getName(), this );
550            m_editorManager.initialize( props );
551
552            m_progressManager   = new ProgressManager();
553
554            // Initialize the authentication, authorization, user and acl managers
555            m_authenticationManager.initialize( this, props );
556            m_authorizationManager.initialize( this, props );
557            m_userManager.initialize( this, props );
558            m_groupManager.initialize( this, props );
559            m_aclManager = getAclManager();
560
561            // Start the Workflow manager
562            m_workflowMgr = ClassUtil.getMappedObject(WorkflowManager.class.getName());
563            m_workflowMgr.initialize(this, props);
564            m_tasksManager = ClassUtil.getMappedObject(TasksManager.class.getName());
565
566            m_internationalizationManager = ClassUtil.getMappedObject(InternationalizationManager.class.getName(),this);
567            m_templateManager = ClassUtil.getMappedObject(TemplateManager.class.getName(), this, props );
568
569            // Since we want to use a page filters initilize() method
570            // as a engine startup listener where we can initialize global event listeners,
571            // it must be called lastly, so that all object references in the engine
572            // are availabe to the initialize() method
573            m_filterManager = ClassUtil.getMappedObject(FilterManager.class.getName(), this, props );
574
575            m_adminBeanManager = ClassUtil.getMappedObject(AdminBeanManager.class.getName(),this);
576
577            // RenderingManager depends on FilterManager events.
578            m_renderingManager.initialize( this, props );
579
580            //
581            //  ReferenceManager has the side effect of loading all
582            //  pages.  Therefore after this point, all page attributes
583            //  are available.
584            //
585            //  initReferenceManager is indirectly using m_filterManager, therefore
586            //  it has to be called after it was initialized.
587            //
588            initReferenceManager();
589
590            //
591            //  Hook the different manager routines into the system.
592            //
593            m_filterManager.addPageFilter(m_referenceManager, -1001 );
594            m_filterManager.addPageFilter(m_searchManager, -1002 );
595        }
596
597        catch( RuntimeException e )
598        {
599            // RuntimeExceptions may occur here, even if they shouldn't.
600            log.fatal( "Failed to start managers.", e );
601            throw new WikiException( "Failed to start managers: " + e.getMessage(), e );
602        }
603        catch (ClassNotFoundException e)
604        {
605            log.fatal( "JSPWiki could not start, URLConstructor was not found: " + e.getMessage(), e );
606            throw new WikiException(e.getMessage(), e );
607        }
608        catch (InstantiationException e)
609        {
610            log.fatal( "JSPWiki could not start, URLConstructor could not be instantiated: " + e.getMessage(), e );
611            throw new WikiException(e.getMessage(), e );
612        }
613        catch (IllegalAccessException e)
614        {
615            log.fatal( "JSPWiki could not start, URLConstructor cannot be accessed: " + e.getMessage(), e );
616            throw new WikiException(e.getMessage(), e );
617        }
618        catch( Exception e )
619        {
620            // Final catch-all for everything
621            log.fatal( "JSPWiki could not start, due to an unknown exception when starting.",e );
622            throw new WikiException( "Failed to start. Caused by: " + e.getMessage() +
623                                     "; please check log files for better information.", e );
624        }
625
626        //
627        //  Initialize the good-to-have-but-not-fatal modules.
628        //
629        try {
630            if( TextUtil.getBooleanProperty( props, RSSGenerator.PROP_GENERATE_RSS,false ) ) {
631                m_rssGenerator = ClassUtil.getMappedObject( RSSGenerator.class.getName(), this, props );
632            }
633
634            m_pageRenamer = ClassUtil.getMappedObject( PageRenamer.class.getName(), this, props );
635        } catch( final Exception e ) {
636            log.error( "Unable to start RSS generator - JSPWiki will still work, but there will be no RSS feed.", e );
637        }
638
639        // Start the RSS generator & generator thread
640        if( m_rssGenerator != null )
641        {
642            m_rssFile = TextUtil.getStringProperty( props,
643                    RSSGenerator.PROP_RSSFILE, "rss.rdf" );
644            File rssFile=null;
645            if (m_rssFile.startsWith(File.separator))
646            {
647                // honor absolute pathnames:
648                rssFile = new File(m_rssFile );
649            }
650            else
651            {
652                // relative path names are anchored from the webapp root path:
653                rssFile = new File( getRootPath(), m_rssFile );
654            }
655            int rssInterval = TextUtil.getIntegerProperty( props,
656                    RSSGenerator.PROP_INTERVAL, 3600 );
657            RSSThread rssThread = new RSSThread( this, rssFile, rssInterval );
658            rssThread.start();
659        }
660
661        fireEvent( WikiEngineEvent.INITIALIZED ); // initialization complete
662
663        log.info("WikiEngine configured.");
664        m_isConfigured = true;
665    }

1413    /**
1414     *  Helper method for doing the HTML translation.
1415     *
1416     *  @param context The WikiContext in which to do the conversion
1417     *  @param pagedata The data to render
1418     *  @param localLinkHook Is called whenever a wiki link is found
1419     *  @param extLinkHook   Is called whenever an external link is found
1420     *  @param parseAccessRules Parse the access rules if we encounter them
1421     *  @param justParse Just parses the pagedata, does not actually render.  In this case, this methods an empty string.
1422     *  @return HTML-rendered page text.
1423     */
1424    public String textToHTML( final WikiContext context,
1425                              String pagedata,
1426                              final StringTransmutator localLinkHook,
1427                              final StringTransmutator extLinkHook,
1428                              final StringTransmutator attLinkHook,
1429                              final boolean            parseAccessRules,
1430                              final boolean            justParse ) {
1431        String result = "";
1432
1433        if( pagedata == null ) {
1434            log.error("NULL pagedata to textToHTML()");
1435            return null;
1436        }
1437
1438        final boolean runFilters = "true".equals(m_variableManager.getValue(context,PROP_RUNFILTERS,"true"));
1439
1440        try {
1441            final StopWatch sw = new StopWatch();
1442            sw.start();
1443
1444            if( runFilters && m_filterManager != null ) {
1445                pagedata = m_filterManager.doPreTranslateFiltering( context, pagedata );
1446            }
1447
1448            final MarkupParser mp = m_renderingManager.getParser( context, pagedata );
1449            mp.addLocalLinkHook( localLinkHook );
1450            mp.addExternalLinkHook( extLinkHook );
1451            mp.addAttachmentLinkHook( attLinkHook );
1452
1453            if( !parseAccessRules ) {
1454                mp.disableAccessRules();
1455            }
1456
1457            final WikiDocument doc = mp.parse();
1458
1459            //  In some cases it's better just to parse, not to render
1460            if( !justParse ) {
1461                result = m_renderingManager.getHTML( context, doc );
1462
1463                if( runFilters && m_filterManager != null ) {
1464                    result = m_filterManager.doPostTranslateFiltering( context, result );
1465                }
1466            }
1467
1468            sw.stop();
1469
1470            if( log.isDebugEnabled() ) {
1471                log.debug( "Page " + context.getRealPage().getName() + " rendered, took " + sw );
1472            }
1473        } catch( final IOException e ) {
1474            log.error( "Failed to scan page data: ", e );
1475        } catch( final FilterException e ) {
1476            log.error( "page filter threw exception: ", e );
1477            // FIXME: Don't yet know what to do
1478        }
1479
1480        return result;
1481    }

1484    /**
1485     *  Writes the WikiText of a page into the page repository. If the <code>jspwiki.properties</code> file contains
1486     *  the property <code>jspwiki.approver.workflow.saveWikiPage</code> and its value resolves to a valid user,
1487     *  {@link org.apache.wiki.auth.authorize.Group} or {@link org.apache.wiki.auth.authorize.Role}, this method will
1488     *  place a {@link org.apache.wiki.workflow.Decision} in the approver's workflow inbox and throw a
1489     *  {@link org.apache.wiki.workflow.DecisionRequiredException}. If the submitting user is authenticated and the
1490     *  page save is rejected, a notification will be placed in the user's decision queue.
1491     *
1492     *  @since 2.1.28
1493     *  @param context The current WikiContext
1494     *  @param text    The Wiki markup for the page.
1495     *  @throws WikiException if the save operation encounters an error during the save operation. If the page-save
1496     *  operation requires approval, the exception will be of type {@link org.apache.wiki.workflow.DecisionRequiredException}.
1497     *  Individual PageFilters, such as the {@link org.apache.wiki.filters.SpamFilter} may also throw a
1498     *  {@link org.apache.wiki.api.exceptions.RedirectException}.
1499     */
1500    public void saveText( WikiContext context, String text ) throws WikiException {
1501        // Check if page data actually changed; bail if not
1502        WikiPage page = context.getPage();
1503        String oldText = getPureText( page );
1504        String proposedText = TextUtil.normalizePostData( text );
1505        if ( oldText != null && oldText.equals( proposedText ) ) {
1506            return;
1507        }
1508
1509        // Check if creation of empty pages is allowed; bail if not
1510        boolean allowEmpty = TextUtil.getBooleanProperty( m_properties, PROP_ALLOW_CREATION_OF_EMPTY_PAGES, false );
1511        if ( !allowEmpty && !pageExists( page ) && text.trim().equals( "" ) ) {
1512            return;
1513        }
1514
1515        // Create approval workflow for page save; add the diffed, proposed and old text versions as
1516        // Facts for the approver (if approval is required). If submitter is authenticated, any reject
1517        // messages will appear in his/her workflow inbox.
1518        WorkflowBuilder builder = WorkflowBuilder.getBuilder( this );
1519        Principal submitter = context.getCurrentUser();
1520        Step prepTask = m_tasksManager.buildPreSaveWikiPageTask( context, proposedText );
1521        Step completionTask = m_tasksManager.buildSaveWikiPageTask();
1522        String diffText = m_differenceManager.makeDiff( context, oldText, proposedText );
1523        boolean isAuthenticated = context.getWikiSession().isAuthenticated();
1524        Fact[] facts = new Fact[ 5 ];
1525        facts[ 0 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_PAGE_NAME, page.getName() );
1526        facts[ 1 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_DIFF_TEXT, diffText );
1527        facts[ 2 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_PROPOSED_TEXT, proposedText );
1528        facts[ 3 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_CURRENT_TEXT, oldText);
1529        facts[ 4 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_IS_AUTHENTICATED, Boolean.valueOf( isAuthenticated ) );
1530        String rejectKey = isAuthenticated ? WorkflowManager.WF_WP_SAVE_REJECT_MESSAGE_KEY : null;
1531        Workflow workflow = builder.buildApprovalWorkflow( submitter,
1532                                                           WorkflowManager.WF_WP_SAVE_APPROVER,
1533                                                           prepTask,
1534                                                           WorkflowManager.WF_WP_SAVE_DECISION_MESSAGE_KEY,
1535                                                           facts,
1536                                                           completionTask,
1537                                                           rejectKey );
1538        m_workflowMgr.start( workflow );
1539
1540        // Let callers know if the page-save requires approval
1541        if ( workflow.getCurrentStep() instanceof Decision ) {
1542            throw new DecisionRequiredException( "The page contents must be approved before they become active." );
1543        }
1544    }

1861                    }
1862                }
