039/**
040 * Manages authentication activities for a Engine: user login, logout, and credential refreshes. This class uses JAAS to determine how
041 * users log in.
042 * <p>
043 * The login procedure is protected in addition by a mechanism which prevents a hacker to try and force-guess passwords by slowing down
044 * attempts to log in into the same account.  Every login attempt is recorded, and stored for a while (currently ten minutes), and each
045 * login attempt during that time incurs a penalty of 2^login attempts milliseconds - that is, 10 login attempts incur a login penalty
046 * of 1.024 seconds. The delay is currently capped to 20 seconds.
047 * 
048 * @since 2.3
049 */
050public interface AuthenticationManager {
051
052    /** If this jspwiki.properties property is <code>true</code>, logs the IP address of the editor on saving. */
053    String PROP_STOREIPADDRESS = "jspwiki.storeIPAddress";
054    
055    /** If this jspwiki.properties property is <code>true</code>, allow cookies to be used for authentication. */
056    String PROP_ALLOW_COOKIE_AUTH = "jspwiki.cookieAuthentication";
057    
058    /** Whether logins should be throttled to limit brute-forcing attempts. Defaults to true. */
059    String PROP_LOGIN_THROTTLING = "jspwiki.login.throttling";
060
061    /** Prefix for LoginModule options key/value pairs. */
062    String PREFIX_LOGIN_MODULE_OPTIONS = "jspwiki.loginModule.options.";
063
064    /** If this jspwiki.properties property is <code>true</code>, allow cookies to be used to assert identities. */
065    String PROP_ALLOW_COOKIE_ASSERTIONS = "jspwiki.cookieAssertions";
066
067    /** The {@link LoginModule} to use for custom authentication. */
068    String PROP_LOGIN_MODULE = "jspwiki.loginModule.class";
069
070    /**
071     * Creates an AuthenticationManager instance for the given Engine and
072     * the specified set of properties. All initialization for the modules is
073     * done here.
074     * @param engine the wiki engine
075     * @param props the properties used to initialize the wiki engine
076     * @throws WikiException if the AuthenticationManager cannot be initialized
077     */
078    void initialize( Engine engine, Properties props ) throws WikiException;
079
080    /**
081     * Returns true if this Engine uses container-managed authentication. This method is used primarily for cosmetic purposes in the
082     * JSP tier, and performs no meaningful security function per se. Delegates to
083     * {@link org.apache.wiki.auth.authorize.WebContainerAuthorizer#isContainerAuthorized()},
084     * if used as the external authorizer; otherwise, returns <code>false</code>.
085     *
086     * @return <code>true</code> if the wiki's authentication is managed by the container, <code>false</code> otherwise
087     */
088    boolean isContainerAuthenticated();
089
090    /**
091     * <p>Logs in the user by attempting to populate a WikiSession Subject from a web servlet request by examining the request
092     *  for the presence of container credentials and user cookies. The processing logic is as follows:
093     * </p>
094     * <ul>
095     * <li>If the WikiSession had previously been unauthenticated, check to see if user has subsequently authenticated. To be considered
096     * "authenticated," the request must supply one of the following (in order of preference): the container <code>userPrincipal</code>,
097     * container <code>remoteUser</code>, or authentication cookie. If the user is authenticated, this method fires event
098     * {@link org.apache.wiki.event.WikiSecurityEvent#LOGIN_AUTHENTICATED} with two parameters: a Principal representing the login principal,
099     * and the current WikiSession. In addition, if the authorizer is of type WebContainerAuthorizer, this method iterates through the
100     * container roles returned by {@link org.apache.wiki.auth.authorize.WebContainerAuthorizer#getRoles()}, tests for membership in each
101     * one, and adds those that pass to the Subject's principal set.</li>
102     * <li>If, after checking for authentication, the WikiSession is still Anonymous, this method next checks to see if the user has
103     * "asserted" an identity by supplying an assertion cookie. If the user is found to be asserted, this method fires event
104     * {@link org.apache.wiki.event.WikiSecurityEvent#LOGIN_ASSERTED} with two parameters: <code>WikiPrincipal(<em>cookievalue</em>)</code>,
105     * and the current WikiSession.</li>
106     * <li>If, after checking for authenticated and asserted status, the  WikiSession is <em>still</em> anonymous, this method fires event
107     * {@link org.apache.wiki.event.WikiSecurityEvent#LOGIN_ANONYMOUS} with two parameters: <code>WikiPrincipal(<em>remoteAddress</em>)</code>,
108     * and the current WikiSession </li>
109     * </ul>
110     *
111     * @param request servlet request for this user
112     * @return always returns <code>true</code> (because anonymous login, at least, will always succeed)
113     * @throws org.apache.wiki.auth.WikiSecurityException if the user cannot be logged in for any reason
114     * @since 2.3
115     */
116    boolean login( HttpServletRequest request ) throws WikiSecurityException;
117    
118    /**
119     * Attempts to perform a WikiSession login for the given username/password combination using JSPWiki's custom authentication mode. In
120     * order to log in, the JAAS LoginModule supplied by the Engine property {@link #PROP_LOGIN_MODULE} will be instantiated, and its
121     * {@link javax.security.auth.spi.LoginModule#initialize(Subject, CallbackHandler, Map, Map)} method will be invoked. By default,
122     * the {@link org.apache.wiki.auth.login.UserDatabaseLoginModule} class will be used. When the LoginModule's <code>initialize</code>
123     * method is invoked, an options Map populated by properties keys prefixed by {@link #PREFIX_LOGIN_MODULE_OPTIONS} will be passed as a
124     * parameter.
125     *
126     * @param session the current wiki session; may not be <code>null</code>.
127     * @param request the user's HTTP request. This parameter may be <code>null</code>, but the configured LoginModule will not have access
128     *                to the HTTP request in this case.
129     * @param username The user name. This is a login name, not a WikiName. In most cases they are the same, but in some cases, they might not be.
130     * @param password the password
131     * @return true, if the username/password is valid
132     * @throws org.apache.wiki.auth.WikiSecurityException if the Authorizer or UserManager cannot be obtained
133     */
134    boolean login( WikiSession session, HttpServletRequest request, String username, String password ) throws WikiSecurityException;
135
136    /**
137     * Logs the user out by retrieving the WikiSession associated with the HttpServletRequest and unbinding all of the Subject's Principals,
138     * except for {@link Role#ALL}, {@link Role#ANONYMOUS}. is a cheap-and-cheerful way to do it without invoking JAAS LoginModules.
139     * The logout operation will also flush the JSESSIONID cookie from the user's browser session, if it was set.
140     *
141     * @param request the current HTTP request
142     */
143    void logout( HttpServletRequest request );
144
145    /**
146     * Determines whether this Engine allows users to assert identities using cookies instead of passwords. This is determined by inspecting
147     * the Engine property {@link #PROP_ALLOW_COOKIE_ASSERTIONS}.
148     *
149     * @return <code>true</code> if cookies are allowed
150     */
151    boolean allowsCookieAssertions();
152
153    /**
154     * Determines whether this Engine allows users to authenticate using cookies instead of passwords. This is determined by inspecting
155     * the Engine property {@link #PROP_ALLOW_COOKIE_AUTH}.
156     *
157     *  @return <code>true</code> if cookies are allowed for authentication
158     *  @since 2.5.62
159     */
160    boolean allowsCookieAuthentication();
161
162    /**
163     * Instantiates and executes a single JAAS {@link LoginModule}, and returns a Set of Principals that results from a successful login.
164     * The LoginModule is instantiated, then its {@link LoginModule#initialize(Subject, CallbackHandler, Map, Map)} method is called. The
165     * parameters passed to <code>initialize</code> is a dummy Subject, an empty shared-state Map, and an options Map the caller supplies.
166     *
167     * @param clazz the LoginModule class to instantiate
168     * @param handler the callback handler to supply to the LoginModule
169     * @param options a Map of key/value strings for initializing the LoginModule
170     * @return the set of Principals returned by the JAAS method {@link Subject#getPrincipals()}
171     * @throws WikiSecurityException if the LoginModule could not be instantiated for any reason
172     */
173    Set< Principal > doJAASLogin( Class< ? extends LoginModule > clazz, CallbackHandler handler, Map< String, String > options) throws WikiSecurityException;
174    
175    /**
176     * Determines whether the supplied Principal is a "role principal".
177     *
178     * @param principal the principal to test
179     * @return {@code true} if the Principal is of type {@link GroupPrincipal} or {@link Role}, {@code false} otherwise.
180     */
181    static boolean isRolePrincipal( final Principal principal ) {
182        return principal instanceof Role || principal instanceof GroupPrincipal;
183    }
184
185    /**
186     * Determines whether the supplied Principal is a "user principal".
187     *
188     * @param principal the principal to test
189     * @return {@code false} if the Principal is of type {@link GroupPrincipal} or {@link Role}, {@code true} otherwise.
190     */
191    static boolean isUserPrincipal( final Principal principal ) {
192        return !isRolePrincipal( principal );
193    }
194
195    /**
196     * Returns the first Principal in a set that isn't a {@link Role} or {@link GroupPrincipal}.
197     *
198     * @param principals the principal set
199     * @return the login principal
200     */
201    default Principal getLoginPrincipal( final Set< Principal > principals ) {
202        for( final Principal principal : principals ) {
203            if ( isUserPrincipal( principal ) ) {
204                return principal;
205            }
206        }
207        return null;
208    }
209
210    // events processing .......................................................
211
212    /**
213     * Registers a WikiEventListener with this instance. This is a convenience method.
214     *
215     * @param listener the event listener
216     */
217    void addWikiEventListener( WikiEventListener listener );
218
219    /**
220     * Un-registers a WikiEventListener with this instance. This is a convenience method.
221     *
222     * @param listener the event listener
223     */
224    void removeWikiEventListener( final WikiEventListener listener );
225
226    /**
227     *  Fires a WikiSecurityEvent of the provided type, Principal and target Object to all registered listeners.
228     *
229     * @see org.apache.wiki.event.WikiSecurityEvent
230     * @param type       the event type to be fired
231     * @param principal  the subject of the event, which may be <code>null</code>
232     * @param target     the changed Object, which may be <code>null</code>
233     */
234    default void fireEvent( final int type, final Principal principal, final Object target ) {
235        if ( WikiEventManager.isListening( this ) ) {
236            WikiEventManager.fireEvent( this, new WikiSecurityEvent( this, type, principal, target ) );
237        }
238    }
