001package org.apache.wiki.tasks.auth;
002
003import org.apache.log4j.Logger;
004import org.apache.wiki.WikiContext;
005import org.apache.wiki.WikiEngine;
006import org.apache.wiki.api.exceptions.WikiException;
007import org.apache.wiki.auth.user.UserProfile;
008import org.apache.wiki.i18n.InternationalizationManager;
009import org.apache.wiki.tasks.TasksManager;
010import org.apache.wiki.util.MailUtil;
011import org.apache.wiki.workflow.Outcome;
012import org.apache.wiki.workflow.Task;
013import org.apache.wiki.workflow.WorkflowManager;
014
015import javax.mail.MessagingException;
016import javax.mail.internet.AddressException;
017import java.util.Locale;
018
019
020/**
021 * Handles the actual profile save action. 
022 */
023public class SaveUserProfileTask extends Task {
024
025    private static final long serialVersionUID = 6994297086560480285L;
026    private static final Logger LOG = Logger.getLogger( SaveUserProfileTask.class );
027    private final WikiEngine m_engine;
028    private final Locale m_loc;
029
030    /**
031     * Constructs a new Task for saving a user profile.
032     * @param engine the wiki engine
033     */
034    public SaveUserProfileTask( final WikiEngine engine, final Locale loc ) {
035        super( TasksManager.USER_PROFILE_SAVE_TASK_MESSAGE_KEY );
036        m_engine = engine;
037        m_loc = loc;
038    }
039
040    /**
041     * Saves the user profile to the user database.
042     * @return {@link org.apache.wiki.workflow.Outcome#STEP_COMPLETE} if the
043     * task completed successfully
044     * @throws WikiException if the save did not complete for some reason
045     */
046    @Override
047    public Outcome execute() throws WikiException {
048        // Retrieve user profile
049        final UserProfile profile = ( UserProfile )getWorkflow().getAttribute( WorkflowManager.WF_UP_CREATE_SAVE_ATTR_SAVED_PROFILE );
050
051        // Save the profile (userdatabase will take care of timestamps for us)
052        m_engine.getUserManager().getUserDatabase().save( profile );
053
054        // Send e-mail if user supplied an e-mail address
055        if ( profile.getEmail() != null ) {
056            try {
057                final InternationalizationManager i18n = m_engine.getInternationalizationManager();
058                final String app = m_engine.getApplicationName();
059                final String to = profile.getEmail();
060                final String subject = i18n.get( InternationalizationManager.DEF_TEMPLATE, m_loc,
061                                                 "notification.createUserProfile.accept.subject", app );
062
063                final String content = i18n.get( InternationalizationManager.DEF_TEMPLATE, m_loc,
064                                                 "notification.createUserProfile.accept.content", app,
065                                                 profile.getLoginName(),
066                                                 profile.getFullname(),
067                                                 profile.getEmail(),
068                                                 m_engine.getURL( WikiContext.LOGIN, null, null ) );
069                MailUtil.sendMessage( m_engine.getWikiProperties(), to, subject, content);
070            } catch ( final AddressException e) {
071                LOG.debug( e.getMessage(), e );
072            } catch ( final MessagingException me ) {
073                LOG.error( "Could not send registration confirmation e-mail. Is the e-mail server running?", me );
074            }
075        }
076
077        return Outcome.STEP_COMPLETE;
078    }
079
080}
