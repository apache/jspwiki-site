019package org.apache.wiki.providers;
020
021import java.io.IOException;
022import java.io.InputStream;
023import java.util.Collection;
024import java.util.Date;
025import java.util.List;
026
027import org.apache.wiki.WikiPage;
028import org.apache.wiki.WikiProvider;
029import org.apache.wiki.api.exceptions.ProviderException;
030import org.apache.wiki.attachment.Attachment;
031import org.apache.wiki.search.QueryItem;
032
033/**
034 *  Defines an attachment provider - a class which is capable of saving
035 *  binary data as attachments.
036 *  <P>
037 *  The difference between this class and WikiPageProvider is that there
038 *  PageProviders handle Unicode text, whereas we handle binary data.
039 *  While there are quite a lot of similarities in how we handle
040 *  things, many providers can really use just one.  In addition,
041 *  since binary files can be really large, we rely on
042 *  Input/OutputStreams.
043 *
044 */
045public interface WikiAttachmentProvider
046    extends WikiProvider
047{
048    /**
049     *  Put new attachment data.
050     *  
051     *  @param att Attachment object to add new data to
052     *  @param data The stream from which the provider should read the data
053     *  @throws IOException If writing fails
054     *  @throws ProviderException If there are other errors.
055     */
056    void putAttachmentData( Attachment att, InputStream data )
057        throws ProviderException,
058               IOException;
059
060    /**
061     *  Get attachment data.
062     *  
063     *  @param att The attachment
064     *  @return An InputStream which you contains the raw data of the object. It's your
065     *          responsibility to close it.
066     *  @throws ProviderException If the attachment cannot be found
067     *  @throws IOException If the attachment cannot be opened
068     */
069
070    InputStream getAttachmentData( Attachment att )
071        throws ProviderException,
072               IOException;
073
074    /**
075     *  Lists all attachments attached to a page.
076     *
077     *  @param page The page to list the attachments from.
078     *  @return A collection of Attachment objects.  May be empty, but never null.
079     *  @throws ProviderException If something goes wrong when listing the attachments.
080     */
081
082    List< Attachment > listAttachments( WikiPage page )
083        throws ProviderException;
084
085    /**
086     * Finds attachments based on the query.
087     * @param query An array of QueryItem objects to search for
088     * @return A Collection of Attachment objects.  May be empty, but never null.
089     */
090    Collection findAttachments( QueryItem[] query );
091
092    /**
093     *  Lists changed attachments since given date.  Can also be used to fetch
094     *  a list of all pages.
095     *  <P>
096     *  This is different from WikiPageProvider, where you basically get a list
097     *  of all pages, then sort them locally.  However, since some providers
098     *  can be more efficient in locating recently changed files (like any database) 
099     *  than our non-optimized Java
100     *  code, it makes more sense to fetch the whole list this way.
101     *  <P>
102     *  To get all files, call this with Date(0L);
103     *
104     *  @param timestamp List all files from this date onward.
105     *  @return A List of Attachment objects, in most-recently-changed first order.
106     *  @throws ProviderException If something goes wrong.
107     */
108    List<Attachment> listAllChanged( Date timestamp )
109        throws ProviderException;
110
111    /**
112     *  Returns info about an attachment.
113     *  
114     *  @param page The parent page
115     *  @param name The name of the attachment
116     *  @param version The version of the attachment (it's okay to use WikiPage.LATEST_VERSION to find the latest one)
117     *  @return An attachment object
118     *  @throws ProviderException If the attachment cannot be found or some other error occurs.
119     */
120    Attachment getAttachmentInfo( WikiPage page, String name, int version )
121        throws ProviderException;
122
123    /**
124     *  Returns version history.  Each element should be
125     *  an Attachment.
126     *  
127     *  @param att The attachment for which to find the version history for.
128     *  @return A List of Attachment objects.
129     */
130    List<Attachment> getVersionHistory( Attachment att );
131
132    /**
133     *  Removes a specific version from the repository.  The implementations
134     *  should really do no more security checks, since that is the domain
135     *  of the AttachmentManager.  Just delete it as efficiently as you can.
136     *
137     *  @since 2.0.19.
138     *
139     *  @param att Attachment to be removed.  The version field is checked, and thus
140     *             only that version is removed.
141     *
142     *  @throws ProviderException If the attachment cannot be removed for some reason.
143     */
144
145    void deleteVersion( Attachment att )
146        throws ProviderException;
147
148    /**
149     *  Removes an entire page from the repository.  The implementations
150     *  should really do no more security checks, since that is the domain
151     *  of the AttachmentManager.  Just delete it as efficiently as you can.  You should also
152     *  delete any auxiliary files and directories that belong to this attachment, 
153     *  IF they were created
154     *  by this provider.
155     *
156     *  @since 2.0.17.
157     *
158     *  @param att Attachment to delete.
159     *
160     *  @throws ProviderException If the page could not be removed for some reason.
161     */
162    void deleteAttachment( Attachment att )
163        throws ProviderException;
164   
165    /**
166     * Move all the attachments for a given page so that they are attached to a
167     * new page.
168     *
169     * @param oldParent Name of the page we are to move the attachments from.
170     * @param newParent Name of the page we are to move the attachments to.
171     *
172     * @throws ProviderException If the attachments could not be moved for some
173     *                           reason.
174     */
175    void moveAttachmentsForPage( String oldParent,
176                                        String newParent )
177        throws ProviderException;
178}
