019package org.apache.wiki.auth.authorize;
020
021import org.apache.wiki.api.core.Context;
022import org.apache.wiki.api.core.Session;
023import org.apache.wiki.auth.Authorizer;
024import org.apache.wiki.auth.NoSuchPrincipalException;
025import org.apache.wiki.auth.WikiSecurityException;
026import org.apache.wiki.event.WikiEventListener;
027import org.apache.wiki.event.WikiEventManager;
028import org.apache.wiki.event.WikiSecurityEvent;
029
030import javax.servlet.http.HttpServletRequest;
031
032
033/**
034 * <p>
035 * Facade class for storing, retrieving and managing wiki groups on behalf of AuthorizationManager, JSPs and other presentation-layer
036 * classes. GroupManager works in collaboration with a back-end {@link GroupDatabase}, which persists groups to permanent storage.
037 * </p>
038 * <p>
039 * <em>Note: prior to JSPWiki 2.4.19, GroupManager was an interface; it is now a concrete, final class. The aspects of GroupManager
040 * which previously extracted group information from storage (e.g., wiki pages) have been refactored into the GroupDatabase interface.</em>
041 * </p>
042 * @since 2.4.19
043 */
044public interface GroupManager extends Authorizer, WikiEventListener {
045
046    /** Key used for adding UI messages to a user's Session. */
047    String MESSAGES_KEY = "group";
048
049    String PROP_GROUPDATABASE = "jspwiki.groupdatabase";
050
051    /**
052     * Returns the Group matching a given name. If the group cannot be found, this method throws a <code>NoSuchPrincipalException</code>.
053     *
054     * @param name the name of the group to find
055     * @return the group
056     * @throws NoSuchPrincipalException if the group cannot be found
057     */
058    Group getGroup( final String name ) throws NoSuchPrincipalException;
059
060    /**
061     * Returns the current external {@link GroupDatabase} in use. This method is guaranteed to return a properly-initialized GroupDatabase,
062     * unless it could not be initialized. In that case, this method throws a {@link org.apache.wiki.api.exceptions.WikiException}. The
063     * GroupDatabase is lazily initialized.
064     *
065     * @throws org.apache.wiki.auth.WikiSecurityException if the GroupDatabase could not be initialized
066     * @return the current GroupDatabase
067     * @since 2.3
068     */
069    GroupDatabase getGroupDatabase() throws WikiSecurityException;
070
071    /**
072     * <p>
073     * Extracts group name and members from passed parameters and populates an existing Group with them. The Group will either be a copy of
074     * an existing Group (if one can be found), or a new, unregistered Group (if not). Optionally, this method can throw a
075     * WikiSecurityException if the Group does not yet exist in the GroupManager cache.
076     * </p>
077     * <p>
078     * The <code>group</code> parameter in the HTTP request contains the Group name to look up and populate. The <code>members</code>
079     * parameter contains the member list. If these differ from those in the existing group, the passed values override the old values.
080     * </p>
081     * <p>
082     * This method does not commit the new Group to the GroupManager cache. To do that, use {@link #setGroup(Session, Group)}.
083     * </p>
084     * @param name the name of the group to construct
085     * @param memberLine the line of text containing the group membership list
086     * @param create whether this method should create a new, empty Group if one with the requested name is not found. If <code>false</code>,
087     *            groups that do not exist will cause a <code>NoSuchPrincipalException</code> to be thrown
088     * @return a new, populated group
089     * @see org.apache.wiki.auth.authorize.Group#RESTRICTED_GROUPNAMES
090     * @throws WikiSecurityException if the group name isn't allowed, or if <code>create</code> is <code>false</code>
091     *                               and the Group named <code>name</code> does not exist
092     */
093    Group parseGroup( String name, String memberLine, boolean create ) throws WikiSecurityException;
094
095    /**
096     * <p>
097     * Extracts group name and members from the HTTP request and populates an existing Group with them. The Group will either be a copy of
098     * an existing Group (if one can be found), or a new, unregistered Group (if not). Optionally, this method can throw a
099     * WikiSecurityException if the Group does not yet exist in the GroupManager cache.
100     * </p>
101     * <p>
102     * The <code>group</code> parameter in the HTTP request contains the Group name to look up and populate. The <code>members</code>
103     * parameter contains the member list. If these differ from those in the existing group, the passed values override the old values.
104     * </p>
105     * <p>
106     * This method does not commit the new Group to the GroupManager cache. To do that, use {@link #setGroup(Session, Group)}.
107     * </p>
108     * @param context the current wiki context
109     * @param create whether this method should create a new, empty Group if one with the requested name is not found. If <code>false</code>,
110     *            groups that do not exist will cause a <code>NoSuchPrincipalException</code> to be thrown
111     * @return a new, populated group
112     * @throws WikiSecurityException if the group name isn't allowed, or if <code>create</code> is <code>false</code>
113     *                               and the Group does not exist
114     */
115    default Group parseGroup( final Context context, final boolean create ) throws WikiSecurityException {
116        // Extract parameters
117        final HttpServletRequest request = context.getHttpRequest();
118        final String name = request.getParameter( "group" );
119        final String memberLine = request.getParameter( "members" );
120
121        // Create the named group; we pass on any NoSuchPrincipalExceptions
122        // that may be thrown if create == false, or WikiSecurityExceptions
123        final Group group = parseGroup( name, memberLine, create );
124
125        // If no members, add the current user by default
126        if( group.members().length == 0 ) {
127            group.add( context.getWikiSession().getUserPrincipal() );
128        }
129
130        return group;
131    }
132
133    /**
134     * Removes a named Group from the group database. If not found, throws a <code>NoSuchPrincipalException</code>. After removal, this
135     * method will commit the delete to the back-end group database. It will also fire a
136     * {@link org.apache.wiki.event.WikiSecurityEvent#GROUP_REMOVE} event with the GroupManager instance as the source and the Group as target.
137     * If <code>index</code> is <code>null</code>, this method throws an {@link IllegalArgumentException}.
138     *
139     * @param index the group to remove
140     * @throws WikiSecurityException if the Group cannot be removed by the back-end
141     * @see org.apache.wiki.auth.authorize.GroupDatabase#delete(Group)
142     */
143    void removeGroup( final String index ) throws WikiSecurityException;
144
145    /**
146     * <p>
147     * Saves the {@link Group} created by a user in a wiki session. This method registers the Group with the GroupManager and saves it to
148     * the back-end database. If an existing Group with the same name already exists, the new group will overwrite it. After saving the
149     * Group, the group database changes are committed.
150     * </p>
151     * <p>
152     * This method fires the following events:
153     * </p>
154     * <ul>
155     * <li><strong>When creating a new Group</strong>, this method fires a {@link org.apache.wiki.event.WikiSecurityEvent#GROUP_ADD} with
156     * the GroupManager instance as its source and the new Group as the target.</li>
157     * <li><strong>When overwriting an existing Group</strong>, this method fires a new
158     * {@link org.apache.wiki.event.WikiSecurityEvent#GROUP_REMOVE} with this GroupManager instance as the source, and the new Group as the
159     * target. It then fires a {@link org.apache.wiki.event.WikiSecurityEvent#GROUP_ADD} event with the same source and target.</li>
160     * </ul>
161     * <p>
162     * In addition, if the save or commit actions fail, this method will attempt to restore the older version of the wiki group if it
163     * exists. This will result in a <code>GROUP_REMOVE</code> event (for the new version of the Group) followed by a <code>GROUP_ADD</code>
164     * event (to indicate restoration of the old version).
165     * </p>
166     * <p>
167     * This method will register the new Group with the GroupManager. For example, {@link org.apache.wiki.auth.AuthenticationManager}
168     * attaches each Session as a GroupManager listener. Thus, the act of registering a Group with <code>setGroup</code> means that
169     * all Sessions will automatically receive group add/change/delete events immediately.
170     * </p>
171     *
172     * @param session the wiki session, which may not be <code>null</code>
173     * @param group the Group, which may not be <code>null</code>
174     * @throws WikiSecurityException if the Group cannot be saved by the back-end
175     */
176    void setGroup( final Session session, final Group group ) throws WikiSecurityException;
177
178    /**
179     * Validates a Group, and appends any errors to the session errors list. Any validation errors are added to the wiki session's messages
180     * collection (see {@link Session#getMessages()}.
181     *
182     * @param context the current wiki context
183     * @param group the supplied Group
184     */
185    void validateGroup( final Context context, final Group group );
186
187    /**
188     * Checks if a String is blank or a restricted Group name, and if it is, appends an error to the Session's message list.
189     *
190     * @param context the wiki context
191     * @param name the Group name to test
192     * @throws WikiSecurityException if <code>session</code> is <code>null</code> or the Group name is illegal
193     * @see Group#RESTRICTED_GROUPNAMES
194     */
195    void checkGroupName( final Context context, final String name ) throws WikiSecurityException;
196
197    // events processing .......................................................
198
199    /**
200     * Registers a WikiEventListener with this instance. This is a convenience method.
201     *
202     * @param listener the event listener
203     */
204    void addWikiEventListener( WikiEventListener listener );
205
206    /**
207     * Un-registers a WikiEventListener with this instance. This is a convenience method.
208     *
209     * @param listener the event listener
210     */
211    void removeWikiEventListener( WikiEventListener listener );
212
213    /**
214     *  Fires a WikiSecurityEvent of the provided type, Principal and target Object to all registered listeners.
215     *
216     * @see org.apache.wiki.event.WikiSecurityEvent
217     * @param type       the event type to be fired
218     * @param target     the changed Object, which may be <code>null</code>
219     */
220    default void fireEvent( final int type, final Object target ) {
221        if( WikiEventManager.isListening( this ) ) {
222            WikiEventManager.fireEvent( this, new WikiSecurityEvent( this, type, target ) );
223        }
224    }
