019package org.apache.wiki.ui;
020
021import org.apache.wiki.WikiContext;
022import org.apache.wiki.api.core.Session;
023import org.apache.wiki.i18n.InternationalizationManager;
024import org.apache.wiki.preferences.Preferences;
025
026import java.text.MessageFormat;
027import java.util.ResourceBundle;
028import java.util.regex.Matcher;
029import java.util.regex.Pattern;
030
031/**
032 * Provides basic validation services for HTTP parameters. Three standard validators are provided: email address, identifier and
033 * standard input. Standard input validator will reject any HTML-like input, and any of a number of special characters. ID validator
034 * rejects HTML and quoted strings, and a couple of special characters.
035 *
036 * @since 2.3.54
037 */
038public final class InputValidator {
039    /** Standard input validator. */
040    public static final int STANDARD       = 0;
041
042    /** Input validator for e-mail addresses. **/
043    public static final int EMAIL          = 1;
044
045    /** @since 2.4.82 */
046    public static final int ID             = 2;
047
048    protected static final Pattern EMAIL_PATTERN  = Pattern.compile( "^[0-9a-zA-Z-_\\.\\+]+@([0-9a-zA-Z-_]+\\.)+[a-zA-Z]+$" );
049
050    protected static final Pattern UNSAFE_PATTERN = Pattern.compile( "[\\x00\\r\\n\\x0f\"':<>\\[\\];#&@\\xff{}\\$%\\\\]" );
051
052    /** Used when checking against IDs such as a full name when saving groups.
053     *  @since 2.4.82 */
054    protected static final Pattern ID_PATTERN     = Pattern.compile( "[\\x00\\r\\n\\x0f\"'<>;&\\xff{}]" );
055
056    private final String           m_form;
057
058    private final Session          m_session;
059
060    private final WikiContext      m_context;
061
062    /**
063     * Constructs a new input validator for a specific form and wiki session. When validation errors are detected, they will be added to
064     * the wiki session's messages.
065     *
066     * @param form the ID or name of the form this validator should be associated with
067     * @param context the wiki context
068     */
069    public InputValidator( final String form, final WikiContext context ) {
070        m_form = form;
071        m_context = context;
072        m_session = context.getWikiSession();
073    }
074
075    /**
076     * Validates a string against the {@link #STANDARD} validator and additionally checks that the value is not <code>null</code> or blank.
077     *
078     * @param input the string to validate
079     * @param label the label for the string or field ("E-mail address")
080     * @return returns <code>true</code> if valid, <code>false</code> otherwise
081     */
082    public boolean validateNotNull( final String input, final String label ) {
083        return validateNotNull( input, label, STANDARD );
084    }
085
086    /**
087     * Validates a string against a particular pattern type and additionally checks that the value is not <code>null</code> or blank.
088     * Delegates to {@link #validate(String, String, int)}.
089     *
090     * @param input the string to validate
091     * @param label the label for the string or field ("E-mail address")
092     * @param type the pattern type to use (<em>e.g.</em>, {@link #STANDARD}, {@link #EMAIL}.
093     * @return returns <code>true</code> if valid, <code>false</code> otherwise
094     */
095    public boolean validateNotNull( final String input, final String label, final int type ) {
096        if ( isBlank( input ) ) {
097            final ResourceBundle rb = Preferences.getBundle( m_context, InternationalizationManager.CORE_BUNDLE );
098            m_session.addMessage( m_form, MessageFormat.format( rb.getString("validate.cantbenull"), label ) );
099            return false;
100        }
101        return validate( input, label, type ) && !isBlank( input );
102    }
103
104    /**
105     * Validates a string against a particular pattern type: e-mail address, standard HTML input, etc. Note that a blank or null string will
106     * always validate.
107     *
108     * @param input the string to validate
109     * @param label the label for the string or field ("E-mail address")
110     * @param type the target pattern to validate against ({@link #STANDARD}, {@link #EMAIL})
111     * @return returns <code>true</code> if valid, <code>false</code> otherwise
112     */
113    public boolean validate( final String input, final String label, final int type ) {
114        // If blank, it's valid
115        if ( isBlank( input ) ) {
116            return true;
117        }
118
119        final ResourceBundle rb = Preferences.getBundle( m_context, InternationalizationManager.CORE_BUNDLE );
120
121        // Otherwise, see if it matches the pattern for the target type
122        final Matcher matcher;
123        final boolean valid;
124        switch( type ) {
125        case STANDARD:
126            matcher = UNSAFE_PATTERN.matcher( input );
127            valid = !matcher.find();
128            if ( !valid ) {
129                // MessageTag already invokes replaceEntities()
130                // Object[] args = { label, "&quot;&#39;&lt;&gt;;&amp;[]#\\@{}%$" };
131                final Object[] args = { label, "\'\"<>;&[]#\\@{}%$" };
132                m_session.addMessage( m_form, MessageFormat.format( rb.getString( "validate.unsafechars" ), args ) );
133            }
134            return valid;
135        case EMAIL:
136            matcher = EMAIL_PATTERN.matcher( input );
137            valid = matcher.matches();
138            if ( !valid ) {
139                final Object[] args = { label };
140                m_session.addMessage( m_form, MessageFormat.format( rb.getString( "validate.invalidemail" ), args ) );
141            }
142            return valid;
143        case ID:
144            matcher = ID_PATTERN.matcher( input );
145            valid = !matcher.find();
146            if ( !valid ) {
147                // MessageTag already invokes replaceEntities()
148                // Object[] args = { label, "&quot;&#39;&lt;&gt;;&amp;{}" };
149                final Object[] args = { label, "\'\"<>;&{}" };
150                m_session.addMessage( m_form, MessageFormat.format( rb.getString( "validate.unsafechars" ), args ) );
151            }
152            return valid;
153         default:
154             break;
155        }
156        throw new IllegalArgumentException( "Invalid input type." );
157    }
158
159    /**
160     * Returns <code>true</code> if a supplied string is null or blank
161     *
162     * @param input the string to check
163     * @return <code>true</code> if <code>null</code> or blank (zero-length); <code>false</code> otherwise
164     */
165    public static boolean isBlank( final String input ) {
166        return input == null || input.trim().length() < 1;
167    }
168
