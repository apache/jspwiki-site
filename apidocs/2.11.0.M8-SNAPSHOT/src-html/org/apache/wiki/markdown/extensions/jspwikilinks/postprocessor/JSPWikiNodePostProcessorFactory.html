019package org.apache.wiki.markdown.extensions.jspwikilinks.postprocessor;
020
021import com.vladsch.flexmark.ast.Link;
022import com.vladsch.flexmark.parser.block.NodePostProcessor;
023import com.vladsch.flexmark.parser.block.NodePostProcessorFactory;
024import com.vladsch.flexmark.util.ast.Document;
025import com.vladsch.flexmark.util.data.DataHolder;
026import org.apache.oro.text.regex.Pattern;
027import org.apache.wiki.api.core.Context;
028
029import java.util.List;
030
031
032/**
033 * Simple {@link NodePostProcessorFactory} to instantiate {@link JSPWikiLinkNodePostProcessor}s.
034 */
035public class JSPWikiNodePostProcessorFactory extends NodePostProcessorFactory {
036
037    private final Context m_context;
038    private final boolean isImageInlining;
039    private final List< Pattern > inlineImagePatterns;
040
041    public JSPWikiNodePostProcessorFactory( final Context m_context,
042                                            final DataHolder options,
043                                            final boolean isImageInlining,
044                                            final List< Pattern > inlineImagePatterns ) {
045        super( true );
046        addNodes( Link.class ); // needs to be called before create( Document )
047        this.m_context = m_context;
048        this.isImageInlining = isImageInlining;
049        this.inlineImagePatterns = inlineImagePatterns;
050    }
051
052    /**
053     * {@inheritDoc}
054     */
055    @Override
056    public NodePostProcessor apply( final Document document ) {
057        return new JSPWikiLinkNodePostProcessor( m_context, document, isImageInlining, inlineImagePatterns );
058    }
059
060}
