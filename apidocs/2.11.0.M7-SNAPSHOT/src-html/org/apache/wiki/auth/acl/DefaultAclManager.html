050/**
051 * Default implementation that parses Acls from wiki page markup.
052 *
053 * @since 2.3
054 */
055public class DefaultAclManager implements AclManager {
056
057    private static final Logger log = Logger.getLogger(DefaultAclManager.class);
058
059    private AuthorizationManager m_auth = null;
060    private Engine m_engine = null;
061    private static final String PERM_REGEX = "("
062                                              + PagePermission.COMMENT_ACTION + "|"
063                                              + PagePermission.DELETE_ACTION  + "|"
064                                              + PagePermission.EDIT_ACTION    + "|"
065                                              + PagePermission.MODIFY_ACTION  + "|"
066                                              + PagePermission.RENAME_ACTION  + "|"
067                                              + PagePermission.UPLOAD_ACTION  + "|"
068                                              + PagePermission.VIEW_ACTION    +
069                                             ")";
070    private static final String ACL_REGEX = "\\[\\{\\s*ALLOW\\s+" + PERM_REGEX + "\\s*(.*?)\\s*\\}\\]";
071
072    /**
073     * Identifies ACL strings in wiki text; the first group is the action (view, edit) and
074     * the second is the list of Principals separated by commas. The overall match is
075     * the ACL string from [{ to }].
076     */
077    public static final Pattern ACL_PATTERN = Pattern.compile( ACL_REGEX );
078
079    /** {@inheritDoc} */
080    @Override
081    public void initialize( final Engine engine, final Properties props ) {
082        m_auth = engine.getManager( AuthorizationManager.class );
083        m_engine = engine;
084    }
085
086    /** {@inheritDoc} */
087    @Override
088    public Acl parseAcl( final WikiPage page, final String ruleLine ) throws WikiSecurityException {
089        Acl acl = page.getAcl();
090        if (acl == null) {
091            acl = new AclImpl();
092        }
093
094        try {
095            final StringTokenizer fieldToks = new StringTokenizer(ruleLine);
096            fieldToks.nextToken();
097            final String actions = fieldToks.nextToken();
098
099            while( fieldToks.hasMoreTokens() ) {
100                final String principalName = fieldToks.nextToken(",").trim();
101                final Principal principal = m_auth.resolvePrincipal(principalName);
102                final AclEntry oldEntry = acl.getEntry(principal);
103
104                if( oldEntry != null ) {
105                    log.debug( "Adding to old acl list: " + principal + ", " + actions );
106                    oldEntry.addPermission( PermissionFactory.getPagePermission( page, actions ) );
107                } else {
108                    log.debug( "Adding new acl entry for " + actions );
109                    final AclEntry entry = new AclEntryImpl();
110                    entry.setPrincipal( principal );
111                    entry.addPermission( PermissionFactory.getPagePermission( page, actions ) );
112
113                    acl.addEntry( entry );
114                }
115            }
116
117            page.setAcl( acl );
118            log.debug( acl.toString() );
119        } catch( final NoSuchElementException nsee ) {
120            log.warn( "Invalid access rule: " + ruleLine + " - defaults will be used." );
121            throw new WikiSecurityException( "Invalid access rule: " + ruleLine, nsee );
122        } catch( final IllegalArgumentException iae ) {
123            throw new WikiSecurityException("Invalid permission type: " + ruleLine, iae);
124        }
125
126        return acl;
127    }
128
129
130    /** {@inheritDoc} */
131    @Override
132    public Acl getPermissions( final WikiPage page ) {
133        //  Does the page already have cached ACLs?
134        Acl acl = page.getAcl();
135        log.debug( "page=" + page.getName() + "\n" + acl );
136
137        if( acl == null ) {
138            //  If null, try the parent.
139            if( page instanceof Attachment ) {
140                final WikiPage parent = ( WikiPage )m_engine.getManager( PageManager.class ).getPage( ( ( Attachment ) page ).getParentName() );
141                acl = getPermissions(parent);
142            } else {
143                //  Or, try parsing the page
144                final WikiContext ctx = new WikiContext( m_engine, page );
145                ctx.setVariable( Context.VAR_EXECUTE_PLUGINS, Boolean.FALSE );
146                m_engine.getManager( RenderingManager.class ).getHTML(ctx, page);
147
148                if (page.getAcl() == null) {
149                    page.setAcl( new AclImpl() );
150                }
151                acl = page.getAcl();
152            }
153        }
154
155        return acl;
156    }
157
158    /** {@inheritDoc} */
159    @Override
160    public void setPermissions( final WikiPage page, final Acl acl ) throws WikiSecurityException {
161        final PageManager pageManager = m_engine.getManager( PageManager.class );
162
163        // Forcibly expire any page locks
164        final PageLock lock = pageManager.getCurrentLock( page );
165        if( lock != null ) {
166            pageManager.unlockPage( lock );
167        }
168
169        // Remove all of the existing ACLs.
170        final String pageText = m_engine.getManager( PageManager.class ).getPureText( page );
171        final Matcher matcher = DefaultAclManager.ACL_PATTERN.matcher( pageText );
172        final String cleansedText = matcher.replaceAll("" );
173        final String newText = DefaultAclManager.printAcl( page.getAcl() ) + cleansedText;
174        try {
175            pageManager.putPageText( page, newText );
176        } catch( final ProviderException e ) {
177            throw new WikiSecurityException( "Could not set Acl. Reason: ProviderExcpetion " + e.getMessage(), e );
178        }
179    }
180
181    /**
182     * Generates an ACL string for inclusion in a wiki page, based on a supplied Acl object. All of the permissions in this Acl are
183     * assumed to apply to the same page scope. The names of the pages are ignored; only the actions and principals matter.
184     *
185     * @param acl the ACL
186     * @return the ACL string
187     */
188    protected static String printAcl( final Acl acl ) {
189        // Extract the ACL entries into a Map with keys == permissions, values == principals
190        final Map< String, List< Principal > > permissionPrincipals = new TreeMap<>();
191        final Enumeration< AclEntry > entries = acl.entries();
192        while( entries.hasMoreElements() ) {
193            final AclEntry entry = entries.nextElement();
194            final Principal principal = entry.getPrincipal();
195            final Enumeration< Permission > permissions = entry.permissions();
196            while( permissions.hasMoreElements() ) {
197                final Permission permission = permissions.nextElement();
198                List< Principal > principals = permissionPrincipals.get( permission.getActions() );
199                if (principals == null) {
200                    principals = new ArrayList<>();
201                    final String action = permission.getActions();
202                    if( action.indexOf(',') != -1 ) {
203                        throw new IllegalStateException("AclEntry permission cannot have multiple targets.");
204                    }
205                    permissionPrincipals.put( action, principals );
206                }
207                principals.add( principal );
208            }
209        }
210
211        // Now, iterate through each permission in the map and generate an ACL string
212        final StringBuilder s = new StringBuilder();
213        for( final Map.Entry< String, List< Principal > > entry : permissionPrincipals.entrySet() ) {
214            final String action = entry.getKey();
215            final List< Principal > principals = entry.getValue();
216            principals.sort( new PrincipalComparator() );
217            s.append( "[{ALLOW " ).append( action ).append( " " );
218            for( int i = 0; i < principals.size(); i++ ) {
219                final Principal principal = principals.get( i );
220                s.append( principal.getName() );
221                if( i < ( principals.size() - 1 ) ) {
222                    s.append( "," );
223                }
224            }
225            s.append( "}]\n" );
226        }
227        return s.toString();
228    }
