040/**
041 *  Defines an editor manager.  An editor can be added by adding a suitable JSP file under templates/default/editors
042 *  If you want your editor to include any scripts or something, you can simply request it by adding the following in your
043 *  {@code ini/jspwiki_module.xml}:
044 *
045 *  <pre>
046 *  &lt;modules>
047 *   &lt;editor name="myeditor">
048 *       &lt;author>Janne Jalkanen&lt;/author>
049 *       &lt;script>foo.js&lt;/script>
050 *       &lt;stylesheet>foo.css&lt;/stylesheet>
051 *       &lt;path>editors/myeditor.jsp&lt;/path>
052 *   &lt;/editor>
053 *  &lt;/modules>
054 *  </pre>
055 *
056 *  @since 2.4
057 */
058public class DefaultEditorManager extends BaseModuleManager implements EditorManager {
059
060    private Map< String, WikiEditorInfo > m_editors;
061
062    private static final Logger log = Logger.getLogger( DefaultEditorManager.class );
063
064    public DefaultEditorManager( final Engine engine ) {
065        super( engine );
066    }
067
068    /** {@inheritDoc} */
069    @Override
070    public void initialize( final Properties props ) {
071        registerEditors();
072    }
073
074    /** This method goes through the jspwiki_module.xml files and hunts for editors. Any editors found are put in the registry. */
075    private void registerEditors() {
076        log.info( "Registering editor modules" );
077        m_editors = new HashMap<>();
078
079        // Register all editors which have created a resource containing its properties. Get all resources of all modules
080        final List< Element > editors = XmlUtil.parse( PLUGIN_RESOURCE_LOCATION, "/modules/editor" );
081        for( final Element pluginEl : editors ) {
082            final String name = pluginEl.getAttributeValue( "name" );
083            final WikiEditorInfo info = WikiEditorInfo.newInstance( name, pluginEl );
084
085            if( checkCompatibility( info ) ) {
086                m_editors.put( name, info );
087                log.debug( "Registered editor " + name );
088            } else {
089                log.info( "Editor '" + name + "' not compatible with this version of JSPWiki." );
090            }
091        }
092    }
093
094    /** {@inheritDoc} */
095    @Override
096    public String getEditorName( final WikiContext context ) {
097        if( context.getRequestContext().equals( WikiContext.PREVIEW ) ) {
098            return EDITOR_PREVIEW;
099        }
100
101        // User has set an editor in preferences
102        String editor = Preferences.getPreference( context, PARA_EDITOR );
103
104        /* FIXME: actual default 'editor' property is read by the Preferences class */
105        if( editor == null ) {
106            // or use the default editor in jspwiki.properties
107            try {
108                editor = m_engine.getManager( VariableManager.class ).getValue( context, PROP_EDITORTYPE );
109            } catch( final NoSuchVariableException e ) {} // This is fine
110        }
111
112        if( editor != null ) {
113            final String[] editorlist = getEditorList();
114            editor = editor.trim();
115            for( final String s : editorlist ) {
116                if( s.equalsIgnoreCase( editor ) ) {
117                    return s;
118                }
119            }
120        }
121
122        return EDITOR_PLAIN;
123    }
124
125    /** {@inheritDoc} */
126    @Override
127    public String[] getEditorList() {
128        final String[] editors = new String[ m_editors.size() ];
129        final Set< String > keys = m_editors.keySet();
130
131        return keys.toArray( editors );
132    }
133
134    /** {@inheritDoc} */
135    @Override
136    public String getEditorPath( final WikiContext context ) {
137        final String editor = getEditorName( context );
138        final WikiEditorInfo ed = m_editors.get( editor );
139        final String path;
140        if( ed != null ) {
141            path = ed.getPath();
142        } else {
143            path = "editors/"+editor+".jsp";
144        }
145
146        return path;
147    }
148
149    /**  Contains info about an editor. */
150    private static final class WikiEditorInfo extends WikiModuleInfo {
151        private String m_path;
152
153        protected static WikiEditorInfo newInstance( final String name, final Element el ) {
154            if( name == null || name.length() == 0 ) {
155                return null;
156            }
157            final WikiEditorInfo info = new WikiEditorInfo( name );
158            info.initializeFromXML( el );
159            return info;
160        }
161
162        /** {@inheritDoc} */
163        @Override
164        protected void initializeFromXML( final Element el ) {
165            super.initializeFromXML( el );
166            m_path = el.getChildText("path");
167        }
168
169        private WikiEditorInfo( final String name ) {
170            super( name );
171        }
172
173        public String getPath() {
174            return m_path;
175        }
176    }
177
178    /** {@inheritDoc} */
179    @Override
180    public Collection< WikiModuleInfo > modules() {
181        return modules( m_editors.values().iterator() );
182    }
183
184    /** {@inheritDoc} */
185    @Override
186    public WikiEditorInfo getModuleInfo( final String moduleName ) {
187        return m_editors.get( moduleName );
188    }
189
190}
