019package org.apache.wiki.ui;
020
021import java.security.Permission;
022
023/**
024 * <p>Represents a logical "unit of work" that includes a request context, JSP, URLPattern, content template and (optionally) a target and
025 * required security permission. Examples of Commands include "view a page," "create a group," and "edit user preferences." </p>
026 * <p> Commands come in two flavors: "static" and "targeted." </p>
027 * <ul>
028 * <li><strong>Static commands</strong> are exactly what they sound like: static. They are <code>final</code>, threadsafe, and immutable.
029 * They have no intrinsic idea of the context they are acting in. For example, the static command {@link PageCommand#VIEW} embodies the
030 * idea of viewing a page &#8212; but exactly <em>which</em> page is left undefined. Static commands exist so that they can be freely
031 * shared and passed around without incurring the penalties of object creation. Static commands are a lot like naked request contexts
032 * ("edit", "view", etc.) except that they include additional, essential properties such as the associated URL pattern and content JSP.</li>
033 * <li><strong>Targeted commands</strong> "decorate" static commands by scoping a static Command at a specific target such as a WikiPage or
034 * GroupPrincipal. Targeted commands are created by calling an existing Command's {@link #targetedCommand(Object)} and supplying the target
035 * object. Implementing classes generally require a specific target type. For example, the {@link PageCommand} class requires that the
036 * target object be of type {@link org.apache.wiki.WikiPage}.</li>
037 * </ul>
038 * <p> Concrete implementations of Command include: </p>
039 * <ul>
040 * <li><strong>PageCommand</strong>: commands for editing, renaming, and viewing pages</li>
041 * <li><strong>GroupCommand</strong>: commands for viewing, editing and
042 * deleting wiki groups</li>
043 * <li><strong>WikiCommand</strong>: commands for wiki-wide operations such as
044 * creating groups, editing preferences and profiles, and logging in/out</li>
045 * <li><strong>RedirectCommand</strong>: commands for redirections to off-site
046 * special pages</li>
047 * </ul>
048 * <p>
049 * For a given targeted Command, its {@link #getTarget()} method will return a non-<code>null</code> value. In addition, its
050 * {@link #requiredPermission()} method will generally also return a non-<code>null</code> value. It is each implementation's responsibility
051 * to construct and store the correct Permission for a given Command and Target. For example, when PageCommand.VIEW is targeted at the
052 * WikiPage <code>Main</code>, the Command's associated permission is <code>PagePermission "<em>theWiki</em>:Main", "view".</code></p>
053 * <p>Static Commands, and targeted Commands that do not require specific permissions to execute, return a <code>null</code> result for
054 * {@link #requiredPermission()}.</p>
055 * @since 2.4.22
056 */
057public interface Command {
058
059    /**
060     * Creates and returns a targeted Command by combining a target, such as a WikiPage or GroupPrincipal into the existing Command.
061     * Subclasses should check to make sure the supplied <code>target</code> object is of the correct type. This method is guaranteed
062     * to return a non-<code>null</code> Command (unless the target is an incorrect type).
063     *
064     * @param target the object to combine, such as a GroupPrincipal or WikiPage
065     * @return the new, targeted Command
066     * @throws IllegalArgumentException if the target is not of the correct type
067     */
068    Command targetedCommand( Object target );
069
070    /**
071     * Returns the content template associated with a Command, such as <code>PreferencesContent.jsp</code>. For Commands that are not
072     * page-related, this method will always return <code>null</code>. <em>Calling methods should always check to see if the result
073     * of this method is <code>null</code></em>.
074     *
075     * @return the content template
076     */
077    String getContentTemplate();
078
079    /**
080     * Returns the JSP associated with the Command. The JSP is a "local" JSP within the JSPWiki webapp; it is not a general HTTP URL.
081     * If it exists, the JSP will be expressed relative to the webapp root, without a leading slash. This method is guaranteed to return
082     * a non-<code>null</code> result, although in some cases the result may be an empty string.
083     *
084     * @return the JSP or url associated with the wiki command
085     */
086    String getJSP();
087
088    /**
089     * Returns the human-friendly name for this command.
090     *
091     * @return the name
092     */
093    String getName();
094
095    /**
096     * Returns the request context associated with this Command. This method is guaranteed to return a non-<code>null</code> String.
097     * @return the request context
098     */
099    String getRequestContext();
100
101    /**
102     * Returns the Permission required to successfully execute this Command. If no Permission is requred, this method returns
103     * <code>null</code>. For example, the static command {@link PageCommand#VIEW} doesn't require a permission because it isn't referring
104     * to a particular WikiPage. However, if this command targets a WikiPage called <code>Main</code>(via
105     * {@link PageCommand#targetedCommand(Object)}, the resulting Command would require the permission
106     * <code>PagePermission "<em>yourWiki</em>:Main", "view"</code>.
107     *
108     * @return the required permission, or <code>null</code> if not required
109     */
110    Permission requiredPermission();
111
112    /**
113     * Returns the target associated with a Command, if it was created with one. Commands created with {@link #targetedCommand(Object)} will
114     * <em>always</em> return a non-<code>null</code> object. <em>Calling methods should always check to see if the result of this method
115     * is <code>null</code></em>.
116     *
117     * @return the wiki page
118     */
119    Object getTarget();
120
121    /**
122     * Returns the URL pattern associated with this Command. This method is guaranteed to return a non-<code>null</code> String.
123     *
124     * @return the URL pattern
125     */
126    String getURLPattern();
127
128}
