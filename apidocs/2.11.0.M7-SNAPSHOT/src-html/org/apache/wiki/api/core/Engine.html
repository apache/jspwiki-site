019package org.apache.wiki.api.core;
020
021import org.apache.wiki.WatchDog;
022import org.apache.wiki.api.exceptions.ProviderException;
023import org.apache.wiki.event.WikiEventListener;
024
025import javax.servlet.ServletContext;
026import java.nio.charset.Charset;
027import java.util.Collection;
028import java.util.Date;
029import java.util.Properties;
030
031
032public interface Engine {
033
034    /** The default inlining pattern.  Currently "*.png" */
035    String DEFAULT_INLINEPATTERN = "*.png";
036
037    /** The name used for the default template. The value is {@value}. */
038    String DEFAULT_TEMPLATE_NAME = "default";
039
040    /** Property for application name */
041    String PROP_APPNAME = "jspwiki.applicationName";
042
043    /** This property defines the inline image pattern.  It's current value is {@value} */
044    String PROP_INLINEIMAGEPTRN = "jspwiki.translatorReader.inlinePattern";
045
046    /** Property start for any interwiki reference. */
047    String PROP_INTERWIKIREF = "jspwiki.interWikiRef.";
048
049    /** If true, then the user name will be stored with the page data.*/
050    String PROP_STOREUSERNAME= "jspwiki.storeUserName";
051
052    /** Define the used encoding.  Currently supported are ISO-8859-1 and UTF-8 */
053    String PROP_ENCODING = "jspwiki.encoding";
054
055    /** Do not use encoding in WikiJSPFilter, default is false for most servers.
056     Double negative, cause for most servers you don't need the property */
057    String PROP_NO_FILTER_ENCODING = "jspwiki.nofilterencoding";
058
059    /** Property name for where the jspwiki work directory should be.
060     If not specified, reverts to ${java.tmpdir}. */
061    String PROP_WORKDIR = "jspwiki.workDir";
062
063    /** The name of the cookie that gets stored to the user browser. */
064    String PREFS_COOKIE_NAME = "JSPWikiUserProfile";
065
066    /** Property name for the "match english plurals" -hack. */
067    String PROP_MATCHPLURALS = "jspwiki.translatorReader.matchEnglishPlurals";
068
069    /** Property name for the template that is used. */
070    String PROP_TEMPLATEDIR = "jspwiki.templateDir";
071
072    /** Property name for the default front page. */
073    String PROP_FRONTPAGE = "jspwiki.frontPage";
074
075    /** Property name for setting the url generator instance */
076    String PROP_URLCONSTRUCTOR = "jspwiki.urlConstructor";
077
078    /** The name of the property containing the ACLManager implementing class. The value is {@value}. */
079    String PROP_ACL_MANAGER_IMPL = "jspwiki.aclManager";
080
081    /** If this property is set to false, we don't allow the creation of empty pages */
082    String PROP_ALLOW_CREATION_OF_EMPTY_PAGES = "jspwiki.allowCreationOfEmptyPages";
083
084    /**
085     * Retrieves the requested object instantiated by the Engine.
086     *
087     * @param manager requested object instantiated by the Engine.
088     * @param <T> type of the requested object.
089     * @return requested object instantiated by the Engine, {@code null} if not available.
090     */
091    < T > T getManager( Class< T > manager );
092
093    /**
094     * check if the WikiEngine has been configured.
095     *
096     * @return {@code true} if it has, {@code false} otherwise.
097     */
098    boolean isConfigured();
099
100    /**
101     *  Returns the set of properties that the WikiEngine was initialized with.  Note that this method returns a direct reference, so it's
102     *  possible to manipulate the properties.  However, this is not advised unless you really know what you're doing.
103     *
104     *  @return The wiki properties
105     */
106    Properties getWikiProperties();
107
108    /**
109     *  Returns the JSPWiki working directory set with "jspwiki.workDir".
110     *
111     *  @since 2.1.100
112     *  @return The working directory.
113     */
114    String getWorkDir();
115
116    /**
117     *  Returns the current template directory.
118     *
119     *  @since 1.9.20
120     *  @return The template directory as initialized by the engine.
121     */
122    String getTemplateDir();
123
124    /**
125     *  Returns the moment when this engine was started.
126     *
127     *  @since 2.0.15.
128     *  @return The start time of this wiki.
129     */
130    Date getStartTime();
131
132    /**
133     *  Returns the base URL, telling where this Wiki actually lives.
134     *
135     *  @since 1.6.1
136     *  @return The Base URL.
137     */
138    String getBaseURL();
139
140    /**
141     *  Returns the URL of the global RSS file.  May be null, if the RSS file generation is not operational.
142     *
143     *  @since 1.7.10
144     *  @return The global RSS url
145     */
146    String getGlobalRSSURL();
147
148    /**
149     *  Returns an URL to some other Wiki that we know.
150     *
151     *  @param  wikiName The name of the other wiki.
152     *  @return null, if no such reference was found.
153     */
154    String getInterWikiURL( String wikiName );
155
156    /**
157     *  Returns an URL if a WikiContext is not available.
158     *
159     *  @param context The WikiContext (VIEW, EDIT, etc...)
160     *  @param pageName Name of the page, as usual
161     *  @param params List of parameters. May be null, if no parameters.
162     *  @return An URL (absolute or relative).
163     */
164    String getURL( String context, String pageName, String params );
165
166    /**
167     *  Returns the default front page, if no page is used.
168     *
169     *  @return The front page name.
170     */
171    String getFrontPage();
172
173    /**
174     *  Returns the ServletContext that this particular WikiEngine was initialized with. <strong>It may return {@code null}</strong>,
175     *  if the Engine is not running inside a servlet container!
176     *
177     *  @since 1.7.10
178     *  @return ServletContext of the Engine, or {@code null}.
179     */
180    ServletContext getServletContext();
181
182    /**
183     *  Returns a collection of all supported InterWiki links.
184     *
185     *  @return A Collection of Strings.
186     */
187    Collection< String > getAllInterWikiLinks();
188
189    /**
190     *  Returns a collection of all image types that get inlined.
191     *
192     *  @return A Collection of Strings with a regexp pattern.
193     */
194    Collection< String > getAllInlinedImagePatterns();
195
196    /**
197     *  <p>If the page is a special page, then returns a direct URL to that page. Otherwise returns <code>null</code>.
198     *  This method delegates requests to {@link org.apache.wiki.ui.CommandResolver#getSpecialPageReference(String)}.</p>
199     *  <p>Special pages are defined in jspwiki.properties using the jspwiki.specialPage setting. They're typically used to give Wiki page
200     *  names to e.g. custom JSP pages.</p>
201     *
202     *  @param original The page to check
203     *  @return A reference to the page, or null, if there's no special page.
204     */
205    String getSpecialPageReference( String original );
206
207    /**
208     *  Returns the name of the application.
209     *
210     *  @return A string describing the name of this application.
211     */
212    String getApplicationName();
213
214    /**
215     *  Returns the root path.  The root path is where the WikiEngine is located in the file system.
216     *
217     *  @since 2.2
218     *  @return A path to where the Wiki is installed in the local filesystem.
219     */
220    String getRootPath();
221
222    /**
223     *  Returns the correct page name, or null, if no such page can be found.  Aliases are considered. This method simply delegates to
224     *  {@link org.apache.wiki.ui.CommandResolver#getFinalPageName(String)}.
225     *
226     *  @since 2.0
227     *  @param page Page name.
228     *  @return The rewritten page name, or null, if the page does not exist.
229     *  @throws ProviderException If something goes wrong in the backend.
230     */
231    String getFinalPageName( String page ) throws ProviderException;
232
233    /**
234     *  Turns a WikiName into something that can be called through using an URL.
235     *
236     *  @since 1.4.1
237     *  @param pagename A name. Can be actually any string.
238     *  @return A properly encoded name.
239     *  @see #decodeName(String)
240     */
241    String encodeName( String pagename );
242
243    /**
244     *  Decodes a URL-encoded request back to regular life.  This properly heeds the encoding as defined in the settings file.
245     *
246     *  @param pagerequest The URL-encoded string to decode
247     *  @return A decoded string.
248     *  @see #encodeName(String)
249     */
250    String decodeName( String pagerequest );
251
252    /**
253     *  Returns the IANA name of the character set encoding we're supposed to be using right now.
254     *
255     *  @since 1.5.3
256     *  @return The content encoding (either UTF-8 or ISO-8859-1).
257     */
258    Charset getContentEncoding();
259
260    /**
261     * Registers a WikiEventListener with this instance.
262     *
263     * @param listener the event listener
264     */
265    void addWikiEventListener( WikiEventListener listener );
266
267    /**
268     * Un-registers a WikiEventListener with this instance.
269     *
270     * @param listener the event listener
271     */
272    void removeWikiEventListener( WikiEventListener listener );
273
274    /**
275     * Adds an attribute to the engine for the duration of this engine.  The value is not persisted.
276     *
277     * @since 2.4.91
278     * @param key the attribute name
279     * @param value the value
280     */
281    void setAttribute( String key, Object value );
282
283    /**
284     *  Gets an attribute from the engine.
285     *
286     *  @param key the attribute name
287     *  @return the value
288     */
289    < T > T getAttribute( String key );
290
291    /**
292     *  Removes an attribute.
293     *
294     *  @param key The key of the attribute to remove.
295     *  @return The previous attribute, if it existed.
296     */
297    < T > T removeAttribute( String key );
298
299    /**
300     *  Returns a WatchDog for current thread.
301     *
302     *  @return The current thread WatchDog.
303     *  @since 2.4.92
304     */
305    WatchDog getCurrentWatchDog();
306
307    /**
308     * Signals that the Engine will be shut down by the servlet container.
309     */
310    void shutdown();
311
312}
