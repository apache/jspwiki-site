019package org.apache.wiki.pages;
020
021import com.codeborne.selenide.Configuration;
022import com.codeborne.selenide.Selenide;
023import org.apache.wiki.its.environment.Env;
024
025
026/**
027 * Builder for {@link Page} Objects.
028 */
029public class PageBuilder {
030
031    final String url;
032    final Page page;
033
034    PageBuilder( final String url, final Page page ) {
035        this.url = url;
036        this.page = page;
037    }
038
039    /**
040     * Sets up Selenide and opens the requested URL, allowing chaining additional operations over the given {@link Page} instance.
041     *
042     * @param page {@link Page} instance, in order to allow chaining operations.
043     * @return {@link Page} instance given on the {@link PageBuilder} constructor.
044     */
045    public < T extends Page > T openAs( final T page ) {
046        Configuration.headless = Env.TESTS_CONFIG_HEADLESS;
047        Configuration.fastSetValue = true; // default value seems to not send `[` or `{` characters to input controls. weird.
048        Configuration.reportsFolder = Env.TESTS_CONFIG_REPORTS_FOLDER;
049        Configuration.startMaximized = Env.TESTS_CONFIG_START_MAXIMIZED;
050        Configuration.downloadsFolder = Env.TESTS_CONFIG_DOWNLOADS_FOLDER;
051        System.setProperty( "wdm.targetPath", Env.TESTS_CONFIG_WDM_TARGET_PATH );
052        
053        Selenide.open( url );
054        return page;
055    }
056
057}
