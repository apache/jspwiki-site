050/**
051 * Logs in an user based on a cookie stored in the user's computer.  The cookie
052 * information is stored in the <code>jspwiki.workDir</code>, under the directory
053 * {@value #COOKIE_DIR}.  For security purposes it is a very, very good idea
054 * to prevent access to this directory by everyone except the web server process;
055 * otherwise people having read access to this directory may be able to spoof
056 * other users.
057 * <p>
058 * The cookie directory is scrubbed of old entries at regular intervals.
059 * <p>
060 * This module must be used with a CallbackHandler (such as
061 * {@link WebContainerCallbackHandler}) that supports the following Callback
062 * types:
063 * </p>
064 *  <ol>
065 *  <li>{@link HttpRequestCallback}- supplies the cookie, which should contain
066 *      an unique id for fetching the UID.</li>
067 *  <li>{@link WikiEngineCallback} - allows access to the Engine itself.
068 *  </ol>
069 * <p>
070 * After authentication, a generic WikiPrincipal based on the username will be
071 * created and associated with the Subject.
072 * </p>
073 * @see javax.security.auth.spi.LoginModule#commit()
074 * @see CookieAssertionLoginModule
075 * @since 2.5.62
076 */
077public class CookieAuthenticationLoginModule extends AbstractLoginModule {
078
079    private static final Logger log = LogManager.getLogger( CookieAuthenticationLoginModule.class );
080    private static final String LOGIN_COOKIE_NAME = "JSPWikiUID";
081
082    /** The directory name under which the cookies are stored.  The value is {@value}. */
083    protected static final String COOKIE_DIR = "logincookies";
084
085    /**
086     * User property for setting how long the cookie is stored on the user's computer.
087     * The value is {@value}.  The default expiry time is 14 days.
088     */
089    public static final String PROP_LOGIN_EXPIRY_DAYS = "jspwiki.cookieAuthentication.expiry";
090
091    /**
092     * Built-in value for storing the cookie.
093     */
094    private static final int DEFAULT_EXPIRY_DAYS = 14;
095
096    private static long c_lastScrubTime;
097
098    /**
099     * Describes how often we scrub the cookieDir directory.
100     */
101    private static final long SCRUB_PERIOD = 60 * 60 * 1000L; // In milliseconds
102
103    /**
104     * @see javax.security.auth.spi.LoginModule#login()
105     * {@inheritDoc}
106     */
107    @Override
108    public boolean login() throws LoginException {
109        // Otherwise, let's go and look for the cookie!
110        final HttpRequestCallback hcb = new HttpRequestCallback();
111        final WikiEngineCallback wcb = new WikiEngineCallback();
112
113        final Callback[] callbacks = new Callback[] { hcb, wcb };
114
115        try {
116            m_handler.handle( callbacks );
117
118            final HttpServletRequest request = hcb.getRequest();
119            final String uid = getLoginCookie( request );
120
121            if( uid != null ) {
122                final Engine engine = wcb.getEngine();
123                final File cookieFile = getCookieFile( engine, uid );
124
125                if( cookieFile != null && cookieFile.exists() && cookieFile.canRead() ) {
126
127                    try( final Reader in = new BufferedReader( new InputStreamReader( new FileInputStream( cookieFile ), StandardCharsets.UTF_8 ) ) ) {
128                        final String username = FileUtil.readContents( in );
129
130                        if( log.isDebugEnabled() ) {
131                            log.debug( "Logged in cookie authenticated name=" + username );
132                        }
133
134                        // If login succeeds, commit these principals/roles
135                        m_principals.add( new WikiPrincipal( username, WikiPrincipal.LOGIN_NAME ) );
136
137                        //
138                        //  Tag the file so that we know that it has been accessed recently.
139                        //
140                        return cookieFile.setLastModified( System.currentTimeMillis() );
141
142                    } catch( final IOException e ) {
143                        return false;
144                    }
145                }
146            }
147        } catch( final IOException e ) {
148            final String message = "IO exception; disallowing login.";
149            log.error( message, e );
150            throw new LoginException( message );
151        } catch( final UnsupportedCallbackException e ) {
152            final String message = "Unable to handle callback; disallowing login.";
153            log.error( message, e );
154            throw new LoginException( message );
155        }
156
157        return false;
158    }
159
160    /**
161     * Attempts to locate the cookie file.
162     *
163     * @param engine Engine
164     * @param uid    An unique ID fetched from the user cookie
165     * @return A File handle, or null, if there was a problem.
166     */
167    private static File getCookieFile( final Engine engine, final String uid ) {
168        final File cookieDir = new File( engine.getWorkDir(), COOKIE_DIR );
169
170        if( !cookieDir.exists() ) {
171            cookieDir.mkdirs();
172        }
173
174        if( !cookieDir.canRead() ) {
175            log.error( "Cannot read from cookie directory!" + cookieDir.getAbsolutePath() );
176            return null;
177        }
178
179        if( !cookieDir.canWrite() ) {
180            log.error( "Cannot write to cookie directory!" + cookieDir.getAbsolutePath() );
181            return null;
182        }
183
184        //
185        //  Scrub away old files
186        //
187        final long now = System.currentTimeMillis();
188
189        if( now > ( c_lastScrubTime + SCRUB_PERIOD ) ) {
190            scrub( TextUtil.getIntegerProperty( engine.getWikiProperties(), PROP_LOGIN_EXPIRY_DAYS, DEFAULT_EXPIRY_DAYS ), cookieDir );
191            c_lastScrubTime = now;
192        }
193
194        //
195        //  Find the cookie file
196        //
197        return new File( cookieDir, uid );
198    }
199
200    /**
201     * Extracts the login cookie UID from the servlet request.
202     *
203     * @param request The HttpServletRequest
204     * @return The UID value from the cookie, or null, if no such cookie exists.
205     */
206    private static String getLoginCookie( final HttpServletRequest request ) {
207        return HttpUtil.retrieveCookieValue( request, LOGIN_COOKIE_NAME );
208    }
209
210    /**
211     * Sets a login cookie based on properties set by the user.  This method also
212     * creates the cookie uid-username mapping in the work directory.
213     *
214     * @param engine   The Engine
215     * @param response The HttpServletResponse
216     * @param username The username for whom to create the cookie.
217     */
218    public static void setLoginCookie( final Engine engine, final HttpServletResponse response, final String username ) {
219        final UUID uid = UUID.randomUUID();
220        final int days = TextUtil.getIntegerProperty( engine.getWikiProperties(), PROP_LOGIN_EXPIRY_DAYS, DEFAULT_EXPIRY_DAYS );
221        final Cookie userId = getLoginCookie( uid.toString() );
222        userId.setMaxAge( days * 24 * 60 * 60 );
223        response.addCookie( userId );
224
225        final File cf = getCookieFile( engine, uid.toString() );
226        if( cf != null ) {
227            //  Write the cookie content to the cookie store file.
228            try( final Writer out = new BufferedWriter( new OutputStreamWriter( new FileOutputStream( cf ), StandardCharsets.UTF_8 ) ) ) {
229                FileUtil.copyContents( new StringReader( username ), out );
230
231                if( log.isDebugEnabled() ) {
232                    log.debug( "Created login cookie for user " + username + " for " + days + " days" );
233                }
234
235            } catch( final IOException ex ) {
236                log.error( "Unable to create cookie file to store user id: " + uid );
237            }
238        }
239    }
240
241    /**
242     * Clears away the login cookie, and removes the uid-username mapping file as well.
243     *
244     * @param engine   Engine
245     * @param request  Servlet request
246     * @param response Servlet response
247     */
248    public static void clearLoginCookie( final Engine engine, final HttpServletRequest request, final HttpServletResponse response ) {
249        final Cookie userId = getLoginCookie( "" );
250        userId.setMaxAge( 0 );
251        response.addCookie( userId );
252
253        final String uid = getLoginCookie( request );
254
255        if( uid != null ) {
256            final File cf = getCookieFile( engine, uid );
257
258            if( cf != null ) {
259                if( !cf.delete() ) {
260                    log.debug( "Error deleting cookie login " + uid );
261                }
262            }
263        }
264    }
265
266    /**
267     * Helper function to get secure LOGIN cookie
268     *
269     * @param: value of the cookie
270     */
271    private static Cookie getLoginCookie( final String value ) {
272        final Cookie c = new Cookie( LOGIN_COOKIE_NAME, value );
273        c.setHttpOnly( true );  //no browser access
274        c.setSecure( true ); //only access via encrypted https allowed
275        return c;
276    }
277
278    /**
279     * Goes through the cookie directory and removes any obsolete files.
280     * The scrubbing takes place one day after the cookie was supposed to expire.
281     * However, if the user has logged in during the expiry period, the expiry is
282     * reset, and the cookie file left here.
283     *
284     * @param days
285     * @param cookieDir
286     */
287    private static synchronized void scrub( final int days, final File cookieDir ) {
288        log.debug( "Scrubbing cookieDir..." );
289        final File[] files = cookieDir.listFiles();
290        final long obsoleteDateLimit = System.currentTimeMillis() - ( ( long )days + 1 ) * 24 * 60 * 60 * 1000L;
291        int deleteCount = 0;
292
293        for( int i = 0; i < files.length; i++ ) {
294            final File f = files[ i ];
295            final long lastModified = f.lastModified();
296            if( lastModified < obsoleteDateLimit ) {
297                if( f.delete() ) {
298                    deleteCount++;
299                } else {
300                    log.debug( "Error deleting cookie login with index " + i );
301                }
302            }
303        }
304
305        log.debug( "Removed " + deleteCount + " obsolete cookie logins" );
306    }
