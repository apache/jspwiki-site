019package org.apache.wiki.url;
020
021import org.apache.wiki.api.core.Engine;
022
023import javax.servlet.http.HttpServletRequest;
024import java.io.IOException;
025import java.nio.charset.Charset;
026import java.util.Properties;
027
028
029/**
030 *  Provides an interface through which JSPWiki constructs URLs. JSPWiki calls the methods of this interface whenever an URL
031 *  that points to any JSPWiki internals is required.  For example, if you need to find an URL to the editor page for page
032 *  "TextFormattingRules", you would call makeURL( WikiContext.EDIT, "TextFormattingRules", false, null );
033 *
034 *  @since 2.2
035 */
036public interface URLConstructor {
037    /**
038     *  Initializes.  Note that the engine is not fully initialized at this
039     *  point, so don't do anything fancy here - use lazy init, if you have to.
040     *
041     *  @param  engine The Engine that this URLConstructor belongs to
042     *  @param properties Properties used to initialize
043     */
044    void initialize( Engine engine, Properties properties );
045
046    /**
047     *  Constructs the URL with a bunch of parameters.
048     *
049     *  @param context The request context (@see WikiContext) that you want the URL for
050     *  @param name The page name (or in case of WikiContext.NONE, the auxiliary JSP page
051     *              or resource you want to point at).  This must be URL encoded.  Null is NOT safe.
052     *  @param parameters An URL parameter string (these must be URL-encoded, and separated with &amp;amp;)
053     *  @return An URL pointing to the resource.  Must never return null - throw an InternalWikiException  if something goes wrong.
054     */
055    String makeURL( String context, String name, String parameters );
056
057    /**
058     *  Should parse the "page" parameter from the actual request. This is essentially the reverse of makeURL() - whenever
059     *  a request constructed by calls to makeURL() is passed to this routine, it MUST be able to parse the resource name
060     *  (WikiPage, Attachment, other resource) from the request.
061     *
062     *  @param context In which request context the request was made (this should help in parsing)
063     *  @param request The HTTP request that was used when coming here
064     *  @param encoding The encoding with which the request was made (UTF-8 or ISO-8859-1).
065     *  @return This method must return the name of the resource.
066     *  @throws IOException If parsing failes
067     */
068    String parsePage( String context, HttpServletRequest request, Charset encoding ) throws IOException;
069
070    /**
071     *  Returns information which JSP page should continue handling this type of request.
072     *
073     * @param request The HTTP Request that was used to end up in this page.
074     * @return "Wiki.jsp", "PageInfo.jsp", etc.  Just return the name, JSPWiki will figure out the page.
075     */
076    String getForwardPage( HttpServletRequest request );
077
078    /**
079     *  Takes the name of the page from the request URI. The initial slash is also removed.  If there is no page, returns null.
080     *
081     *  @param request The request to parse
082     *  @param encoding The encoding to use
083     *
084     *  @return a parsed page name, or null, if it cannot be found
085     */
086    static String parsePageFromURL( final HttpServletRequest request, final Charset encoding ) {
087        final String name = request.getPathInfo();
088        if( name == null || name.length() <= 1 ) {
089            return null;
090        } else if( name.charAt(0) == '/' ) {
091            return name.substring(1);
092        }
093
094        //  This is required, because by default all URLs are handled as Latin1, even if they are really UTF-8.
095        // name = TextUtil.urlDecode( name, encoding );
096
097        return name;
098    }
099
100}
