019package org.apache.wiki.render;
020
021import org.apache.log4j.Logger;
022import org.apache.wiki.StringTransmutator;
023import org.apache.wiki.api.core.Context;
024import org.apache.wiki.api.core.Engine;
025import org.apache.wiki.api.core.Page;
026import org.apache.wiki.api.exceptions.WikiException;
027import org.apache.wiki.api.providers.PageProvider;
028import org.apache.wiki.event.WikiEventListener;
029import org.apache.wiki.modules.InternalModule;
030import org.apache.wiki.parser.MarkupParser;
031import org.apache.wiki.parser.WikiDocument;
032
033import java.io.IOException;
034import java.util.Properties;
035
036
037/**
038 *  This class provides a facade towards the differing rendering routines.  You should use the routines in this manager
039 *  instead of the ones in Engine, if you don't want the different side effects to occur - such as WikiFilters.
040 *  <p>
041 *  This class also manages a rendering cache, i.e. documents are stored between calls. You may control the cache by
042 *  tweaking the ehcache.xml file.
043 *  <p>
044 *
045 *  @since  2.4
046 */
047public interface RenderingManager extends WikiEventListener, InternalModule {
048
049    /** markup parser property. */
050    String PROP_PARSER = "jspwiki.renderingManager.markupParser";
051
052    /** default renderer property. */
053    String PROP_RENDERER = "jspwiki.renderingManager.renderer";
054
055    /** default wysiwyg renderer property. */
056    String PROP_WYSIWYG_RENDERER = "jspwiki.renderingManager.renderer.wysiwyg";
057
058    String PROP_BEAUTIFYTITLE = "jspwiki.breakTitleWithSpaces";
059
060    /** Name of the regular page cache. */
061    String DOCUMENTCACHE_NAME = "jspwiki.renderingCache";
062
063    /**
064     *  Initializes the RenderingManager.
065     *  Checks for cache size settings, initializes the document cache.
066     *  Looks for alternative WikiRenderers, initializes one, or the default
067     *  XHTMLRenderer, for use.
068     *
069     *  @param engine A Engine instance.
070     *  @param properties A list of properties to get parameters from.
071     *  @throws WikiException If the manager could not be initialized.
072     */
073    void initialize( Engine engine, Properties properties ) throws WikiException;
074
075    /**
076     *  Beautifies the title of the page by appending spaces in suitable places, if the user has so decreed in the properties when
077     *  constructing this Engine.  However, attachment names are only beautified by the name.
078     *
079     *  @param title The title to beautify
080     *  @return A beautified title (or, if beautification is off, returns the title without modification)
081     *  @since 1.7.11, moved to PageManager on 2.11.0
082     */
083    String beautifyTitle( String title );
084
085    /**
086     *  Beautifies the title of the page by appending non-breaking spaces in suitable places.  This is really suitable only for HTML output,
087     *  as it uses the &amp;nbsp; -character.
088     *
089     *  @param title The title to beautify
090     *  @return A beautified title.
091     *  @since 2.1.127
092     */
093    String beautifyTitleNoBreak( String title );
094
095    /**
096     *  Returns the wiki Parser
097     *  @param pagedata the page data
098     *  @return A MarkupParser instance.
099     */
100    MarkupParser getParser( Context context, String pagedata );
101
102    /**
103     *  Returns a cached document, if one is found.
104     *
105     * @param context the wiki context
106     * @param pagedata the page data
107     * @return the rendered wiki document
108     */
109    WikiDocument getRenderedDocument( Context context, String pagedata );
110
111    /**
112     * Returns a WikiRenderer instance, initialized with the given context and doc. The object is an XHTMLRenderer,
113     * unless overridden in jspwiki.properties with PROP_RENDERER.
114     *
115     * @param context The WikiContext
116     * @param doc The document to render
117     * @return A WikiRenderer for this document, or null, if no such renderer could be instantiated.
118     */
119    WikiRenderer getRenderer( Context context, WikiDocument doc );
120
121    /**
122     * Returns a WikiRenderer instance meant for WYSIWYG editing, initialized with the given
123     * context and doc. The object is an WysiwygEditingRenderer, unless overridden
124     * in jspwiki.properties with PROP_WYSIWYG_RENDERER.
125     *
126     * @param context The WikiContext
127     * @param doc The document to render
128     * @return A WikiRenderer instance meant for WYSIWYG editing, for this document, or null, if no such renderer could be instantiated.
129     */
130    WikiRenderer getWysiwygRenderer( Context context, WikiDocument doc );
131
132    /**
133     *  Simply renders a WikiDocument to a String.  This version does not get the document from the cache - in fact, it does
134     *  not cache the document at all.  This is very useful, if you have something that you want to render outside the caching
135     *  routines.  Because the cache is based on full pages, and the cache keys are based on names, use this routine if you're
136     *  rendering anything for yourself.
137     *
138     *  @param context The WikiContext to render in
139     *  @param doc A proper WikiDocument
140     *  @return Rendered HTML.
141     *  @throws IOException If the WikiDocument is poorly formed.
142     */
143    String getHTML( Context context, WikiDocument doc ) throws IOException;
144
145    /**
146     *  Returns the converted HTML of the page using a different context than the default context.
147     *
148     *  @param  context A WikiContext in which you wish to render this page in.
149     *  @param  page WikiPage reference.
150     *  @return HTML-rendered version of the page.
151     */
152    String getHTML( Context context, Page page );
153
154    /**
155     *  Returns the converted HTML of the page's specific version. The version must be a positive integer, otherwise the current
156     *  version is returned.
157     *
158     *  @param pagename WikiName of the page to convert.
159     *  @param version Version number to fetch
160     *  @return HTML-rendered page text.
161     */
162    String getHTML( String pagename, int version );
163
164    /**
165     *   Convenience method for rendering, using the default parser and renderer.  Note that you can't use this method
166     *   to do any arbitrary rendering, as the pagedata MUST be the data from the that the WikiContext refers to - this
167     *   method caches the HTML internally, and will return the cached version.  If the pagedata is different from what
168     *   was cached, will re-render and store the pagedata into the internal cache.
169     *
170     *   @param context the wiki context
171     *   @param pagedata the page data
172     *   @return XHTML data.
173     */
174    default String getHTML( final Context context, final String pagedata ) {
175        try {
176            final WikiDocument doc = getRenderedDocument( context, pagedata );
177            return getHTML( context, doc );
178        } catch( final IOException e ) {
179            Logger.getLogger( RenderingManager.class ).error("Unable to parse", e );
180        }
181
182        return null;
183    }
184
185    /**
186     *  Returns the converted HTML of the page.
187     *
188     *  @param page WikiName of the page to convert.
189     *  @return HTML-rendered version of the page.
190     */
191    default String getHTML( final String page ) {
192        return getHTML( page, PageProvider.LATEST_VERSION );
193    }
194
195    /**
196     *  Converts raw page data to HTML.
197     *
198     *  @param pagedata Raw page data to convert to HTML
199     *  @param context  The WikiContext in which the page is to be rendered
200     *  @return Rendered page text
201     */
202    String textToHTML( Context context, String pagedata );
203
204    /**
205     *  Helper method for doing the HTML translation.
206     *
207     *  @param context The WikiContext in which to do the conversion
208     *  @param pagedata The data to render
209     *  @param localLinkHook Is called whenever a wiki link is found
210     *  @param extLinkHook   Is called whenever an external link is found
211     *  @param parseAccessRules Parse the access rules if we encounter them
212     *  @param justParse Just parses the pagedata, does not actually render.  In this case, this methods an empty string.
213     *  @return HTML-rendered page text.
214     */
215    String textToHTML( Context context,
216                       String pagedata,
217                       StringTransmutator localLinkHook,
218                       StringTransmutator extLinkHook,
219                       StringTransmutator attLinkHook,
220                       boolean parseAccessRules,
221                       boolean justParse );
222
223    /**
224     *  Just convert WikiText to HTML.
225     *
226     *  @param context The WikiContext in which to do the conversion
227     *  @param pagedata The data to render
228     *  @param localLinkHook Is called whenever a wiki link is found
229     *  @param extLinkHook   Is called whenever an external link is found
230     *
231     *  @return HTML-rendered page text.
232     */
233    default String textToHTML( final Context context,
234                               final String pagedata,
235                               final StringTransmutator localLinkHook,
236                               final StringTransmutator extLinkHook ) {
237        return textToHTML( context, pagedata, localLinkHook, extLinkHook, null, true, false );
238    }
239
240    /**
241     *  Just convert WikiText to HTML.
242     *
243     *  @param context The WikiContext in which to do the conversion
244     *  @param pagedata The data to render
245     *  @param localLinkHook Is called whenever a wiki link is found
246     *  @param extLinkHook   Is called whenever an external link is found
247     *  @param attLinkHook   Is called whenever an attachment link is found
248     *  @return HTML-rendered page text.
249     */
250    default String textToHTML( final Context context,
251                               final String pagedata,
252                               final StringTransmutator localLinkHook,
253                               final StringTransmutator extLinkHook,
254                               final StringTransmutator attLinkHook ) {
255        return textToHTML( context, pagedata, localLinkHook, extLinkHook, attLinkHook, true, false );
256    }
