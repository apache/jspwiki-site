019package org.apache.wiki.api.core;
020
021import javax.servlet.http.HttpServletRequest;
022import java.security.Principal;
023
024
025/**
026 *  <p>Provides state information throughout the processing of a page.  A Context is born when the JSP pages that are the main entry
027 *  points, are invoked.  The JSPWiki engine creates the new Context, which basically holds information about the page, the
028 *  handling engine, and in which context (view, edit, etc) the call was done.</p>
029 *  <p>A Context also provides request-specific variables, which can be used to communicate between plugins on the same page, or
030 *  between different instances of the same plugin. A Context variable is valid until the processing of the WikiPage has ended. For
031 *  an example, please see the Counter plugin.</p>
032 *  <p>When a Context is created, it automatically associates a {@link Session} object with the user's
033 *  HttpSession. The Session contains information about the user's authentication status, and is consulted by {@link #getCurrentUser()}
034 *  object.</p>
035 *  <p>Do not cache the WikiPage object that you get from the WikiContext; always use getPage()!</p>
036 *
037 *  @see org.apache.wiki.plugin.Counter
038 */
039public interface Context extends Cloneable, Command {
040
041    String ATTR_CONTEXT = "jspwiki.context";
042
043    /**
044     *  Variable name which tells whether plugins should be executed or not. Value can be either {@code Boolean.TRUE} or
045     *  {@code Boolean.FALSE}. While not set it's value is {@code null}.
046     */
047    String VAR_EXECUTE_PLUGINS = "_PluginContent.execute";
048
049    /** Name of the variable which is set to Boolean.TRUE or Boolean.FALSE depending on whether WYSIWYG is currently in effect. */
050    String VAR_WYSIWYG_EDITOR_MODE = "WYSIWYG_EDITOR_MODE";
051
052    /**
053     *  Returns the WikiPage that is being handled.
054     *
055     *  @return the WikiPage which was fetched.
056     */
057    Page getPage();
058
059    /**
060     *  Sets the WikiPage that is being handled.
061     *
062     *  @param wikiPage The wikipage
063     *  @since 2.1.37.
064     */
065    void setPage( Page wikiPage );
066
067    /**
068     *  Gets a reference to the real WikiPage whose content is currently being rendered. If your plugin e.g. does some variable setting, be
069     *  aware that if it is embedded in the LeftMenu or some other WikiPage added with InsertPageTag, you should consider what you want to
070     *  do - do you wish to really reference the "master" WikiPage or the included page.
071     *  <p>
072     *  For example, in the default template, there is a WikiPage called "LeftMenu". Whenever you access a page, e.g. "Main", the master
073     *  WikiPage will be Main, and that's what the getPage() will return - regardless of whether your plugin resides on the LeftMenu or on
074     *  the Main page.  However, getRealPage() will return "LeftMenu".
075     *
076     *  @return A reference to the real page.
077     *  @see org.apache.wiki.tags.InsertPageTag
078     *  @see org.apache.wiki.parser.JSPWikiMarkupParser
079     */
080    Page getRealPage();
081
082    /**
083     *  Sets a reference to the real WikiPage whose content is currently being rendered.
084     *  <p>
085     *  Sometimes you may want to render the WikiPage using some other page's context. In those cases, it is highly recommended that you set
086     *  the setRealPage() to point at the real WikiPage you are rendering.  Please see InsertPageTag for an example.
087     *  <p>
088     *  Also, if your plugin e.g. does some variable setting, be aware that if it is embedded in the LeftMenu or some other WikiPage added
089     *  with InsertPageTag, you should consider what you want to do - do you wish to really reference the "master" WikiPage or the included
090     *  page.
091     *
092     *  @param wikiPage  The real WikiPage which is being rendered.
093     *  @return The previous real page
094     *  @since 2.3.14
095     *  @see org.apache.wiki.tags.InsertPageTag
096     */
097    Page setRealPage( Page wikiPagePage );
098
099    /**
100     *  Returns the handling engine.
101     *
102     *  @return The wikiengine owning this context.
103     */
104    Engine getEngine();
105
106    /**
107     *  Sets the request context.  See above for the different request contexts (VIEW, EDIT, etc.)
108     *
109     *  @param context The request context (one of the predefined contexts.)
110     */
111    void setRequestContext( String context );
112
113    /**
114     *  Gets a previously set variable.
115     *
116     *  @param key The variable name.
117     *  @return The variable contents.
118     */
119    < T > T getVariable( String key );
120
121    /**
122     *  Sets a variable.  The variable is valid while the WikiContext is valid, i.e. while WikiPage processing continues.  The variable data
123     *  is discarded once the WikiPage processing is finished.
124     *
125     *  @param key The variable name.
126     *  @param data The variable value.
127     */
128    void setVariable( String key, Object data );
129
130    /**
131     * This is just a simple helper method which will first check the context if there is already an override in place, and if there is not,
132     * it will then check the given properties.
133     *
134     * @param key What key are we searching for?
135     * @param defValue Default value for the boolean
136     * @return {@code true} or {@code false}.
137     */
138    boolean getBooleanWikiProperty( String key, boolean defValue );
139
140    /**
141     *  This method will safely return any HTTP parameters that might have been defined.  You should use this method instead
142     *  of peeking directly into the result of getHttpRequest(), since this method is smart enough to do all of the right things,
143     *  figure out UTF-8 encoded parameters, etc.
144     *
145     *  @since 2.0.13.
146     *  @param paramName Parameter name to look for.
147     *  @return HTTP parameter, or null, if no such parameter existed.
148     */
149    String getHttpParameter( String paramName );
150
151    /**
152     *  If the request did originate from a HTTP request, then the HTTP request can be fetched here.  However, it the request
153     *  did NOT originate from a HTTP request, then this method will return null, and YOU SHOULD CHECK FOR IT!
154     *
155     *  @return Null, if no HTTP request was done.
156     *  @since 2.0.13.
157     */
158    HttpServletRequest getHttpRequest();
159
160    /**
161     *  Sets the template to be used for this request.
162     *
163     *  @param dir The template name
164     *  @since 2.1.15.
165     */
166    void setTemplate( String dir );
167
168    /**
169     *  Gets the template that is to be used throughout this request.
170     *
171     *  @since 2.1.15.
172     *  @return template name
173     */
174    String getTemplate();
175
176    /**
177     *  Returns the Session associated with the context. This method is guaranteed to always return a valid Session.
178     *  If this context was constructed without an associated HttpServletRequest, it will return a guest session.
179     *
180     *  @return The Session associate with this context.
181     */
182    Session getWikiSession();
183
184    /**
185     *  Convenience method that gets the current user. Delegates the lookup to the Session associated with this Context.
186     *  May return null, in case the current user has not yet been determined; or this is an internal system. If the Session has not
187     *  been set, <em>always</em> returns null.
188     *
189     *  @return The current user; or maybe null in case of internal calls.
190     */
191    Principal getCurrentUser();
192
193    /**
194     *  Returns true, if the current user has administrative permissions (i.e. the omnipotent AllPermission).
195     *
196     *  @since 2.4.46
197     *  @return true, if the user has all permissions.
198     */
199    boolean hasAdminPermissions();
200
201    /**
202     *  A shortcut to generate a VIEW url.
203     *
204     *  @param WikiPage The WikiPage to which to link.
205     *  @return An URL to the page.  This honours the current absolute/relative setting.
206     */
207    String getViewURL( String WikiPage );
208
209    /**
210     *  Figure out to which WikiPage we are really going to.  Considers special WikiPage names from the jspwiki.properties, and possible aliases.
211     *
212     *  @return A complete URL to the new WikiPage to redirect to
213     *  @since 2.2
214     */
215    String getRedirectURL();
216
217    /**
218     * Returns the Command associated with this Context.
219     *
220     * @return the command
221     */
222    Command getCommand();
223
224    /**
225     *  Creates an URL for the given request context.
226     *
227     *  @param context e.g. WikiContext.EDIT
228     *  @param page The WikiPage to which to link
229     *  @return An URL to the page.
230     */
231    default String getURL( final String context, final String page ) {
232        return getURL( context, page, null );
233    }
234
235    /**
236     *  Returns an URL from a page. It this Context instance was constructed with an actual HttpServletRequest, we will attempt to
237     *  construct the URL using HttpUtil, which preserves the HTTPS portion if it was used.
238     *
239     *  @param context The request context (e.g. WikiContext.UPLOAD)
240     *  @param page The WikiPage to which to link
241     *  @param params A list of parameters, separated with "&amp;"
242     *
243     *  @return An URL to the given context and page.
244     */
245    default String getURL( final String context, final String page, final String params ) {
246        // FIXME: is rather slow
247        return getEngine().getURL( context, page, params );
248    }
249
250    /** {@inheritDoc} */
251    Context clone();
252
253}
