057/**
058 *  This class provides a facade towards the differing rendering routines.  You should use the routines in this manager
059 *  instead of the ones in Engine, if you don't want the different side effects to occur - such as WikiFilters.
060 *  <p>
061 *  This class also manages a rendering cache, i.e. documents are stored between calls. You may control the cache by
062 *  tweaking the ehcache.xml file.
063 *  <p>
064 *
065 *  @since  2.4
066 */
067public class DefaultRenderingManager implements RenderingManager {
068
069    private static final Logger log = Logger.getLogger( DefaultRenderingManager.class );
070
071    /** The capacity of the caches, if you want something else, tweak ehcache.xml. */
072    private static final int    DEFAULT_CACHESIZE     = 1_000;
073    private static final String VERSION_DELIMITER     = "::";
074
075    /** The name of the default renderer. */
076    private static final String DEFAULT_PARSER = JSPWikiMarkupParser.class.getName();
077    /** The name of the default renderer. */
078    private static final String DEFAULT_RENDERER = XHTMLRenderer.class.getName();
079    /** The name of the default WYSIWYG renderer. */
080    private static final String DEFAULT_WYSIWYG_RENDERER = WysiwygEditingRenderer.class.getName();
081
082    private Engine m_engine;
083
084    private boolean m_useCache = true;
085    private final CacheManager m_cacheManager = CacheManager.getInstance();
086    private final int m_cacheExpiryPeriod = 24*60*60; // This can be relatively long
087
088    /** If true, all titles will be cleaned. */
089    private boolean m_beautifyTitle = false;
090
091    /** Stores the WikiDocuments that have been cached. */
092    private Cache m_documentCache;
093
094    private Constructor< ? > m_rendererConstructor;
095    private Constructor< ? > m_rendererWysiwygConstructor;
096    private String m_markupParserClass = DEFAULT_PARSER;

098    /**
099     *  {@inheritDoc}
100     */
101    @Override
102    public void initialize( final Engine engine, final Properties properties ) throws WikiException {
103        m_engine = engine;
104        m_markupParserClass = properties.getProperty( PROP_PARSER, DEFAULT_PARSER );
105        if( !ClassUtil.assignable( m_markupParserClass, MarkupParser.class.getName() ) ) {
106            log.warn( m_markupParserClass + " does not subclass " + MarkupParser.class.getName() + " reverting to default markup parser." );
107            m_markupParserClass = DEFAULT_PARSER;
108        }
109        log.info( "Using " + m_markupParserClass + " as markup parser." );
110
111        m_beautifyTitle  = TextUtil.getBooleanProperty( properties, PROP_BEAUTIFYTITLE, m_beautifyTitle );
112        m_useCache = "true".equals( properties.getProperty( PageManager.PROP_USECACHE ) );
113
114        if( m_useCache ) {
115            final String documentCacheName = engine.getApplicationName() + "." + DOCUMENTCACHE_NAME;
116            if (m_cacheManager.cacheExists(documentCacheName)) {
117                m_documentCache = m_cacheManager.getCache(documentCacheName);
118            } else {
119                log.info( "cache with name " + documentCacheName + " not found in ehcache.xml, creating it with defaults." );
120                m_documentCache = new Cache( documentCacheName, DEFAULT_CACHESIZE, false, false, m_cacheExpiryPeriod, m_cacheExpiryPeriod );
121                m_cacheManager.addCache( m_documentCache );
122            }
123        }
124
125        final String renderImplName = properties.getProperty( PROP_RENDERER, DEFAULT_RENDERER );
126        final String renderWysiwygImplName = properties.getProperty( PROP_WYSIWYG_RENDERER, DEFAULT_WYSIWYG_RENDERER );
127
128        final Class< ? >[] rendererParams = { WikiContext.class, WikiDocument.class };
129        m_rendererConstructor = initRenderer( renderImplName, rendererParams );
130        m_rendererWysiwygConstructor = initRenderer( renderWysiwygImplName, rendererParams );
131
132        log.info( "Rendering content with " + renderImplName + "." );
133
134        WikiEventManager.getInstance().addWikiEventListener( m_engine.getManager( FilterManager.class ),this );
135    }
136
137    private Constructor< ? > initRenderer( final String renderImplName, final Class< ? >[] rendererParams ) throws WikiException {
138        Constructor< ? > c = null;
139        try {
140            final Class< ? > clazz = Class.forName( renderImplName );
141            c = clazz.getConstructor( rendererParams );
142        } catch( final ClassNotFoundException e ) {
143            log.error( "Unable to find WikiRenderer implementation " + renderImplName );
144        } catch( final SecurityException e ) {
145            log.error( "Unable to access the WikiRenderer(WikiContext,WikiDocument) constructor for "  + renderImplName );
146        } catch( final NoSuchMethodException e ) {
147            log.error( "Unable to locate the WikiRenderer(WikiContext,WikiDocument) constructor for "  + renderImplName );
148        }
149        if( c == null ) {
150            throw new WikiException( "Failed to get WikiRenderer '" + renderImplName + "'." );
151        }
152        return c;
153    }

155    /**
156     * {@inheritDoc}
157     */
158    @Override
159    public String beautifyTitle( final String title ) {
160        if( m_beautifyTitle ) {
161            try {
162                final Attachment att = m_engine.getManager( AttachmentManager.class ).getAttachmentInfo( title );
163                if( att == null ) {
164                    return TextUtil.beautifyString( title );
165                }
166
167                final String parent = TextUtil.beautifyString( att.getParentName() );
168                return parent + "/" + att.getFileName();
169            } catch( final ProviderException e ) {
170                return title;
171            }
172        }
173
174        return title;
175    }

192    @Override
193    public MarkupParser getParser( final WikiContext context, final String pagedata ) {
194        try {
195            return ClassUtil.getMappedObject( m_markupParserClass, context, new StringReader( pagedata ) );
196        } catch( final ReflectiveOperationException | IllegalArgumentException e ) {
197            log.error( "unable to get an instance of " + m_markupParserClass + " (" + e.getMessage() + "), returning default markup parser.", e );
198            return new JSPWikiMarkupParser( context, new StringReader( pagedata ) );
199        }
200    }

202    /**
203     *  {@inheritDoc}
204     */
205    @Override
206    // FIXME: The cache management policy is not very good: deleted/changed pages should be detected better.
207    public WikiDocument getRenderedDocument( final WikiContext context, final String pagedata ) {
208        final String pageid = context.getRealPage().getName() + VERSION_DELIMITER +
209                              context.getRealPage().getVersion() + VERSION_DELIMITER +
210                              context.getVariable( WikiContext.VAR_EXECUTE_PLUGINS );
211
212        if( useCache( context ) ) {
213            final Element element = m_documentCache.get( pageid );
214            if ( element != null ) {
215                final WikiDocument doc = ( WikiDocument )element.getObjectValue();
216
217                //
218                //  This check is needed in case the different filters have actually changed the page data.
219                //  FIXME: Figure out a faster method
220                if( pagedata.equals( doc.getPageData() ) ) {
221                    if( log.isDebugEnabled() ) {
222                        log.debug( "Using cached HTML for page " + pageid );
223                    }
224                    return doc;
225                }
226            } else if( log.isDebugEnabled() ) {
227                log.debug( "Re-rendering and storing " + pageid );
228            }
229        }
230
231        //  Refresh the data content
232        try {
233            final MarkupParser parser = getParser( context, pagedata );
234            final WikiDocument doc = parser.parse();
235            doc.setPageData( pagedata );
236            if( useCache( context ) ) {
237                m_documentCache.put( new Element( pageid, doc ) );
238            }
239            return doc;
240        } catch( final IOException ex ) {
241            log.error( "Unable to parse", ex );
242        }
243
244        return null;
245    }
246
247    boolean useCache( final WikiContext context ) {
248        return m_useCache && WikiContext.VIEW.equals( context.getRequestContext() );
249    }

251    /**
252     *  {@inheritDoc}
253     */
254    @Override
255    public String getHTML( final WikiContext context, final WikiDocument doc ) throws IOException {
256        final Boolean wysiwygVariable = ( Boolean )context.getVariable( WikiContext.VAR_WYSIWYG_EDITOR_MODE );
257        final boolean wysiwygEditorMode;
258        if( wysiwygVariable != null ) {
259            wysiwygEditorMode = wysiwygVariable;
260        } else {
261            wysiwygEditorMode = false;
262        }
263        final WikiRenderer rend;
264        if( wysiwygEditorMode ) {
265            rend = getWysiwygRenderer( context, doc );
266        } else {
267            rend = getRenderer( context, doc );
268        }
269
270        return rend.getString();
271    }

282    /**
283     *  Returns the converted HTML of the page's specific version. The version must be a positive integer, otherwise the current
284     *  version is returned.
285     *
286     *  @param pagename WikiName of the page to convert.
287     *  @param version Version number to fetch
288     *  @return HTML-rendered page text.
289     */
290    @Override public String getHTML( final String pagename, final int version ) {
291        final WikiPage page = m_engine.getManager( PageManager.class ).getPage( pagename, version );
292        final WikiContext context = new WikiContext( m_engine, page );
293        context.setRequestContext( WikiContext.NONE );
294        return getHTML( context, page );
295    }

297    /**
298     *  {@inheritDoc}
299     */
300    @Override
301    public String textToHTML( final WikiContext context, String pagedata ) {
302        String result = "";
303
304        final boolean runFilters = "true".equals( m_engine.getManager( VariableManager.class ).getValue( context,VariableManager.VAR_RUNFILTERS,"true" ) );
305
306        final StopWatch sw = new StopWatch();
307        sw.start();
308        try {
309            if( runFilters ) {
310                pagedata = m_engine.getManager( FilterManager.class ).doPreTranslateFiltering( context, pagedata );
311            }
312
313            result = getHTML( context, pagedata );
314
315            if( runFilters ) {
316                result = m_engine.getManager( FilterManager.class ).doPostTranslateFiltering( context, result );
317            }
318        } catch( final FilterException e ) {
319            log.error( "page filter threw exception: ", e );
320            // FIXME: Don't yet know what to do
321        }
322        sw.stop();
323        if( log.isDebugEnabled() ) {
324            log.debug( "Page " + context.getRealPage().getName() + " rendered, took " + sw );
325        }
326
327        return result;
328    }

330    /**
331     *  {@inheritDoc}
332     */
333    @Override
334    public String textToHTML( final WikiContext context,
335                              String pagedata,
336                              final StringTransmutator localLinkHook,
337                              final StringTransmutator extLinkHook,
338                              final StringTransmutator attLinkHook,
339                              final boolean parseAccessRules,
340                              final boolean justParse ) {
341        String result = "";
342
343        if( pagedata == null ) {
344            log.error("NULL pagedata to textToHTML()");
345            return null;
346        }
347
348        final boolean runFilters = "true".equals( m_engine.getManager( VariableManager.class ).getValue( context, VariableManager.VAR_RUNFILTERS,"true" ) );
349
350        try {
351            final StopWatch sw = new StopWatch();
352            sw.start();
353
354            if( runFilters && m_engine.getManager( FilterManager.class ) != null ) {
355                pagedata = m_engine.getManager( FilterManager.class ).doPreTranslateFiltering( context, pagedata );
356            }
357
358            final MarkupParser mp = getParser( context, pagedata );
359            mp.addLocalLinkHook( localLinkHook );
360            mp.addExternalLinkHook( extLinkHook );
361            mp.addAttachmentLinkHook( attLinkHook );
362
363            if( !parseAccessRules ) {
364                mp.disableAccessRules();
365            }
366
367            final WikiDocument doc = mp.parse();
368
369            //  In some cases it's better just to parse, not to render
370            if( !justParse ) {
371                result = getHTML( context, doc );
372
373                if( runFilters && m_engine.getManager( FilterManager.class ) != null ) {
374                    result = m_engine.getManager( FilterManager.class ).doPostTranslateFiltering( context, result );
375                }
376            }
377
378            sw.stop();
379
380            if( log.isDebugEnabled() ) {
381                log.debug( "Page " + context.getRealPage().getName() + " rendered, took " + sw );
382            }
383        } catch( final IOException e ) {
384            log.error( "Failed to scan page data: ", e );
385        } catch( final FilterException e ) {
386            log.error( "page filter threw exception: ", e );
387            // FIXME: Don't yet know what to do
388        }
389
390        return result;
391    }

411    @SuppressWarnings("unchecked")
412    private < T extends WikiRenderer > T getRenderer( final Object[] params, final Constructor<?> rendererConstructor ) {
413        try {
414            return ( T )rendererConstructor.newInstance( params );
415        } catch( final Exception e ) {
416            log.error( "Unable to create WikiRenderer", e );
417        }
418        return null;

421    /**
422     * {@inheritDoc}
423     *
424     * <p>Flushes the document cache in response to a POST_SAVE_BEGIN event.
425     *
426     * @see WikiEventListener#actionPerformed(WikiEvent)
427     */
428    @Override
429    public void actionPerformed( final WikiEvent event ) {
430        log.debug( "event received: " + event.toString() );
431        if( m_useCache ) {
432            if( ( event instanceof WikiPageEvent ) && ( event.getType() == WikiPageEvent.POST_SAVE_BEGIN ) ) {
433                if( m_documentCache != null ) {
434                    final String pageName = ( ( WikiPageEvent ) event ).getPageName();
435                    m_documentCache.remove( pageName );
436                    final Collection< String > referringPages = m_engine.getManager( ReferenceManager.class ).findReferrers( pageName );
437
438                    //
439                    //  Flush also those pages that refer to this page (if an nonexistent page
440                    //  appears, we need to flush the HTML that refers to the now-existent page)
441                    //
442                    if( referringPages != null ) {
443                        for( final String page : referringPages ) {
444                            if( log.isDebugEnabled() ) {
445                                log.debug( "Flushing latest version of " + page );
446                            }
447                            // as there is a new version of the page expire both plugin and pluginless versions of the old page
448                            m_documentCache.remove( page + VERSION_DELIMITER + WikiPageProvider.LATEST_VERSION  + VERSION_DELIMITER + Boolean.FALSE );
449                            m_documentCache.remove( page + VERSION_DELIMITER + WikiPageProvider.LATEST_VERSION  + VERSION_DELIMITER + Boolean.TRUE );
450                            m_documentCache.remove( page + VERSION_DELIMITER + WikiPageProvider.LATEST_VERSION  + VERSION_DELIMITER + null );
451                        }
452                    }
453                }
454            }
455        }
456    }
