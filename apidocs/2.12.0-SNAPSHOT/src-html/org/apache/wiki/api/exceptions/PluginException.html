019package org.apache.wiki.api.exceptions;
020
021
022/**
023 *  Provides a generic PluginException.  This is the kind of an exception that the plugins should throw.
024 */
025public class PluginException  extends WikiException {
026
027    private static final long serialVersionUID = -289900047240960332L;
028
029    private final Throwable m_throwable;
030
031    /**
032     *  Create a PluginException.
033     *
034     *  @param message exception message.
035     */
036    public PluginException( final String message ) {
037        super( message );
038        m_throwable = null;
039    }
040
041    /**
042     *  Create a PluginException with the given original exception wrapped.
043     *
044     *  @param message exception message.
045     *  @param original The original exception.
046     */
047    public PluginException( final String message, final Throwable original ) {
048        super( message, original );
049        m_throwable = original;
050    }
051
052    /**
053     *  Return the original exception.
054     *
055     *  @return The original exception.
056     */
057    public Throwable getRootThrowable() {
058        return m_throwable;
059    }
060
061}
