065/**
066 * This plugin counts the number of times a page has been viewed.<br/>
067 * Parameters:
068 * <ul>
069 * <li>count=yes|no</li>
070 * <li>show=none|count|list</li>
071 * <li>entries=maximum number of list entries to be returned</li>
072 * <li>min=minimum page count to be listed</li>
073 * <li>max=maximum page count to be listed</li>
074 * <li>sort=name|count</li>
075 * </ul>
076 * Default values:<br/>
077 * <code>show=none  sort=name</code>
078 * 
079 * @since 2.8
080 */
081public class PageViewPlugin extends AbstractReferralPlugin implements Plugin, InitializablePlugin {
082

133    /**
134     * Initialize the PageViewPlugin and its singleton.
135     * 
136     * @param engine The wiki engine.
137     */
138    @Override
139    public void initialize( final Engine engine ) {
140        log.info( "initializing PageViewPlugin" );
141        synchronized( this ) {
142            if( c_singleton == null ) {
143                c_singleton = new PageViewManager();
144            }
145            c_singleton.initialize( engine );
146        }
147    }
148
149    /**
150     * Cleanup the singleton reference.
151     */
152    private void cleanup() {
153        log.info( "cleaning up PageView Manager" );
154        c_singleton = null;
155    }

157    /**
158     *  {@inheritDoc}
159     */
160    @Override
161    public String execute( final Context context, final Map< String, String > params ) throws PluginException {
162        final PageViewManager manager = c_singleton;
163        String result = STR_EMPTY;
164
165        if( manager != null ) {
166            result = manager.execute( context, params );
167        }
168
169        return result;
170    }

172    /**
173     * Page view manager, handling all storage.
174     */
175    public final class PageViewManager implements WikiEventListener {
176        /** Are we initialized? */
177        private boolean m_initialized;
178
179        /** The page counters. */
180        private Map<String, Counter> m_counters;
181
182        /** The page counters in storage format. */
183        private Properties m_storage;
184
185        /** Are all changes stored? */
186        private boolean m_dirty;
187
188        /** The page count storage background thread. */
189        private Thread m_pageCountSaveThread;
190
191        /** The work directory. */
192        private String m_workDir;
193
194        /** Comparator for descending sort on page count. */
195        private final Comparator< Object > m_compareCountDescending = ( o1, o2 ) -> {
196            final int v1 = getCount( o1 );
197            final int v2 = getCount( o2 );
198            return ( v1 == v2 ) ? ( ( String )o1 ).compareTo( ( String )o2 ) : ( v1 < v2 ) ? 1 : -1;
199        };
200
201        /**
202         * Initialize the page view manager.
203         * 
204         * @param engine The wiki engine.
205         */
206        public synchronized void initialize( final Engine engine ) {
207            log.info( "initializing PageView Manager" );
208            m_workDir = engine.getWorkDir();
209            engine.addWikiEventListener( this );
210            if( m_counters == null ) {
211                // Load the counters into a collection
212                m_storage = new Properties();
213                m_counters = new TreeMap<>();
214
215                loadCounters();
216            }
217
218            // backup counters every 5 minutes
219            if( m_pageCountSaveThread == null ) {
220                m_pageCountSaveThread = new CounterSaveThread( engine, 5 * STORAGE_INTERVAL, this );
221                m_pageCountSaveThread.start();
222            }
223
224            m_initialized = true;
225        }
226
227        /**
228         * Handle the shutdown event via the page counter thread.
229         */
230        private synchronized void handleShutdown() {
231            log.info( "handleShutdown: The counter store thread was shut down." );
232
233            cleanup();
234
235            if( m_counters != null ) {
236
237                m_dirty = true;
238                storeCounters();
239
240                m_counters.clear();
241                m_counters = null;
242
243                m_storage.clear();
244                m_storage = null;
245            }
246
247            m_initialized = false;
248
249            m_pageCountSaveThread = null;
250        }
251
252        /**
253         * Inspect wiki events for shutdown.
254         * 
255         * @param event The wiki event to inspect.
256         */
257        @Override
258        public void actionPerformed( final WikiEvent event ) {
259            if( event instanceof WikiEngineEvent ) {
260                if( event.getType() == WikiEngineEvent.SHUTDOWN ) {
261                    log.info( "Detected wiki engine shutdown" );
262                    handleShutdown();
263                }
264            } else if( ( event instanceof WikiPageRenameEvent ) && ( event.getType() == WikiPageRenameEvent.PAGE_RENAMED ) ) {
265                final String oldPageName = ( ( WikiPageRenameEvent )event ).getOldPageName();
266                final String newPageName = ( ( WikiPageRenameEvent )event ).getNewPageName();
267                final Counter oldCounter = m_counters.get( oldPageName );
268                if( oldCounter != null ) {
269                    m_storage.remove( oldPageName );
270                    m_counters.put( newPageName, oldCounter );
271                    m_storage.setProperty( newPageName, oldCounter.toString() );
272                    m_counters.remove( oldPageName );
273                    m_dirty = true;
274                }
275            } else if( ( event instanceof WikiPageEvent ) && ( event.getType() == WikiPageEvent.PAGE_DELETED ) ) {
276                final String pageName = ( ( WikiPageEvent )event ).getPageName();
277                m_storage.remove( pageName );
278                m_counters.remove( pageName );
279            }
280        }

282        /**
283         * Count a page hit, present a pages' counter or output a list of page counts.
284         * 
285         * @param context the wiki context
286         * @param params the plugin parameters
287         * @return String Wiki page snippet
288         * @throws PluginException Malformed pattern parameter.
289         */
290        public String execute( final Context context, final Map< String, String > params ) throws PluginException {
291            final Engine engine = context.getEngine();
292            final Page page = context.getPage();
293            String result = STR_EMPTY;
294
295            if( page != null ) {
296                // get parameters
297                final String pagename = page.getName();
298                String count = params.get( PARAM_COUNT );
299                final String show = params.get( PARAM_SHOW );
300                int entries = TextUtil.parseIntParameter( params.get( PARAM_MAX_ENTRIES ), Integer.MAX_VALUE );
301                final int max = TextUtil.parseIntParameter( params.get( PARAM_MAX_COUNT ), Integer.MAX_VALUE );
302                final int min = TextUtil.parseIntParameter( params.get( PARAM_MIN_COUNT ), Integer.MIN_VALUE );
303                final String sort = params.get( PARAM_SORT );
304                final String body = params.get( DefaultPluginManager.PARAM_BODY );
305                final Pattern[] exclude = compileGlobs( PARAM_EXCLUDE, params.get( PARAM_EXCLUDE ) );
306                final Pattern[] include = compileGlobs( PARAM_INCLUDE, params.get( PARAM_INCLUDE ) );
307                final Pattern[] refer = compileGlobs( PARAM_REFER, params.get( PARAM_REFER ) );
308                final PatternMatcher matcher = (null != exclude || null != include || null != refer) ? new Perl5Matcher() : null;
309                boolean increment = false;
310
311                // increment counter?
312                if( STR_YES.equals( count ) ) {
313                    increment = true;
314                } else {
315                    count = null;
316                }
317
318                // default increment counter?
319                if( ( show == null || STR_NONE.equals( show ) ) && count == null ) {
320                    increment = true;
321                }
322
323                // filter on referring pages?
324                Collection< String > referrers = null;
325
326                if( refer != null ) {
327                    final ReferenceManager refManager = engine.getManager( ReferenceManager.class );
328                    for( final String name : refManager.findCreated() ) {
329                        boolean use = false;
330                        for( int n = 0; !use && n < refer.length; n++ ) {
331                            use = matcher.matches( name, refer[ n ] );
332                        }
333
334                        if( use ) {
335                            final Collection< String > refs = engine.getManager( ReferenceManager.class ).findReferrers( name );
336                            if( refs != null && !refs.isEmpty() ) {
337                                if( referrers == null ) {
338                                    referrers = new HashSet<>();
339                                }
340                                referrers.addAll( refs );
341                            }
342                        }
343                    }
344                }
345
346                synchronized( this ) {
347                    Counter counter = m_counters.get( pagename );
348
349                    // only count in view mode, keep storage values in sync
350                    if( increment && ContextEnum.PAGE_VIEW.getRequestContext().equalsIgnoreCase( context.getRequestContext() ) ) {
351                        if( counter == null ) {
352                            counter = new Counter();
353                            m_counters.put( pagename, counter );
354                        }
355                        counter.increment();
356                        m_storage.setProperty( pagename, counter.toString() );
357                        m_dirty = true;
358                    }
359
360                    if( show == null || STR_NONE.equals( show ) ) {
361                        // nothing to show
362
363                    } else if( PARAM_COUNT.equals( show ) ) {
364                        // show page count
365                        if( counter == null ) {
366                            counter = new Counter();
367                            m_counters.put( pagename, counter );
368                            m_storage.setProperty( pagename, counter.toString() );
369                            m_dirty = true;
370                        }
371                        result = counter.toString();
372
373                    } else if( body != null && 0 < body.length() && STR_LIST.equals( show ) ) {
374                        // show list of counts
375                        String header = STR_EMPTY;
376                        String line = body;
377                        String footer = STR_EMPTY;
378                        int start = body.indexOf( STR_SEPARATOR );
379
380                        // split body into header, line, footer on ---- separator
381                        if( 0 < start ) {
382                            header = body.substring( 0, start );
383                            start = skipWhitespace( start + STR_SEPARATOR.length(), body );
384                            int end = body.indexOf( STR_SEPARATOR, start );
385                            if( start >= end ) {
386                                line = body.substring( start );
387                            } else {
388                                line = body.substring( start, end );
389                                end = skipWhitespace( end + STR_SEPARATOR.length(), body );
390                                footer = body.substring( end );
391                            }
392                        }
393
394                        // sort on name or count?
395                        Map< String, Counter > sorted = m_counters;
396                        if( PARAM_COUNT.equals( sort ) ) {
397                            sorted = new TreeMap<>( m_compareCountDescending );
398                            sorted.putAll( m_counters );
399                        }
400
401                        // build a messagebuffer with the list in wiki markup
402                        final StringBuffer buf = new StringBuffer( header );
403                        final MessageFormat fmt = new MessageFormat( line );
404                        final Object[] args = new Object[] { pagename, STR_EMPTY, STR_EMPTY };
405                        final Iterator< Entry< String, Counter > > iter = sorted.entrySet().iterator();
406
407                        while( 0 < entries && iter.hasNext() ) {
408                            final Entry< String, Counter > entry = iter.next();
409                            final String name = entry.getKey();
410
411                            // check minimum/maximum count
412                            final int value = entry.getValue().getValue();
413                            boolean use = min <= value && value <= max;
414
415                            // did we specify a refer-to page?
416                            if( use && referrers != null ) {
417                                use = referrers.contains( name );
418                            }
419
420                            // did we specify what pages to include?
421                            if( use && include != null ) {
422                                use = false;
423
424                                for( int n = 0; !use && n < include.length; n++ ) {
425                                    use = matcher.matches( name, include[ n ] );
426                                }
427                            }
428
429                            // did we specify what pages to exclude?
430                            if( use && null != exclude ) {
431                                for( int n = 0; use && n < exclude.length; n++ ) {
432                                    use &= !matcher.matches( name, exclude[ n ] );
433                                }
434                            }
435
436                            if( use ) {
437                                args[ 1 ] = engine.getManager( RenderingManager.class ).beautifyTitle( name );
438                                args[ 2 ] = entry.getValue();
439
440                                fmt.format( args, buf, null );
441
442                                entries--;
443                            }
444                        }
445                        buf.append( footer );
446
447                        // let the engine render the list
448                        result = engine.getManager( RenderingManager.class ).textToHTML( context, buf.toString() );
449                    }
450                }
451            }
452            return result;
453        }

455        /**
456         * Compile regexp parameter.
457         * 
458         * @param name The name of the parameter.
459         * @param value The parameter value.
460         * @return Pattern[] The compiled patterns, or <code>null</code>.
461         * @throws PluginException On malformed patterns.
462         */
463        private Pattern[] compileGlobs( final String name, final String value ) throws PluginException {
464            Pattern[] result = null;
465            if( value != null && 0 < value.length() && !STR_GLOBSTAR.equals( value ) ) {
466                try {
467                    final PatternCompiler pc = new GlobCompiler();
468                    final String[] ptrns = StringUtils.split( value, STR_COMMA );
469                    result = new Pattern[ ptrns.length ];
470
471                    for( int n = 0; n < ptrns.length; n++ ) {
472                        result[ n ] = pc.compile( ptrns[ n ] );
473                    }
474                } catch( final MalformedPatternException e ) {
475                    throw new PluginException( "Parameter " + name + " has a malformed pattern: " + e.getMessage() );
476                }
477            }
478
479            return result;
480        }

482        /**
483         * Adjust offset skipping whitespace.
484         * 
485         * @param offset The offset in value to adjust.
486         * @param value String in which offset points.
487         * @return int Adjusted offset into value.
488         */
489        private int skipWhitespace( int offset, final String value ) {
490            while( Character.isWhitespace( value.charAt( offset ) ) ) {
491                offset++;
492            }
493            return offset;
494        }

496        /**
497         * Retrieve a page count.
498         * 
499         * @return int The page count for the given key.
500         * @param key the key for the Counter
501         */
502        protected int getCount( final Object key )
503        {
504            return m_counters.get( key ).getValue();
505        }
506
507        /**
508         * Load the page view counters from file.
509         */
510        private void loadCounters() {
511            if( m_counters != null && m_storage != null ) {
512                log.info( "Loading counters." );
513                synchronized( this ) {
514                    try( final InputStream fis = new FileInputStream( new File( m_workDir, COUNTER_PAGE ) ) ) {
515                        m_storage.load( fis );
516                    } catch( final IOException ioe ) {
517                        log.error( "Can't load page counter store: " + ioe.getMessage() + " , will create a new one!" );
518                    }
519
520                    // Copy the collection into a sorted map
521                    for( final Entry< ?, ? > entry : m_storage.entrySet() ) {
522                        m_counters.put( ( String )entry.getKey(), new Counter( ( String )entry.getValue() ) );
523                    }
524                    
525                    log.info( "Loaded " + m_counters.size() + " counter values." );
526                }
527            }
528        }
529
530        /**
531         * Save the page view counters to file.
532         */
533        protected void storeCounters() {
534            if( m_counters != null && m_storage != null && m_dirty ) {
535                log.info( "Storing " + m_counters.size() + " counter values." );
536                synchronized( this ) {
537                    // Write out the collection of counters
538                    try( final OutputStream fos = new FileOutputStream( new File( m_workDir, COUNTER_PAGE ) ) ) {
539                        m_storage.store( fos, "\n# The number of times each page has been viewed.\n# Do not modify.\n" );
540                        fos.flush();
541
542                        m_dirty = false;
543                    } catch( final IOException ioe ) {
544                        log.error( "Couldn't store counters values: " + ioe.getMessage() );
545                    }
546                }
547            }
548        }
549
550        /**
551         * Is the given thread still current?
552         *
553         * @param thrd thread that can be the current background thread.
554         * @return boolean <code>true</code> if the thread is still the current background thread.
555         */
556        private synchronized boolean isRunning( final Thread thrd )
557        {
558            return m_initialized && thrd == m_pageCountSaveThread;
559        }

563    /** Counter for page hits collection. */
564    private static final class Counter {
565
566        /** The count value. */
567        private int m_count;
568
569        /**
570         * Create a new counter.
571         */
572        public Counter() {
573        }
574
575        /**
576         * Create and initialize a new counter.
577         * 
578         * @param value Count value.
579         */
580        public Counter( final String value )
581        {
582            setValue( value );
583        }
584
585        /**
586         * Increment counter.
587         */
588        public void increment()
589        {
590            m_count++;
591        }
592
593        /**
594         * Get the count value.
595         * 
596         * @return int
597         */
598        public int getValue()
599        {
600            return m_count;
601        }
602
603        /**
604         * Set the count value.
605         * 
606         * @param value String representation of the count.
607         */
608        public void setValue( final String value )
609        {
610            m_count = NumberUtils.toInt( value );
611        }
612
613        /**
614         * @return String String representation of the count.
615         */
616        @Override
617        public String toString()
618        {
619            return String.valueOf( m_count );
620        }

624    /**
625     * Background thread storing the page counters.
626     */
627    static final class CounterSaveThread extends WikiBackgroundThread {
628
629        /** The page view manager. */
630        private final PageViewManager m_manager;
631
632        /**
633         * Create a wiki background thread to store the page counters.
634         * 
635         * @param engine The wiki engine.
636         * @param interval Delay in seconds between saves.
637         * @param pageViewManager page view manager.
638         */
639        public CounterSaveThread( final Engine engine, final int interval, final PageViewManager pageViewManager ) {
640            super( engine, interval );
641            if( pageViewManager == null ) {
642                throw new IllegalArgumentException( "Manager cannot be null" );
643            }
644
645            m_manager = pageViewManager;
646        }
647
648        /**
649         * Save the page counters to file.
650         */
651        @Override
652        public void backgroundTask() {
653            if( m_manager.isRunning( this ) ) {
654                m_manager.storeCounters();
655            }
656        }
