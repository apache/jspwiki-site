019package org.apache.wiki.plugin;
020
021import org.apache.wiki.WikiContext;
022import org.apache.wiki.WikiEngine;
023import org.apache.wiki.WikiPage;
024import org.apache.wiki.api.exceptions.PluginException;
025import org.apache.wiki.api.exceptions.ProviderException;
026import org.apache.wiki.api.plugin.WikiPlugin;
027import org.apache.wiki.auth.AuthorizationManager;
028import org.apache.wiki.auth.permissions.PermissionFactory;
029import org.apache.wiki.preferences.Preferences;
030import org.apache.wiki.util.HttpUtil;
031import org.apache.wiki.util.TextUtil;
032
033import java.util.ArrayList;
034import java.util.List;
035import java.util.Map;
036import java.util.ResourceBundle;
037
038
039/**
040 *  Inserts page contents.  Muchos thanks to Scott Hurlbert for the initial code.
041 *
042 *  <p>Parameters : </p>
043 *  <ul>
044 *  <li><b>page</b> - the name of the page to be inserted</li>
045 *  <li><b>style</b> - the style to use</li>
046 *  <li><b>maxlength</b> - the maximum length of the page to be inserted (page contents)</li>
047 *  <li><b>class</b> - the class to use</li>
048 *  <li><b>section</b> - the section of the page that has to be inserted (separated by "----"</li>
049 *  <li><b>default</b> - the text to insert if the requested page does not exist</li>
050 *  </ul>
051 *
052 *  @since 2.1.37
053 */
054public class InsertPage
055    implements WikiPlugin
056{
057    /** Parameter name for setting the page.  Value is <tt>{@value}</tt>. */
058    public static final String PARAM_PAGENAME  = "page";
059    /** Parameter name for setting the style.  Value is <tt>{@value}</tt>. */
060    public static final String PARAM_STYLE     = "style";
061    /** Parameter name for setting the maxlength.  Value is <tt>{@value}</tt>. */
062    public static final String PARAM_MAXLENGTH = "maxlength";
063    /** Parameter name for setting the class.  Value is <tt>{@value}</tt>. */
064    public static final String PARAM_CLASS     = "class";
065    /** Parameter name for setting the show option.  Value is <tt>{@value}</tt>. */
066    public static final String PARAM_SHOW   = "show";
067    /** Parameter name for setting the section.  Value is <tt>{@value}</tt>. */
068    public static final String PARAM_SECTION   = "section";
069    /** Parameter name for setting the default.  Value is <tt>{@value}</tt>. */
070    public static final String PARAM_DEFAULT   = "default";
071
072    private static final String DEFAULT_STYLE = "";
073
074    private static final String ONCE_COOKIE = "JSPWiki.Once.";
075
076    /** This attribute is stashed in the WikiContext to make sure that we don't
077     *  have circular references.
078     */
079    public static final String ATTR_RECURSE    = "org.apache.wiki.plugin.InsertPage.recurseCheck";
080
081    /**
082     *  {@inheritDoc}
083     */
084    @SuppressWarnings("unchecked")
085    public String execute( WikiContext context, Map<String, String> params )
086        throws PluginException
087    {
088        WikiEngine engine = context.getEngine();
089
090        StringBuilder res = new StringBuilder();
091
092        String clazz        = params.get( PARAM_CLASS );
093        String includedPage = params.get( PARAM_PAGENAME );
094        String style        = params.get( PARAM_STYLE );
095        Boolean showOnce    = "once".equals( params.get( PARAM_SHOW ) );
096        String defaultstr   = params.get( PARAM_DEFAULT );
097        int    section      = TextUtil.parseIntParameter(params.get( PARAM_SECTION ), -1 );
098        int    maxlen       = TextUtil.parseIntParameter(params.get( PARAM_MAXLENGTH ), -1 );
099
100        ResourceBundle rb = Preferences.getBundle( context, WikiPlugin.CORE_PLUGINS_RESOURCEBUNDLE );
101
102
103        if( style == null ) style = DEFAULT_STYLE;
104
105        if( maxlen == -1 ) maxlen = Integer.MAX_VALUE;
106
107        if( includedPage != null )
108        {
109            WikiPage page;
110            try {
111                final String pageName = engine.getFinalPageName( includedPage );
112                if( pageName != null ) {
113                    page = engine.getPageManager().getPage( pageName );
114                } else {
115                    page = engine.getPageManager().getPage( includedPage );
116                }
117            } catch( final ProviderException e ) {
118                res.append( "<span class=\"error\">Page could not be found by the page provider.</span>" );
119                return res.toString();
120            }
121
122            if( page != null )
123            {
124                //
125                //  Check for recursivity
126                //
127
128                List<String> previousIncludes = (List<String>)context.getVariable( ATTR_RECURSE );
129
130                if( previousIncludes != null )
131                {
132                    if( previousIncludes.contains( page.getName() ) )
133                    {
134                        return "<span class=\"error\">Error: Circular reference - you can't include a page in itself!</span>";
135                    }
136                }
137                else
138                {
139                    previousIncludes = new ArrayList<String>();
140                }
141
142                //
143                // Check for permissions
144                //
145                AuthorizationManager mgr = engine.getAuthorizationManager();
146
147                if( !mgr.checkPermission( context.getWikiSession(),
148                                          PermissionFactory.getPagePermission( page, "view") ) )
149                {
150                    res.append("<span class=\"error\">You do not have permission to view this included page.</span>");
151                    return res.toString();
152                }
153
154                //
155                // Show Once
156                // Check for page-cookie, only include page if cookie is not yet set
157                //
158                String cookieName = "";
159
160                if( showOnce )
161                {
162                    cookieName = ONCE_COOKIE +
163                                 TextUtil.urlEncodeUTF8( page.getName() )
164                                         .replaceAll( "\\+", "%20" );
165
166                    if( HttpUtil.retrieveCookieValue( context.getHttpRequest(),
167                                                      cookieName ) != null )
168                    {
169                        return "";  //silent exit
170                    }
171
172                }
173
174                // move here, after premature exit points (permissions, page-cookie)
175                previousIncludes.add( page.getName() );
176                context.setVariable( ATTR_RECURSE, previousIncludes );
177
178                /**
179                 *  We want inclusion to occur within the context of
180                 *  its own page, because we need the links to be correct.
181                 */
182
183                WikiContext includedContext = (WikiContext) context.clone();
184                includedContext.setPage( page );
185
186                String pageData = engine.getPageManager().getPureText( page );
187                String moreLink = "";
188
189                if( section != -1 ) {
190                    try {
191                        pageData = TextUtil.getSection( pageData, section );
192                    } catch( final IllegalArgumentException e ) {
193                        throw new PluginException( e.getMessage() );
194                    }
195                }
196
197                if( pageData.length() > maxlen )
198                {
199                    pageData = pageData.substring( 0, maxlen )+" ...";
200                    moreLink = "<p><a href=\""+context.getURL(WikiContext.VIEW,includedPage)+"\">"+rb.getString("insertpage.more")+"</a></p>";
201                }
202
203                res.append("<div class=\"inserted-page ");
204                if( clazz != null ) res.append( clazz );
205                if( !style.equals(DEFAULT_STYLE) ) res.append("\" style=\""+style );
206                if( showOnce ) res.append("\" data-once=\""+cookieName );
207                res.append("\" >");
208
209                res.append( engine.textToHTML( includedContext, pageData ) );
210                res.append( moreLink );
211
212                res.append("</div>");
213
214                //
215                //  Remove the name from the stack; we're now done with this.
216                //
217                previousIncludes.remove( page.getName() );
218                context.setVariable( ATTR_RECURSE, previousIncludes );
219            }
220            else
221            {
222                if( defaultstr != null )
223                {
224                    res.append( defaultstr );
225                }
226                else
227                {
228                    res.append("There is no page called '"+includedPage+"'.  Would you like to ");
229                    res.append("<a href=\""+context.getURL( WikiContext.EDIT, includedPage )+"\">create it?</a>");
230                }
231            }
232        }
233        else
234        {
235            res.append("<span class=\"error\">");
236            res.append("You have to define a page!");
237            res.append("</span>");
238        }
239
240        return res.toString();
241    }
