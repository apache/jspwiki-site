040/**
041 * <p>
042 * Monitor class that tracks running Workflows. The WorkflowManager also keeps track of the names of
043 * users or groups expected to approve particular Workflows.
044 * </p>
045 */
046public class DefaultWorkflowManager implements WorkflowManager {
047
048    private static final Logger LOG = Logger.getLogger( DefaultWorkflowManager.class );
049    static final String SERIALIZATION_FILE = "wkflmgr.ser";
050
051    /** We use this also a generic serialization id */
052    private static final long serialVersionUID = 6L;
053
054    DecisionQueue m_queue = new DecisionQueue();
055    Set< Workflow > m_workflows;
056    final Map< String, Principal > m_approvers;
057    List< Workflow > m_completed;
058    private Engine m_engine = null;
059
060    /**
061     * Constructs a new WorkflowManager, with an empty workflow cache.
062     */
063    public DefaultWorkflowManager() {
064        m_workflows = ConcurrentHashMap.newKeySet();
065        m_approvers = new ConcurrentHashMap<>();
066        m_completed = new CopyOnWriteArrayList<>();
067        WikiEventEmitter.attach( this );
068    }
069
070    /**
071     * {@inheritDoc}
072     */
073    @Override
074    public Set< Workflow > getWorkflows() {
075        final Set< Workflow > workflows = ConcurrentHashMap.newKeySet();
076        workflows.addAll( m_workflows );
077        return workflows;
078    }
079
080    /**
081     * {@inheritDoc}
082     */
083    @Override
084    public List< Workflow > getCompletedWorkflows() {
085        return new CopyOnWriteArrayList< >( m_completed );
086    }
087
088    /**
089     * {@inheritDoc}
090     *
091     * Any properties that begin with {@link #PROPERTY_APPROVER_PREFIX} will be assumed to be Decisions that require approval. For a given
092     * property key, everything after the prefix denotes the Decision's message key. The property value indicates the Principal (Role,
093     * GroupPrincipal, WikiPrincipal) that must approve the Decision. For example, if the property key/value pair is
094     * {@code jspwiki.approver.workflow.saveWikiPage=Admin}, the Decision's message key is <code>workflow.saveWikiPage</code>. The Principal
095     * <code>Admin</code> will be resolved via {@link org.apache.wiki.auth.AuthorizationManager#resolvePrincipal(String)}.
096     */
097    @Override
098    public void initialize( final Engine engine, final Properties props ) {
099        m_engine = engine;
100
101        // Identify the workflows requiring approvals
102        for( final Object o : props.keySet() ) {
103            final String prop = ( String )o;
104            if( prop.startsWith( PROPERTY_APPROVER_PREFIX ) ) {
105                // For the key, everything after the prefix is the workflow name
106                final String key = prop.substring( PROPERTY_APPROVER_PREFIX.length() );
107                if( key.length() > 0 ) {
108                    // Only use non-null/non-blank approvers
109                    final String approver = props.getProperty( prop );
110                    if( approver != null && approver.length() > 0 ) {
111                        m_approvers.put( key, new UnresolvedPrincipal( approver ) );
112                    }
113                }
114            }
115        }
116
117        unserializeFromDisk( new File( m_engine.getWorkDir(), SERIALIZATION_FILE ) );
118    }
119
120    /**
121     *  Reads the serialized data from the disk back to memory.
122     *
123     * @return the date when the data was last written on disk or {@code 0} if there has been problems reading from disk.
124     */
125    @SuppressWarnings( "unchecked" )
126    synchronized long unserializeFromDisk( final File f ) {
127        long saved = 0L;
128        final StopWatch sw = new StopWatch();
129        sw.start();
130        try( final ObjectInputStream in = new ObjectInputStream( new BufferedInputStream( new FileInputStream( f ) ) ) ) {
131            final long ver = in.readLong();
132            if( ver != serialVersionUID ) {
133                LOG.warn( "File format has changed; Unable to recover workflows and decision queue from disk." );
134            } else {
135                saved        = in.readLong();
136                m_workflows  = ( Set< Workflow > )in.readObject();
137                m_queue      = ( DecisionQueue )in.readObject();
138                m_completed  = ( List< Workflow > )in.readObject();
139                LOG.debug( "Read serialized data successfully in " + sw );
140            }
141        } catch( final IOException | ClassNotFoundException e ) {
142            LOG.error( "unable to recover from disk workflows and decision queue: " + e.getMessage(), e );
143        }
144        sw.stop();
145
146        return saved;
147    }
148
149    /**
150     *  Serializes workflows and decisionqueue to disk.  The format is private, don't touch it.
151     */
152    synchronized void serializeToDisk( final File f ) {
153        try( final ObjectOutputStream out = new ObjectOutputStream( new BufferedOutputStream( new FileOutputStream( f ) ) ) ) {
154            final StopWatch sw = new StopWatch();
155            sw.start();
156
157            out.writeLong( serialVersionUID );
158            out.writeLong( System.currentTimeMillis() ); // Timestamp
159            out.writeObject( m_workflows );
160            out.writeObject( m_queue );
161            out.writeObject( m_completed );
162
163            sw.stop();
164
165            LOG.debug( "serialization done - took " + sw );
166        } catch( final IOException ioe ) {
167            LOG.error( "Unable to serialize!", ioe );
168        }
169    }
170
171    /**
172     * {@inheritDoc}
173     */
174    @Override
175    public boolean requiresApproval( final String messageKey ) {
176        return  m_approvers.containsKey( messageKey );
177    }
178
179    /**
180     * {@inheritDoc}
181     */
182    @Override
183    public Principal getApprover( final String messageKey ) throws WikiException {
184        Principal approver = m_approvers.get( messageKey );
185        if ( approver == null ) {
186            throw new WikiException( "Workflow '" + messageKey + "' does not require approval." );
187        }
188
189        // Try to resolve UnresolvedPrincipals
190        if ( approver instanceof UnresolvedPrincipal ) {
191            final String name = approver.getName();
192            approver = m_engine.getManager( AuthorizationManager.class ).resolvePrincipal( name );
193
194            // If still unresolved, throw exception; otherwise, freshen our cache
195            if ( approver instanceof UnresolvedPrincipal ) {
196                throw new WikiException( "Workflow approver '" + name + "' cannot not be resolved." );
197            }
198
199            m_approvers.put( messageKey, approver );
200        }
201        return approver;
202    }
203
204    /**
205     * Protected helper method that returns the associated Engine
206     *
207     * @return the wiki engine
208     */
209    protected Engine getEngine() {
210        if ( m_engine == null ) {
211            throw new IllegalStateException( "Engine cannot be null; please initialize WorkflowManager first." );
212        }
213        return m_engine;
214    }
215
216    /**
217     * Returns the DecisionQueue associated with this WorkflowManager
218     *
219     * @return the decision queue
220     */
221    @Override
222    public DecisionQueue getDecisionQueue() {
223        return m_queue;
224    }
225
226    /**
227     * {@inheritDoc}
228     */
229    @Override
230    public List< Workflow > getOwnerWorkflows( final Session session ) {
231        final List< Workflow > workflows = new ArrayList<>();
232        if ( session.isAuthenticated() ) {
233            final Principal[] sessionPrincipals = session.getPrincipals();
234            for( final Workflow w : m_workflows ) {
235                final Principal owner = w.getOwner();
236                for ( final Principal sessionPrincipal : sessionPrincipals ) {
237                    if ( sessionPrincipal.equals( owner ) ) {
238                        workflows.add( w );
239                        break;
240                    }
241                }
242            }
243        }
244        return workflows;
245    }
246
247    /**
248     * Listens for {@link WorkflowEvent} objects emitted by Workflows. In particular, this method listens for {@link WorkflowEvent#CREATED},
249     * {@link WorkflowEvent#ABORTED}, {@link WorkflowEvent#COMPLETED} and {@link WorkflowEvent#DQ_REMOVAL} events. If a workflow is created,
250     * it is automatically added to the cache. If one is aborted or completed, it is automatically removed. If a removal from decision queue
251     * is issued, the current step from workflow, which is assumed to be a {@link Decision}, is removed from the {@link DecisionQueue}.
252     * 
253     * @param event the event passed to this listener
254     */
255    @Override
256    public void actionPerformed( final WikiEvent event ) {
257        if( event instanceof WorkflowEvent ) {
258            if( event.getSrc() instanceof Workflow ) {
259                final Workflow workflow = event.getSrc();
260                switch( event.getType() ) {
261                // Remove from manager
262                case WorkflowEvent.ABORTED   :
263                case WorkflowEvent.COMPLETED : remove( workflow ); break;
264                // Add to manager
265                case WorkflowEvent.CREATED   : add( workflow ); break;
266                default: break;
267                }
268            } else if( event.getSrc() instanceof Decision ) {
269                final Decision decision = event.getSrc();
270                switch( event.getType() ) {
271                // Add to DecisionQueue
272                case WorkflowEvent.DQ_ADDITION : addToDecisionQueue( decision ); break;
273                // Remove from DecisionQueue
274                case WorkflowEvent.DQ_REMOVAL  : removeFromDecisionQueue( decision, event.getArg( 0, Context.class ) ); break;
275                default: break;
276                }
277            }
278            serializeToDisk( new File( m_engine.getWorkDir(), SERIALIZATION_FILE ) );
279        }
280    }
281
282    /**
283     * Protected helper method that adds a newly created Workflow to the cache, and sets its {@code workflowManager} and
284     * {@code Id} properties if not set.
285     *
286     * @param workflow the workflow to add
287     */
288    protected void add( final Workflow workflow ) {
289        m_workflows.add( workflow );
290    }
291
292    /**
293     * Protected helper method that removes a specified Workflow from the cache, and moves it to the workflow history list. This method
294     * defensively checks to see if the workflow has not yet been removed.
295     *
296     * @param workflow the workflow to remove
297     */
298    protected void remove( final Workflow workflow ) {
299        if( m_workflows.contains( workflow ) ) {
300            m_workflows.remove( workflow );
301            m_completed.add( workflow );
302        }
303    }
304
305    protected void removeFromDecisionQueue( final Decision decision, final Context context ) {
306        // If current workflow is waiting for input, restart it and remove Decision from DecisionQueue
307        final int workflowId = decision.getWorkflowId();
308        final Optional< Workflow > optw = m_workflows.stream().filter( w -> w.getId() == workflowId ).findAny();
309        if( optw.isPresent() ) {
310            final Workflow w = optw.get();
311            if( w.getCurrentState() == Workflow.WAITING && decision.equals( w.getCurrentStep() ) ) {
312                getDecisionQueue().remove( decision );
313                // Restart workflow
314                try {
315                    w.restart( context );
316                } catch( final WikiException e ) {
317                    LOG.error( "restarting workflow #" + w.getId() + " caused " + e.getMessage(), e );
318                }
319            }
320        }
321    }
322
323    protected void addToDecisionQueue( final Decision decision ) {
324        getDecisionQueue().add( decision );
325    }
