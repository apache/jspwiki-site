040/**
041 *  Provides facilities for handling attachments.  All attachment handling goes through this class.
042 *  <p>
043 *  The AttachmentManager provides a facade towards the current WikiAttachmentProvider that is in use.
044 *  It is created by the Engine as a singleton object, and can be requested through the Engine.
045 *
046 *  @since 1.9.28
047 */
048public interface AttachmentManager {
049
050    /** The property name for defining the attachment provider class name. */
051    String PROP_PROVIDER = "jspwiki.attachmentProvider";
052
053    /** The maximum size of attachments that can be uploaded. */
054    String PROP_MAXSIZE  = "jspwiki.attachment.maxsize";
055
056    /** A space-separated list of attachment types which can be uploaded */
057    String PROP_ALLOWEDEXTENSIONS = "jspwiki.attachment.allowed";
058
059    /** A space-separated list of attachment types which cannot be uploaded */
060    String PROP_FORBIDDENEXTENSIONS = "jspwiki.attachment.forbidden";
061
062    /** A space-separated list of attachment types which never will open in the browser. */
063    String PROP_FORCEDOWNLOAD = "jspwiki.attachment.forceDownload";
064
065    /** Name of the page cache. */
066    String CACHE_NAME = "jspwiki.dynamicAttachmentCache";
067
068    /** The capacity of the cache, if you want something else, tweak ehcache.xml. */
069    int DEFAULT_CACHECAPACITY = 1_000;
070
071    /**
072     *  Returns true, if attachments are enabled and running.
073     *
074     *  @return A boolean value indicating whether attachment functionality is enabled.
075     */
076    boolean attachmentsEnabled();
077
078    /**
079     *  Gets info on a particular attachment, latest version.
080     *
081     *  @param name A full attachment name.
082     *  @return Attachment, or null, if no such attachment exists.
083     *  @throws ProviderException If something goes wrong.
084     */
085    default Attachment getAttachmentInfo( final String name ) throws ProviderException {
086        return getAttachmentInfo( name, WikiProvider.LATEST_VERSION );
087    }
088
089    /**
090     *  Gets info on a particular attachment with the given version.
091     *
092     *  @param name A full attachment name.
093     *  @param version A version number.
094     *  @return Attachment, or null, if no such attachment or version exists.
095     *  @throws ProviderException If something goes wrong.
096     */
097
098    default Attachment getAttachmentInfo( final String name, final int version ) throws ProviderException {
099        if( name == null ) {
100            return null;
101        }
102
103        return getAttachmentInfo( null, name, version );
104    }
105
106    /**
107     *  Figures out the full attachment name from the context and attachment name.
108     *
109     *  @param context The current WikiContext
110     *  @param attachmentname The file name of the attachment.
111     *  @return Attachment, or null, if no such attachment exists.
112     *  @throws ProviderException If something goes wrong.
113     */
114    default Attachment getAttachmentInfo( final Context context, final String attachmentname ) throws ProviderException {
115        return getAttachmentInfo( context, attachmentname, WikiProvider.LATEST_VERSION );
116    }
117
118    /**
119     *  Figures out the full attachment name from the context and attachment name.
120     *
121     *  @param context The current WikiContext
122     *  @param attachmentname The file name of the attachment.
123     *  @param version A particular version.
124     *  @return Attachment, or null, if no such attachment or version exists.
125     *  @throws ProviderException If something goes wrong.
126     */
127    Attachment getAttachmentInfo( Context context, String attachmentname, int version ) throws ProviderException;
128
129    /**
130     *  Figures out the full attachment name from the context and attachment name.
131     *
132     *  @param context The current WikiContext
133     *  @param attachmentname The file name of the attachment.
134     *  @return Attachment, or null, if no such attachment exists.
135     */
136    String getAttachmentInfoName( Context context, String attachmentname );
137
138    /**
139     *  Returns the list of attachments associated with a given wiki page. If there are no attachments, returns an empty Collection.
140     *
141     *  @param wikipage The wiki page from which you are seeking attachments for.
142     *  @return a valid collection of attachments.
143     *  @throws ProviderException If there was something wrong in the backend.
144     */
145    List< Attachment > listAttachments( Page wikipage ) throws ProviderException;
146
147    /**
148     *  Returns true, if the page has any attachments at all.  This is a convenience method.
149     *
150     *  @param wikipage The wiki page from which you are seeking attachments for.
151     *  @return True, if the page has attachments, else false.
152     */
153    default boolean hasAttachments( final Page wikipage ) {
154        try {
155            return listAttachments( wikipage ).size() > 0;
156        } catch( final Exception e ) {
157            Logger.getLogger( AttachmentManager.class ).info( e.getMessage(), e );
158        }
159
160        return false;
161    }
162
163    /**
164     *  Check if attachement link should force a download iso opening the attachment in the browser.
165     *
166     *  @param name  Name of attachment to be checked
167     *  @return true, if the attachment should be downloaded when clicking the link
168     *  @since 2.11.0 M4
169    */
170    boolean forceDownload( String name );
171
172    /**
173     *  Finds a (real) attachment from the repository as an {@link InputStream}.
174     *
175     *  @param att Attachment
176     *  @return An InputStream to read from. May return null, if attachments are disabled.
177     *  @throws IOException If the stream cannot be opened
178     *  @throws ProviderException If the backend fails due to some other reason.
179     */
180    default InputStream getAttachmentStream( final Attachment att ) throws IOException, ProviderException {
181        return getAttachmentStream( null, att );
182    }
183
184    /**
185     *  Returns an attachment stream using the particular WikiContext. This method should be used instead of
186     *  {@link #getAttachmentStream(Attachment)}, since it also allows the DynamicAttachments to function.
187     *
188     *  @param ctx The Wiki Context
189     *  @param att The Attachment to find
190     *  @return An InputStream.  May return null, if attachments are disabled.  You must take care of closing it.
191     *  @throws ProviderException If the backend fails due to some reason
192     *  @throws IOException If the stream cannot be opened
193     */
194    InputStream getAttachmentStream( WikiContext ctx, Attachment att ) throws ProviderException, IOException;
195
196    /**
197     *  Stores a dynamic attachment.  Unlike storeAttachment(), this just stores the attachment in the memory.
198     *
199     *  @param ctx A WikiContext
200     *  @param att An attachment to store
201     */
202    void storeDynamicAttachment( WikiContext ctx, DynamicAttachment att );
203
204    /**
205     *  Finds a DynamicAttachment.  Normally, you should just use {@link #getAttachmentInfo(String)} , since that will find also
206     *  {@link DynamicAttachment}s.
207     *
208     *  @param name The name of the attachment to look for
209     *  @return An Attachment, or null.
210     *  @see #getAttachmentInfo(String)
211     */
212    DynamicAttachment getDynamicAttachment( String name );
213
214    /**
215     *  Stores an attachment that lives in the given file. If the attachment did not exist previously, this method will create it.
216     *  If it did exist, it stores a new version.
217     *
218     *  @param att Attachment to store this under.
219     *  @param source A file to read from.
220     *
221     *  @throws IOException If writing the attachment failed.
222     *  @throws ProviderException If something else went wrong.
223     */
224    default void storeAttachment( final Attachment att, final File source ) throws IOException, ProviderException {
225        try( final FileInputStream in = new FileInputStream( source ) ) {
226            storeAttachment( att, in );
227        }
228    }
229
230    /**
231     *  Stores an attachment directly from a stream. If the attachment did not exist previously, this method will create it.
232     *  If it did exist, it stores a new version.
233     *
234     *  @param att Attachment to store this under.
235     *  @param in  InputStream from which the attachment contents will be read.
236     *  @throws IOException If writing the attachment failed.
237     *  @throws ProviderException If something else went wrong.
238     */
239    void storeAttachment( Attachment att, InputStream in ) throws IOException, ProviderException;
240
241    /**
242     *  Returns a list of versions of the attachment.
243     *
244     *  @param attachmentName A fully qualified name of the attachment.
245     *
246     *  @return A list of Attachments.  May return null, if attachments are
247     *          disabled.
248     *  @throws ProviderException If the provider fails for some reason.
249     */
250    List< Attachment > getVersionHistory( String attachmentName ) throws ProviderException;
251
252    /**
253     *  Returns a collection of Attachments, containing each and every attachment that is in this Wiki.
254     *
255     *  @return A collection of attachments.  If attachments are disabled, will return an empty collection.
256     *  @throws ProviderException If something went wrong with the backend
257     */
258    Collection< Attachment > getAllAttachments() throws ProviderException;
259
260    /**
261     *  Returns the current attachment provider.
262     *
263     *  @return The current provider.  May be null, if attachments are disabled.
264     */
265    AttachmentProvider getCurrentProvider();
266
267    /**
268     *  Deletes the given attachment version.
269     *
270     *  @param att The attachment to delete
271     *  @throws ProviderException If something goes wrong with the backend.
272     */
273    void deleteVersion( Attachment att ) throws ProviderException;
274
275    /**
276     *  Deletes all versions of the given attachment.
277     *
278     *  @param att The Attachment to delete.
279     *  @throws ProviderException if something goes wrong with the backend.
280     */
281    void deleteAttachment( Attachment att ) throws ProviderException;
282
283    /**
284     *  Validates the filename and makes sure it is legal.  It trims and splits and replaces bad characters.
285     *
286     *  @param filename file name to validate.
287     *  @return A validated name with annoying characters replaced.
288     *  @throws WikiException If the filename is not legal (e.g. empty)
289     */
290    static String validateFileName( String filename ) throws WikiException {
291        if( filename == null || filename.trim().length() == 0 ) {
292            Logger.getLogger( AttachmentManager.class ).error( "Empty file name given." );
293
294            // the caller should catch the exception and use the exception text as an i18n key
295            throw new WikiException(  "attach.empty.file" );
296        }
297
298        //  Some browser send the full path info with the filename, so we need
299        //  to remove it here by simply splitting along slashes and then taking the path.
300        final String[] splitpath = filename.split( "[/\\\\]" );
301        filename = splitpath[splitpath.length-1];
302
303        // Should help with IE 5.22 on OSX
304        filename = filename.trim();
305
306        // If file name ends with .jsp or .jspf, the user is being naughty!
307        if( filename.toLowerCase().endsWith( ".jsp" ) || filename.toLowerCase().endsWith( ".jspf" ) ) {
308            Logger.getLogger( AttachmentManager.class )
309                  .info( "Attempt to upload a file with a .jsp/.jspf extension.  In certain cases this " +
310                         "can trigger unwanted security side effects, so we're preventing it." );
311
312            // the caller should catch the exception and use the exception text as an i18n key
313            throw new WikiException(  "attach.unwanted.file"  );
314        }
315
316        //  Remove any characters that might be a problem. Most importantly - characters that might stop processing of the URL.
317        return StringUtils.replaceChars( filename, "#?\"'", "____" );
318    }
