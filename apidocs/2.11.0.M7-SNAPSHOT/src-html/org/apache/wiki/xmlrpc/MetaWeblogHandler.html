042/**
043 *  Provides handlers for all RPC routines of the MetaWeblog API.
044 *  <P>
045 *  JSPWiki does not support categories, and therefore we always return
046 *  an empty list for getCategories().  Note also that this API is not
047 *  suitable for general Wiki editing, since JSPWiki formats the entries
048 *  in a wiki-compatible manner.  And you cannot choose your page names
049 *  either.  Since 2.1.94 the entire MetaWeblog API is supported.
050 *
051 *  @since 2.1.7
052 */
053
054public class MetaWeblogHandler implements WikiRPCHandler {
055
056    private static final Logger log = Logger.getLogger( MetaWeblogHandler.class );
057
058    private WikiContext m_context;
059
060    /**
061     *  {@inheritDoc}
062     */
063    public void initialize( final WikiContext context )
064    {
065        m_context = context;
066    }

068    /**
069     *  Does a quick check against the current user
070     *  and does he have permissions to do the stuff
071     *  that he really wants to.
072     *  <p>
073     *  If there is no authentication enabled, returns normally.
074     *
075     *  @throws XmlRpcException with the correct error message, if auth fails.
076     */
077    private void checkPermissions( final WikiPage page,
078                                   final String username,
079                                   final String password,
080                                   final String permission ) throws XmlRpcException {
081        try {
082            final AuthenticationManager amm = m_context.getEngine().getAuthenticationManager();
083            final AuthorizationManager mgr = m_context.getEngine().getAuthorizationManager();
084
085            if( amm.login( m_context.getWikiSession(), m_context.getHttpRequest(), username, password ) ) {
086                if( !mgr.checkPermission( m_context.getWikiSession(), PermissionFactory.getPagePermission( page, permission ) ) ) {
087                    throw new XmlRpcException( 1, "No permission" );
088                }
089            } else {
090                throw new XmlRpcException( 1, "Unknown login" );
091            }
092        } catch( final WikiSecurityException e ) {
093            throw new XmlRpcException( 1, e.getMessage(), e );
094        }
095    }

097    /**
098     *  JSPWiki does not support categories, therefore JSPWiki always returns an empty list for categories.
099     *
100     *  @param blogid The id of the blog.
101     *  @param username The username to use
102     *  @param password The password
103     *  @throws XmlRpcException If something goes wrong
104     *  @return An empty hashtable.
105     */
106    public Hashtable< Object, Object > getCategories( final String blogid, final String username, final String password )  throws XmlRpcException {
107        final WikiPage page = m_context.getEngine().getPageManager().getPage( blogid );
108        checkPermissions( page, username, password, "view" );
109        return new Hashtable<>();
110    }

112    private String getURL( final String page ) {
113        return m_context.getEngine().getURL( WikiContext.VIEW, page,null );
114    }

116    /**
117     *  Takes a wiki page, and creates a metaWeblog struct out of it.
118     *
119     *  @param page The actual entry page
120     *  @return A metaWeblog entry struct.
121     */
122    private Hashtable< String,Object > makeEntry( final WikiPage page ) {
123        final WikiPage firstVersion = m_context.getEngine().getPageManager().getPage( page.getName(), 1 );
124        final Hashtable< String, Object > ht = new Hashtable<>();
125        ht.put( "dateCreated", firstVersion.getLastModified() );
126        ht.put( "link", getURL(page.getName() ) );
127        ht.put( "permaLink", getURL(page.getName() ) );
128        ht.put( "postid", page.getName() );
129        ht.put( "userid", page.getAuthor() );
130
131        final String pageText = m_context.getEngine().getPageManager().getText(page.getName());
132        final int firstLine = pageText.indexOf('\n');
133
134        String title = "";
135        if( firstLine > 0 ) {
136            title = pageText.substring( 0, firstLine );
137        }
138
139        if( title.trim().length() == 0 ) {
140            title = page.getName();
141        }
142
143        // Remove wiki formatting
144        while( title.startsWith("!") ) {
145            title = title.substring(1);
146        }
147
148        ht.put("title", title);
149        ht.put("description", pageText);
150
151        return ht;
152    }

154    /**
155     *  Returns a list of the recent posts to this weblog.
156     *
157     *  @param blogid The id of the blog.
158     *  @param username The username to use
159     *  @param password The password
160     *  @param numberOfPosts How many posts to find
161     *  @throws XmlRpcException If something goes wrong
162     *  @return As per MetaweblogAPI specification
163     */
164    // FIXME: The implementation is suboptimal, as it goes through all of the blog entries.
165    public Hashtable getRecentPosts( final String blogid, final String username, final String password, final int numberOfPosts ) throws XmlRpcException {
166        final Hashtable<String, Hashtable<String, Object>> result = new Hashtable<>();
167        log.info( "metaWeblog.getRecentPosts() called");
168        final WikiPage page = m_context.getEngine().getPageManager().getPage( blogid );
169        checkPermissions( page, username, password, "view" );
170
171        final WeblogPlugin plugin = new WeblogPlugin();
172        final List< WikiPage > changed = plugin.findBlogEntries( m_context.getEngine(), blogid, new Date( 0L ), new Date() );
173        changed.sort( new PageTimeComparator() );
174
175        int items = 0;
176        for( final Iterator< WikiPage > i = changed.iterator(); i.hasNext() && items < numberOfPosts; items++ ) {
177            final WikiPage p = i.next();
178            result.put( "entry", makeEntry( p ) );
179        }
180
181        return result;
182    }

184    /**
185     *  Adds a new post to the blog.
186     *
187     *  @param blogid The id of the blog.
188     *  @param username The username to use
189     *  @param password The password
190     *  @param content As per Metaweblogapi contract
191     *  @param publish This parameter is ignored for JSPWiki.
192     *  @return Returns an empty string
193     *  @throws XmlRpcException If something goes wrong
194     */
195    public String newPost( final String blogid,
196                           final String username,
197                           final String password,
198                           final Hashtable< String, Object > content,
199                           final boolean publish ) throws XmlRpcException {
200        log.info("metaWeblog.newPost() called");
201        final WikiEngine engine = m_context.getEngine();
202        final WikiPage page = engine.getPageManager().getPage( blogid );
203        checkPermissions( page, username, password, "createPages" );
204
205        try {
206            final WeblogEntryPlugin plugin = new WeblogEntryPlugin();
207            final String pageName = plugin.getNewEntryPage( engine, blogid );
208            final WikiPage entryPage = new WikiPage( engine, pageName );
209            entryPage.setAuthor( username );
210
211            final WikiContext context = new WikiContext( engine, entryPage );
212            final StringBuilder text = new StringBuilder();
213            text.append( "!" ).append( content.get( "title" ) );
214            text.append( "\n\n" );
215            text.append( content.get("description") );
216
217            log.debug("Writing entry: "+text);
218
219            engine.getPageManager().saveText( context, text.toString() );
220        } catch( final Exception e ) {
221            log.error("Failed to create weblog entry",e);
222            throw new XmlRpcException( 0, "Failed to create weblog entry: "+e.getMessage() );
223        }
224
225        return ""; // FIXME:
226    }

228    /**
229     *  Creates an attachment and adds it to the blog.  The attachment
230     *  is created into the main blog page, not the actual post page,
231     *  because we do not know it at this point.
232     *
233     *  @param blogid The id of the blog.
234     *  @param username The username to use
235     *  @param password The password
236     *  @param content As per the MetaweblogAPI contract
237     *  @return As per the MetaweblogAPI contract
238     *  @throws XmlRpcException If something goes wrong
239     */
240    public Hashtable< String, Object > newMediaObject( final String blogid,
241                                                       final String username,
242                                                       final String password,
243                                                       final Hashtable< String, Object > content ) throws XmlRpcException {
244        final WikiEngine engine = m_context.getEngine();
245        final String url;
246
247        log.info( "metaWeblog.newMediaObject() called" );
248
249        final WikiPage page = engine.getPageManager().getPage( blogid );
250        checkPermissions( page, username, password, "upload" );
251
252        final String name = (String) content.get( "name" );
253        final byte[] data = (byte[]) content.get( "bits" );
254
255        final AttachmentManager attmgr = engine.getAttachmentManager();
256
257        try {
258            final Attachment att = new Attachment( engine, blogid, name );
259            att.setAuthor( username );
260            attmgr.storeAttachment( att, new ByteArrayInputStream( data ) );
261
262            url = engine.getURL( WikiContext.ATTACH, att.getName(), null );
263        } catch( final Exception e ) {
264            log.error( "Failed to upload attachment", e );
265            throw new XmlRpcException( 0, "Failed to upload media object: "+e.getMessage() );
266        }
267
268        final Hashtable< String, Object > result = new Hashtable<>();
269        result.put("url", url);
270
271        return result;
272    }

275    /**
276     *  Allows the user to edit a post.  It does not allow general editability of wiki pages, because of the limitations of the metaWeblog API.
277     */
278    boolean editPost( final String postid,
279                      final String username,
280                      final String password,
281                      final Hashtable< String,Object > content,
282                      final boolean publish ) throws XmlRpcException {
283        final WikiEngine engine = m_context.getEngine();
284        log.info("metaWeblog.editPost("+postid+") called");
285
286        // FIXME: Is postid correct?  Should we determine it from the page name?
287        final WikiPage page = engine.getPageManager().getPage( postid );
288        checkPermissions( page, username, password, "edit" );
289
290        try {
291            final WikiPage entryPage = (WikiPage)page.clone();
292            entryPage.setAuthor( username );
293
294            final WikiContext context = new WikiContext( engine, entryPage );
295
296            final StringBuilder text = new StringBuilder();
297            text.append( "!" ).append( content.get( "title" ) );
298            text.append( "\n\n" );
299            text.append( content.get("description") );
300
301            log.debug("Updating entry: "+text);
302
303            engine.getPageManager().saveText( context, text.toString() );
304        } catch( final Exception e ) {
305            log.error("Failed to create weblog entry",e);
306            throw new XmlRpcException( 0, "Failed to update weblog entry: "+e.getMessage() );
307        }
308
309        return true;
310    }
