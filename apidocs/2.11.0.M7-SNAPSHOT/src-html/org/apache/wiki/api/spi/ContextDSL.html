001package org.apache.wiki.api.spi;
002
003import org.apache.wiki.api.core.Command;
004import org.apache.wiki.api.core.Context;
005import org.apache.wiki.api.core.Engine;
006import org.apache.wiki.api.core.Page;
007
008import javax.servlet.http.HttpServletRequest;
009
010
011public class ContextDSL {
012
013    private final ContextSPI contextSPI;
014
015    ContextDSL( final ContextSPI contextSPI ) {
016        this.contextSPI = contextSPI;
017    }
018
019    /**
020     *  Create a new Context for the given Page.
021     *
022     *  @param engine The Engine that is handling the request.
023     *  @param page The Page. If you want to create a Context for an older version of a page, you must use this method.
024     */
025    public Context create( final Engine engine, final Page page ) {
026        return contextSPI.create( engine, page );
027    }
028
029    /**
030     * <p>Creates a new Context for the given Engine, Command and HttpServletRequest.</p>
031     * <p>This constructor will also look up the HttpSession associated with the request, and determine if a Session object is present.
032     * If not, a new one is created.</p>
033     *
034     * @param engine The Engine that is handling the request
035     * @param request The HttpServletRequest that should be associated with this context. This parameter may be <code>null</code>.
036     * @param command the command
037     */
038    public Context create( final Engine engine, final HttpServletRequest request, final Command command ) {
039        return contextSPI.create( engine, request, command );
040    }
041
042    /**
043     * Creates a new Context for the given Engine, Page and HttpServletRequest.
044     *
045     * @param engine The Engine that is handling the request
046     * @param request The HttpServletRequest that should be associated with this context. This parameter may be <code>null</code>.
047     * @param page The WikiPage. If you want to create a WikiContext for an older version of a page, you must supply this parameter
048     */
049    public Context create( final Engine engine, final HttpServletRequest request, final Page page ) {
050        return contextSPI.create( engine, request, page );
051    }
052
053    /**
054     *  Creates a new Context from a supplied HTTP request, using a default wiki context.
055     *
056     *  @param engine The Engine that is handling the request
057     *  @param request the HTTP request
058     *  @param requestContext the default context to use
059     */
060    public Context create( final Engine engine, final HttpServletRequest request, final String requestContext ) {
061        return contextSPI.create( engine, request, requestContext );
062    }
063
064}
