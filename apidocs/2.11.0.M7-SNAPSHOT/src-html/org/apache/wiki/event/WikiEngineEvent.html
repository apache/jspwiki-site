020package org.apache.wiki.event;
021
022import org.apache.wiki.WikiEngine;
023
024/**
025  * WikiEngineEvent indicates a change in the state of the WikiEngine.
026  * 
027  * @see     org.apache.wiki.event.WikiEvent
028  * @since   2.4.20
029  */
030public class WikiEngineEvent extends WikiEvent {
031
032    private static final long serialVersionUID = 1829433967558773970L;
033
034    /** Indicates a WikiEngine initialization event, fired as the  wiki service is being initialized (in progress). */
035    public static final int INITIALIZING   = -1;
036
037    /** Indicates a WikiEngine initialized event, fired after the  wiki service is fully available. */
038    public static final int INITIALIZED    = 0;
039
040    /** Indicates a WikiEngine closing event, fired as a signal that the wiki service is shutting down. */
041    public static final int SHUTDOWN       = 1;
042
043    /**
044     * Indicates a WikiEngine stopped event, fired after halting the wiki service.
045     * A WikiEngine in this state is not expected to provide further services.
046     */
047    public static final int STOPPED        = 2;
048
049    private WikiEngine m_engine;
050
051     /**
052      *  Constructs an instance of this event.
053      *
054      * @param eventSource  the Object that is the source of the event, which <b>must</b> be the WikiEngine. If it is not, this
055      * method thows a ClassCastException
056      * @param type the event type
057      */
058    public WikiEngineEvent( final Object eventSource, final int type ) {
059        super( eventSource, type );
060        m_engine = ( WikiEngine )eventSource;
061    }
062
063    /**
064     *  Sets the type of this event.
065     *
066     * @param type the type of this WikiEngineEvent.
067     */
068    protected void setType( final int type ) {
069        if( type >= INITIALIZING && type <= STOPPED ) {
070            super.setType( type );
071        } else {
072            super.setType( ERROR );
073        }
074    }
075
076    /**
077     *  Returns the WikiEngine that spawned this event.
078     *
079     * @return  the WikiEngine that spawned this event.
080     */
081    public WikiEngine getEngine() {
082        return m_engine;
083    }
084
085    /**
086     * Returns <code>true</code> if the int value is a WikiPageEvent type.
087     *
088     * @param type the event type
089     * @return the result
090     */
091    public static boolean isValidType( final int type ) {
092        return type >= INITIALIZING && type <= STOPPED;
093    }
094
095    /**
096     *  Returns a textual representation of the event type.
097     *
098     * @return a String representation of the type
099     */
100    public final String eventName() {
101        switch ( getType() ) {
102            case INITIALIZING: return "INITIALIZING";
103            case INITIALIZED:  return "INITIALIZED";
104            case SHUTDOWN:     return "SHUTDOWN";
105            case STOPPED:      return "STOPPED";
106            default:           return super.eventName();
107        }
108    }
109
110    /**
111     *  Returns a human-readable description of the event type.
112     *
113     * @return a String description of the type
114     */
115    public final String getTypeDescription() {
116        switch ( getType() ) {
117            case INITIALIZING: return "wiki engine initializing";
118            case INITIALIZED:  return "wiki engine initialized";
119            case SHUTDOWN:     return "wiki engine shutting down";
120            case STOPPED:      return "wiki engine stopped";
121            default:           return super.getTypeDescription();
122        }
123    }
124
125}
