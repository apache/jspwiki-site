052/**
053 *  Provides basic, versioning attachments.
054 *
055 *  <PRE>
056 *   Structure is as follows:
057 *      attachment_dir/
058 *         ThisPage/
059 *            attachment.doc/
060 *               attachment.properties
061 *               1.doc
062 *               2.doc
063 *               3.doc
064 *            picture.png/
065 *               attachment.properties
066 *               1.png
067 *               2.png
068 *         ThatPage/
069 *            picture.png/
070 *               attachment.properties
071 *               1.png
072 *             
073 *  </PRE>
074 *
075 *  The names of the directories will be URLencoded.
076 *  <p>
077 *  "attachment.properties" consists of the following items:
078 *  <UL>
079 *   <LI>1.author = author name for version 1 (etc)
080 *  </UL>
081 */
082public class BasicAttachmentProvider implements AttachmentProvider {
083
084    private Engine m_engine;
085    private String m_storageDir;

087    /*
088     * Disable client cache for files with patterns
089     * since 2.5.96
090     */
091    private Pattern m_disableCache;
092    
093    /** The property name for specifying which attachments are not cached.  Value is <tt>{@value}</tt>. */
094    public static final String PROP_DISABLECACHE = "jspwiki.basicAttachmentProvider.disableCache";
095
096    /** The name of the property file. */
097    public static final String PROPERTY_FILE = "attachment.properties";
098
099    /** The default extension for the page attachment directory name. */
100    public static final String DIR_EXTENSION = "-att";
101    
102    /** The default extension for the attachment directory. */
103    public static final String ATTDIR_EXTENSION = "-dir";
104    
105    private static final Logger log = LogManager.getLogger( BasicAttachmentProvider.class );

107    /**
108     *  {@inheritDoc}
109     */
110    @Override
111    public void initialize( final Engine engine, final Properties properties ) throws NoRequiredPropertyException, IOException {
112        m_engine = engine;
113        m_storageDir = TextUtil.getCanonicalFilePathProperty( properties, PROP_STORAGEDIR,
114                                                       System.getProperty("user.home") + File.separator + "jspwiki-files");
115
116        final String patternString = engine.getWikiProperties().getProperty( PROP_DISABLECACHE );
117        if ( patternString != null ) {
118            m_disableCache = Pattern.compile(patternString);
119        }
120
121        //  Check if the directory exists - if it doesn't, create it.
122        final File f = new File( m_storageDir );
123        if( !f.exists() ) {
124            f.mkdirs();
125        }
126
127        // Some sanity checks
128        if( !f.exists() ) {
129            throw new IOException( "Could not find or create attachment storage directory '" + m_storageDir + "'" );
130        }
131
132        if( !f.canWrite() ) {
133            throw new IOException( "Cannot write to the attachment storage directory '" + m_storageDir + "'" );
134        }
135
136        if( !f.isDirectory() ) {
137            throw new IOException( "Your attachment storage points to a file, not a directory: '" + m_storageDir + "'" );
138        }
139    }

141    /**
142     *  Finds storage dir, and if it exists, makes sure that it is valid.
143     *
144     *  @param wikipage Page to which this attachment is attached.
145     */
146    private File findPageDir( String wikipage ) throws ProviderException {
147        wikipage = mangleName( wikipage );
148
149        final File f = new File( m_storageDir, wikipage + DIR_EXTENSION );
150        if( f.exists() && !f.isDirectory() ) {
151            throw new ProviderException( "Storage dir '" + f.getAbsolutePath() + "' is not a directory!" );
152        }
153
154        return f;
155    }
156
157    private static String mangleName( final String wikiname ) {
158        return TextUtil.urlEncodeUTF8( wikiname );
159    }
160
161    private static String unmangleName( final String filename )
162    {
163        return TextUtil.urlDecodeUTF8( filename );
164    }

166    /**
167     *  Finds the dir in which the attachment lives.
168     */
169    private File findAttachmentDir( final Attachment att ) throws ProviderException {
170        File f = new File( findPageDir( att.getParentName() ), mangleName( att.getFileName() + ATTDIR_EXTENSION ) );
171
172        //  Migration code for earlier versions of JSPWiki. Originally, we used plain filename.  Then we realized we need
173        //  to urlencode it.  Then we realized that we have to use a postfix to make sure illegal file names are never formed.
174        if( !f.exists() ) {
175            File oldf = new File( findPageDir( att.getParentName() ), mangleName( att.getFileName() ) );
176            if( oldf.exists() ) {
177                f = oldf;
178            } else {
179                oldf = new File( findPageDir( att.getParentName() ), att.getFileName() );
180                if( oldf.exists() ) {
181                    f = oldf;
182                }
183            }
184        }
185
186        return f;
187    }

189    /**
190     * Goes through the repository and decides which version is the newest one in that directory.
191     *
192     * @return Latest version number in the repository, or 0, if there is no page in the repository.
193     */
194    private int findLatestVersion( final Attachment att ) throws ProviderException {
195        final File attDir  = findAttachmentDir( att );
196        final String[] pages = attDir.list( new AttachmentVersionFilter() );
197        if( pages == null ) {
198            return 0; // No such thing found.
199        }
200
201        int version = 0;
202        for( final String page : pages ) {
203            final int cutpoint = page.indexOf( '.' );
204            final String pageNum = ( cutpoint > 0 ) ? page.substring( 0, cutpoint ) : page;
205
206            try {
207                final int res = Integer.parseInt( pageNum );
208
209                if( res > version ) {
210                    version = res;
211                }
212            } catch( final NumberFormatException e ) {
213            } // It's okay to skip these.
214        }
215
216        return version;
217    }

219    /**
220     *  Returns the file extension.  For example "test.png" returns "png".
221     *  <p>
222     *  If file has no extension, will return "bin"
223     *  
224     *  @param filename The file name to check
225     *  @return The extension.  If no extension is found, returns "bin".
226     */
227    protected static String getFileExtension( final String filename ) {
228        String fileExt = "bin";
229
230        final int dot = filename.lastIndexOf('.');
231        if( dot >= 0 && dot < filename.length()-1 ) {
232            fileExt = mangleName( filename.substring( dot+1 ) );
233        }
234
235        return fileExt;
236    }

238    /**
239     *  Writes the page properties back to the file system.
240     *  Note that it WILL overwrite any previous properties.
241     */
242    private void putPageProperties( final Attachment att, final Properties properties ) throws IOException, ProviderException {
243        final File attDir = findAttachmentDir( att );
244        final File propertyFile = new File( attDir, PROPERTY_FILE );
245        try( final OutputStream out = new FileOutputStream( propertyFile ) ) {
246            properties.store( out, " JSPWiki page properties for " + att.getName() + ". DO NOT MODIFY!" );
247        }
248    }
249
250    /**
251     *  Reads page properties from the file system.
252     */
253    private Properties getPageProperties( final Attachment att ) throws IOException, ProviderException {
254        final Properties props = new Properties();
255        final File propertyFile = new File( findAttachmentDir(att), PROPERTY_FILE );
256        if( propertyFile.exists() ) {
257            try( final InputStream in = new FileInputStream( propertyFile ) ) {
258                props.load( in );
259            }
260        }
261        
262        return props;
263    }

265    /**
266     *  {@inheritDoc}
267     */
268    @Override
269    public void putAttachmentData( final Attachment att, final InputStream data ) throws ProviderException, IOException {
270        final File attDir = findAttachmentDir( att );
271
272        if( !attDir.exists() ) {
273            attDir.mkdirs();
274        }
275        final int latestVersion = findLatestVersion( att );
276        final int versionNumber = latestVersion + 1;
277
278        final File newfile = new File( attDir, versionNumber + "." + getFileExtension( att.getFileName() ) );
279        try( final OutputStream out = new FileOutputStream( newfile ) ) {
280            log.info( "Uploading attachment " + att.getFileName() + " to page " + att.getParentName() );
281            log.info( "Saving attachment contents to " + newfile.getAbsolutePath() );
282            FileUtil.copyContents( data, out );
283
284            final Properties props = getPageProperties( att );
285
286            String author = att.getAuthor();
287            if( author == null ) {
288                author = "unknown"; // FIXME: Should be localized, but cannot due to missing WikiContext
289            }
290            props.setProperty( versionNumber + ".author", author );
291
292            final String changeNote = att.getAttribute( Page.CHANGENOTE );
293            if( changeNote != null ) {
294                props.setProperty( versionNumber + ".changenote", changeNote );
295            }
296            
297            putPageProperties( att, props );
298        } catch( final IOException e ) {
299            log.error( "Could not save attachment data: ", e );
300            throw (IOException) e.fillInStackTrace();
301        }
302    }

312    private File findFile( final File dir, final Attachment att ) throws FileNotFoundException, ProviderException {
313        int version = att.getVersion();
314        if( version == WikiProvider.LATEST_VERSION ) {
315            version = findLatestVersion( att );
316        }
317
318        final String ext = getFileExtension( att.getFileName() );
319        File f = new File( dir, version + "." + ext );
320
321        if( !f.exists() ) {
322            if( "bin".equals( ext ) ) {
323                final File fOld = new File( dir, version + "." );
324                if( fOld.exists() ) {
325                    f = fOld;
326                }
327            }
328            if( !f.exists() ) {
329                throw new FileNotFoundException( "No such file: " + f.getAbsolutePath() + " exists." );
330            }
331        }
332
333        return f;
334    }

340    public InputStream getAttachmentData( final Attachment att ) throws IOException, ProviderException {
341        final File attDir = findAttachmentDir( att );
342        try {
343            final File f = findFile( attDir, att );
344            return new FileInputStream( f );
345        } catch( final FileNotFoundException e ) {
346            log.error( "File not found: " + e.getMessage() );
347            throw new ProviderException( "No such page was found." );

351    /**
352     *  {@inheritDoc}
353     */
354    @Override
355    public List< Attachment > listAttachments( final Page page ) throws ProviderException {
356        final List< Attachment > result = new ArrayList<>();
357        final File dir = findPageDir( page.getName() );
358        final String[] attachments = dir.list();
359        if( attachments != null ) {
360            //  We now have a list of all potential attachments in the directory.
361            for( final String attachment : attachments ) {
362                final File f = new File( dir, attachment );
363                if( f.isDirectory() ) {
364                    String attachmentName = unmangleName( attachment );
365
366                    //  Is it a new-stylea attachment directory?  If yes, we'll just deduce the name.  If not, however,
367                    //  we'll check if there's a suitable property file in the directory.
368                    if( attachmentName.endsWith( ATTDIR_EXTENSION ) ) {
369                        attachmentName = attachmentName.substring( 0, attachmentName.length() - ATTDIR_EXTENSION.length() );
370                    } else {
371                        final File propFile = new File( f, PROPERTY_FILE );
372                        if( !propFile.exists() ) {
373                            //  This is not obviously a JSPWiki attachment, so let's just skip it.
374                            continue;
375                        }
376                    }
377
378                    final Attachment att = getAttachmentInfo( page, attachmentName, WikiProvider.LATEST_VERSION );
379                    //  Sanity check - shouldn't really be happening, unless you mess with the repository directly.
380                    if( att == null ) {
381                        throw new ProviderException( "Attachment disappeared while reading information:"
382                                + " if you did not touch the repository, there is a serious bug somewhere. " + "Attachment = " + attachment
383                                + ", decoded = " + attachmentName );
384                    }
385
386                    result.add( att );
387                }
388            }
389        }
390
391        return result;
392    }

405    // FIXME: Very unoptimized.
406    @Override
407    public List< Attachment > listAllChanged( final Date timestamp ) throws ProviderException {
408        final File attDir = new File( m_storageDir );
409        if( !attDir.exists() ) {
410            throw new ProviderException( "Specified attachment directory " + m_storageDir + " does not exist!" );
411        }
412
413        final ArrayList< Attachment > list = new ArrayList<>();
414        final String[] pagesWithAttachments = attDir.list( new AttachmentFilter() );
415
416        if( pagesWithAttachments != null ) {
417            for( final String pagesWithAttachment : pagesWithAttachments ) {
418                String pageId = unmangleName( pagesWithAttachment );
419                pageId = pageId.substring( 0, pageId.length() - DIR_EXTENSION.length() );
420
421                final Collection< Attachment > c = listAttachments( Wiki.contents().page( m_engine, pageId ) );
422                for( final Attachment att : c ) {
423                    if( att.getLastModified().after( timestamp ) ) {
424                        list.add( att );
425                    }
426                }
427            }
428        }
429
430        list.sort( new PageTimeComparator() );
431
432        return list;
433    }

435    /**
436     *  {@inheritDoc}
437     */
438    @Override
439    public Attachment getAttachmentInfo( final Page page, final String name, int version ) throws ProviderException {
440        final Attachment att = new org.apache.wiki.attachment.Attachment( m_engine, page.getName(), name );
441        final File dir = findAttachmentDir( att );
442        if( !dir.exists() ) {
443            // log.debug("Attachment dir not found - thus no attachment can exist.");
444            return null;
445        }
446        
447        if( version == WikiProvider.LATEST_VERSION ) {
448            version = findLatestVersion(att);
449        }
450
451        att.setVersion( version );
452        
453        // Should attachment be cachable by the client (browser)?
454        if( m_disableCache != null ) {
455            final Matcher matcher = m_disableCache.matcher( name );
456            if( matcher.matches() ) {
457                att.setCacheable( false );
458            }
459        }
460
461        // System.out.println("Fetching info on version "+version);
462        try {
463            final Properties props = getPageProperties( att );
464            att.setAuthor( props.getProperty( version+".author" ) );
465            final String changeNote = props.getProperty( version+".changenote" );
466            if( changeNote != null ) {
467                att.setAttribute( Page.CHANGENOTE, changeNote );
468            }
469
470            final File f = findFile( dir, att );
471            att.setSize( f.length() );
472            att.setLastModified( new Date( f.lastModified() ) );
473        } catch( final FileNotFoundException e ) {
474            log.error( "Can't get attachment properties for " + att, e );
475            return null;
476        } catch( final IOException e ) {
477            log.error("Can't read page properties", e );
478            throw new ProviderException("Cannot read page properties: "+e.getMessage());
479        }
480        // FIXME: Check for existence of this particular version.
481
482        return att;
483    }

485    /**
486     *  {@inheritDoc}
487     */
488    @Override
489    public List< Attachment > getVersionHistory( final Attachment att ) {
490        final ArrayList< Attachment > list = new ArrayList<>();
491        try {
492            final int latest = findLatestVersion( att );
493            for( int i = latest; i >= 1; i-- ) {
494                final Attachment a = getAttachmentInfo( Wiki.contents().page( m_engine, att.getParentName() ), att.getFileName(), i );
495                if( a != null ) {
496                    list.add( a );
497                }
498            }
499        } catch( final ProviderException e ) {
500            log.error( "Getting version history failed for page: " + att, e );
501            // FIXME: Should this fail?
502        }
503
504        return list;
505    }

519    public void deleteAttachment( final Attachment att ) throws ProviderException {
520        final File dir = findAttachmentDir( att );
521        final String[] files = dir.list();
522        for( final String s : files ) {
523            final File file = new File( dir.getAbsolutePath() + "/" + s );

525        }

557    /**
558     *  {@inheritDoc}
559     */
560    @Override
561    public void moveAttachmentsForPage( final String oldParent, final String newParent ) throws ProviderException {
562        final File srcDir = findPageDir( oldParent );
563        final File destDir = findPageDir( newParent );
564
565        log.debug( "Trying to move all attachments from " + srcDir + " to " + destDir );
566
567        // If it exists, we're overwriting an old page (this has already been confirmed at a higher level), so delete any existing attachments.
568        if( destDir.exists() ) {
569            log.error( "Page rename failed because target directory " + destDir + " exists" );
570        } else {
571            // destDir.getParentFile().mkdir();
572            srcDir.renameTo( destDir );
573        }
574    }
