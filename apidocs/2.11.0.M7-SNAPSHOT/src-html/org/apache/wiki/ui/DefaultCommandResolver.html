038/**
039 * <p>Default implementation for {@link CommandResolver}</p>
040 *
041 * @since 2.4.22
042 */
043public final class DefaultCommandResolver implements CommandResolver {
044
045    /** Private map with request contexts as keys, Commands as values */
046    private static final Map< String, Command > CONTEXTS;
047
048    /** Private map with JSPs as keys, Commands as values */
049    private static final Map< String, Command > JSPS;
050
051    /* Store the JSP-to-Command and context-to-Command mappings */
052    static {
053        CONTEXTS = new HashMap<>();
054        JSPS = new HashMap<>();
055        final Command[] commands = AbstractCommand.allCommands();
056        for( final Command command : commands ) {
057            JSPS.put( command.getJSP(), command );
058            CONTEXTS.put( command.getRequestContext(), command );
059        }
060    }
061
062    private static final Logger LOG = Logger.getLogger( DefaultCommandResolver.class );
063
064    private final WikiEngine m_engine;
065
066    /** If true, we'll also consider english plurals (+s) a match. */
067    private final boolean m_matchEnglishPlurals;
068
069    /** Stores special page names as keys, and Commands as values. */
070    private final Map<String, Command> m_specialPages;
071
072    /**
073     * Constructs a CommandResolver for a given WikiEngine. This constructor will extract the special page references for this wiki and
074     * store them in a cache used for resolution.
075     *
076     * @param engine the wiki engine
077     * @param properties the properties used to initialize the wiki
078     */
079    public DefaultCommandResolver( final WikiEngine engine, final Properties properties ) {
080        m_engine = engine;
081        m_specialPages = new HashMap<>();
082
083        // Skim through the properties and look for anything with the "special page" prefix. Create maps that allow us look up
084        // the correct Command based on special page name. If a matching command isn't found, create a RedirectCommand.
085        for( final String key : properties.stringPropertyNames() ) {
086            if ( key.startsWith( PROP_SPECIALPAGE ) ) {
087                String specialPage = key.substring( PROP_SPECIALPAGE.length() );
088                String jsp = properties.getProperty( key );
089                if ( jsp != null ) {
090                    specialPage = specialPage.trim();
091                    jsp = jsp.trim();
092                    Command command = JSPS.get( jsp );
093                    if ( command == null ) {
094                        final Command redirect = RedirectCommand.REDIRECT;
095                        command = redirect.targetedCommand( jsp );
096                    }
097                    m_specialPages.put( specialPage, command );
098                }
099            }
100        }
101
102        // Do we match plurals?
103        m_matchEnglishPlurals = TextUtil.getBooleanProperty( properties, WikiEngine.PROP_MATCHPLURALS, true );
104    }
105
106    /**
107     * {@inheritDoc}
108     */
109    @Override
110    public Command findCommand( final HttpServletRequest request, final String defaultContext ) {
111        // Corner case if request is null
112        if ( request == null ) {
113            return CommandResolver.findCommand( defaultContext );
114        }
115
116        Command command = null;
117
118        // Determine the name of the page (which may be null)
119        String pageName = extractPageFromParameter( defaultContext, request );
120
121        // Can we find a special-page command matching the extracted page?
122        if ( pageName != null ) {
123            command = m_specialPages.get( pageName );
124        }
125
126        // If we haven't found a matching command yet, extract the JSP path and compare to our list of special pages
127        if ( command == null ) {
128            command = extractCommandFromPath( request );
129
130            // Otherwise: use the default context
131            if ( command == null ) {
132                command = CONTEXTS.get( defaultContext );
133                if ( command == null ) {
134                    throw new IllegalArgumentException( "Wiki context " + defaultContext + " is illegal." );
135                }
136            }
137        }
138
139        // For PageCommand.VIEW, default to front page if a page wasn't supplied
140        if( PageCommand.VIEW.equals( command ) && pageName == null ) {
141            pageName = m_engine.getFrontPage();
142        }
143
144        // These next blocks handle targeting requirements
145
146        // If we were passed a page parameter, try to resolve it
147        if ( command instanceof PageCommand && pageName != null ) {
148            // If there's a matching WikiPage, "wrap" the command
149            final WikiPage page = resolvePage( request, pageName );
150            return command.targetedCommand( page );
151        }
152
153        // If "create group" command, target this wiki
154        final String wiki = m_engine.getApplicationName();
155        if ( WikiCommand.CREATE_GROUP.equals( command ) ) {
156            return WikiCommand.CREATE_GROUP.targetedCommand( wiki );
157        }
158
159        // If group command, see if we were passed a group name
160        if( command instanceof GroupCommand ) {
161            String groupName = request.getParameter( "group" );
162            groupName = TextUtil.replaceEntities( groupName );
163            if ( groupName != null && groupName.length() > 0 ) {
164                final GroupPrincipal group = new GroupPrincipal( groupName );
165                return command.targetedCommand( group );
166            }
167        }
168
169        // No page provided; return an "ordinary" command
170        return command;
171    }

173    /**
174     * {@inheritDoc}
175     */
176    @Override
177    public String getFinalPageName( final String page ) throws ProviderException {
178        boolean isThere = simplePageExists( page );
179        String  finalName = page;
180
181        if ( !isThere && m_matchEnglishPlurals ) {
182            if ( page.endsWith( "s" ) ) {
183                finalName = page.substring( 0, page.length() - 1 );
184            } else {
185                finalName += "s";
186            }
187
188            isThere = simplePageExists( finalName );
189        }
190
191        if( !isThere ) {
192            finalName = MarkupParser.wikifyLink( page );
193            isThere = simplePageExists(finalName);
194
195            if( !isThere && m_matchEnglishPlurals ) {
196                if( finalName.endsWith( "s" ) ) {
197                    finalName = finalName.substring( 0, finalName.length() - 1 );
198                } else {
199                    finalName += "s";
200                }
201
202                isThere = simplePageExists( finalName );
203            }
204        }
205
206        return isThere ? finalName : null;
207    }

209    /**
210     * {@inheritDoc}
211     */
212    @Override
213    public String getSpecialPageReference( final String page ) {
214        final Command command = m_specialPages.get( page );
215        if ( command != null ) {
216            return m_engine.getURLConstructor().makeURL( command.getRequestContext(), command.getURLPattern(), null );
217        }
218
219        return null;
220    }
221
222    /**
223     * Extracts a Command based on the JSP path of an HTTP request. If the JSP requested matches a Command's <code>getJSP()</code>
224     * value, that Command is returned.
225     *
226     * @param request the HTTP request
227     * @return the resolved Command, or <code>null</code> if not found
228     */
229    protected Command extractCommandFromPath( final HttpServletRequest request ) {
230        String jsp = request.getServletPath();
231
232        // Take everything to right of initial / and left of # or ?
233        final int hashMark = jsp.indexOf( '#' );
234        if ( hashMark != -1 ) {
235            jsp = jsp.substring( 0, hashMark );
236        }
237        final int questionMark = jsp.indexOf( '?' );
238        if ( questionMark != -1 ) {
239            jsp = jsp.substring( 0, questionMark );
240        }
241        if ( jsp.startsWith( "/" ) ) {
242            jsp = jsp.substring( 1 );
243        }
244
245        // Find special page reference?
246        for( final Map.Entry< String, Command > entry : m_specialPages.entrySet() ) {
247            final Command specialCommand = entry.getValue();
248            if( specialCommand.getJSP().equals( jsp ) ) {
249                return specialCommand;
250            }
251        }
252
253        // Still haven't found a matching command? Ok, see if we match against our standard list of JSPs
254        if ( jsp.length() > 0 && JSPS.containsKey( jsp ) ) {
255            return JSPS.get( jsp );
256        }
257
258        return null;
259    }
260
261    /**
262     * {@inheritDoc}
263     */
264    @Override
265    public String extractPageFromParameter( final String requestContext, final HttpServletRequest request ) {
266        // Extract the page name from the URL directly
267        try {
268            String page = m_engine.getURLConstructor().parsePage( requestContext, request, m_engine.getContentEncoding() );
269            if ( page != null ) {
270                try {
271                    // Look for singular/plural variants; if one not found, take the one the user supplied
272                    final String finalPage = getFinalPageName( page );
273                    if ( finalPage != null ) {
274                        page = finalPage;
275                    }
276                } catch( final ProviderException e ) {
277                    // FIXME: Should not ignore!
278                }
279                return page;
280            }
281        } catch( final IOException e ) {
282            LOG.error( "Unable to create context", e );
283            throw new InternalWikiException( "Big internal booboo, please check logs." , e );
284        }
285
286        // Didn't resolve; return null
287        return null;
288    }
289
290    /**
291     * {@inheritDoc}
292     */
293    @Override
294    public WikiPage resolvePage( final HttpServletRequest request, String page ) {
295        // See if the user included a version parameter
296        int version = WikiProvider.LATEST_VERSION;
297        final String rev = request.getParameter( "version" );
298        if ( rev != null ) {
299            try {
300                version = Integer.parseInt( rev );
301            } catch( final NumberFormatException e ) {
302                // This happens a lot with bots or other guys who are trying to test if we are vulnerable to e.g. XSS attacks.  We catch
303                // it here so that the admin does not get tons of mail.
304            }
305        }
306
307        WikiPage wikipage = m_engine.getPageManager().getPage( page, version );
308        if ( wikipage == null ) {
309            page = MarkupParser.cleanLink( page );
310            wikipage = new WikiPage( m_engine, page );
311        }
312        return wikipage;
313    }
314
315    /**
316     * Determines whether a "page" exists by examining the list of special pages and querying the page manager.
317     *
318     * @param page the page to seek
319     * @return <code>true</code> if the page exists, <code>false</code> otherwise
320     * @throws ProviderException if the underlyng page provider that locates pages
321     * throws an exception
322     */
323    protected boolean simplePageExists( final String page ) throws ProviderException {
324        if ( m_specialPages.containsKey( page ) ) {
325            return true;
326        }
327        return m_engine.getPageManager().pageExists( page );
328    }
