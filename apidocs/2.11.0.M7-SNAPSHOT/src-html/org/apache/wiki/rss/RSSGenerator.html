037/**
038 *  The master class for generating different kinds of Feeds (including RSS1.0, 2.0 and Atom).
039 *  <p>
040 *  This class can produce quite a few different styles of feeds.  The following modes are available:
041 *  
042 *  <ul>
043 *  <li><b>wiki</b> - All the changes to the given page are enumerated and announced as diffs.</li>
044 *  <li><b>full</b> - Each page is only considered once.  This produces a very RecentChanges-style feed,
045 *                   where each page is only listed once, even if it has changed multiple times.</li>
046 *  <li><b>blog</b> - Each page change is assumed to be a blog entry, so no diffs are produced, but
047 *                    the page content is always completely in the entry in rendered HTML.</li>
048 *
049 *  @since  1.7.5.
050 */
051// FIXME: Limit diff and page content size.
052// FIXME3.0: This class would need a bit of refactoring.  Method names, e.g. are confusing.

134     *  Does the required formatting and entity replacement for XML.
135     *  
136     *  @param s String to format.
137     *  @return A formatted string.
138     */
139    // FIXME: Replicates Feed.format().
140    public static String format( String s ) {
141        s = TextUtil.replaceString( s, "&", "&amp;" );
142        s = TextUtil.replaceString( s, "<", "&lt;" );
143        s = TextUtil.replaceString( s, "]]>", "]]&gt;" );
144
145        return s.trim();
146    }

148    private String getAuthor( final WikiPage page ) {
149        String author = page.getAuthor();
150        if( author == null ) {
151            author = "An unknown author";
152        }
153
154        return author;
155    }

157    private String getAttachmentDescription( final Attachment att ) {
158        final String author = getAuthor( att );
159        final StringBuilder sb = new StringBuilder();
160
161        if( att.getVersion() != 1 ) {
162            sb.append( author ).append( " uploaded a new version of this attachment on " ).append( att.getLastModified() );
163        } else {
164            sb.append( author ).append( " created this attachment on " ).append( att.getLastModified() );
165        }
166
167        sb.append( "<br /><hr /><br />" )
168          .append( "Parent page: <a href=\"" )
169          .append( m_engine.getURL( WikiContext.VIEW, att.getParentName(), null ) )
170          .append( "\">" ).append( att.getParentName() ).append( "</a><br />" )
171          .append( "Info page: <a href=\"" )
172          .append( m_engine.getURL( WikiContext.INFO, att.getName(), null ) )
173          .append( "\">" ).append( att.getName() ).append( "</a>" );
174
175        return sb.toString();
176    }
177
178    private String getPageDescription( final WikiPage page ) {
179        final StringBuilder buf = new StringBuilder();
180        final String author = getAuthor( page );
181        final WikiContext ctx = new WikiContext( m_engine, page );
182        if( page.getVersion() > 1 ) {
183            final String diff = m_engine.getDifferenceManager().getDiff( ctx,
184                                                                page.getVersion() - 1, // FIXME: Will fail when non-contiguous versions
185                                                                         page.getVersion() );
186
187            buf.append( author ).append( " changed this page on " ).append( page.getLastModified() ).append( ":<br /><hr /><br />" );
188            buf.append( diff );
189        } else {
190            buf.append( author ).append( " created this page on " ).append( page.getLastModified() ).append( ":<br /><hr /><br />" );
191            buf.append( m_engine.getRenderingManager().getHTML( page.getName() ) );
192        }
193
194        return buf.toString();
195    }
196
197    private String getEntryDescription( final WikiPage page ) {
198        final String res;
199        if( page instanceof Attachment ) {
200            res = getAttachmentDescription( (Attachment)page );
201        } else {
202            res = getPageDescription( page );
203        }
204
205        return res;
206    }

211        return page.getName() + ", version " + page.getVersion();

214    /**
215     *  Generates the RSS resource.  You probably want to output this result into a file or something, or serve as output from a servlet.
216     *  
217     *  @return A RSS 1.0 feed in the "full" mode.
218     */
219    public String generate() {
220        final WikiContext context = new WikiContext( m_engine, new WikiPage( m_engine, "__DUMMY" ) );
221        context.setRequestContext( WikiContext.RSS );
222        final Feed feed = new RSS10Feed( context );
223        return "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n" + generateFullWikiRSS( context, feed );
224    }

226    /**
227     * Returns the content type of this RSS feed.
228     *
229     * @since 2.3.15
230     * @param mode the RSS mode: {@link #RSS10}, {@link #RSS20} or {@link #ATOM}.
231     * @return the content type
232     */
233    public static String getContentType( final String mode ) {
234        if( mode.equals( RSS10 ) || mode.equals( RSS20 ) ) {
235            return "application/rss+xml";
236        } else if( mode.equals( ATOM ) ) {
237            return "application/atom+xml";
238        }
239
240        return "application/octet-stream"; // Unknown type
241    }

243    /**
244     * Generates a feed based on a context and list of changes.
245     *
246     * @param wikiContext The WikiContext
247     * @param changed A list of Entry objects
248     * @param mode The mode (wiki/blog)
249     * @param type The type (RSS10, RSS20, ATOM).  Default is RSS 1.0
250     * @return Fully formed XML.
251     * @throws IllegalArgumentException If an illegal mode is given.
252     */
253    public String generateFeed( final WikiContext wikiContext, final List< WikiPage > changed, final String mode, final String type ) throws IllegalArgumentException {
254        final Feed feed;
255        final String res;
256
257        if( ATOM.equals(type) ) {
258            feed = new AtomFeed( wikiContext );
259        } else if( RSS20.equals( type ) ) {
260            feed = new RSS20Feed( wikiContext );
261        } else {
262            feed = new RSS10Feed( wikiContext );
263        }
264
265        feed.setMode( mode );
266
267        if( MODE_BLOG.equals( mode ) ) {
268            res = generateBlogRSS( wikiContext, changed, feed );
269        } else if( MODE_FULL.equals(mode) ) {
270            res = generateFullWikiRSS( wikiContext, feed );
271        } else if( MODE_WIKI.equals(mode) ) {
272            res = generateWikiPageRSS( wikiContext, changed, feed );
273        } else {
274            throw new IllegalArgumentException( "Invalid value for feed mode: "+mode );
275        }
276
277        return res;
278    }

301    /**
302     *  Generates an RSS feed for the entire wiki.  Each item should be an instance of the RSSItem class.
303     *  
304     *  @param wikiContext A WikiContext
305     *  @param feed A Feed to generate the feed to.
306     *  @return feed.getString().
307     */
308    protected String generateFullWikiRSS( final WikiContext wikiContext, final Feed feed ) {
309        feed.setChannelTitle( m_engine.getApplicationName() );
310        feed.setFeedURL( m_engine.getBaseURL() );
311        feed.setChannelLanguage( m_channelLanguage );
312        feed.setChannelDescription( m_channelDescription );
313
314        final Set< WikiPage > changed = m_engine.getPageManager().getRecentChanges();
315
316        final WikiSession session = WikiSession.guestSession( m_engine );
317        int items = 0;
318        for( final Iterator< WikiPage > i = changed.iterator(); i.hasNext() && items < 15; items++ ) {
319            final WikiPage page = i.next();
320
321            //  Check if the anonymous user has view access to this page.
322            if( !m_engine.getAuthorizationManager().checkPermission(session, new PagePermission(page,PagePermission.VIEW_ACTION) ) ) {
323                // No permission, skip to the next one.
324                continue;
325            }
326
327            final String url;
328            if( page instanceof Attachment ) {
329                url = m_engine.getURL( WikiContext.ATTACH, page.getName(),null );
330            } else {
331                url = m_engine.getURL( WikiContext.VIEW, page.getName(), null );
332            }
333
334            final Entry e = new Entry();
335            e.setPage( page );
336            e.setURL( url );
337            e.setTitle( page.getName() );
338            e.setContent( getEntryDescription(page) );
339            e.setAuthor( getAuthor(page) );
340
341            feed.addEntry( e );
342        }
343
344        return feed.getString();
345    }

347    /**
348     * Create RSS/Atom as if this page was a wikipage (in contrast to Blog mode).
349     *
350     * @param wikiContext The WikiContext
351     * @param changed A List of changed WikiPages.
352     * @param feed A Feed object to fill.
353     * @return the RSS representation of the wiki context
354     */
355    protected String generateWikiPageRSS( final WikiContext wikiContext, final List< WikiPage > changed, final Feed feed ) {
356        feed.setChannelTitle( m_engine.getApplicationName()+": "+wikiContext.getPage().getName() );
357        feed.setFeedURL( wikiContext.getViewURL( wikiContext.getPage().getName() ) );
358        final String language = m_engine.getVariableManager().getVariable( wikiContext, PROP_CHANNEL_LANGUAGE );
359
360        if( language != null ) {
361            feed.setChannelLanguage( language );
362        } else {
363            feed.setChannelLanguage( m_channelLanguage );
364        }
365        final String channelDescription = m_engine.getVariableManager().getVariable( wikiContext, PROP_CHANNEL_DESCRIPTION );
366
367        if( channelDescription != null ) {
368            feed.setChannelDescription( channelDescription );
369        }
370
371        changed.sort( new PageTimeComparator() );
372
373        int items = 0;
374        for( final Iterator< WikiPage > i = changed.iterator(); i.hasNext() && items < 15; items++ ) {
375            final WikiPage page = i.next();
376            final Entry e = new Entry();
377            e.setPage( page );
378            String url;
379
380            if( page instanceof Attachment ) {
381                url = m_engine.getURL( WikiContext.ATTACH, page.getName(), "version=" + page.getVersion() );
382            } else {
383                url = m_engine.getURL( WikiContext.VIEW, page.getName(), "version=" + page.getVersion() );
384            }
385
386            // Unfortunately, this is needed because the code will again go through replacement conversion
387            url = TextUtil.replaceString( url, "&amp;", "&" );
388            e.setURL( url );
389            e.setTitle( getEntryTitle(page) );
390            e.setContent( getEntryDescription(page) );
391            e.setAuthor( getAuthor(page) );
392
393            feed.addEntry( e );
394        }
395
396        return feed.getString();
397    }

400    /**
401     *  Creates RSS from modifications as if this page was a blog (using the WeblogPlugin).
402     *
403     *  @param wikiContext The WikiContext, as usual.
404     *  @param changed A list of the changed pages.
405     *  @param feed A valid Feed object.  The feed will be used to create the RSS/Atom, depending on which kind of an object you want to put in it.
406     *  @return A String of valid RSS or Atom.
407     */
408    protected String generateBlogRSS( final WikiContext wikiContext, final List< WikiPage > changed, final Feed feed ) {
409        if( log.isDebugEnabled() ) {
410            log.debug( "Generating RSS for blog, size=" + changed.size() );
411        }
412
413        final String ctitle = m_engine.getVariableManager().getVariable( wikiContext, PROP_CHANNEL_TITLE );
414        if( ctitle != null ) {
415            feed.setChannelTitle( ctitle );
416        } else {
417            feed.setChannelTitle( m_engine.getApplicationName() + ":" + wikiContext.getPage().getName() );
418        }
419
420        feed.setFeedURL( wikiContext.getViewURL( wikiContext.getPage().getName() ) );
421
422        final String language = m_engine.getVariableManager().getVariable( wikiContext, PROP_CHANNEL_LANGUAGE );
423        if( language != null ) {
424            feed.setChannelLanguage( language );
425        } else {
426            feed.setChannelLanguage( m_channelLanguage );
427        }
428
429        final String channelDescription = m_engine.getVariableManager().getVariable( wikiContext, PROP_CHANNEL_DESCRIPTION );
430        if( channelDescription != null ) {
431            feed.setChannelDescription( channelDescription );
432        }
433
434        changed.sort( new PageTimeComparator() );
435
436        int items = 0;
437        for( final Iterator< WikiPage > i = changed.iterator(); i.hasNext() && items < 15; items++ ) {
438            final WikiPage page = i.next();
439            final Entry e = new Entry();
440            e.setPage( page );
441            final String url;
442
443            if( page instanceof Attachment ) {
444                url = m_engine.getURL( WikiContext.ATTACH, page.getName(),null );
445            } else {
446                url = m_engine.getURL( WikiContext.VIEW, page.getName(),null );
447            }
448
449            e.setURL( url );
450
451            //  Title
452            String pageText = m_engine.getPageManager().getPureText( page.getName(), WikiProvider.LATEST_VERSION );
453
454            String title = "";
455            final int firstLine = pageText.indexOf('\n');
456
457            if( firstLine > 0 ) {
458                title = pageText.substring( 0, firstLine ).trim();
459            }
460
461            if( title.length() == 0 ) {
462                title = page.getName();
463            }
464
465            // Remove wiki formatting
466            while( title.startsWith("!") ) {
467                title = title.substring(1);
468            }
469
470            e.setTitle( title );
471
472            //  Description
473            if( firstLine > 0 ) {
474                int maxlen = pageText.length();
475                if( maxlen > MAX_CHARACTERS ) {
476                    maxlen = MAX_CHARACTERS;
477                }
478                pageText = m_engine.getRenderingManager().textToHTML( wikiContext, pageText.substring( firstLine + 1, maxlen ).trim() );
479                if( maxlen == MAX_CHARACTERS ) {
480                    pageText += "...";
481                }
482                e.setContent( pageText );
483            } else {
484                e.setContent( title );
485            }
486            e.setAuthor( getAuthor(page) );
487            feed.addEntry( e );
488        }
489
490        return feed.getString();
491    }
