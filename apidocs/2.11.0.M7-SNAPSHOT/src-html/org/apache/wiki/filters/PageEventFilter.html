020package org.apache.wiki.filters;
021
022import org.apache.wiki.api.filters.BasicPageFilter;
023import org.apache.wiki.api.exceptions.FilterException;
024
025import org.apache.wiki.WikiContext;
026import org.apache.wiki.event.WikiEventListener;
027import org.apache.wiki.event.WikiEventManager;
028import org.apache.wiki.event.WikiPageEvent;
029
030import java.util.Properties;
031
032/**
033  * Fires WikiPageEvents for page events.
034  * <p>
035  * Adding a PageEventFilter to the FilterManager will automatically attach an event delegate with the WikiEventManager to provide for
036  * firing and listener management. All that remains is then adding the listener to the filter via the WikiEventManager. This is quite
037  * simple:
038  * </p>
039  * <pre>
040  *    PageEventFilter filter = new PageEventFilter();
041  *    engine.getFilterManager().addPageFilter(filter,5000);
042  *    // attach listener to filter
043  *    WikiEventManager.addWikiEventListener(filter,listener);
044  * </pre>
045  * <p>
046  * This class provides convenience methods for adding and removing WikiEventListeners.
047  * </p>
048  *
049  * @see org.apache.wiki.event.WikiEventManager
050  */
051public class PageEventFilter extends BasicPageFilter {
052
053    /**
054     * Called whenever a new PageFilter is instantiated and reset.
055     */
056    public void initialize( final Properties properties ) throws FilterException {
057        //
058    }
059
060    /**
061     * This method is called whenever a page has been loaded from the provider, but not yet been sent through the TranslatorReader.
062     * Note that you cannot do HTML translation here, because TranslatorReader is likely to escape it.
063     *
064     * @param wikiContext The current wikicontext.
065     * @param content WikiMarkup.
066     */
067    public String preTranslate( final WikiContext wikiContext, final String content ) {
068        fireEvent( WikiPageEvent.PRE_TRANSLATE, wikiContext );
069        return content;
070    }
071
072
073    /**
074     * This method is called after a page has been fed through the TranslatorReader, so anything you are seeing here is translated content.
075     * If you want to do any of your own WikiMarkup2HTML translation, do it here.
076     */
077    public String postTranslate( final WikiContext wikiContext, final String htmlContent ) {
078        fireEvent( WikiPageEvent.POST_TRANSLATE, wikiContext );
079        return htmlContent;
080    }
081
082
083    /**
084      * This method is called before the page has been saved to the PageProvider.
085      */
086    public String preSave( final WikiContext wikiContext, final String content ) {
087        fireEvent( WikiPageEvent.PRE_SAVE, wikiContext );
088        return content;
089    }
090
091
092    /**
093      * This method is called after the page has been successfully saved. If the saving fails for any reason, then this method will not
094      * be called.
095      * <p>
096      * Since the result is discarded from this method, this is only useful for things like counters, etc.
097      */
098    public void postSave( final WikiContext wikiContext, final String content ) {
099        fireEvent( WikiPageEvent.POST_SAVE, wikiContext );
100    }
101
102
103    // events processing .......................................................
104
105    /**
106     *  Registers a WikiEventListener with this instance. This is a convenience method.
107     *
108     * @param listener the event listener
109     */
110    public final synchronized void addWikiEventListener( final WikiEventListener listener ) {
111        WikiEventManager.addWikiEventListener( this, listener );
112    }
113
114    /**
115     *  Un-registers a WikiEventListener with this instance. This is a convenience method.
116     *
117     * @param listener the event listener
118     */
119    public final synchronized void removeWikiEventListener( final WikiEventListener listener ) {
120        WikiEventManager.removeWikiEventListener( this, listener );
121    }
122
123    /**
124     *  Fires a WikiPageEvent of the provided type and page name to all registered listeners. Only <tt>PAGE_LOCK</tt> and
125     *  <tt>PAGE_UNLOCK</tt> event types will fire an event; other event types are ignored.
126     *
127     * @see org.apache.wiki.event.WikiPageEvent
128     * @param type      the WikiPageEvent type to be fired.
129     * @param context   the WikiContext of the event.
130     */
131    protected final void fireEvent( final int type, final WikiContext context ) {
132        if( WikiEventManager.isListening(this ) && WikiPageEvent.isValidType( type ) ) {
133            final WikiPageEvent event = new WikiPageEvent( context.getEngine(), type, context.getPage().getName() );
134            WikiEventManager.fireEvent(this, event );
135        }
136    }
137
138}
