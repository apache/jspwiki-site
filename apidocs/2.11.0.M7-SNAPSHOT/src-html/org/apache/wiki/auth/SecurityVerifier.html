137    /**
138     * Constructs a new SecurityVerifier for a supplied WikiEngine and WikiSession.
139     * @param engine the wiki engine
140     * @param session the wiki session (typically, that of an administrator)
141     */
142    public SecurityVerifier( final Engine engine, final WikiSession session ) {
143        m_engine = engine;
144        m_session = session;
145        m_session.clearMessages();
146        verifyJaas();
147        verifyPolicy();
148        try {
149            verifyPolicyAndContainerRoles();
150        } catch( final WikiException e ) {
151            m_session.addMessage( ERROR_ROLES, e.getMessage() );
152        }
153        verifyGroupDatabase();
154        verifyUserDatabase();
155    }

169    /**
170     * Formats and returns an HTML table containing sample permissions and what
171     * roles are allowed to have them. This method will throw an
172     * {@link IllegalStateException} if the authorizer is not of type
173     * {@link org.apache.wiki.auth.authorize.WebContainerAuthorizer}
174     * @return the formatted HTML table containing the result of the tests
175     */
176    public String policyRoleTable()
177    {
178        final Principal[] roles = m_policyPrincipals;
179        final String wiki = m_engine.getApplicationName();
180
181        final String[] pages = new String[]
182        { "Main", "Index", "GroupTest", "GroupAdmin" };
183        final String[] pageActions = new String[]
184        { "view", "edit", "modify", "rename", "delete" };
185
186        final String[] groups = new String[]
187        { "Admin", "TestGroup", "Foo" };
188        final String[] groupActions = new String[]
189        { "view", "edit", null, null, "delete" };
190
191        // Calculate column widths
192        final String colWidth;
193        if( pageActions.length > 0 && roles.length > 0 ) {
194            colWidth = ( 67f / ( pageActions.length * roles.length ) ) + "%";
195        } else {
196            colWidth = "67%";
197        }
198
199        final StringBuilder s = new StringBuilder();
200
201        // Write the table header
202        s.append( "<table class=\"wikitable\" border=\"1\">\n" );
203        s.append( "  <colgroup span=\"1\" width=\"33%\"/>\n" );
204        s.append( "  <colgroup span=\"" + pageActions.length * roles.length + "\" width=\"" + colWidth
205                + "\" align=\"center\"/>\n" );
206        s.append( "  <tr>\n" );
207        s.append( "    <th rowspan=\"2\" valign=\"bottom\">Permission</th>\n" );
208        for( int i = 0; i < roles.length; i++ )
209        {
210            s.append( "    <th colspan=\"" + pageActions.length + "\" title=\"" + roles[i].getClass().getName() + "\">"
211                    + roles[i].getName() + "</th>\n" );
212        }
213        s.append( "  </tr>\n" );
214
215        // Print a column for each role
216        s.append( "  <tr>\n" );
217        for( int i = 0; i < roles.length; i++ )
218        {
219            for( final String pageAction : pageActions )
220            {
221                final String action = pageAction.substring( 0, 1 );
222                s.append( "    <th title=\"" + pageAction + "\">" + action + "</th>\n" );
223            }
224        }
225        s.append( "  </tr>\n" );
226
227        // Write page permission tests first
228        for( final String page : pages ) {
229            s.append( "  <tr>\n" );
230            s.append( "    <td>PagePermission \"" + wiki + ":" + page + "\"</td>\n" );
231            for( final Principal role : roles ) {
232                for( final String pageAction : pageActions ) {
233                    final Permission permission = PermissionFactory.getPagePermission( wiki + ":" + page, pageAction );
234                    s.append( printPermissionTest( permission, role, 1 ) );
235                }
236            }
237            s.append( "  </tr>\n" );
238        }
239
240        // Now do the group tests
241        for( final String group : groups ) {
242            s.append( "  <tr>\n" );
243            s.append( "    <td>GroupPermission \"" + wiki + ":" + group + "\"</td>\n" );
244            for( final Principal role : roles ) {
245                for( final String groupAction : groupActions ) {
246                    Permission permission = null;
247                    if( groupAction != null ) {
248                        permission = new GroupPermission( wiki + ":" + group, groupAction );
249                    }
250                    s.append( printPermissionTest( permission, role, 1 ) );
251                }
252            }
253            s.append( "  </tr>\n" );
254        }
255
256
257        // Now check the wiki-wide permissions
258        final String[] wikiPerms = new String[] { "createGroups", "createPages", "login", "editPreferences", "editProfile" };
259        for( final String wikiPerm : wikiPerms ) {
260            s.append( "  <tr>\n" );
261            s.append( "    <td>WikiPermission \"" + wiki + "\",\"" + wikiPerm + "\"</td>\n" );
262            for( final Principal role : roles ) {
263                final Permission permission = new WikiPermission( wiki, wikiPerm );
264                s.append( printPermissionTest( permission, role, pageActions.length ) );
265            }
266            s.append( "  </tr>\n" );
267        }
268
269        // Lastly, check for AllPermission
270        s.append( "  <tr>\n" );
271        s.append( "    <td>AllPermission \"" + wiki + "\"</td>\n" );
272        for( final Principal role : roles )
273        {
274            final Permission permission = new AllPermission( wiki );
275            s.append( printPermissionTest( permission, role, pageActions.length ) );
276        }
277        s.append( "  </tr>\n" );
278
279        // We're done!
280        s.append( "</table>" );
281        return s.toString();
282    }

284    /**
285     * Prints a &lt;td&gt; HTML element with the results of a permission test.
286     * @param permission the permission to format
287     * @param principal
288     * @param cols
289     */
290    private String printPermissionTest( final Permission permission, final Principal principal, final int cols ) {
291        final StringBuilder s = new StringBuilder();
292        if( permission == null ) {
293            s.append( "    <td colspan=\"" + cols + "\" align=\"center\" title=\"N/A\">" );
294            s.append( "&nbsp;</td>\n" );
295        } else {
296            final boolean allowed = verifyStaticPermission( principal, permission );
297            s.append( "    <td colspan=\"" + cols + "\" align=\"center\" title=\"" );
298            s.append( allowed ? "ALLOW: " : "DENY: " );
299            s.append( permission.getClass().getName() );
300            s.append( " &quot;" );
301            s.append( permission.getName() );
302            s.append( "&quot;" );
303            if ( permission.getName() != null )
304            {
305                s.append( ",&quot;" );
306                s.append( permission.getActions() );
307                s.append( "&quot;" );
308            }
309            s.append( " " );
310            s.append( principal.getClass().getName() );
311            s.append( " &quot;" );
312            s.append( principal.getName() );
313            s.append( "&quot;" );
314            s.append( "\"" );
315            s.append( allowed ? BG_GREEN + ">" : BG_RED + ">" );
316            s.append( "&nbsp;</td>\n" );
317        }
318        return s.toString();
319    }

321    /**
322     * Formats and returns an HTML table containing the roles the web container
323     * is aware of, and whether each role maps to particular JSPs. This method
324     * throws an {@link IllegalStateException} if the authorizer is not of type
325     * {@link org.apache.wiki.auth.authorize.WebContainerAuthorizer}
326     * @return the formatted HTML table containing the result of the tests
327     * @throws WikiException if tests fail for unexpected reasons
328     */
329    public String containerRoleTable() throws WikiException {
330        final AuthorizationManager authorizationManager = m_engine.getManager( AuthorizationManager.class );
331        final Authorizer authorizer = authorizationManager.getAuthorizer();
332
333        // If authorizer not WebContainerAuthorizer, print error message
334        if ( !( authorizer instanceof WebContainerAuthorizer ) ) {
335            throw new IllegalStateException( "Authorizer should be WebContainerAuthorizer" );
336        }
337
338        // Now, print a table with JSP pages listed on the left, and
339        // an evaluation of each pages' constraints for each role
340        // we discovered
341        final StringBuilder s = new StringBuilder();
342        final Principal[] roles = authorizer.getRoles();
343        s.append( "<table class=\"wikitable\" border=\"1\">\n" );
344        s.append( "<thead>\n" );
345        s.append( "  <tr>\n" );
346        s.append( "    <th rowspan=\"2\">Action</th>\n" );
347        s.append( "    <th rowspan=\"2\">Page</th>\n" );
348        s.append( "    <th colspan=\"" + roles.length + 1 + "\">Roles</th>\n" );
349        s.append( "  </tr>\n" );
350        s.append( "  <tr>\n" );
351        s.append( "    <th>Anonymous</th>\n" );
352        for( final Principal role : roles ) {
353            s.append( "    <th>" + role.getName() + "</th>\n" );
354        }
355        s.append( "</tr>\n" );
356        s.append( "</thead>\n" );
357        s.append( "<tbody>\n" );
358
359        final WebContainerAuthorizer wca = (WebContainerAuthorizer) authorizer;
360        for( int i = 0; i < CONTAINER_ACTIONS.length; i++ ) {
361            final String action = CONTAINER_ACTIONS[i];
362            final String jsp = CONTAINER_JSPS[i];
363
364            // Print whether the page is constrained for each role
365            final boolean allowsAnonymous = !wca.isConstrained( jsp, Role.ALL );
366            s.append( "  <tr>\n" );
367            s.append( "    <td>" + action + "</td>\n" );
368            s.append( "    <td>" + jsp + "</td>\n" );
369            s.append( "    <td title=\"" );
370            s.append( allowsAnonymous ? "ALLOW: " : "DENY: " );
371            s.append( jsp );
372            s.append( " Anonymous" );
373            s.append( "\"" );
374            s.append( allowsAnonymous ? BG_GREEN + ">" : BG_RED + ">" );
375            s.append( "&nbsp;</td>\n" );
376            for( final Principal role : roles )
377            {
378                final boolean allowed = allowsAnonymous || wca.isConstrained( jsp, (Role)role );
379                s.append( "    <td title=\"" );
380                s.append( allowed ? "ALLOW: " : "DENY: " );
381                s.append( jsp );
382                s.append( " " );
383                s.append( role.getClass().getName() );
384                s.append( " &quot;" );
385                s.append( role.getName() );
386                s.append( "&quot;" );
387                s.append( "\"" );
388                s.append( allowed ? BG_GREEN + ">" : BG_RED + ">" );
389                s.append( "&nbsp;</td>\n" );
390            }
391            s.append( "  </tr>\n" );
392        }
393
394        s.append( "</tbody>\n" );
395        s.append( "</table>\n" );
396        return s.toString();
397    }

423    /**
424     * Verifies that the roles given in the security policy are reflected by the
425     * container <code>web.xml</code> file.
426     * @throws WikiException if the web authorizer cannot verify the roles
427     */
428    protected void verifyPolicyAndContainerRoles() throws WikiException
429    {
430        final Authorizer authorizer = m_engine.getManager( AuthorizationManager.class ).getAuthorizer();
431        final Principal[] containerRoles = authorizer.getRoles();
432        boolean missing = false;
433        for( final Principal principal : m_policyPrincipals )
434        {
435            if ( principal instanceof Role )
436            {
437                final Role role = (Role) principal;
438                final boolean isContainerRole = ArrayUtils.contains( containerRoles, role );
439                if ( !Role.isBuiltInRole( role ) && !isContainerRole )
440                {
441                    m_session.addMessage( ERROR_ROLES, "Role '" + role.getName() + "' is defined in security policy but not in web.xml." );
442                    missing = true;
443                }
444            }
445        }
446        if ( !missing )
447        {
448            m_session.addMessage( INFO_ROLES, "Every non-standard role defined in the security policy was also found in web.xml." );
449        }
450    }

452    /**
453     * Verifies that the group datbase was initialized properly, and that
454     * user add and delete operations work as they should.
455     */
456    protected void verifyGroupDatabase()
457    {
458        final GroupManager mgr = m_engine.getManager( GroupManager.class );
459        GroupDatabase db = null;
460        try {
461            db = m_engine.getManager( GroupManager.class ).getGroupDatabase();
462        } catch ( final WikiSecurityException e ) {
463            m_session.addMessage( ERROR_GROUPS, "Could not retrieve GroupManager: " + e.getMessage() );
464        }
465
466        // Check for obvious error conditions
467        if ( mgr == null || db == null ) {
468            if ( mgr == null ) {
469                m_session.addMessage( ERROR_GROUPS, "GroupManager is null; JSPWiki could not initialize it. Check the error logs." );
470            }
471            if ( db == null ) {
472                m_session.addMessage( ERROR_GROUPS, "GroupDatabase is null; JSPWiki could not initialize it. Check the error logs." );
473            }
474            return;
475        }
476
477        // Everything initialized OK...
478
479        // Tell user what class of database this is.
480        m_session.addMessage( INFO_GROUPS, "GroupDatabase is of type '" + db.getClass().getName() + "'. It appears to be initialized properly." );
481
482        // Now, see how many groups we have.
483        final int oldGroupCount;
484        try {
485            final Group[] groups = db.groups();
486            oldGroupCount = groups.length;
487            m_session.addMessage( INFO_GROUPS, "The group database contains " + oldGroupCount + " groups." );
488        } catch( final WikiSecurityException e ) {
489            m_session.addMessage( ERROR_GROUPS, "Could not obtain a list of current groups: " + e.getMessage() );
490            return;
491        }
492
493        // Try adding a bogus group with random name
494        final String name = "TestGroup" + System.currentTimeMillis();
495        final Group group;
496        try {
497            // Create dummy test group
498            group = mgr.parseGroup( name, "", true );
499            final Principal user = new WikiPrincipal( "TestUser" );
500            group.add( user );
501            db.save( group, new WikiPrincipal( "SecurityVerifier" ) );
502
503            // Make sure the group saved successfully
504            if( db.groups().length == oldGroupCount ) {
505                m_session.addMessage( ERROR_GROUPS, "Could not add a test group to the database." );
506                return;
507            }
508            m_session.addMessage( INFO_GROUPS, "The group database allows new groups to be created, as it should." );
509        } catch( final WikiSecurityException e ) {
510            m_session.addMessage( ERROR_GROUPS, "Could not add a group to the database: " + e.getMessage() );
511            return;
512        }
513
514        // Now delete the group; should be back to old count
515        try {
516            db.delete( group );
517            if( db.groups().length != oldGroupCount ) {
518                m_session.addMessage( ERROR_GROUPS, "Could not delete a test group from the database." );
519                return;
520            }
521            m_session.addMessage( INFO_GROUPS, "The group database allows groups to be deleted, as it should." );
522        } catch( final WikiSecurityException e ) {
523            m_session.addMessage( ERROR_GROUPS, "Could not delete a test group from the database: " + e.getMessage() );
524            return;
525        }
526
527        m_session.addMessage( INFO_GROUPS, "The group database configuration looks fine." );
528    }

530    /**
531     * Verfies the JAAS configuration. The configuration is valid if value of the
532     * <code>jspwiki.properties<code> property
533     * {@value org.apache.wiki.auth.AuthenticationManager#PROP_LOGIN_MODULE}
534     * resolves to a valid class on the classpath.
535     */
536    protected void verifyJaas() {
537        // Verify that the specified JAAS moduie corresponds to a class we can load successfully.
538        final String jaasClass = m_engine.getWikiProperties().getProperty( AuthenticationManager.PROP_LOGIN_MODULE );
539        if( jaasClass == null || jaasClass.length() == 0 ) {
540            m_session.addMessage( ERROR_JAAS, "The value of the '" + AuthenticationManager.PROP_LOGIN_MODULE
541                    + "' property was null or blank. This is a fatal error. This value should be set to a valid LoginModule implementation "
542                    + "on the classpath." );
543            return;
544        }
545
546        // See if we can find the LoginModule on the classpath
547        Class< ? > c = null;
548        try {
549            m_session.addMessage( INFO_JAAS,
550                    "The property '" + AuthenticationManager.PROP_LOGIN_MODULE + "' specified the class '" + jaasClass + ".'" );
551            c = Class.forName( jaasClass );
552        } catch( final ClassNotFoundException e ) {
553            m_session.addMessage( ERROR_JAAS, "We could not find the the class '" + jaasClass + "' on the " + "classpath. This is fatal error." );
554        }
555
556        // Is the specified class actually a LoginModule?
557        if( LoginModule.class.isAssignableFrom( c ) ) {
558            m_session.addMessage( INFO_JAAS, "We found the the class '" + jaasClass + "' on the classpath, and it is a LoginModule implementation. Good!" );
559        } else {
560            m_session.addMessage( ERROR_JAAS, "We found the the class '" + jaasClass + "' on the classpath, but it does not seem to be LoginModule implementation! This is fatal error." );
561        }
562    }

564    /**
565     * Looks up a file name based on a JRE system property and returns the associated
566     * File object if it exists. This method adds messages with the topic prefix 
567     * {@link #ERROR} and {@link #INFO} as appropriate, with the suffix matching the 
568     * supplied property.
569     * @param property the system property to look up
570     * @return the file object, or <code>null</code> if not found
571     */
572    protected File getFileFromProperty( final String property )
573    {
574        String propertyValue = null;
575        try
576        {
577            propertyValue = System.getProperty( property );
578            if ( propertyValue == null )
579            {
580                m_session.addMessage( "Error." + property, "The system property '" + property + "' is null." );
581                return null;
582            }
583
584            //
585            //  It's also possible to use "==" to mark a property.  We remove that
586            //  here so that we can actually find the property file, then.
587            //
588            if( propertyValue.startsWith("=") )
589            {
590                propertyValue = propertyValue.substring(1);
591            }
592
593            try
594            {
595                m_session.addMessage( "Info." + property, "The system property '" + property + "' is set to: "
596                        + propertyValue + "." );
597
598                // Prepend a file: prefix if not there already
599                if ( !propertyValue.startsWith( "file:" ) )
600                {
601                  propertyValue = "file:" + propertyValue;
602                }
603                final URL url = new URL( propertyValue );
604                final File file = new File( url.getPath() );
605                if ( file.exists() )
606                {
607                    m_session.addMessage( "Info." + property, "File '" + propertyValue + "' exists in the filesystem." );
608                    return file;
609                }
610            }
611            catch( final MalformedURLException e )
612            {
613                // Swallow exception because we can't find it anyway
614            }
615            m_session.addMessage( "Error." + property, "File '" + propertyValue
616                    + "' doesn't seem to exist. This might be a problem." );
617            return null;
618        }
619        catch( final SecurityException e )
620        {
621            m_session.addMessage( "Error." + property, "We could not read system property '" + property
622                    + "'. This is probably because you are running with a security manager." );
623            return null;
624        }
625    }

627    /**
628     * Verfies the Java security policy configuration. The configuration is
629     * valid if value of the local policy (at <code>WEB-INF/jspwiki.policy</code>
630     * resolves to an existing file, and the policy file contained therein
631     * represents a valid policy.
632     */
633    @SuppressWarnings("unchecked")
634    protected void verifyPolicy() {
635        // Look up the policy file and set the status text.
636        final URL policyURL = m_engine.findConfigFile( AuthorizationManager.DEFAULT_POLICY );
637        String path = policyURL.getPath();
638        if ( path.startsWith("file:") ) {
639            path = path.substring( 5 );
640        }
641        final File policyFile = new File( path );
642
643        // Next, verify the policy
644        try {
645            // Get the file
646            final PolicyReader policy = new PolicyReader( policyFile );
647            m_session.addMessage( INFO_POLICY, "The security policy '" + policy.getFile() + "' exists." );
648
649            // See if there is a keystore that's valid
650            final KeyStore ks = policy.getKeyStore();
651            if ( ks == null ) {
652                m_session.addMessage( WARNING_POLICY,
653                    "Policy file does not have a keystore... at least not one that we can locate. If your policy file " +
654                    "does not contain any 'signedBy' blocks, this is probably ok." );
655            } else {
656                m_session.addMessage( INFO_POLICY,
657                    "The security policy specifies a keystore, and we were able to locate it in the filesystem." );
658            }
659
660            // Verify the file
661            policy.read();
662            final List<Exception> errors = policy.getMessages();
663            if ( errors.size() > 0 ) {
664                for( final Exception e : errors ) {
665                    m_session.addMessage( ERROR_POLICY, e.getMessage() );
666                }
667            } else {
668                m_session.addMessage( INFO_POLICY, "The security policy looks fine." );
669                m_isSecurityPolicyConfigured = true;
670            }
671
672            // Stash the unique principals mentioned in the file,
673            // plus our standard roles.
674            final Set<Principal> principals = new LinkedHashSet<>();
675            principals.add( Role.ALL );
676            principals.add( Role.ANONYMOUS );
677            principals.add( Role.ASSERTED );
678            principals.add( Role.AUTHENTICATED );
679            final ProtectionDomain[] domains = policy.getProtectionDomains();
680            for ( final ProtectionDomain domain : domains ) {
681                for( final Principal principal : domain.getPrincipals() ) {
682                    principals.add( principal );
683                }
684            }
685            m_policyPrincipals = principals.toArray( new Principal[principals.size()] );
686        } catch( final IOException e ) {
687            m_session.addMessage( ERROR_POLICY, e.getMessage() );
688        }
689    }

691    /**
692     * Verifies that a particular Principal possesses a Permission, as defined
693     * in the security policy file.
694     * @param principal the principal
695     * @param permission the permission
696     * @return the result, based on consultation with the active Java security
697     *         policy
698     */
699    protected boolean verifyStaticPermission( final Principal principal, final Permission permission )
700    {
701        final Subject subject = new Subject();
702        subject.getPrincipals().add( principal );
703        final boolean allowedByGlobalPolicy = (Boolean)
704            Subject.doAsPrivileged( subject, ( PrivilegedAction< Object > )() -> {
705                try {
706                    AccessController.checkPermission( permission );
707                    return Boolean.TRUE;
708                } catch( final AccessControlException e ) {
709                    return Boolean.FALSE;
710                }
711            }, null );
712
713        if ( allowedByGlobalPolicy )
714        {
715            return true;
716        }
717
718        // Check local policy
719        final Principal[] principals = new Principal[]{ principal };
720        return m_engine.getManager( AuthorizationManager.class ).allowedByLocalPolicy( principals, permission );
721    }

723    /**
724     * Verifies that the user datbase was initialized properly, and that
725     * user add and delete operations work as they should.
726     */
727    protected void verifyUserDatabase()
728    {
729        final UserDatabase db = m_engine.getManager( UserManager.class ).getUserDatabase();
730
731        // Check for obvious error conditions
732        if ( db == null )
733        {
734            m_session.addMessage( ERROR_DB, "UserDatabase is null; JSPWiki could not " +
735                    "initialize it. Check the error logs." );
736            return;
737        }
738
739        if ( db instanceof DummyUserDatabase )
740        {
741            m_session.addMessage( ERROR_DB, "UserDatabase is DummyUserDatabase; JSPWiki " +
742                    "may not have been able to initialize the database you supplied in " +
743                    "jspwiki.properties, or you left the 'jspwiki.userdatabase' property " +
744                    "blank. Check the error logs." );
745        }
746
747        // Tell user what class of database this is.
748        m_session.addMessage( INFO_DB, "UserDatabase is of type '" + db.getClass().getName() +
749                                       "'. It appears to be initialized properly." );
750
751        // Now, see how many users we have.
752        final int oldUserCount;
753        try {
754            final Principal[] users = db.getWikiNames();
755            oldUserCount = users.length;
756            m_session.addMessage( INFO_DB, "The user database contains " + oldUserCount + " users." );
757        } catch( final WikiSecurityException e ) {
758            m_session.addMessage( ERROR_DB, "Could not obtain a list of current users: " + e.getMessage() );
759            return;
760        }
761
762        // Try adding a bogus user with random name
763        final String loginName = "TestUser" + System.currentTimeMillis();
764        try {
765            final UserProfile profile = db.newProfile();
766            profile.setEmail( "jspwiki.tests@mailinator.com" );
767            profile.setLoginName( loginName );
768            profile.setFullname( "FullName" + loginName );
769            profile.setPassword( "password" );
770            db.save( profile );
771
772            // Make sure the profile saved successfully
773            if( db.getWikiNames().length == oldUserCount ) {
774                m_session.addMessage( ERROR_DB, "Could not add a test user to the database." );
775                return;
776            }
777            m_session.addMessage( INFO_DB, "The user database allows new users to be created, as it should." );
778        } catch( final WikiSecurityException e ) {
779            m_session.addMessage( ERROR_DB, "Could not add a test user to the database: " + e.getMessage() );
780            return;
781        }
782
783        // Now delete the profile; should be back to old count
784        try {
785            db.deleteByLoginName( loginName );
786            if( db.getWikiNames().length != oldUserCount ) {
787                m_session.addMessage( ERROR_DB, "Could not delete a test user from the database." );
788                return;
789            }
790            m_session.addMessage( INFO_DB, "The user database allows users to be deleted, as it should." );
791        } catch( final WikiSecurityException e ) {
792            m_session.addMessage( ERROR_DB, "Could not delete a test user to the database: " + e.getMessage() );
793            return;
794        }
795
796        m_session.addMessage( INFO_DB, "The user database configuration looks fine." );
797    }
