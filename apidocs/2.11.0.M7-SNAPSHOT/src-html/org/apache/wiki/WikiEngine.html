083/**
084 *  Provides Wiki services to the JSP page.
085 *
086 *  <P>
087 *  This is the main interface through which everything should go.
088 *
089 *  <P>

091 *  by using the WikiEngine.getInstance() method.  Never create a new
092 *  WikiEngine() from scratch, unless you're writing tests.

095 *  you should always get it using the WikiEngine.getInstance() method.

311    // FIXME: Potential make-things-easier thingy here: no need to fetch the wikiengine anymore
312    //        Wiki.jsp.jspInit() [really old code]; it's probably even faster to fetch it
313    //        using this method every time than go to pageContext.getAttribute().
314
315    public static synchronized WikiEngine getInstance( final ServletContext context, Properties props ) throws InternalWikiException {
316        WikiEngine engine = ( WikiEngine )context.getAttribute( ATTR_WIKIENGINE );
317
318        if( engine == null ) {
319            final String appid = Integer.toString(context.hashCode()); //FIXME: Kludge, use real type.
320            context.log(" Assigning new engine to "+appid);
321            try {
322                if( props == null ) {
323                    props = PropertyReader.loadWebAppProps( context );
324                }
325
326                engine = new WikiEngine( context, appid, props );
327                context.setAttribute( ATTR_WIKIENGINE, engine );
328            } catch( final Exception e ) {
329                context.log( "ERROR: Failed to create a Wiki engine: "+e.getMessage() );
330                log.error( "ERROR: Failed to create a Wiki engine, stacktrace follows " , e);
331                throw new InternalWikiException( "No wiki engine, check logs." , e);
332            }
333
334        }
335
336        return engine;
337    }

351    /**
352     *  Instantiate using this method when you're running as a servlet and
353     *  WikiEngine will figure out where to look for the property
354     *  file.
355     *  Do not use this method - use WikiEngine.getInstance() instead.
356     *
357     *  @param context A ServletContext.
358     *  @param appid   An Application ID.  This application is an unique random string which
359     *                 is used to recognize this WikiEngine.
360     *  @param props   The WikiEngine configuration.
361     *  @throws WikiException If the WikiEngine construction fails.
362     */
363    protected WikiEngine( final ServletContext context, final String appid, final Properties props ) throws WikiException {
364        m_servletContext = context;
365        m_appid          = appid;
366
367        // Stash the WikiEngine in the servlet context
368        if ( context != null ) {
369            context.setAttribute( ATTR_WIKIENGINE,  this );
370            m_rootPath = context.getRealPath("/");
371        }
372
373        try {
374            //  Note: May be null, if JSPWiki has been deployed in a WAR file.
375            initialize( props );
376            log.info( "Root path for this Wiki is: '" + m_rootPath + "'" );
377        } catch( final Exception e ) {
378            final String msg = Release.APPNAME+": Unable to load and setup properties from jspwiki.properties. "+e.getMessage();
379            if ( context != null ) {
380                context.log( msg );
381            }
382            throw new WikiException( msg, e );
383        }
384    }

386    /**
387     *  Does all the real initialization.
388     */
389    private void initialize( final Properties props ) throws WikiException {
390        m_startTime  = new Date();
391        m_properties = props;
392
393        //
394        //  Initialize log4j.  However, make sure that we don't initialize it multiple times.
395        //  By default we load the log4j config statements from jspwiki.properties, unless
396        //  the property jspwiki.use.external.logconfig=true, in that case we let log4j figure out the
397        //  logging configuration.
398        //
399        if( !c_configured ) {
400            final String useExternalLogConfig = TextUtil.getStringProperty( props,"jspwiki.use.external.logconfig","false" );
401            if( useExternalLogConfig == null || useExternalLogConfig.equals( "false" ) ) {
402                PropertyConfigurator.configure( props );
403            }
404            c_configured = true;
405        }
406
407        log.info( "*******************************************" );
408        log.info( Release.APPNAME + " " + Release.getVersionString() + " starting. Whee!" );
409
410        fireEvent( WikiEngineEvent.INITIALIZING ); // begin initialization
411
412        log.debug("Java version: "+System.getProperty("java.runtime.version"));
413        log.debug("Java vendor: "+System.getProperty("java.vm.vendor"));
414        log.debug("OS: "+System.getProperty("os.name")+" "+System.getProperty("os.version")+" "+System.getProperty("os.arch"));
415        log.debug("Default server locale: "+Locale.getDefault());
416        log.debug("Default server timezone: "+TimeZone.getDefault().getDisplayName(true, TimeZone.LONG));
417
418        if( m_servletContext != null )
419        {
420            log.info("Servlet container: "+m_servletContext.getServerInfo() );
421            if( m_servletContext.getMajorVersion() < 2 ||
422                (m_servletContext.getMajorVersion() == 2 && m_servletContext.getMinorVersion() < 4) )
423            {
424                throw new InternalWikiException("I require a container which supports at least version 2.4 of Servlet specification");
425            }
426        }
427
428        log.debug("Configuring WikiEngine...");
429
430        //  Initializes the CommandResolver
431        m_commandResolver  = new CommandResolver( this, props );
432
433        //
434        //  Create and find the default working directory.
435        //
436        m_workDir = TextUtil.getStringProperty( props, PROP_WORKDIR, null );
437
438        if( m_workDir == null ) {
439            m_workDir = System.getProperty("java.io.tmpdir", ".");
440            m_workDir += File.separator+Release.APPNAME+"-"+m_appid;
441        }
442
443        try {
444            final File f = new File( m_workDir );
445            f.mkdirs();
446
447            //
448            //  A bunch of sanity checks
449            //
450            if( !f.exists() ) {
451                throw new WikiException("Work directory does not exist: "+m_workDir);
452            }
453            if( !f.canRead() ) {
454                throw new WikiException("No permission to read work directory: "+m_workDir);
455            }
456            if( !f.canWrite() ) {
457                throw new WikiException("No permission to write to work directory: "+m_workDir);
458            }
459            if( !f.isDirectory() ) {
460                throw new WikiException("jspwiki.workDir does not point to a directory: "+m_workDir);
461            }
462        } catch( final SecurityException e ) {
463            log.fatal( "Unable to find or create the working directory: "+m_workDir, e );
464            throw new IllegalArgumentException( "Unable to find or create the working dir: " + m_workDir, e );
465        }
466
467        log.info( "JSPWiki working directory is '" + m_workDir + "'" );
468
469        m_saveUserInfo   = TextUtil.getBooleanProperty( props, PROP_STOREUSERNAME, m_saveUserInfo );
470        m_useUTF8        = StandardCharsets.UTF_8.name().equals( TextUtil.getStringProperty( props, PROP_ENCODING, StandardCharsets.ISO_8859_1.name() ) );
471        m_templateDir    = TextUtil.getStringProperty( props, PROP_TEMPLATEDIR, "default" );
472        enforceValidTemplateDirectory();
473        m_frontPage      = TextUtil.getStringProperty( props, PROP_FRONTPAGE,   "Main" );
474
475        //
476        //  Initialize the important modules.  Any exception thrown by the
477        //  managers means that we will not start up.
478        //
479
480        // FIXME: This part of the code is getting unwieldy.  We must think
481        //        of a better way to do the startup-sequence.
482        try {
483            final Class< ? > urlclass = ClassUtil.findClass( "org.apache.wiki.url",
484                                                             TextUtil.getStringProperty( props, PROP_URLCONSTRUCTOR, "DefaultURLConstructor" ) );
485            m_urlConstructor = ( URLConstructor ) urlclass.getDeclaredConstructor().newInstance();
486            m_urlConstructor.initialize( this, props );
487
488            m_pageManager           = ClassUtil.getMappedObject( PageManager.class.getName(), this, props );
489            m_pluginManager         = ClassUtil.getMappedObject( PluginManager.class.getName(), this, props );
490            m_differenceManager     = ClassUtil.getMappedObject( DifferenceManager.class.getName(), this, props );
491            m_attachmentManager     = ClassUtil.getMappedObject( AttachmentManager.class.getName(), this, props );
492            m_variableManager       = ClassUtil.getMappedObject( VariableManager.class.getName(), props );
493            m_renderingManager      = ClassUtil.getMappedObject( RenderingManager.class.getName() );
494            m_searchManager         = ClassUtil.getMappedObject( SearchManager.class.getName(), this, props );
495            m_authenticationManager = ClassUtil.getMappedObject( AuthenticationManager.class.getName() );
496            m_authorizationManager  = ClassUtil.getMappedObject( AuthorizationManager.class.getName() );
497            m_userManager           = ClassUtil.getMappedObject( UserManager.class.getName() );
498            m_groupManager          = ClassUtil.getMappedObject( GroupManager.class.getName() );
499            m_editorManager         = ClassUtil.getMappedObject( EditorManager.class.getName(), this );
500            m_editorManager.initialize( props );
501
502            m_progressManager   = new ProgressManager();
503
504            // Initialize the authentication, authorization, user and acl managers
505            m_authenticationManager.initialize( this, props );
506            m_authorizationManager.initialize( this, props );
507            m_userManager.initialize( this, props );
508            m_groupManager.initialize( this, props );
509            m_aclManager = getAclManager();
510
511            // Start the Workflow manager
512            m_workflowMgr = ClassUtil.getMappedObject(WorkflowManager.class.getName());
513            m_workflowMgr.initialize(this, props);
514            m_tasksManager = ClassUtil.getMappedObject(TasksManager.class.getName());
515
516            m_internationalizationManager = ClassUtil.getMappedObject(InternationalizationManager.class.getName(),this);
517            m_templateManager = ClassUtil.getMappedObject(TemplateManager.class.getName(), this, props );
518
519            // Since we want to use a page filters initilize() method
520            // as a engine startup listener where we can initialize global event listeners,
521            // it must be called lastly, so that all object references in the engine
522            // are availabe to the initialize() method
523            m_filterManager = ClassUtil.getMappedObject(FilterManager.class.getName(), this, props );
524
525            m_adminBeanManager = ClassUtil.getMappedObject(AdminBeanManager.class.getName(),this);
526
527            // RenderingManager depends on FilterManager events.
528            m_renderingManager.initialize( this, props );
529
530            //
531            //  ReferenceManager has the side effect of loading all
532            //  pages.  Therefore after this point, all page attributes
533            //  are available.
534            //
535            //  initReferenceManager is indirectly using m_filterManager, therefore
536            //  it has to be called after it was initialized.
537            //
538            initReferenceManager();
539
540            //
541            //  Hook the different manager routines into the system.
542            //
543            m_filterManager.addPageFilter(m_referenceManager, -1001 );
544            m_filterManager.addPageFilter(m_searchManager, -1002 );
545        } catch( final RuntimeException e ) {
546            // RuntimeExceptions may occur here, even if they shouldn't.
547            log.fatal( "Failed to start managers.", e );
548            throw new WikiException( "Failed to start managers: " + e.getMessage(), e );
549        } catch( final ClassNotFoundException e ) {
550            log.fatal( "JSPWiki could not start, URLConstructor was not found: " + e.getMessage(), e );
551            throw new WikiException( e.getMessage(), e );
552        } catch( final InstantiationException e ) {
553            log.fatal( "JSPWiki could not start, URLConstructor could not be instantiated: " + e.getMessage(), e );
554            throw new WikiException( e.getMessage(), e );
555        } catch( final IllegalAccessException e ) {
556            log.fatal( "JSPWiki could not start, URLConstructor cannot be accessed: " + e.getMessage(), e );
557            throw new WikiException( e.getMessage(), e );
558        } catch( final Exception e ) {
559            // Final catch-all for everything
560            log.fatal( "JSPWiki could not start, due to an unknown exception when starting.",e );
561            throw new WikiException( "Failed to start. Caused by: " + e.getMessage() +
562                                     "; please check log files for better information.", e );
563        }
564
565        //
566        //  Initialize the good-to-have-but-not-fatal modules.
567        //
568        try {
569            if( TextUtil.getBooleanProperty( props, RSSGenerator.PROP_GENERATE_RSS,false ) ) {
570                m_rssGenerator = ClassUtil.getMappedObject( RSSGenerator.class.getName(), this, props );
571            }
572
573            m_pageRenamer = ClassUtil.getMappedObject( PageRenamer.class.getName(), this, props );
574        } catch( final Exception e ) {
575            log.error( "Unable to start RSS generator - JSPWiki will still work, but there will be no RSS feed.", e );
576        }
577
578        // Start the RSS generator & generator thread
579        if( m_rssGenerator != null ) {
580            m_rssFile = TextUtil.getStringProperty( props, RSSGenerator.PROP_RSSFILE, "rss.rdf" );
581            final File rssFile;
582            if( m_rssFile.startsWith( File.separator ) ) { // honor absolute pathnames:
583                rssFile = new File(m_rssFile );
584            } else { // relative path names are anchored from the webapp root path:
585                rssFile = new File( getRootPath(), m_rssFile );
586            }
587            final int rssInterval = TextUtil.getIntegerProperty( props, RSSGenerator.PROP_INTERVAL, 3600 );
588            final RSSThread rssThread = new RSSThread( this, rssFile, rssInterval );
589            rssThread.start();
590        }
591
592        fireEvent( WikiEngineEvent.INITIALIZED ); // initialization complete
593
594        log.info("WikiEngine configured.");
595        m_isConfigured = true;
596    }

598    /**
599     * Checks if the template directory specified in the wiki's properties actually exists. If it doesn't, then {@code m_templateDir} is
600     * set to {@link #DEFAULT_TEMPLATE_NAME}.
601     * <p>
602     * This checks the existence of the <tt>ViewTemplate.jsp</tt> file, which exists in every template using {@code m_servletContext.getRealPath("/")}.
603     * <p>
604     * {@code m_servletContext.getRealPath("/")} can return {@code null} on certain servers/conditions (f.ex, packed wars), an extra check
605     * against {@code m_servletContext.getResource} is made.
606     */
607    void enforceValidTemplateDirectory() {
608        if( m_servletContext != null ) {
609            final String viewTemplate = "templates" + File.separator + getTemplateDir() + File.separator + "ViewTemplate.jsp";
610            boolean exists = new File( m_servletContext.getRealPath("/") + viewTemplate ).exists();
611            if( !exists ) {
612                try {
613                    final URL url = m_servletContext.getResource( viewTemplate );
614                    exists = url != null && StringUtils.isNotEmpty( url.getFile() );
615                } catch( final MalformedURLException e ) {
616                    log.warn( "template not found with viewTemplate " + viewTemplate );
617                }
618            }
619            if( !exists ) {
620                log.warn( getTemplateDir() + " template not found, updating WikiEngine's default template to " + DEFAULT_TEMPLATE_NAME );
621                m_templateDir = DEFAULT_TEMPLATE_NAME;
622            }
623        }
624    }

627     *  Initializes the reference manager. Scans all existing WikiPages for
628     *  internal links and adds them to the ReferenceManager object.
629     *
630     *  @throws WikiException If the reference manager initialization fails.
631     */
632    public void initReferenceManager() throws WikiException {
633        try {
634            final ArrayList<WikiPage> pages = new ArrayList<>();
635            pages.addAll( m_pageManager.getAllPages() );
636            pages.addAll( m_attachmentManager.getAllAttachments() );
637
638            // Build a new manager with default key lists.
639            if( m_referenceManager == null ) {
640                m_referenceManager = ClassUtil.getMappedObject(ReferenceManager.class.getName(), this );
641                m_referenceManager.initialize( pages );
642            }
643
644        } catch( final ProviderException e ) {
645            log.fatal("PageProvider is unable to list pages: ", e);
646        } catch( final ReflectiveOperationException | IllegalArgumentException e ) {
647            throw new WikiException( "Could not instantiate ReferenceManager: " + e.getMessage(), e );
648        }
649    }
