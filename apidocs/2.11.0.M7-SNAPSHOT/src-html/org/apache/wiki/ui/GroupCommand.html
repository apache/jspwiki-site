019package org.apache.wiki.ui;
020
021import org.apache.wiki.api.core.Command;
022import org.apache.wiki.auth.GroupPrincipal;
023import org.apache.wiki.auth.permissions.GroupPermission;
024
025import java.security.Permission;
026
027/**
028 * <p>Defines Commands for viewing, editing and deleting wiki groups. GroupCommands can be combined with GroupPrincipals to produce
029 * targeted Commands.</p>
030 * <p>This class is not <code>final</code>; it may be extended in the future.</p>
031 * @since 2.4.22
032 */
033public final class GroupCommand extends AbstractCommand {
034
035    /** GroupCommand for deleting a group. */
036    public static final Command DELETE_GROUP = new GroupCommand( "deleteGroup",
037                                                       "%uDeleteGroup.jsp?group=%n",
038                                                  null,
039                                                          null,
040                                                                 GroupPermission.DELETE_ACTION );
041
042    /** GroupCommand for editing a group. */
043       public static final Command EDIT_GROUP = new GroupCommand( "editGroup",
044                                                        "%uEditGroup.jsp?group=%n",
045                                                   "EditGroupContent.jsp",
046                                                            null,
047                                                                 GroupPermission.EDIT_ACTION );
048
049       /** GroupCommand for viewing a group. */
050    public static final Command VIEW_GROUP = new GroupCommand( "viewGroup",
051                                                     "%uGroup.jsp?group=%n",
052                                                "GroupContent.jsp",
053                                                        null,
054                                                               GroupPermission.VIEW_ACTION );
055
056    private final String m_action;
057    
058    private final Permission m_permission;
059    
060    /**
061     * Constructs a new Command with a specified wiki context, URL pattern, type, and content template. The WikiPage for this command is
062     * initialized to <code>null</code>.
063     *
064     * @param requestContext the request context
065     * @param urlPattern the URL pattern
066     * @param target the target of this command (a GroupPrincipal representing a Group); may be <code>null</code>
067     * @param action the action used to construct a suitable GroupPermission
068     * @param contentTemplate the content template; may be <code>null</code>
069     * @throws IllegalArgumentException if the request content, URL pattern, or type is <code>null</code>
070     */
071    private GroupCommand( final String requestContext,
072                          final String urlPattern,
073                          final String contentTemplate,
074                          final GroupPrincipal target,
075                          final String action ) {
076        super( requestContext, urlPattern, contentTemplate, target );
077        m_action = action;
078        if ( target == null || m_action == null ) {
079            m_permission = null;
080        } else {
081            m_permission = new GroupPermission( target.getName(), action );
082        }
083    }
084
085    /**
086     * Creates and returns a targeted Command by combining a GroupPrincipal with this Command. The supplied <code>target</code> object
087     * must be non-<code>null</code> and of type GroupPrincipal. If the target is not of the correct type, this method throws an
088     * {@link IllegalArgumentException}.
089     *
090     * @param target the GroupPrincipal to combine into the current Command
091     * @return the new, targeted command
092     */
093    public Command targetedCommand( final Object target ) {
094        if( !( target instanceof GroupPrincipal ) ) {
095            throw new IllegalArgumentException( "Target must non-null and of type GroupPrincipal." );
096        }
097        return new GroupCommand( getRequestContext(), getURLPattern(), getContentTemplate(), ( GroupPrincipal )target, m_action );
098    }
099    
100    /**
101     * Returns the name of the command, which will either be the target (if specified), or the "friendly name" for the JSP.
102     *
103     * @return the name
104     * @see org.apache.wiki.api.core.Command#getName()
105     */
106    public String getName() {
107        final Object target = getTarget();
108        if ( target == null ) {
109            return getJSPFriendlyName();
110        }
111        return ( ( GroupPrincipal ) target ).getName();
112    }
113
114    /**
115     * Returns the permission required to execute this GroupCommand.
116     *
117     * @return the permission
118     * @see org.apache.wiki.api.core.Command#requiredPermission()
119     */
120    public Permission requiredPermission() {
121        return m_permission;
122    }
123
124}
