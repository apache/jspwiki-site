019package org.apache.wiki.filters;
020
021import org.apache.log4j.Logger;
022import org.apache.wiki.WikiContext;
023import org.apache.wiki.WikiEngine;
024import org.apache.wiki.api.core.Engine;
025import org.apache.wiki.api.filters.BasicPageFilter;
026import org.apache.xmlrpc.AsyncCallback;
027import org.apache.xmlrpc.XmlRpcClient;
028
029import java.net.MalformedURLException;
030import java.net.URL;
031import java.util.Hashtable;
032import java.util.Properties;
033import java.util.Vector;
034
035/**
036 *  A very dumb class that pings weblogs.com on each save.  INTERNAL USE ONLY SO FAR! Look, but don't use as-is.
037 */
038// FIXME: Needs to figure out when only weblogs have been saved.
039// FIXME: rpc endpoint must be configurable
040// FIXME: Should really be settable per-page.
041// FIXME: Weblog name has been set to stone
042public class PingWeblogsComFilter extends BasicPageFilter {
043
044    private static final Logger log = Logger.getLogger( PingWeblogsComFilter.class );
045
046    private String m_pingURL;
047
048    /**
049     *  The property name for the URL to ping.  Value is <tt>{@value}</tt>.
050     */
051    public static final String PROP_PINGURL = "pingurl";
052
053    /**
054     *  {@inheritDoc}
055     */
056    @Override public void initialize( final WikiEngine engine, final Properties props ) {
057        m_pingURL = props.getProperty( PROP_PINGURL, "http://rpc.weblogs.com/RPC2" );
058    }
059
060    /**
061     *  {@inheritDoc}
062     */
063    @Override public void postSave( final WikiContext context, final String pagecontent ) {
064        String blogName = context.getPage().getName();
065        final Engine engine   = context.getEngine();
066
067        final int blogentryTxt = blogName.indexOf("_blogentry_");
068        if( blogentryTxt == -1 ) {
069            return; // This is not a weblog entry.
070        }
071        
072        blogName = blogName.substring( 0, blogentryTxt );
073
074        if( blogName.equals( engine.getFrontPage() ) ) {
075            blogName = null;
076        }
077
078        try {
079            final XmlRpcClient xmlrpc = new XmlRpcClient(m_pingURL);
080            final Vector< String > params = new Vector<>();
081            params.addElement( "The Butt Ugly Weblog" ); // FIXME: Must be settable
082            params.addElement( engine.getURL( WikiContext.VIEW, blogName, null ) );
083
084            if( log.isDebugEnabled() ) {
085                log.debug( "Pinging weblogs.com with URL: " + engine.getURL( WikiContext.VIEW, blogName, null ) );
086            }
087
088            xmlrpc.executeAsync("weblogUpdates.ping", params, 
089                                new AsyncCallback() {
090                                    @Override public void handleError( final Exception ex, final URL url, final String method ) {
091                                        log.error( "Unable to execute weblogs.com ping to URL: " + url.toString(), ex );
092                                    }
093
094                                    @Override public void handleResult( final Object result, final URL url, final String method ) {
095                                        @SuppressWarnings("unchecked")
096                                        final Hashtable< String, Object > res = (Hashtable < String, Object > ) result;
097
098                                        final Boolean flerror = (Boolean)res.get("flerror");
099                                        final String  msg     = (String)res.get("message");
100
101                                        if( flerror ) {
102                                            log.error("Failed to ping: "+msg);
103                                        }
104
105                                        log.info("Weblogs.com has been pinged.");
106                                    }
107                                }
108                                );
109        } catch( final MalformedURLException e ) {
110            log.error("Malformed URL",e);
111        }
112    }
113}
