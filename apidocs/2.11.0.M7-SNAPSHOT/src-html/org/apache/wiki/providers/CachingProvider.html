048/**
049 *  Provides a caching page provider.  This class rests on top of a real provider class and provides a cache to speed things up.  Only
050 *  if the cache copy of the page text has expired, we fetch it from the provider.
051 *  <p>
052 *  This class does not detect if someone has modified the page externally, not through JSPWiki routines.
053 *  <p>
054 *  Heavily based on ideas by Chris Brooking.
055 *  <p>
056 *  Since 2.10 uses the Ehcache library.
057 *
058 *  @since 1.6.4
059 */
060// FIXME: Synchronization is a bit inconsistent in places.
061// FIXME: A part of the stuff is now redundant, since we could easily use the text cache for a lot of things.  RefactorMe.
062public class CachingProvider implements PageProvider {

094    // The default settings of the caches, if you want something else, provide an "ehcache.xml" file
095    // Please note that JSPWiki ships with a default "ehcache.xml" in the classpath
096    public static final int   DEFAULT_CACHECAPACITY   = 1000; // Good most wikis
097    public static final int   DEFAULT_CACHETIMETOLIVESECONDS = 24*3600;
098    public static final int   DEFAULT_CACHETIMETOIDLESECONDS = 24*3600;

100    /**
101     *  {@inheritDoc}
102     */
103    @Override
104    public void initialize( final Engine engine, final Properties properties ) throws NoRequiredPropertyException, IOException {
105        log.debug("Initing CachingProvider");
106
107        // engine is used for getting the search engine
108        m_engine = engine;
109
110        final String cacheName = engine.getApplicationName() + "." + CACHE_NAME;
111        if (m_cacheManager.cacheExists(cacheName)) {
112            m_cache = m_cacheManager.getCache(cacheName);
113        } else {
114            log.info("cache with name " + cacheName +  " not found in ehcache.xml, creating it with defaults.");
115            m_cache = new Cache(cacheName, DEFAULT_CACHECAPACITY, false, false, DEFAULT_CACHETIMETOLIVESECONDS, DEFAULT_CACHETIMETOIDLESECONDS);
116            m_cacheManager.addCache(m_cache);
117        }
118
119        final String textCacheName = engine.getApplicationName() + "." + TEXTCACHE_NAME;
120        if (m_cacheManager.cacheExists(textCacheName)) {
121            m_textCache= m_cacheManager.getCache(textCacheName);
122        } else {
123            log.info("cache with name " + textCacheName +  " not found in ehcache.xml, creating it with defaults.");
124            m_textCache = new Cache(textCacheName, DEFAULT_CACHECAPACITY, false, false, DEFAULT_CACHETIMETOLIVESECONDS, DEFAULT_CACHETIMETOIDLESECONDS);
125            m_cacheManager.addCache(m_textCache);
126        }
127
128        final String historyCacheName = engine.getApplicationName() + "." + HISTORYCACHE_NAME;
129        if (m_cacheManager.cacheExists(historyCacheName)) {
130            m_historyCache= m_cacheManager.getCache(historyCacheName);
131        } else {
132            log.info("cache with name " + historyCacheName +  " not found in ehcache.xml, creating it with defaults.");
133            m_historyCache = new Cache(historyCacheName, DEFAULT_CACHECAPACITY, false, false, DEFAULT_CACHETIMETOLIVESECONDS, DEFAULT_CACHETIMETOIDLESECONDS);
134            m_cacheManager.addCache(m_historyCache);
135        }
136
137        //
138        // m_cache.getCacheEventNotificationService().registerListener(new CacheItemCollector());
139
140        //
141        //  Find and initialize real provider.
142        //
143        final String classname;
144        try {
145            classname = TextUtil.getRequiredProperty( properties, PageManager.PROP_PAGEPROVIDER );
146        } catch( final NoSuchElementException e ) {
147            throw new NoRequiredPropertyException( e.getMessage(), PageManager.PROP_PAGEPROVIDER );
148        }
149
150        try {
151            final Class< ? > providerclass = ClassUtil.findClass( "org.apache.wiki.providers", classname );
152            m_provider = ( PageProvider )providerclass.newInstance();
153
154            log.debug( "Initializing real provider class " + m_provider );
155            m_provider.initialize( engine, properties );
156        } catch( final ClassNotFoundException e ) {
157            log.error( "Unable to locate provider class " + classname, e );
158            throw new IllegalArgumentException( "no provider class", e );
159        } catch( final InstantiationException e ) {
160            log.error( "Unable to create provider class " + classname, e );
161            throw new IllegalArgumentException( "faulty provider class", e );
162        } catch( final IllegalAccessException e ) {
163            log.error( "Illegal access to provider class " + classname, e );
164            throw new IllegalArgumentException( "illegal provider class", e );
165        }
166    }

168    private Page getPageInfoFromCache( final String name) throws ProviderException {
169        // Sanity check; seems to occur sometimes
170        if( name == null ) {
171            return null;
172        }
173
174        final Element cacheElement = m_cache.get( name );
175        if( cacheElement == null ) {
176            final Page refreshed = m_provider.getPageInfo( name, PageProvider.LATEST_VERSION );
177            if( refreshed != null ) {
178                m_cache.put( new Element( name, refreshed ) );
179                return refreshed;
180            } else {
181                // page does not exist anywhere
182                return null;
183            }
184        }
185        return ( Page )cacheElement.getObjectValue();
186    }

189    /**
190     *  {@inheritDoc}
191     */
192    @Override
193    public boolean pageExists( final String pageName, final int version ) {
194        if( pageName == null ) {
195            return false;
196        }
197
198        final Page p;
199        try {
200            p = getPageInfoFromCache( pageName );
201        } catch( final ProviderException e ) {
202            log.info( "Provider failed while trying to check if page exists: " + pageName );
203            return false;
204        }
205
206        if( p != null ) {
207            final int latestVersion = p.getVersion();
208            if( version == latestVersion || version == LATEST_VERSION ) {
209                return true;
210            }
211
212            return m_provider.pageExists( pageName, version );
213        }
214
215        try {
216            return getPageInfo( pageName, version ) != null;
217        } catch( final ProviderException e ) {
218        }
219
220        return false;
221    }

223    /**
224     *  {@inheritDoc}
225     */
226    @Override
227    public boolean pageExists( final String pageName ) {
228        if( pageName == null ) {
229            return false;
230        }
231
232        final Page p;
233        try {
234            p = getPageInfoFromCache( pageName );
235        } catch( final ProviderException e ) {
236            log.info( "Provider failed while trying to check if page exists: " + pageName );
237            return false;
238        }
239
240        //  A null item means that the page either does not exist, or has not yet been cached; a non-null means that the page does exist.
241        if( p != null ) {
242            return true;
243        }
244
245        //  If we have a list of all pages in memory, then any page not in the cache must be non-existent.
246        if( m_gotall ) {
247            return false;
248        }
249
250        //  We could add the page to the cache here as well, but in order to understand whether that is a good thing or not we would
251        //  need to analyze the JSPWiki calling patterns extensively.  Presumably it would be a good thing if pageExists() is called
252        //  many times before the first getPageText() is called, and the whole page is cached.
253        return m_provider.pageExists( pageName );
254    }

256    /**
257     *  {@inheritDoc}
258     */
259    @Override
260    public String getPageText( final String pageName, final int version ) throws ProviderException {
261        if( pageName == null ) {
262            return null;
263        }
264
265        final String result;
266        if( version == PageProvider.LATEST_VERSION ) {
267            result = getTextFromCache( pageName );
268        } else {
269            final Page p = getPageInfoFromCache( pageName );
270
271            //  Or is this the latest version fetched by version number?
272            if( p != null && p.getVersion() == version ) {
273                result = getTextFromCache( pageName );
274            } else {
275                result = m_provider.getPageText( pageName, version );
276            }
277        }
278
279        return result;
280    }

283    private String getTextFromCache( final String pageName ) throws ProviderException {
284        if (pageName == null) {
285            return null;
286        }
287
288        final String text;
289        final Element cacheElement = m_textCache.get(pageName);
290        if( cacheElement != null ) {
291            m_cacheHits++;
292            return ( String )cacheElement.getObjectValue();
293        }
294        if( pageExists( pageName ) ) {
295            text = m_provider.getPageText( pageName, PageProvider.LATEST_VERSION );
296            m_textCache.put( new Element( pageName, text ) );
297            m_cacheMisses++;
298            return text;
299        }
300        //page not found (not in cache, not by real provider)
301        return  null;
302    }

304    /**
305     *  {@inheritDoc}
306     */
307    @Override
308    public void putPageText( final Page page, final String text ) throws ProviderException {
309        synchronized( this ) {
310            m_provider.putPageText( page, text );
311            page.setLastModified( new Date() );
312
313            // Refresh caches properly
314            m_cache.remove( page.getName() );
315            m_textCache.remove( page.getName() );
316            m_historyCache.remove( page.getName() );
317
318            getPageInfoFromCache( page.getName() );
319        }
320    }

322    /**
323     *  {@inheritDoc}
324     */
325    @Override
326    public Collection< Page > getAllPages() throws ProviderException {
327        final Collection< Page > all;
328
329        if ( !m_gotall ) {
330            all = m_provider.getAllPages();
331
332            // Make sure that all pages are in the cache.
333            synchronized( this ) {
334                for( final Page p : all ) {
335                    m_cache.put( new Element( p.getName(), p ) );
336                }
337
338                m_gotall = true;
339            }
340        } else {
341            @SuppressWarnings("unchecked") final List< String > keys = m_cache.getKeysWithExpiryCheck();
342            all = new TreeSet<>();
343            for( final String key : keys ) {
344                final Element element = m_cache.get( key );
345                final Page cachedPage = ( Page )element.getObjectValue();
346                if( cachedPage != null ) {
347                    all.add( cachedPage );
348                }
349            }
350        }
351
352        if( all.size() >= m_cache.getCacheConfiguration().getMaxEntriesLocalHeap() ) {
353            log.warn( "seems " + m_cache.getName() + " can't hold all pages from your page repository, " +
354                      "so we're delegating on the underlying provider instead. Please consider increasing " +
355                      "your cache sizes on ehcache.xml to avoid this behaviour" );
356            return m_provider.getAllPages();
357        }
358
359        return all;
360    }

388    //  FIXME: Kludge: make sure that the page is also parsed and it gets all the necessary variables.
389    private void refreshMetadata( final Page page ) {
390        if( page != null && !page.hasMetadata() ) {
391            final RenderingManager mgr = m_engine.getManager( RenderingManager.class );
392            try {
393                final String data = m_provider.getPageText( page.getName(), page.getVersion() );
394                final WikiContext ctx = new WikiContext( m_engine, page );
395                final MarkupParser parser = mgr.getParser( ctx, data );
396
397                parser.parse();
398            } catch( final Exception ex ) {
399                log.debug( "Failed to retrieve variables for wikipage " + page );
400            }
401        }
402    }

404    /**
405     *  {@inheritDoc}
406     */
407    @Override
408    public Page getPageInfo( final String pageName, final int version ) throws ProviderException {
409        final Page page;
410        final Page cached = getPageInfoFromCache( pageName );
411        final int latestcached = ( cached != null ) ? cached.getVersion() : Integer.MIN_VALUE;
412        if( version == PageProvider.LATEST_VERSION || version == latestcached ) {
413            if( cached == null ) {
414                final Page data = m_provider.getPageInfo( pageName, version );
415                if( data != null ) {
416                    m_cache.put( new Element( pageName, data ) );
417                }
418                page = data;
419            } else {
420                page = cached;
421            }
422        } else {
423            // We do not cache old versions.
424            page = m_provider.getPageInfo( pageName, version );
425        }
426        refreshMetadata( page );
427        return page;
428    }

430    /**
431     *  {@inheritDoc}
432     */
433    @SuppressWarnings("unchecked")
434    @Override
435    public List< Page > getVersionHistory( final String pageName) throws ProviderException {
436        final List< Page > history;
437        if( pageName == null ) {
438            return null;
439        }
440        final Element element = m_historyCache.get( pageName );
441        if( element != null ) {
442            m_historyCacheHits++;
443            history = ( List< Page > )element.getObjectValue();
444        } else {
445            history = m_provider.getVersionHistory( pageName );
446            m_historyCache.put( new Element( pageName, history ) );
447            m_historyCacheMisses++;
448        }
449
450        return history;
451    }

467    /**
468     *  {@inheritDoc}
469     */
470    @Override
471    public void deleteVersion( final String pageName, final int version ) throws ProviderException {
472        //  Luckily, this is such a rare operation it is okay to synchronize against the whole thing.
473        synchronized( this ) {
474            final Page cached = getPageInfoFromCache( pageName );
475            final int latestcached = ( cached != null ) ? cached.getVersion() : Integer.MIN_VALUE;
476
477            //  If we have this version cached, remove from cache.
478            if( version == PageProvider.LATEST_VERSION || version == latestcached ) {
479                m_cache.remove( pageName );
480                m_textCache.remove( pageName );
481            }
482
483            m_provider.deleteVersion( pageName, version );
484            m_historyCache.remove( pageName );
485        }
486    }

492    public void deletePage( final String pageName ) throws ProviderException {
493        //  See note in deleteVersion().
494        synchronized( this ) {
495            m_cache.put( new Element( pageName, null ) );
496            m_textCache.put( new Element( pageName, null ) );
497            m_historyCache.put( new Element( pageName, null ) );
498            m_provider.deletePage( pageName );
499        }

502    /**
503     *  {@inheritDoc}
504     */
505    @Override
506    public void movePage( final String from, final String to ) throws ProviderException {
507        m_provider.movePage( from, to );
508
509        synchronized( this ) {
510            // Clear any cached version of the old page and new page
511            m_cache.remove( from );
512            m_textCache.remove( from );
513            m_historyCache.remove( from );
514            log.debug( "Removing to page " + to + " from cache" );
515            m_cache.remove( to );
516            m_textCache.remove( to );
517            m_historyCache.remove( to );
518        }
519    }
