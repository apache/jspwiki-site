019package org.apache.wiki.providers;
020
021import java.util.Collection;
022import java.util.Date;
023import java.util.List;
024
025import org.apache.wiki.WikiPage;
026import org.apache.wiki.WikiProvider;
027import org.apache.wiki.api.exceptions.ProviderException;
028import org.apache.wiki.search.QueryItem;
029
030/**
031 *  Each Wiki page provider should implement this interface.
032 *  <P>
033 *  You can build whatever page providers based on this, just
034 *  leave the unused methods do something useful.
035 *  <P>
036 *  WikiPageProvider uses Strings and ints to refer to pages.  This may
037 *  be a bit odd, since WikiAttachmentProviders all use Attachment
038 *  instead of name/version.  We will perhaps modify these in the
039 *  future.  In the mean time, name/version is quite sufficient.
040 *  <P>
041 *  FIXME: In reality we should have an AbstractWikiPageProvider,
042 *  which would provide intelligent backups for subclasses.
043 *
044 */
045public interface WikiPageProvider
046    extends WikiProvider
047{
048    /**
049     *  Attempts to save the page text for page "page".  Note that the
050     *  provider creates a new version regardless of what the version
051     *  parameter of the WikiPage is.
052     *  
053     *  @param page The WikiPage to save
054     *  @param text The text to save.
055     *  @throws ProviderException If something goes wrong.
056     */
057    void putPageText( WikiPage page, String text )
058        throws ProviderException;
059
060    /**
061     *  Return true, if page exists.
062     *  
063     *  @param page The page name.
064     *  @return true, if the page exists; false otherwise.
065     */
066
067    boolean pageExists( String page );
068
069    /**
070     * Return true, if page with a particular version exists.
071     *
072     * @param page    The page name to check for
073     * @param version The version to check
074     * @return True, if page exists; false otherwise.
075     */
076
077    boolean pageExists(String page, int version);
078
079    /**
080     *  Finds pages based on the query.   Only applicable to providers
081     *  which implement the FastSearch interface.  Otherwise JSPWiki
082     *  will use its internal cache.
083     *  <p>
084     *  This method should really be a part of the FastSearch IF.
085     *  
086     *  @param query An array of QueryItems to match
087     *  @return A Collection of WikiPages.
088     */
089    Collection findPages( QueryItem[] query );
090
091    /**
092     *  Returns info about the page.
093     *  
094     *  @return A filled WikiPage.
095     *  @param page The page name
096     *  @param version The version number
097     *  @throws ProviderException If something goes wrong.
098     */
099    WikiPage getPageInfo( String page, int version )
100        throws ProviderException;
101
102    /**
103     *  Returns all pages.  Each element in the returned
104     *  Collection should be a WikiPage.
105     *  
106     *  @return A collection of WikiPages
107     *  @throws ProviderException If something goes wrong.
108     */
109
110    Collection getAllPages()
111        throws ProviderException;
112
113    /**
114     *  Gets a list of recent changes.
115     *  
116     *  @param date The date to check from
117     *  @return A Collection of WikiPages
118     *  @since 1.6.4
119     */
120
121    Collection getAllChangedSince( Date date );
122
123    /**
124     *  Gets the number of pages.
125     *  
126     *  @return The number of pages in the repository
127     *  @throws ProviderException If something goes wrong
128     *  @since 1.6.4
129     */
130
131    int getPageCount()
132        throws ProviderException;
133
134    /**
135     *  Returns version history.  Each element should be
136     *  a WikiPage.
137     *
138     *  @param page The name of the page to get the history from.
139     *  @return A collection of WikiPages.
140     *  @throws ProviderException If something goes wrong.
141     */
142    List< WikiPage > getVersionHistory( String page ) throws ProviderException;
143
144    /**
145     *  Gets a specific version out of the repository.
146     *
147     *  @param page Name of the page to fetch.
148     *  @param version Version of the page to fetch.
149     *
150     *  @return The content of the page, or null, if the page does not exist.
151     *  @throws ProviderException If something goes wrong.
152     */
153
154    String getPageText( String page, int version )
155        throws ProviderException;
156
157    /**
158     *  Removes a specific version from the repository.  The implementations
159     *  should really do no more security checks, since that is the domain
160     *  of the PageManager.  Just delete it as efficiently as you can.
161     *
162     *  @since 2.0.17.
163     *
164     *  @param pageName Name of the page to be removed.
165     *  @param version  Version of the page to be removed.  May be LATEST_VERSION.
166     *
167     *  @throws ProviderException If the page cannot be removed for some reason.
168     */
169
170    void deleteVersion( String pageName, int version )
171        throws ProviderException;
172
173    /**
174     *  Removes an entire page from the repository.  The implementations
175     *  should really do no more security checks, since that is the domain
176     *  of the PageManager.  Just delete it as efficiently as you can.  You should also
177     *  delete any auxiliary files that belong to this page, IF they were created
178     *  by this provider.
179     *
180     *  <P>The reason why this is named differently from
181     *  deleteVersion() (logically, this method should be an
182     *  overloaded version) is that I want to be absolutely sure I
183     *  don't accidentally use the wrong method.  With overloading
184     *  something like that happens sometimes...
185     *
186     *  @since 2.0.17.
187     *
188     *  @param pageName Name of the page to be removed completely.
189     *
190     *  @throws ProviderException If the page could not be removed for some reason.
191     */
192    void deletePage( String pageName )
193        throws ProviderException;
194
195     
196     /**
197      * Move a page
198      *
199      * @param from  Name of the page to move.
200      * @param to    New name of the page.
201      *
202      * @throws ProviderException If the page could not be moved for some reason.
203      */
204     void movePage(String from, String to)
205         throws ProviderException;
206
207}
