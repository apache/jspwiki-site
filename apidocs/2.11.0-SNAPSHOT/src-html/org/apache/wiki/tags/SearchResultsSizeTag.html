019package org.apache.wiki.tags;
020
021import java.io.IOException;
022import java.util.Collection;
023import javax.servlet.jsp.PageContext;
024
025/**
026 *  Outputs the size of the search results list, if it contains any items.
027 *  Otherwise outputs an empty string.
028 *
029 *  @since 2.0
030 */
031public class SearchResultsSizeTag
032    extends WikiTagBase
033{
034    private static final long serialVersionUID = 0L;
035    
036    public final int doWikiStartTag() throws IOException {
037        final Collection< ? > list = ( Collection< ? > )pageContext.getAttribute( "searchresults", PageContext.REQUEST_SCOPE );
038        if( list != null ) {
039            pageContext.getOut().print( list.size() );
040        }
041
042        return SKIP_BODY;
043    }
044}
