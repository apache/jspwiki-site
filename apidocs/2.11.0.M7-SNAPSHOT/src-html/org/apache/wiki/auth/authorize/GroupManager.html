019package org.apache.wiki.auth.authorize;

021import org.apache.commons.lang3.ArrayUtils;
022import org.apache.wiki.WikiContext;
023import org.apache.wiki.api.core.Session;
024import org.apache.wiki.auth.Authorizer;
025import org.apache.wiki.auth.NoSuchPrincipalException;
026import org.apache.wiki.auth.WikiSecurityException;
027import org.apache.wiki.event.WikiEventListener;
028import org.apache.wiki.event.WikiEventManager;
029import org.apache.wiki.event.WikiSecurityEvent;
030import org.apache.wiki.ui.InputValidator;
031
032import javax.servlet.http.HttpServletRequest;
033import java.security.Principal;

036/**
037 * <p>
038 * Facade class for storing, retrieving and managing wiki groups on behalf of AuthorizationManager, JSPs and other presentation-layer
039 * classes. GroupManager works in collaboration with a back-end {@link GroupDatabase}, which persists groups to permanent storage.
040 * </p>
041 * <p>
042 * <em>Note: prior to JSPWiki 2.4.19, GroupManager was an interface; it is now a concrete, final class. The aspects of GroupManager
043 * which previously extracted group information from storage (e.g., wiki pages) have been refactored into the GroupDatabase interface.</em>
044 * </p>
045 * @since 2.4.19
046 */
047public interface GroupManager extends Authorizer, WikiEventListener {
048
049    /** Key used for adding UI messages to a user's Session. */
050    String MESSAGES_KEY = "group";
051
052    String PROP_GROUPDATABASE = "jspwiki.groupdatabase";
053
054    /**
055     * Returns the Group matching a given name. If the group cannot be found, this method throws a <code>NoSuchPrincipalException</code>.
056     *
057     * @param name the name of the group to find
058     * @return the group
059     * @throws NoSuchPrincipalException if the group cannot be found
060     */
061    Group getGroup( final String name ) throws NoSuchPrincipalException;
062
063    /**
064     * Returns the current external {@link GroupDatabase} in use. This method is guaranteed to return a properly-initialized GroupDatabase,
065     * unless it could not be initialized. In that case, this method throws a {@link org.apache.wiki.api.exceptions.WikiException}. The
066     * GroupDatabase is lazily initialized.
067     *
068     * @throws org.apache.wiki.auth.WikiSecurityException if the GroupDatabase could not be initialized
069     * @return the current GroupDatabase
070     * @since 2.3
071     */
072    GroupDatabase getGroupDatabase() throws WikiSecurityException;
073
074    /**
075     * <p>
076     * Extracts group name and members from passed parameters and populates an existing Group with them. The Group will either be a copy of
077     * an existing Group (if one can be found), or a new, unregistered Group (if not). Optionally, this method can throw a
078     * WikiSecurityException if the Group does not yet exist in the GroupManager cache.
079     * </p>
080     * <p>
081     * The <code>group</code> parameter in the HTTP request contains the Group name to look up and populate. The <code>members</code>
082     * parameter contains the member list. If these differ from those in the existing group, the passed values override the old values.
083     * </p>
084     * <p>
085     * This method does not commit the new Group to the GroupManager cache. To do that, use {@link #setGroup(Session, Group)}.
086     * </p>
087     * @param name the name of the group to construct
088     * @param memberLine the line of text containing the group membership list
089     * @param create whether this method should create a new, empty Group if one with the requested name is not found. If <code>false</code>,
090     *            groups that do not exist will cause a <code>NoSuchPrincipalException</code> to be thrown
091     * @return a new, populated group
092     * @see org.apache.wiki.auth.authorize.Group#RESTRICTED_GROUPNAMES
093     * @throws WikiSecurityException if the group name isn't allowed, or if <code>create</code> is <code>false</code>
094     *                               and the Group named <code>name</code> does not exist
095     */
096    Group parseGroup( String name, String memberLine, boolean create ) throws WikiSecurityException;
097
098    /**
099     * <p>
100     * Extracts group name and members from the HTTP request and populates an existing Group with them. The Group will either be a copy of
101     * an existing Group (if one can be found), or a new, unregistered Group (if not). Optionally, this method can throw a
102     * WikiSecurityException if the Group does not yet exist in the GroupManager cache.
103     * </p>
104     * <p>
105     * The <code>group</code> parameter in the HTTP request contains the Group name to look up and populate. The <code>members</code>
106     * parameter contains the member list. If these differ from those in the existing group, the passed values override the old values.
107     * </p>
108     * <p>
109     * This method does not commit the new Group to the GroupManager cache. To do that, use {@link #setGroup(Session, Group)}.
110     * </p>
111     * @param context the current wiki context
112     * @param create whether this method should create a new, empty Group if one with the requested name is not found. If <code>false</code>,
113     *            groups that do not exist will cause a <code>NoSuchPrincipalException</code> to be thrown
114     * @return a new, populated group
115     * @throws WikiSecurityException if the group name isn't allowed, or if <code>create</code> is <code>false</code>
116     *                               and the Group does not exist
117     */
118    default Group parseGroup( final WikiContext context, final boolean create ) throws WikiSecurityException {
119        // Extract parameters
120        final HttpServletRequest request = context.getHttpRequest();
121        final String name = request.getParameter( "group" );
122        final String memberLine = request.getParameter( "members" );
123
124        // Create the named group; we pass on any NoSuchPrincipalExceptions
125        // that may be thrown if create == false, or WikiSecurityExceptions
126        final Group group = parseGroup( name, memberLine, create );
127
128        // If no members, add the current user by default
129        if( group.members().length == 0 ) {
130            group.add( context.getWikiSession().getUserPrincipal() );
131        }
132
133        return group;
134    }
135
136    /**
137     * Removes a named Group from the group database. If not found, throws a <code>NoSuchPrincipalException</code>. After removal, this
138     * method will commit the delete to the back-end group database. It will also fire a
139     * {@link org.apache.wiki.event.WikiSecurityEvent#GROUP_REMOVE} event with the GroupManager instance as the source and the Group as target.
140     * If <code>index</code> is <code>null</code>, this method throws an {@link IllegalArgumentException}.
141     *
142     * @param index the group to remove
143     * @throws WikiSecurityException if the Group cannot be removed by the back-end
144     * @see org.apache.wiki.auth.authorize.GroupDatabase#delete(Group)
145     */
146    void removeGroup( final String index ) throws WikiSecurityException;
147
148    /**
149     * <p>
150     * Saves the {@link Group} created by a user in a wiki session. This method registers the Group with the GroupManager and saves it to
151     * the back-end database. If an existing Group with the same name already exists, the new group will overwrite it. After saving the
152     * Group, the group database changes are committed.
153     * </p>
154     * <p>
155     * This method fires the following events:
156     * </p>
157     * <ul>
158     * <li><strong>When creating a new Group</strong>, this method fires a {@link org.apache.wiki.event.WikiSecurityEvent#GROUP_ADD} with
159     * the GroupManager instance as its source and the new Group as the target.</li>
160     * <li><strong>When overwriting an existing Group</strong>, this method fires a new
161     * {@link org.apache.wiki.event.WikiSecurityEvent#GROUP_REMOVE} with this GroupManager instance as the source, and the new Group as the
162     * target. It then fires a {@link org.apache.wiki.event.WikiSecurityEvent#GROUP_ADD} event with the same source and target.</li>
163     * </ul>
164     * <p>
165     * In addition, if the save or commit actions fail, this method will attempt to restore the older version of the wiki group if it
166     * exists. This will result in a <code>GROUP_REMOVE</code> event (for the new version of the Group) followed by a <code>GROUP_ADD</code>
167     * event (to indicate restoration of the old version).
168     * </p>
169     * <p>
170     * This method will register the new Group with the GroupManager. For example, {@link org.apache.wiki.auth.AuthenticationManager}
171     * attaches each Session as a GroupManager listener. Thus, the act of registering a Group with <code>setGroup</code> means that
172     * all Sessions will automatically receive group add/change/delete events immediately.
173     * </p>
174     *
175     * @param session the wiki session, which may not be <code>null</code>
176     * @param group the Group, which may not be <code>null</code>
177     * @throws WikiSecurityException if the Group cannot be saved by the back-end
178     */
179    void setGroup( final Session session, final Group group ) throws WikiSecurityException;
180
181    /**
182     * Validates a Group, and appends any errors to the session errors list. Any validation errors are added to the wiki session's messages
183     * collection (see {@link Session#getMessages()}.
184     *
185     * @param context the current wiki context
186     * @param group the supplied Group
187     */
188    default void validateGroup( final WikiContext context, final Group group ) {
189        final InputValidator validator = new InputValidator( MESSAGES_KEY, context );
190
191        // Name cannot be null or one of the restricted names
192        try {
193            checkGroupName( context, group.getName() );
194        } catch( final WikiSecurityException e ) {
195        }
196
197        // Member names must be "safe" strings
198        final Principal[] members = group.members();
199        for( final Principal member : members ) {
200            validator.validateNotNull( member.getName(), "Full name", InputValidator.ID );
201        }
202    }
203
204
205    /**
206     * Checks if a String is blank or a restricted Group name, and if it is, appends an error to the Session's message list.
207     *
208     * @param context the wiki context
209     * @param name the Group name to test
210     * @throws WikiSecurityException if <code>session</code> is <code>null</code> or the Group name is illegal
211     * @see Group#RESTRICTED_GROUPNAMES
212     */
213    default void checkGroupName( final WikiContext context, final String name ) throws WikiSecurityException {
214        // TODO: groups cannot have the same name as a user
215
216        // Name cannot be null
217        final InputValidator validator = new InputValidator( MESSAGES_KEY, context );
218        validator.validateNotNull( name, "Group name" );
219
220        // Name cannot be one of the restricted names either
221        if( ArrayUtils.contains( Group.RESTRICTED_GROUPNAMES, name ) ) {
222            throw new WikiSecurityException( "The group name '" + name + "' is illegal. Choose another." );
223        }
224    }

228    /**
229     * Registers a WikiEventListener with this instance. This is a convenience method.
230     *
231     * @param listener the event listener
232     */
233    void addWikiEventListener( WikiEventListener listener );
234
235    /**
236     * Un-registers a WikiEventListener with this instance. This is a convenience method.
237     *
238     * @param listener the event listener
239     */
240    void removeWikiEventListener( WikiEventListener listener );
241
242    /**
243     *  Fires a WikiSecurityEvent of the provided type, Principal and target Object to all registered listeners.
244     *
245     * @see org.apache.wiki.event.WikiSecurityEvent
246     * @param type       the event type to be fired
247     * @param target     the changed Object, which may be <code>null</code>
248     */
249    default void fireEvent( final int type, final Object target ) {
250        if( WikiEventManager.isListening( this ) ) {
251            WikiEventManager.fireEvent( this, new WikiSecurityEvent( this, type, target ) );
252        }
253    }
