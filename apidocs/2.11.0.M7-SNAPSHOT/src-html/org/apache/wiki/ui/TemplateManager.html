150    /**
151     *  Tries to locate a given resource from the template directory. If the given resource is not found under the current name, returns the
152     *  path to the corresponding one in the default template.
153     *
154     *  @param sContext The servlet context
155     *  @param name The name of the resource
156     *  @return The name of the resource which was found.
157     */
158    private static String findResource( final ServletContext sContext, final String name ) {
159        String resourceName = name;
160        try( final InputStream is = sContext.getResourceAsStream( resourceName ) ) {
161            if( is == null ) {
162                final String defname = makeFullJSPName( DEFAULT_TEMPLATE, removeTemplatePart( resourceName ) );
163                try( final InputStream iis = sContext.getResourceAsStream( defname ) ) {
164                    resourceName = iis != null ? defname : null;
165                }
166            }
167        } catch( final IOException e ) {
168            log.error( "unable to open " + name + " as resource stream", e );
169        }
170        return resourceName;
171    }

173    /**
174     *  Attempts to find a resource from the given template, and if it's not found
175     *  attempts to locate it from the default template.
176     * @param sContext
177     * @param template
178     * @param name
179     * @return the Resource for the given template and name.
180     */
181    private static String findResource( final ServletContext sContext, final String template, final String name ) {
182        if( name.charAt(0) == '/' ) {
183            // This is already a full path
184            return findResource( sContext, name );
185        }
186        final String fullname = makeFullJSPName( template, name );
187        return findResource( sContext, fullname );
188    }

202    /**
203     *  Removes the template part of a name.
204     */
205    private static String removeTemplatePart( String name ) {
206        int idx = 0;
207        if( name.startsWith( "/" ) ) {
208            idx = 1;
209        }
210
211        idx = name.indexOf('/', idx);
212        if( idx != -1 ) {
213            idx = name.indexOf('/', idx+1); // Find second "/"
214            if( idx != -1 ) {
215                name = name.substring( idx+1 );
216            }
217        }
218
219        if( log.isDebugEnabled() ) {
220            log.debug( "Final name = "+name );
221        }
222        return name;
223    }

236    /**
237     *  Attempts to locate a resource under the given template.  If that template does not exist, or the page does not exist under that
238     *  template, will attempt to locate a similarly named file under the default template.
239     *  <p>
240     *  Even though the name suggests only JSP files can be located, but in fact this method can find also other resources than JSP files.
241     *
242     *  @param pageContext The JSP PageContext
243     *  @param template From which template we should seek initially?
244     *  @param name Which resource are we looking for (e.g. "ViewTemplate.jsp")
245     *  @return path to the JSP page; null, if it was not found.
246     */
247    public String findJSP( final PageContext pageContext, final String template, final String name ) {
248        if( name == null || template == null ) {
249            log.fatal("findJSP() was asked to find a null template or name (" + template + "," + name + ")." + " JSP page '" +
250                      ( ( HttpServletRequest )pageContext.getRequest() ).getRequestURI() + "'" );
251            throw new InternalWikiException( "Illegal arguments to findJSP(); please check logs." );
252        }
253
254        return findResource( pageContext.getServletContext(), template, name );
255    }

258     *  Attempts to locate a resource under the given template.  This matches the functionality findJSP(), but uses the WikiContext as
259     *  the argument.  If there is no servlet context (i.e. this is embedded), will just simply return a best-guess.
260     *  <p>
261     *  This method is typically used to locate any resource, including JSP pages, images, scripts, etc.
262     *
263     *  @since 2.6
264     *  @param ctx the wiki context
265     *  @param template the name of the template to use
266     *  @param name the name of the resource to fine
267     *  @return the path to the resource
268     */
269    public String findResource( final WikiContext ctx, final String template, final String name ) {
270        if( m_engine.getServletContext() != null ) {
271            return findResource( m_engine.getServletContext(), template, name );
272        }
273
274        return getPath(template)+"/"+name;
275    }

322    /**
323     *   Lists the skins available under this template.  Returns an empty Set, if there are no extra skins available.  Note that
324     *   this method does not check whether there is anything actually in the directories, it just lists them.  This may change
325     *   in the future.
326     *
327     *   @param pageContext the JSP PageContext
328     *   @param template The template to search
329     *   @return Set of Strings with the skin names.
330     *   @since 2.3.26
331     */
332    public Set< String > listSkins( final PageContext pageContext, final String template ) {
333        final String place = makeFullJSPName( template, SKIN_DIRECTORY );
334        final ServletContext sContext = pageContext.getServletContext();
335        final Set< String > skinSet = sContext.getResourcePaths( place );
336        final Set< String > resultSet = new TreeSet<>();
337
338        if( log.isDebugEnabled() ) {
339            log.debug( "Listings skins from " + place );
340        }
341
342        if( skinSet != null ) {
343            final String[] skins = skinSet.toArray( new String[]{} );
344            for( final String skin : skins ) {
345                final String[] s = StringUtils.split( skin, "/" );
346                if( s.length > 2 && skin.endsWith( "/" ) ) {
347                    final String skinName = s[ s.length - 1 ];
348                    resultSet.add( skinName );
349                    if( log.isDebugEnabled() ) {
350                        log.debug( "...adding skin '" + skinName + "'" );
351                    }
352                }
353            }
354        }
355
356        return resultSet;
357    }

359    /**
360     * List all installed i18n language properties by classpath searching for files like :
361     *    templates/default_*.properties
362     *    templates/default.properties
363     *
364     * @param pageContext
365     * @return map of installed Languages
366     * @since 2.7.x
367     */
368    public Map< String, String > listLanguages( final PageContext pageContext ) {
369        final Map< String, String > resultMap = new LinkedHashMap<>();
370        final String clientLanguage = pageContext.getRequest().getLocale().toString();
371        final List< String > entries = ClassUtil.classpathEntriesUnder( DIRECTORY );
372        for( String name : entries ) {
373            if ( name.equals( I18NRESOURCE_EN ) || (name.startsWith( I18NRESOURCE_PREFIX ) && name.endsWith( I18NRESOURCE_SUFFIX ) ) ) {
374                if( name.equals( I18NRESOURCE_EN ) ) {
375                    name = I18NRESOURCE_EN_ID;
376                } else {
377                    name = name.substring( I18NRESOURCE_PREFIX.length(), name.lastIndexOf( I18NRESOURCE_SUFFIX ) );
378                }
379                final Locale locale = new Locale( name.substring( 0, 2 ), ( ( name.indexOf( "_" ) == -1 ) ? "" : name.substring( 3, 5 ) ) );
380                String defaultLanguage = "";
381                if( clientLanguage.startsWith( name ) ) {
382                    defaultLanguage = LocaleSupport.getLocalizedMessage( pageContext, I18NDEFAULT_LOCALE );
383                }
384                resultMap.put( name, locale.getDisplayName( locale ) + " " + defaultLanguage );
385            }
386        }
387
388        return resultMap;
389    }

435        }*/
436

439            // dummy format pattern

442

557    /**
558     *  Extract all i18n strings in the javascript domain. (javascript.*) Returns a javascript snippet which defines the LocalizedStings array.
559     *
560     *  @param context the {@link WikiContext}
561     *  @return Javascript snippet which defines the LocalizedStrings array
562     *  @since 2.5.108
563     */
564    private static String getJSLocalizedStrings( final WikiContext context ) {
565        final StringBuilder sb = new StringBuilder();
566        sb.append( "var LocalizedStrings = {\n");
567        final ResourceBundle rb = Preferences.getBundle( context, InternationalizationManager.DEF_TEMPLATE );
568        boolean first = true;
569
570        for( final Enumeration< String > en = rb.getKeys(); en.hasMoreElements(); ) {
571            final String key = en.nextElement();
572            if( key.startsWith("javascript") ) {
573                if( first ) {
574                    first = false;
575                } else {
576                    sb.append( ",\n" );
577                }
578                sb.append( "\""+key+"\":\""+rb.getString(key)+"\"" );
579            }
580        }
581        sb.append("\n};\n");
582
583        return( sb.toString() );
584    }

586    /**
587     *  Adds a resource request to the current request context. The content will be added at the resource-type marker
588     *  (see IncludeResourcesTag) in WikiJSPFilter.
589     *  <p>
590     *  The resources can be of different types.  For RESOURCE_SCRIPT and RESOURCE_STYLESHEET this is an URI path to the resource
591     *  (a script file or an external stylesheet) that needs to be included.  For RESOURCE_INLINECSS the resource should be something
592     *  that can be added between &lt;style>&lt;/style> in the header file (commonheader.jsp).  For RESOURCE_JSFUNCTION it is the name
593     *  of the Javascript function that should be run at page load.
594     *  <p>
595     *  The IncludeResourceTag inserts code in the template files, which is then filled by the WikiFilter after the request has been
596     *  rendered but not yet sent to the recipient.
597     *  <p>
598     *  Note that ALL resource requests get rendered, so this method does not check if the request already exists in the resources.
599     *  Therefore, if you have a plugin which makes a new resource request every time, you'll end up with multiple resource requests
600     *  rendered.  It's thus a good idea to make this request only once during the page life cycle.
601     *
602     *  @param ctx The current wiki context
603     *  @param type What kind of a request should be added?
604     *  @param resource The resource to add.
605     */
606    @SuppressWarnings("unchecked")
607    public static void addResourceRequest( final WikiContext ctx, final String type, final String resource ) {
608        HashMap< String, Vector< String > > resourcemap = ( HashMap< String, Vector< String > > ) ctx.getVariable( RESOURCE_INCLUDES );
609        if( resourcemap == null ) {
610            resourcemap = new HashMap<>();
611        }
612
613        Vector< String > resources = resourcemap.get( type );
614        if( resources == null ) {
615            resources = new Vector<>();
616        }
617
618        String resourceString = null;
619        switch( type ) {
620        case RESOURCE_SCRIPT:
621            resourceString = "<script type='text/javascript' src='" + resource + "'></script>";
622            break;
623        case RESOURCE_STYLESHEET:
624            resourceString = "<link rel='stylesheet' type='text/css' href='" + resource + "' />";
625            break;
626        case RESOURCE_INLINECSS:
627            resourceString = "<style type='text/css'>\n" + resource + "\n</style>\n";
628            break;
629        case RESOURCE_JSFUNCTION:
630        case RESOURCE_HTTPHEADER:
631            resourceString = resource;
632            break;
633        }
634
635        if( resourceString != null ) {
636            resources.add( resourceString );
637        }
638
639        log.debug( "Request to add a resource: " + resourceString );
640
641        resourcemap.put( type, resources );
642        ctx.setVariable( RESOURCE_INCLUDES, resourcemap );
643    }

645    /**
646     *  Returns resource requests for a particular type.  If there are no resources, returns an empty array.
647     *
648     *  @param ctx WikiContext
649     *  @param type The resource request type
650     *  @return a String array for the resource requests
651     */
652    @SuppressWarnings("unchecked")
653    public static String[] getResourceRequests( final WikiContext ctx, final String type ) {
654        final HashMap< String, Vector< String > > hm = ( HashMap< String, Vector< String > > ) ctx.getVariable( RESOURCE_INCLUDES );
655        if( hm == null ) {
656            return new String[0];
657        }
658
659        final Vector<String> resources = hm.get( type );
660        if( resources == null ){
661            return new String[0];
662        }
663
664        final String[] res = new String[resources.size()];
665        return resources.toArray( res );
666    }

668    /**
669     *  Returns all those types that have been requested so far.
670     *
671     * @param ctx the wiki context
672     * @return the array of types requested
673     */
674    @SuppressWarnings("unchecked")
675    public static String[] getResourceTypes( final WikiContext ctx ) {
676        String[] res = new String[0];
677        if( ctx != null ) {
678            final HashMap< String, String > hm = ( HashMap< String, String > ) ctx.getVariable( RESOURCE_INCLUDES );
679            if( hm != null ) {
680                final Set< String > keys = hm.keySet();
681                res = keys.toArray( res );
682            }
683        }
684
685        return res;
686    }
