049/**
050 *  Returns the Recent Changes in the wiki being a date-sorted list of page names.
051 *
052 *  <p>Parameters: </p>
053 *  <ul>
054 *  <li><b>since</b> - show changes from the last n days, for example since=5 shows only the pages that were changed in the last five days</li>
055 *  <li><b>format</b> - (full|compact) : if "full", then display a long version with all possible info. If "compact", then be as compact as possible.</li>
056 *  <li><b>timeFormat</b> - the time format to use, the default is "HH:mm:ss"</li>
057 *  <li><b>dateFormat</b> - the date format to use, the default is "dd.MM.yyyy"</li>
058 *  </ul>
059 */
060public class RecentChangesPlugin extends AbstractReferralPlugin implements Plugin {
061    
062    private static final Logger log = Logger.getLogger( RecentChangesPlugin.class );
063    
064    /** Parameter name for the separator format.  Value is <tt>{@value}</tt>. */
065    public static final String PARAM_FORMAT      = "format";
066    /** Parameter name for the separator timeFormat.  Value is <tt>{@value}</tt>. */
067    public static final String PARAM_TIME_FORMAT = "timeFormat";
068    /** Parameter name for the separator dateFormat.  Value is <tt>{@value}</tt>. */
069    public static final String PARAM_DATE_FORMAT = "dateFormat";
070
071    /** How many days we show by default. */
072    private static final int   DEFAULT_DAYS = 100*365;
073    public static final String DEFAULT_TIME_FORMAT ="HH:mm:ss";
074    public static final String DEFAULT_DATE_FORMAT ="dd.MM.yyyy";
075
076
077    /**
078     * {@inheritDoc}
079     */
080    @Override
081    public String execute( final Context context, final Map< String, String > params ) throws PluginException {
082        final int since = TextUtil.parseIntParameter( params.get( "since" ), DEFAULT_DAYS );
083        String spacing  = "4";
084        boolean showAuthor = true;
085        boolean showChangenote = true;
086        String tablewidth = "4";
087        
088        final Engine engine = context.getEngine();
089
090        //
091        //  Which format we want to see?
092        //
093        if( "compact".equals( params.get( PARAM_FORMAT ) ) ) {
094            spacing  = "0";
095            showAuthor = false;
096            showChangenote = false;
097            tablewidth = "2";
098        }
099
100        final Calendar sincedate = new GregorianCalendar();
101        sincedate.add( Calendar.DAY_OF_MONTH, -since );
102
103        log.debug("Calculating recent changes from "+sincedate.getTime());
104
105        // FIXME: Should really have a since date on the getRecentChanges method.
106        Collection< Page > changes = engine.getManager( PageManager.class ).getRecentChanges();
107        super.initialize( context, params );
108        changes = filterWikiPageCollection( changes );
109        
110        if ( changes != null ) {
111            Date olddate = new Date( 0 );
112
113            final DateFormat fmt = getDateFormat( context, params );
114            final DateFormat tfmt = getTimeFormat( context, params );
115
116            final Element rt = XhtmlUtil.element( XHTML.table );
117            rt.setAttribute( XHTML.ATTR_class, "recentchanges" );
118            rt.setAttribute( XHTML.ATTR_cellpadding, spacing );
119
120            for( final Page pageref : changes ) {
121                final Date lastmod = pageref.getLastModified();
122
123                if( lastmod.before( sincedate.getTime() ) ) {
124                    break;
125                }
126
127                if( !isSameDay( lastmod, olddate ) ) {
128                    final Element row = XhtmlUtil.element( XHTML.tr );
129                    final Element col = XhtmlUtil.element( XHTML.td );
130                    col.setAttribute( XHTML.ATTR_colspan, tablewidth );
131                    col.setAttribute( XHTML.ATTR_class, "date" );
132                    col.addContent( XhtmlUtil.element( XHTML.b, fmt.format( lastmod ) ) );
133
134                    rt.addContent( row );
135                    row.addContent( col );
136                    olddate = lastmod;
137                }
138
139                final String href = context.getURL( pageref instanceof Attachment ? WikiContext.ATTACH : WikiContext.VIEW, pageref.getName() );
140                Element link = XhtmlUtil.link( href, engine.getManager( RenderingManager.class ).beautifyTitle( pageref.getName() ) );
141                final Element row = XhtmlUtil.element( XHTML.tr );
142                final Element col = XhtmlUtil.element( XHTML.td );
143                col.setAttribute( XHTML.ATTR_width, "30%" );
144                col.addContent( link );
145
146                //
147                //  Add the direct link to the attachment info.
148                //
149                if( pageref instanceof Attachment ) {
150                    link = XhtmlUtil.link( context.getURL( WikiContext.INFO, pageref.getName() ), null );
151                    link.setAttribute( XHTML.ATTR_class, "infolink" );
152
153                    final Element img = XhtmlUtil.img( context.getURL( WikiContext.NONE, "images/attachment_small.png" ), null );
154                    link.addContent( img );
155
156                    col.addContent( link );
157                }
158
159                row.addContent( col );
160                rt.addContent( row );
161
162                if( pageref instanceof Attachment ) {
163                    final Element td = XhtmlUtil.element( XHTML.td, tfmt.format( lastmod ) );
164                    td.setAttribute( XHTML.ATTR_class, "lastchange" );
165                    row.addContent( td );
166                } else {
167                    final Element infocol = XhtmlUtil.element( XHTML.td );
168                    infocol.setAttribute( XHTML.ATTR_class, "lastchange" );
169                    infocol.addContent( XhtmlUtil.link( context.getURL( WikiContext.DIFF, pageref.getName(), "r1=-1" ),
170                                                        tfmt.format( lastmod ) ) );
171                    row.addContent( infocol );
172                }
173
174                //
175                //  Display author information.
176                //
177                if( showAuthor ) {
178                    final String author = pageref.getAuthor();
179
180                    final Element authorinfo = XhtmlUtil.element( XHTML.td );
181                    authorinfo.setAttribute( XHTML.ATTR_class, "author" );
182
183                    if( author != null ) {
184                        if( engine.getManager( PageManager.class ).wikiPageExists( author ) ) {
185                            authorinfo.addContent( XhtmlUtil.link( context.getURL( WikiContext.VIEW, author ), author ) );
186                        } else {
187                            authorinfo.addContent( author );
188                        }
189                    } else {
190                        authorinfo.addContent( Preferences.getBundle( context, InternationalizationManager.CORE_BUNDLE )
191                                .getString( "common.unknownauthor" ) );
192                    }
193
194                    row.addContent( authorinfo );
195                }
196
197                // Change note
198                if( showChangenote ) {
199                    final String changenote = pageref.getAttribute( Page.CHANGENOTE );
200                    final Element td_changenote = XhtmlUtil.element( XHTML.td, changenote );
201                    td_changenote.setAttribute( XHTML.ATTR_class, "changenote" );
202                    row.addContent( td_changenote );
203                }
204
205                //  Revert note
206/*                
207                if( context.hasAdminPermissions() )
208                {
209                    row.addElement( new td("Revert") );
210                }
211 */
212            }
213            return XhtmlUtil.serialize( rt, XhtmlUtil.EXPAND_EMPTY_NODES );
214        }
215        return "";
216    }
217
218    
219    private boolean isSameDay( final Date a, final Date b ) {
220        final Calendar aa = Calendar.getInstance(); aa.setTime( a );
221        final Calendar bb = Calendar.getInstance(); bb.setTime( b );
222
223        return aa.get( Calendar.YEAR ) == bb.get( Calendar.YEAR ) 
224            && aa.get( Calendar.DAY_OF_YEAR ) == bb.get( Calendar.DAY_OF_YEAR );
225    }

228    // TODO: Ideally the default behavior should be to return the default format for the default
229    // locale, but that is at odds with the 1st version of this plugin. We seek to preserve the
230    // behaviour of that first version, so to get the default format, the user must explicitly do
231    // something like: dateFormat='' timeformat='' which is a odd, but probably okay.
232    private DateFormat getTimeFormat( final Context context, final Map< String, String > params ) {
233        final String formatString = get( params, DEFAULT_TIME_FORMAT, PARAM_TIME_FORMAT );
234        if( StringUtils.isBlank( formatString ) ) {
235            return Preferences.getDateFormat( context, TimeFormat.TIME );
236        }
237
238        return new SimpleDateFormat( formatString );
239    }
240
241    private DateFormat getDateFormat( final Context context, final Map< String, String > params ) {
242        final String formatString = get( params, DEFAULT_DATE_FORMAT, PARAM_DATE_FORMAT );
243        if( StringUtils.isBlank( formatString ) ) {
244            return Preferences.getDateFormat( context, TimeFormat.DATE );
245        }
246
247        return new SimpleDateFormat( formatString );
248    }
249    
250    private String get( final Map< String, String > params, final String defaultValue, final String paramName ) {
251        final String value = params.get( paramName );
252        return value == null ? defaultValue : value;
253    }
