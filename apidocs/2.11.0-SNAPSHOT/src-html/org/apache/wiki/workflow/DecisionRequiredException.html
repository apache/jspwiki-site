019package org.apache.wiki.workflow;
020
021import org.apache.wiki.api.exceptions.WikiException;
022
023/**
024 * Exception thrown when an activity -- that would otherwise complete silently --
025 * cannot complete because a workflow {@link Decision} is required. The message
026 * string should be a human-readable, internationalized String explaining why
027 * the activity could not complete, or that the activity has been queued for 
028 * review.
029 *
030 */
031public class DecisionRequiredException extends WikiException
032{
033
034    private static final long serialVersionUID = 1L;
035
036    /**
037     * Constructs a new exception.
038     * @param message the message
039     */
040    public DecisionRequiredException(final String message )
041    {
042        super( message );
043    }
044}
