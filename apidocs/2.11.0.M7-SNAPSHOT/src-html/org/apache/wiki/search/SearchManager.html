019package org.apache.wiki.search;
020
021import org.apache.wiki.WikiContext;
022import org.apache.wiki.WikiPage;
023import org.apache.wiki.api.exceptions.ProviderException;
024import org.apache.wiki.api.filters.PageFilter;
025import org.apache.wiki.event.WikiEventListener;
026import org.apache.wiki.modules.InternalModule;
027
028import java.io.IOException;
029import java.util.Collection;
030
031
032/**
033 *  Manages searching the Wiki.
034 *
035 *  @since 2.2.21.
036 */
037public interface SearchManager extends PageFilter, InternalModule, WikiEventListener {
038
039    String DEFAULT_SEARCHPROVIDER = "org.apache.wiki.search.LuceneSearchProvider";
040
041    /** Property name for setting the search provider. Value is <tt>{@value}</tt>. */
042    String PROP_SEARCHPROVIDER = "jspwiki.searchProvider";
043
044    /** The name of the JSON object that manages search. */
045    String JSON_SEARCH = "search";
046
047    /**
048     *  Returns the SearchProvider used.
049     *
050     *  @return The current SearchProvider.
051     */
052    SearchProvider getSearchEngine();
053
054    /**
055     *  Sends a search to the current search provider. The query is is whatever native format the query engine wants to use.
056     *
057     * @param query The query.  Null is safe, and is interpreted as an empty query.
058     * @param wikiContext the context within which to run the search
059     * @return A collection of WikiPages that matched.
060     * @throws ProviderException If the provider fails and a search cannot be completed.
061     * @throws IOException If something else goes wrong.
062     */
063    default Collection< SearchResult > findPages( String query, final WikiContext wikiContext ) throws ProviderException, IOException {
064        if( query == null ) {
065            query = "";
066        }
067        return getSearchEngine().findPages( query, wikiContext );
068    }
069
070    /**
071     *  Removes the page from the search cache (if any).
072     *
073     *  @param page  The page to remove
074     */
075    default void pageRemoved( final WikiPage page ) {
076        getSearchEngine().pageRemoved( page );
077    }
078
079    /**
080     *   Forces the reindex of the given page.
081     *
082     *   @param page The page.
083     */
084    default void reindexPage( final WikiPage page ) {
085        getSearchEngine().reindexPage( page );
086    }
087
088}
