047/**
048 *  Provides a caching attachment provider.  This class rests on top of a
049 *  real provider class and provides a cache to speed things up.  Only the
050 *  Attachment objects are cached; the actual attachment contents are
051 *  fetched always from the provider.
052 *
053 *  @since 2.1.64.
054 */
055
056//        EntryRefreshPolicy for that.
057public class CachingAttachmentProvider implements WikiAttachmentProvider {
058
059    private static final Logger log = Logger.getLogger(CachingAttachmentProvider.class);
060
061    private WikiAttachmentProvider m_provider;
062
063    private CacheManager m_cacheManager = CacheManager.getInstance();
064
065    /** Default cache capacity for now. */
066    public static final int m_capacity = 1000;
067
068    /**
069     *  The cache contains Collection objects which contain Attachment objects.
070     *  The key is the parent wiki page name (String).
071     */
072    private Cache m_cache;
073    /** Name of the attachment cache. */
074    public static final String ATTCOLLCACHE_NAME = "jspwiki.attachmentCollectionsCache";
075
076    /**
077     * This cache contains Attachment objects and is keyed by attachment name.
078     * This provides for quickly giving recently changed attachments (for the RecentChanges plugin)
079     */
080    private Cache m_attCache;
081    /** Name of the attachment cache. */
082    public static final String ATTCACHE_NAME = "jspwiki.attachmentsCache";
083
084    private long m_cacheMisses = 0;
085    private long m_cacheHits = 0;
086
087    /** The extension to append to directory names to denote an attachment directory. */
088    public static final String DIR_EXTENSION   = "-att";
089
090    /** Property that supplies the directory used to store attachments. */
091    public static final String PROP_STORAGEDIR = "jspwiki.basicAttachmentProvider.storageDir";
092
093    private boolean m_gotall = false;
094
095    /**
096     * {@inheritDoc}
097     */
098    @Override
099    public void initialize( final Engine engine, final Properties properties ) throws NoRequiredPropertyException, IOException {
100        log.info("Initing CachingAttachmentProvider");
101
102        final String attCollCacheName = engine.getApplicationName() + "." + ATTCOLLCACHE_NAME;
103        if (m_cacheManager.cacheExists(attCollCacheName)) {
104            m_cache = m_cacheManager.getCache(attCollCacheName);
105        } else {
106            m_cache = new Cache(attCollCacheName, m_capacity, false, false, 0, 0);
107            m_cacheManager.addCache(m_cache);
108        }
109
110        //
111        // cache for the individual Attachment objects, attachment name is key, the Attachment object is the cached object
112        //
113        final String attCacheName = engine.getApplicationName() + "." + ATTCACHE_NAME;
114        if (m_cacheManager.cacheExists(attCacheName)) {
115            m_attCache = m_cacheManager.getCache(attCacheName);
116        } else {
117            m_attCache = new Cache(attCacheName, m_capacity, false, false, 0, 0);
118            m_cacheManager.addCache(m_attCache);
119        }
120        //
121        //  Find and initialize real provider.
122        //
123        final String classname;
124        try {
125            classname = TextUtil.getRequiredProperty( properties, AttachmentManager.PROP_PROVIDER );
126        } catch( final NoSuchElementException e ) {
127            throw new NoRequiredPropertyException( e.getMessage(), AttachmentManager.PROP_PROVIDER );
128        }
129
130        try
131        {
132            final Class<?> providerclass = ClassUtil.findClass( "org.apache.wiki.providers", classname);
133
134            m_provider = (WikiAttachmentProvider)providerclass.newInstance();
135
136            log.debug("Initializing real provider class "+m_provider);
137            m_provider.initialize( engine, properties );
138        }
139        catch( final ClassNotFoundException e )
140        {
141            log.error("Unable to locate provider class "+classname,e);
142            throw new IllegalArgumentException("no provider class", e);
143        }
144        catch( final InstantiationException e )
145        {
146            log.error("Unable to create provider class "+classname,e);
147            throw new IllegalArgumentException("faulty provider class", e);
148        }
149        catch( final IllegalAccessException e )
150        {
151            log.error("Illegal access to provider class "+classname,e);
152            throw new IllegalArgumentException("illegal provider class", e);
153        }
154
155    }

157    /**
158     * {@inheritDoc}
159     */
160    @Override
161    public void putAttachmentData( final Attachment att, final InputStream data )
162        throws ProviderException, IOException {
163        m_provider.putAttachmentData( att, data );
164
165        m_cache.remove(att.getParentName());
166        att.setLastModified(new Date());
167        m_attCache.put(new Element(att.getName(), att));
168    }

171     * {@inheritDoc}
172     */
173    @Override
174    public InputStream getAttachmentData( final Attachment att )
175        throws ProviderException, IOException {
176        return m_provider.getAttachmentData( att );
177    }

179    /**
180     * {@inheritDoc}
181     */
182    @Override
183    public List< Attachment > listAttachments( final WikiPage page) throws ProviderException {
184        log.debug("Listing attachments for " + page);
185        final Element element = m_cache.get(page.getName());
186
187        if (element != null) {
188            @SuppressWarnings("unchecked") final List< Attachment > c = ( List< Attachment > )element.getObjectValue();
189            log.debug("LIST from cache, " + page.getName() + ", size=" + c.size());
190            return cloneCollection(c);
191        }
192
193        log.debug("list NOT in cache, " + page.getName());
194
195        return refresh(page);
196    }
197
198    private <T> List<T> cloneCollection( final Collection<T> c )
199    {
200        final ArrayList<T> list = new ArrayList<>();
201
202        list.addAll( c );
203
204        return list;
205    }

216    /**
217     * {@inheritDoc}
218     */
219    @Override
220    public List<Attachment> listAllChanged( final Date timestamp) throws ProviderException {
221        List< Attachment > all = null;
222        //
223        // we do a one-time build up of the cache, after this the cache is updated for every attachment add/delete
224        if (m_gotall == false) {
225            all = m_provider.listAllChanged(timestamp);
226
227            // Put all pages in the cache :
228
229            synchronized (this) {
230                for ( final Iterator< Attachment > i = all.iterator(); i.hasNext(); ) {
231                    final Attachment att = i.next();
232                    m_attCache.put(new Element(att.getName(), att));
233                }
234                m_gotall = true;
235            }
236        } else {
237            @SuppressWarnings("unchecked") final List< String > keys = m_attCache.getKeysWithExpiryCheck();
238            all = new ArrayList<>();
239            for ( final String key : keys) {
240                final Element element = m_attCache.get(key);
241                final Attachment cachedAttachment = ( Attachment )element.getObjectValue();
242                if (cachedAttachment != null) {
243                    all.add(cachedAttachment);
244                }
245            }
246        }
247
248        return all;
249    }
250
251    /**
252     *  Simply goes through the collection and attempts to locate the
253     *  given attachment of that name.
254     *
255     *  @return null, if no such attachment was in this collection.
256     */
257    private Attachment findAttachmentFromCollection( final Collection< Attachment > c, final String name ) {
258        for( final Attachment att : new ArrayList< >( c ) ) {
259            if( name.equals( att.getFileName() ) ) {
260                return att;
261            }
262        }
263
264        return null;
265    }
266
267    /**
268     *  Refreshes the cache content and updates counters.
269     *
270     *  @return The newly fetched object from the provider.
271     */
272    private List<Attachment> refresh( final WikiPage page ) throws ProviderException
273    {
274        final List<Attachment> c = m_provider.listAttachments( page );
275        m_cache.put(new Element(page.getName(), c));
276
277        return c;
278    }

280    /**
281     * {@inheritDoc}
282     */
283    @SuppressWarnings("unchecked")
284    @Override
285    public Attachment getAttachmentInfo( final WikiPage page, final String name, final int version) throws ProviderException {
286        if (log.isDebugEnabled()) {
287            log.debug("Getting attachments for " + page + ", name=" + name + ", version=" + version);
288        }
289
290        //
291        //  We don't cache previous versions
292        //
293        if (version != WikiProvider.LATEST_VERSION) {
294            log.debug("...we don't cache old versions");
295            return m_provider.getAttachmentInfo(page, name, version);
296        }
297
298        Collection<Attachment> c = null;
299        final Element element =   m_cache.get(page.getName());
300
301        if (element == null) {
302            log.debug(page.getName() + " wasn't in the cache");
303            c = refresh(page);
304
305            if (c == null) {
306                return null; // No such attachment
307            }
308        } else {
309            log.debug(page.getName() + " FOUND in the cache");
310            c = (Collection<Attachment>) element.getObjectValue();
311        }
312
313        return findAttachmentFromCollection(c, name);
314    }

331        // This isn't strictly speaking correct, but it does not really matter

336    /**
337     * {@inheritDoc}
338     */
339    @Override
340    public void deleteAttachment( final Attachment att ) throws ProviderException
341    {
342        m_cache.remove(att.getParentName());
343        m_attCache.remove(att.getName());
344        m_provider.deleteAttachment( att );
345    }

371    /**
372     * {@inheritDoc}
373     */
374    @Override
375    public void moveAttachmentsForPage( final String oldParent, final String newParent ) throws ProviderException
376    {
377        m_provider.moveAttachmentsForPage(oldParent, newParent);
378        m_cache.remove(newParent);
379        m_cache.remove(oldParent);
380
381        //
382        //  This is a kludge to make sure that the pages are removed
383        //  from the other cache as well.
384        //
385        final String checkName = oldParent + "/";
386
387        @SuppressWarnings("unchecked") final List< String > names = m_cache.getKeysWithExpiryCheck();
388        for( final String name : names )
389        {
390            if( name.startsWith( checkName ) )
391            {
392                m_attCache.remove(name);
393            }
394        }
395    }
