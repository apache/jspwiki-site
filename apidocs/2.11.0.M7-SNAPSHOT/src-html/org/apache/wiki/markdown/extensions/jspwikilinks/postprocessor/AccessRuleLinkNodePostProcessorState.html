019package org.apache.wiki.markdown.extensions.jspwikilinks.postprocessor;
020
021import com.vladsch.flexmark.util.ast.Node;
022import com.vladsch.flexmark.util.ast.NodeTracker;
023import org.apache.log4j.Logger;
024import org.apache.wiki.WikiContext;
025import org.apache.wiki.WikiPage;
026import org.apache.wiki.auth.WikiSecurityException;
027import org.apache.wiki.auth.acl.Acl;
028import org.apache.wiki.markdown.nodes.JSPWikiLink;
029import org.apache.wiki.render.RenderingManager;
030
031
032/**
033 * {@link NodePostProcessorState} which further post processes access rules links.
034 */
035public class AccessRuleLinkNodePostProcessorState implements NodePostProcessorState< JSPWikiLink > {
036
037    private static final Logger LOG = Logger.getLogger( AccessRuleLinkNodePostProcessorState.class );
038    private final WikiContext wikiContext;
039    private final boolean m_wysiwygEditorMode;
040
041    public AccessRuleLinkNodePostProcessorState( final WikiContext wikiContext ) {
042        this.wikiContext = wikiContext;
043        final Boolean wysiwygVariable = ( Boolean )wikiContext.getVariable( RenderingManager.WYSIWYG_EDITOR_MODE );
044        m_wysiwygEditorMode = wysiwygVariable != null ? wysiwygVariable.booleanValue() : false;
045    }
046
047    /**
048     * {@inheritDoc}
049     *
050     * @see NodePostProcessorState#process(NodeTracker, Node)
051     */
052    @Override
053    public void process( final NodeTracker state, final JSPWikiLink link ) {
054        String ruleLine = NodePostProcessorStateCommonOperations.inlineLinkTextOnWysiwyg( state, link, m_wysiwygEditorMode );
055        if( wikiContext.getEngine().getRenderingManager().getParser( wikiContext, link.getUrl().toString() ).isParseAccessRules() ) {
056            final WikiPage page = wikiContext.getRealPage();
057            if( ruleLine.startsWith( "{" ) ) {
058                ruleLine = ruleLine.substring( 1 );
059            }
060            if( ruleLine.endsWith( "}" ) ) {
061                ruleLine = ruleLine.substring( 0, ruleLine.length() - 1 );
062            }
063            LOG.debug( "page=" + page.getName() + ", ACL = " + ruleLine );
064
065            try {
066                final Acl acl = wikiContext.getEngine().getAclManager().parseAcl( page, ruleLine );
067                page.setAcl( acl );
068                link.unlink();
069                state.nodeRemoved( link );
070                LOG.debug( acl.toString() );
071            } catch( final WikiSecurityException wse ) {
072                NodePostProcessorStateCommonOperations.makeError( state, link, wse.getMessage() );
073            }
074        }
075    }
076
077}
