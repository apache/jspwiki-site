055/**
056 * <p>Manages {@link DefaultUserProfile} objects using XML files for persistence.
057 * Passwords are hashed using SHA1. User entries are simple <code>&lt;user&gt;</code>
058 * elements under the root. User profile properties are attributes of the
059 * element. For example:</p>
060 * <blockquote><code>
061 * &lt;users&gt;<br/>
062 * &nbsp;&nbsp;&lt;user loginName="janne" fullName="Janne Jalkanen"<br/> 
063 * &nbsp;&nbsp;&nbsp;&nbsp;wikiName="JanneJalkanen" email="janne@ecyrd.com"<br/>
064 * &nbsp;&nbsp;&nbsp;&nbsp;password="{SHA}457b08e825da547c3b77fbc1ff906a1d00a7daee"/&gt;<br/>
065 * &lt;/users&gt;
066 * </code></blockquote> 
067 * <p>In this example, the un-hashed password is <code>myP@5sw0rd</code>. Passwords are hashed without salt.</p>
068 * @since 2.3
069 */
070

111    /**
112     * Looks up and deletes the first {@link UserProfile} in the user database
113     * that matches a profile having a given login name. If the user database
114     * does not contain a user with a matching attribute, throws a
115     * {@link NoSuchPrincipalException}.
116     * @param loginName the login name of the user profile that shall be deleted
117     */
118    public synchronized void deleteByLoginName( String loginName ) throws NoSuchPrincipalException, WikiSecurityException
119    {
120        if ( c_dom == null )
121        {
122            throw new WikiSecurityException( "FATAL: database does not exist" );
123        }
124            
125        NodeList users = c_dom.getDocumentElement().getElementsByTagName( USER_TAG );
126        for( int i = 0; i < users.getLength(); i++ )
127        {
128            Element user = (Element) users.item( i );
129            if ( user.getAttribute( LOGIN_NAME ).equals( loginName ) )
130            {
131                c_dom.getDocumentElement().removeChild(user);
132                
133                // Commit to disk
134                saveDOM();
135                return;
136            }
137        }
138        throw new NoSuchPrincipalException( "Not in database: " + loginName );
139    }

230    /**
231     * Returns all WikiNames that are stored in the UserDatabase
232     * as an array of WikiPrincipal objects. If the database does not
233     * contain any profiles, this method will return a zero-length
234     * array.
235     * @return the WikiNames
236     * @throws WikiSecurityException In case things fail.
237     */
238    public Principal[] getWikiNames() throws WikiSecurityException
239    {
240        if ( c_dom == null )
241        {
242            throw new IllegalStateException( "FATAL: database does not exist" );
243        }
244        SortedSet<Principal> principals = new TreeSet<Principal>();
245        NodeList users = c_dom.getElementsByTagName( USER_TAG );
246        for( int i = 0; i < users.getLength(); i++ )
247        {
248            Element user = (Element) users.item( i );
249            String wikiName = user.getAttribute( WIKI_NAME );
250            if ( wikiName == null )
251            {
252                log.warn( "Detected null wiki name in XMLUserDataBase. Check your user database." );
253            }
254            else
255            {
256                Principal principal = new WikiPrincipal( wikiName, WikiPrincipal.WIKI_NAME );
257                principals.add( principal );
258            }
259        }
260        return principals.toArray( new Principal[principals.size()] );
261    }

263    /**
264     * Initializes the user database based on values from a Properties object.
265     * The properties object must contain a file path to the XML database file
266     * whose key is {@link #PROP_USERDATABASE}.
267     * @see org.apache.wiki.auth.user.UserDatabase#initialize(org.apache.wiki.WikiEngine,
268     *      java.util.Properties)
269     * @throws NoRequiredPropertyException if the user database cannot be located, parsed, or opened
270     */
271    public void initialize( WikiEngine engine, Properties props ) throws NoRequiredPropertyException
272    {
273        File defaultFile = null;
274        if( engine.getRootPath() == null )
275        {
276            log.warn( "Cannot identify JSPWiki root path"  );
277            defaultFile = new File( "WEB-INF/" + DEFAULT_USERDATABASE ).getAbsoluteFile();
278        }
279        else
280        {
281            defaultFile = new File( engine.getRootPath() + "/WEB-INF/" + DEFAULT_USERDATABASE );
282        }
283
284        // Get database file location
285        String file = TextUtil.getStringProperty(props, PROP_USERDATABASE, defaultFile.getAbsolutePath());
286        if( file == null )
287        {
288            log.warn( "XML user database property " + PROP_USERDATABASE + " not found; trying " + defaultFile  );
289            c_file = defaultFile;
290        }
291        else 
292        {
293            c_file = new File( file );
294        }
295
296        log.info("XML user database at "+c_file.getAbsolutePath());
297        
298        buildDOM();
299        sanitizeDOM();
300    }

302    private void buildDOM()
303    {
304        // Read DOM
305        DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
306        factory.setValidating( false );
307        factory.setExpandEntityReferences( false );
308        factory.setIgnoringComments( true );
309        factory.setNamespaceAware( false );
310        try
311        {
312            c_dom = factory.newDocumentBuilder().parse( c_file );
313            log.debug( "Database successfully initialized" );
314            c_lastModified = c_file.lastModified();
315            c_lastCheck    = System.currentTimeMillis();
316        }
317        catch( ParserConfigurationException e )
318        {
319            log.error( "Configuration error: " + e.getMessage() );
320        }
321        catch( SAXException e )
322        {
323            log.error( "SAX error: " + e.getMessage() );
324        }
325        catch( FileNotFoundException e )
326        {
327            log.info("User database not found; creating from scratch...");
328        }
329        catch( IOException e )
330        {
331            log.error( "IO error: " + e.getMessage() );
332        }
333        if ( c_dom == null )
334        {
335            try
336            {
337                //
338                //  Create the DOM from scratch
339                //
340                c_dom = factory.newDocumentBuilder().newDocument();
341                c_dom.appendChild( c_dom.createElement( "users") );
342            }
343            catch( ParserConfigurationException e )
344            {
345                log.fatal( "Could not create in-memory DOM" );
346            }
347        }
348    }

350    private void saveDOM() throws WikiSecurityException
351    {
352        if ( c_dom == null )
353        {
354            log.fatal( "User database doesn't exist in memory." );
355        }
356
357        File newFile = new File( c_file.getAbsolutePath() + ".new" );
358        try
359        {
360            BufferedWriter io = new BufferedWriter( new OutputStreamWriter ( 
361                    new FileOutputStream( newFile ), "UTF-8" ) );
362            
363            // Write the file header and document root
364            io.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
365            io.write("<users>\n");
366            
367            // Write each profile as a <user> node
368            Element root = c_dom.getDocumentElement();
369            NodeList nodes = root.getElementsByTagName( USER_TAG );  
370            for( int i = 0; i < nodes.getLength(); i++ )
371            {
372                Element user = (Element)nodes.item( i );
373                io.write( "    <" + USER_TAG + " ");
374                io.write( UID );
375                io.write( "=\"" + user.getAttribute( UID ) + "\" " );
376                io.write( LOGIN_NAME );
377                io.write( "=\"" + user.getAttribute( LOGIN_NAME ) + "\" " );
378                io.write( WIKI_NAME );
379                io.write( "=\"" + user.getAttribute( WIKI_NAME ) + "\" " );
380                io.write( FULL_NAME );
381                io.write( "=\"" + user.getAttribute( FULL_NAME ) + "\" " );
382                io.write( EMAIL );
383                io.write( "=\"" + user.getAttribute( EMAIL ) + "\" " );
384                io.write( PASSWORD );
385                io.write( "=\"" + user.getAttribute( PASSWORD ) + "\" " );
386                io.write( CREATED );
387                io.write( "=\"" + user.getAttribute( CREATED ) + "\" " );
388                io.write( LAST_MODIFIED );
389                io.write( "=\"" + user.getAttribute( LAST_MODIFIED ) + "\" " );
390                io.write( LOCK_EXPIRY );
391                io.write( "=\"" + user.getAttribute( LOCK_EXPIRY ) + "\" " );
392                io.write( ">" );
393                NodeList attributes = user.getElementsByTagName( ATTRIBUTES_TAG );
394                for ( int j = 0; j < attributes.getLength(); j++ )
395                {
396                    Element attribute = (Element)attributes.item( j );
397                    String value = extractText( attribute );
398                    io.write( "\n        <" + ATTRIBUTES_TAG + ">" );
399                    io.write( value );
400                    io.write( "</" + ATTRIBUTES_TAG + ">" );
401                }
402                io.write("\n    </" +USER_TAG + ">\n");
403            }
404            io.write("</users>");
405            io.close();
406        }
407        catch ( IOException e )
408        {
409            throw new WikiSecurityException( e.getLocalizedMessage(), e );
410        }
411
412        // Copy new file over old version
413        File backup = new File( c_file.getAbsolutePath() + ".old" );
414        if ( backup.exists() )
415        {
416            if ( !backup.delete() )
417            {
418                log.error( "Could not delete old user database backup: " + backup );
419            }
420        }
421        if ( !c_file.renameTo( backup ) )
422        {
423            log.error( "Could not create user database backup: " + backup );
424        }
425        if ( !newFile.renameTo( c_file ) )
426        {
427            log.error( "Could not save database: " + backup + " restoring backup." );
428            if ( !backup.renameTo( c_file ) )
429            {
430                log.error( "Restore failed. Check the file permissions." );
431            }
432            log.error( "Could not save database: " + c_file + ". Check the file permissions" );
433        }
434    }

436    private long c_lastCheck    = 0;
437    private long c_lastModified = 0;
438    
439    private void checkForRefresh()
440    {
441        long time = System.currentTimeMillis();
442        
443        if( time - c_lastCheck > 60*1000L )
444        {
445            long lastModified = c_file.lastModified();
446            
447            if( lastModified > c_lastModified )
448            {
449                buildDOM();
450            }
451        }
452    }

454    /**
455     * @see org.apache.wiki.auth.user.UserDatabase#rename(String, String)
456     */
457    public synchronized void rename(String loginName, String newName) throws NoSuchPrincipalException, DuplicateUserException, WikiSecurityException
458    {
459        if ( c_dom == null )
460        {
461            log.fatal( "Could not rename profile '" + loginName + "'; database does not exist" );
462            throw new IllegalStateException( "FATAL: database does not exist" );
463        }
464        checkForRefresh();
465        
466        // Get the existing user; if not found, throws NoSuchPrincipalException
467        UserProfile profile = findByLoginName( loginName );
468        
469        // Get user with the proposed name; if found, it's a collision
470        try 
471        {
472            UserProfile otherProfile = findByLoginName( newName );
473            if ( otherProfile != null )
474            {
475                throw new DuplicateUserException( "security.error.cannot.rename", newName );
476            }
477        }
478        catch ( NoSuchPrincipalException e )
479        {
480            // Good! That means it's safe to save using the new name
481        }
482        
483        // Find the user with the old login id attribute, and change it
484        NodeList users = c_dom.getElementsByTagName( USER_TAG );
485        for( int i = 0; i < users.getLength(); i++ )
486        {
487            Element user = (Element) users.item( i );
488            if ( user.getAttribute( LOGIN_NAME ).equals( loginName ) )
489            {
490                DateFormat c_format = new SimpleDateFormat( DATE_FORMAT );
491                Date modDate = new Date( System.currentTimeMillis() );
492                setAttribute( user, LOGIN_NAME, newName );
493                setAttribute( user, LAST_MODIFIED, c_format.format( modDate ) );
494                profile.setLoginName( newName );
495                profile.setLastModified( modDate );
496                break;
497            }
498        }
499        
500        // Commit to disk
501        saveDOM();
502    }

504    /**
505     * Saves a {@link UserProfile}to the user database, overwriting the
506     * existing profile if it exists. The user name under which the profile
507     * should be saved is returned by the supplied profile's
508     * {@link UserProfile#getLoginName()}method.
509     * @param profile the user profile to save
510     * @throws WikiSecurityException if the profile cannot be saved
511     */
512    public synchronized void save( UserProfile profile ) throws WikiSecurityException
513    {
514        if ( c_dom == null )
515        {
516            log.fatal( "Could not save profile " + profile + " database does not exist" );
517            throw new IllegalStateException( "FATAL: database does not exist" );
518        }
519        
520        checkForRefresh();
521        
522        DateFormat c_format = new SimpleDateFormat( DATE_FORMAT );
523        String index = profile.getLoginName();
524        NodeList users = c_dom.getElementsByTagName( USER_TAG );
525        Element user = null;
526        for( int i = 0; i < users.getLength(); i++ )
527        {
528            Element currentUser = (Element) users.item( i );
529            if ( currentUser.getAttribute( LOGIN_NAME ).equals( index ) )
530            {
531                user = currentUser;
532                break;
533            }
534        }
535        
536        boolean isNew = false;
537        
538        Date modDate = new Date( System.currentTimeMillis() );
539        if( user == null )
540        {
541            // Create new user node
542            profile.setCreated( modDate );
543            log.info( "Creating new user " + index );
544            user = c_dom.createElement( USER_TAG );
545            c_dom.getDocumentElement().appendChild( user );
546            setAttribute( user, CREATED, c_format.format( profile.getCreated() ) );
547            isNew = true;
548        }
549        else
550        {
551            // To update existing user node, delete old attributes first...
552            NodeList attributes = user.getElementsByTagName( ATTRIBUTES_TAG );
553            for ( int i = 0; i < attributes.getLength(); i++ )
554            {
555                user.removeChild( attributes.item( i ) );
556            }
557        }
558        
559        setAttribute( user, UID, profile.getUid() );
560        setAttribute( user, LAST_MODIFIED, c_format.format( modDate ) );
561        setAttribute( user, LOGIN_NAME, profile.getLoginName() );
562        setAttribute( user, FULL_NAME, profile.getFullname() );
563        setAttribute( user, WIKI_NAME, profile.getWikiName() );
564        setAttribute( user, EMAIL, profile.getEmail() );
565        Date lockExpiry = profile.getLockExpiry();
566        setAttribute( user, LOCK_EXPIRY, lockExpiry == null ? "" : c_format.format( lockExpiry ) );
567
568        // Hash and save the new password if it's different from old one
569        String newPassword = profile.getPassword();
570        if ( newPassword != null && !newPassword.equals( "" ) )
571        {
572            String oldPassword = user.getAttribute( PASSWORD );
573            if ( !oldPassword.equals( newPassword ) )
574            {
575                setAttribute( user, PASSWORD, getHash( newPassword ) );
576            }
577        }
578        
579        // Save the attributes as as Base64 string
580        if ( profile.getAttributes().size() > 0 )
581        {
582            try
583            {
584                String encodedAttributes = Serializer.serializeToBase64( profile.getAttributes() );
585                Element attributes = c_dom.createElement( ATTRIBUTES_TAG );
586                user.appendChild( attributes );
587                Text value = c_dom.createTextNode( encodedAttributes );
588                attributes.appendChild( value );
589            }
590            catch ( IOException e )
591            {
592                throw new WikiSecurityException( "Could not save user profile attribute. Reason: " + e.getMessage(), e );
593            }
594        }
595
596        // Set the profile timestamps
597        if ( isNew )
598        {
599            profile.setCreated( modDate );
600        }
601        profile.setLastModified( modDate );
602        
603        // Commit to disk
604        saveDOM();
605    }

607    /**
608     * Private method that returns the first {@link UserProfile}matching a
609     * &lt;user&gt; element's supplied attribute. This method will also
610     * set the UID if it has not yet been set.
611     * @param matchAttribute
612     * @param index
613     * @return the profile, or <code>null</code> if not found
614     */
615    private UserProfile findByAttribute( String matchAttribute, String index )
616    {
617        if ( c_dom == null )
618        {
619            throw new IllegalStateException( "FATAL: database does not exist" );
620        }
621        
622        checkForRefresh();
623        
624        NodeList users = c_dom.getElementsByTagName( USER_TAG );
625        
626        if( users == null ) return null;
627
628        // check if we have to do a case insensitive compare
629        boolean caseSensitiveCompare = true;
630        if (matchAttribute.equals(EMAIL))
631        {
632            caseSensitiveCompare = false;
633        }
634
635        for( int i = 0; i < users.getLength(); i++ )
636        {
637            Element user = (Element) users.item( i );
638            String userAttribute = user.getAttribute( matchAttribute );
639            if (!caseSensitiveCompare)
640            {
641                userAttribute = StringUtils.lowerCase(userAttribute);
642                index = StringUtils.lowerCase(index);
643            }
644            if ( userAttribute.equals( index ) )
645            {
646                UserProfile profile = newProfile();
647                
648                // Parse basic attributes
649                profile.setUid( user.getAttribute( UID ) );
650                if ( profile.getUid() == null || profile.getUid().length() == 0 )
651                {
652                    profile.setUid( generateUid( this ) );
653                }
654                profile.setLoginName( user.getAttribute( LOGIN_NAME ) );
655                profile.setFullname( user.getAttribute( FULL_NAME ) );
656                profile.setPassword( user.getAttribute( PASSWORD ) );
657                profile.setEmail( user.getAttribute( EMAIL ) );
658                
659                // Get created/modified timestamps
660                String created = user.getAttribute( CREATED );
661                String modified = user.getAttribute( LAST_MODIFIED );
662                profile.setCreated( parseDate( profile, created ) );                  
663                profile.setLastModified( parseDate( profile, modified ) );                  
664                
665                // Is the profile locked?
666                String lockExpiry = user.getAttribute( LOCK_EXPIRY );
667                if ( lockExpiry == null || lockExpiry.length() == 0 )
668                {
669                    profile.setLockExpiry( null );
670                }
671                else
672                {
673                    profile.setLockExpiry( new Date( Long.parseLong( lockExpiry ) ) );
674                }
675                
676                // Extract all of the user's attributes (should only be one attributes tag, but you never know!)
677                NodeList attributes = user.getElementsByTagName( ATTRIBUTES_TAG );
678                for ( int j = 0; j < attributes.getLength(); j++ )
679                {
680                    Element attribute = (Element)attributes.item( j );
681                    String serializedMap = extractText( attribute );
682                    try
683                    {
684                        Map<String,? extends Serializable> map = Serializer.deserializeFromBase64( serializedMap );
685                        profile.getAttributes().putAll( map );
686                    }
687                    catch ( IOException e )
688                    {
689                        log.error( "Could not parse user profile attributes!", e );
690                    }
691                }
692
693                return profile;
694            }
695        }
696        return null;
697    }

699    /**
700     * Extracts all of the text nodes that are immediate children of an Element.
701     * @param element the base element
702     * @return the text nodes that are immediate children of the base element, concatenated together
703     */
704    private String extractText( Element element )
705    {
706        String text = "";
707        if ( element.getChildNodes().getLength() > 0 )
708        {
709            NodeList children = element.getChildNodes();
710            for ( int k = 0; k < children.getLength(); k++ )
711            {
712                Node child = children.item( k );
713                if ( child.getNodeType() == Node.TEXT_NODE )
714                {
715                    text = text + ((Text)child).getData();
716                }
717            }
718        }
719        return text;
720    }

722    /**
723     *  Tries to parse a date using the default format - then, for backwards
724     *  compatibility reasons, tries the platform default.
725     *  
726     *  @param profile
727     *  @param date
728     *  @return A parsed date, or null, if both parse attempts fail.
729     */
730    private Date parseDate( UserProfile profile, String date )
731    {
732        try
733        {
734            DateFormat c_format = new SimpleDateFormat( DATE_FORMAT );
735            return c_format.parse( date );
736        }
737        catch( ParseException e )
738        {
739            try
740            {
741                return DateFormat.getDateTimeInstance().parse( date );
742            }
743            catch ( ParseException e2)
744            {
745                log.warn("Could not parse 'created' or 'lastModified' "
746                    + "attribute for "
747                    + " profile '" + profile.getLoginName() + "'."
748                    + " It may have been tampered with." );
749            }            
750        }
751        return null;
752    }

754    /**
755     * After loading the DOM, this method sanity-checks the dates in the DOM and makes
756     * sure they are formatted properly. This is sort-of hacky, but it should work.
757     */
758    private void sanitizeDOM()
759    {
760        if ( c_dom == null )
761        {
762            throw new IllegalStateException( "FATAL: database does not exist" );
763        }
764        
765        NodeList users = c_dom.getElementsByTagName( USER_TAG );
766        for( int i = 0; i < users.getLength(); i++ )
767        {
768            Element user = (Element) users.item( i );
769            
770            // Sanitize UID (and generate a new one if one does not exist)
771            String uid = user.getAttribute( UID ).trim();
772            if ( uid == null || uid.length() == 0 || "-1".equals( uid ) )
773            {
774                uid = String.valueOf( generateUid( this ) );
775                user.setAttribute( UID, uid );
776            }
777            
778            // Sanitize dates
779            String loginName = user.getAttribute( LOGIN_NAME );
780            String created = user.getAttribute( CREATED );
781            String modified = user.getAttribute( LAST_MODIFIED );
782            DateFormat c_format = new SimpleDateFormat( DATE_FORMAT );
783            try
784            {
785                created = c_format.format( c_format.parse( created ) );
786                modified = c_format.format( c_format.parse( modified ) );
787                user.setAttribute( CREATED,  created );
788                user.setAttribute( LAST_MODIFIED,  modified );
789            }
790            catch( ParseException e )
791            {
792                try
793                {
794                    created = c_format.format( DateFormat.getDateTimeInstance().parse( created ) );
795                    modified = c_format.format( DateFormat.getDateTimeInstance().parse( modified ) );
796                    user.setAttribute( CREATED,  created );
797                    user.setAttribute( LAST_MODIFIED,  modified );
798                }
799                catch ( ParseException e2 )
800                {
801                    log.warn( "Could not parse 'created' or 'lastModified' attribute for profile '" + loginName + "'."
802                            + " It may have been tampered with." );
803                }            
804            }
805        }
806    }
