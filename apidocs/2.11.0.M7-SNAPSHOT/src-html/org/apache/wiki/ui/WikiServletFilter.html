019package org.apache.wiki.ui;
020
021import org.apache.log4j.Logger;
022import org.apache.log4j.NDC;
023import org.apache.wiki.WikiContext;
024import org.apache.wiki.WikiEngine;
025import org.apache.wiki.WikiSession;
026import org.apache.wiki.auth.SessionMonitor;
027import org.apache.wiki.auth.WikiSecurityException;
028
029import javax.servlet.Filter;
030import javax.servlet.FilterChain;
031import javax.servlet.FilterConfig;
032import javax.servlet.ServletContext;
033import javax.servlet.ServletException;
034import javax.servlet.ServletRequest;
035import javax.servlet.ServletResponse;
036import javax.servlet.http.HttpServletRequest;
037import javax.servlet.http.HttpServletRequestWrapper;
038import java.io.IOException;
039import java.io.PrintWriter;
040
041/**
042 * Filter that verifies that the {@link org.apache.wiki.WikiEngine} is running, and sets the authentication status for the user's
043 * WikiSession. Each HTTP request processed by this filter is wrapped by a {@link WikiRequestWrapper}. The wrapper's primary responsibility
044 * is to return the correct <code>userPrincipal</code> and <code>remoteUser</code> for authenticated JSPWiki users (whether authenticated
045 * by container or by JSPWiki's custom system). The wrapper's other responsibility is to incorporate JSPWiki built-in roles
046 * into the role-checking algorithm for {@link  HttpServletRequest#isUserInRole(String)}. Just before the request is wrapped, the method
047 * {@link org.apache.wiki.auth.AuthenticationManager#login(HttpServletRequest)} executes; this method contains all of the logic needed to
048 * grab any user login credentials set by the container or by cookies.
049 */
050public class WikiServletFilter implements Filter {
051
052    private static final Logger log = Logger.getLogger( WikiServletFilter.class );
053    protected WikiEngine m_engine = null;
054
055    /**
056     *  Creates a Wiki Servlet Filter.
057     */
058    public WikiServletFilter()
059    {
060        super();
061    }
062
063    /**
064     * Initializes the WikiServletFilter.
065     * 
066     * @param config The FilterConfig.
067     * @throws ServletException If a WikiEngine cannot be started.
068     */
069    public void init( final FilterConfig config ) throws ServletException {
070        final ServletContext context = config.getServletContext();
071
072        // TODO REMOVEME when resolving JSPWIKI-129
073        if( System.getSecurityManager() != null ) {
074            context.log( "== JSPWIKI WARNING ==   : This container is running with a security manager. JSPWiki does not yet really support that right now. See issue JSPWIKI-129 for details and information on how to proceed." );
075        }
076
077        m_engine = WikiEngine.getInstance( context, null );
078    }
079
080    /**
081     * Destroys the WikiServletFilter.
082     */
083    public void destroy() {
084    }
085
086    /**
087    * Checks that the WikiEngine is running ok, wraps the current HTTP request, and sets the correct authentication state for the users's
088    * WikiSession. First, the method {@link org.apache.wiki.auth.AuthenticationManager#login(HttpServletRequest)}
089    * executes, which sets the authentication state. Then, the request is wrapped with a
090    * {@link WikiRequestWrapper}.
091    * @param request the current HTTP request object
092    * @param response the current HTTP response object
093    * @param chain The Filter chain passed down.
094    * @throws ServletException if {@link org.apache.wiki.auth.AuthenticationManager#login(HttpServletRequest)} fails for any reason
095    * @throws IOException If writing to the servlet response fails. 
096    */
097    public void doFilter( final ServletRequest request, final ServletResponse response, final FilterChain chain ) throws IOException, ServletException {
098        //  Sanity check; it might be true in some conditions, but we need to know where.
099        if( chain == null ) {
100            throw new ServletException("FilterChain is null, even if it should not be.  Please report this to the jspwiki development team.");
101        }
102        
103        if( m_engine == null ) {
104            final PrintWriter out = response.getWriter();
105            out.print("<!DOCTYPE html><html lang=\"en\"><head><title>Fatal problem with JSPWiki</title></head>");
106            out.print("<body>");
107            out.print("<h1>JSPWiki has not been started</h1>");
108            out.print("<p>JSPWiki is not running.  This is probably due to a configuration error in your jspwiki.properties file, ");
109            out.print("or a problem with your servlet container.  Please double-check everything before issuing a bug report ");
110            out.print("at jspwiki.apache.org.</p>");
111            out.print("<p>We apologize for the inconvenience.  No, really, we do.  We're trying to ");
112            out.print("JSPWiki as easy as we can, but there is only so much we have time to test ");
113            out.print("platforms.</p>");
114            out.print( "<p>Please go to the <a href='Install.jsp'>installer</a> to continue.</p>" );
115            out.print("</body></html>");
116            return;
117        }   
118        
119        // If we haven't done so, wrap the request
120        HttpServletRequest httpRequest = ( HttpServletRequest )request;
121        
122        // Set the character encoding
123        httpRequest.setCharacterEncoding( m_engine.getContentEncoding().displayName() );
124        
125        if ( !isWrapped( request ) ) {
126            // Prepare the WikiSession
127            try {
128                m_engine.getAuthenticationManager().login( httpRequest );
129                final WikiSession wikiSession = SessionMonitor.getInstance( m_engine ).find( httpRequest.getSession() );
130                httpRequest = new WikiRequestWrapper( m_engine, httpRequest );
131                if ( log.isDebugEnabled() ) {
132                    log.debug( "Executed security filters for user=" + wikiSession.getLoginPrincipal().getName() + ", path=" + httpRequest.getRequestURI() );
133                }
134            } catch( final WikiSecurityException e ) {
135                throw new ServletException( e );
136            }
137        }
138
139        try {
140            NDC.push( m_engine.getApplicationName()+":"+httpRequest.getRequestURL() );
141            chain.doFilter( httpRequest, response );
142        } finally {
143            NDC.pop();
144            NDC.remove();
145        }
146
147    }
148
149    /**
150     *  Figures out the wiki context from the request.  This method does not create the context if it does not exist.
151     *  
152     *  @param request The request to examine
153     *  @return A valid WikiContext value (or null, if the context could not be located).
154     */
155    protected WikiContext getWikiContext( final ServletRequest request ) {
156        final HttpServletRequest httpRequest = (HttpServletRequest) request;
157        return (WikiContext) httpRequest.getAttribute( WikiContext.ATTR_CONTEXT );
158    }
159
160    /** 
161     * Determines whether the request has been previously wrapped with a WikiRequestWrapper. 
162     * We find the wrapper by recursively unwrapping successive request wrappers, if they have been supplied.
163     * @param request the current HTTP request
164     * @return <code>true</code> if the request has previously been wrapped;
165     * <code>false</code> otherwise
166     */
167    private boolean isWrapped( ServletRequest request ) {
168        while( !(request instanceof WikiRequestWrapper ) && request != null && request instanceof HttpServletRequestWrapper ) {
169            request = ( ( HttpServletRequestWrapper ) request ).getRequest();
170        }
171        return request instanceof WikiRequestWrapper;
172    }
