054/**
055 *  Manages searching the Wiki.
056 *
057 *  @since 2.2.21.
058 */
059public class DefaultSearchManager extends BasicPageFilter implements SearchManager {
060
061    private static final Logger log = Logger.getLogger( DefaultSearchManager.class );
062
063    private SearchProvider m_searchProvider;
064
065    /**
066     *  Creates a new SearchManager.
067     *
068     *  @param engine The Engine that owns this SearchManager.
069     *  @param properties The list of Properties.
070     *  @throws FilterException If it cannot be instantiated.
071     */
072    public DefaultSearchManager( final Engine engine, final Properties properties ) throws FilterException {
073        initialize( engine, properties );
074        WikiEventManager.getInstance().addWikiEventListener( m_engine.getManager( PageManager.class ), this );
075
076        // TODO: Replace with custom annotations. See JSPWIKI-566
077        WikiAjaxDispatcherServlet.registerServlet( JSON_SEARCH, new JSONSearch() );
078    }
079
080    /**
081     *  Provides a JSON AJAX API to the JSPWiki Search Engine.
082     */
083    public class JSONSearch implements WikiAjaxServlet {
084
085        public static final String AJAX_ACTION_SUGGESTIONS = "suggestions";
086        public static final String AJAX_ACTION_PAGES = "pages";
087        public static final int DEFAULT_MAX_RESULTS = 20;
088        public int maxResults = DEFAULT_MAX_RESULTS;
089
090        /** {@inheritDoc} */
091        @Override
092        public String getServletMapping() {
093            return JSON_SEARCH;
094        }
095
096        /** {@inheritDoc} */
097        @Override
098        public void service( final HttpServletRequest req,
099                             final HttpServletResponse resp,
100                             final String actionName,
101                             final List< String > params ) throws IOException {
102            String result = "";
103            if( StringUtils.isNotBlank( actionName ) ) {
104                if( params.size() < 1 ) {
105                    return;
106                }
107                final String itemId = params.get( 0 );
108                log.debug( "itemId=" + itemId );
109                if( params.size() > 1 ) {
110                    final String maxResultsParam = params.get( 1 );
111                    log.debug( "maxResultsParam=" + maxResultsParam );
112                    if( StringUtils.isNotBlank( maxResultsParam ) && StringUtils.isNumeric( maxResultsParam ) ) {
113                        maxResults = Integer.parseInt( maxResultsParam );
114                    }
115                }
116
117                if( actionName.equals( AJAX_ACTION_SUGGESTIONS ) ) {
118                    log.debug( "Calling getSuggestions() START" );
119                    final List< String > callResults = getSuggestions( itemId, maxResults );
120                    log.debug( "Calling getSuggestions() DONE. " + callResults.size() );
121                    result = AjaxUtil.toJson( callResults );
122                } else if( actionName.equals( AJAX_ACTION_PAGES ) ) {
123                    log.debug("Calling findPages() START");
124                    final WikiContext wikiContext = new WikiContext( m_engine, req, WikiContext.VIEW );
125                    final List< Map< String, Object > > callResults = findPages( itemId, maxResults, wikiContext );
126                    log.debug( "Calling findPages() DONE. " + callResults.size() );
127                    result = AjaxUtil.toJson( callResults );
128                }
129            }
130            log.debug( "result=" + result );
131            resp.getWriter().write( result );
132        }
133
134        /**
135         *  Provides a list of suggestions to use for a page name. Currently the algorithm just looks into the value parameter,
136         *  and returns all page names from that.
137         *
138         *  @param wikiName the page name
139         *  @param maxLength maximum number of suggestions
140         *  @return the suggestions
141         */
142        public List< String > getSuggestions( String wikiName, final int maxLength ) {
143            final StopWatch sw = new StopWatch();
144            sw.start();
145            final List< String > list = new ArrayList<>( maxLength );
146            if( wikiName.length() > 0 ) {
147                // split pagename and attachment filename
148                String filename = "";
149                final int pos = wikiName.indexOf("/");
150                if( pos >= 0 ) {
151                    filename = wikiName.substring( pos ).toLowerCase();
152                    wikiName = wikiName.substring( 0, pos );
153                }
154
155                final String cleanWikiName = MarkupParser.cleanLink(wikiName).toLowerCase() + filename;
156                final String oldStyleName = MarkupParser.wikifyLink(wikiName).toLowerCase() + filename;
157                final Set< String > allPages = m_engine.getManager( ReferenceManager.class ).findCreated();
158
159                int counter = 0;
160                for( final Iterator< String > i = allPages.iterator(); i.hasNext() && counter < maxLength; ) {
161                    final String p = i.next();
162                    final String pp = p.toLowerCase();
163                    if( pp.startsWith( cleanWikiName) || pp.startsWith( oldStyleName ) ) {
164                        list.add( p );
165                        counter++;
166                    }
167                }
168            }
169
170            sw.stop();
171            if( log.isDebugEnabled() ) {
172                log.debug( "Suggestion request for " + wikiName + " done in " + sw );
173            }
174            return list;
175        }
176
177        /**
178         *  Performs a full search of pages.
179         *
180         *  @param searchString The query string
181         *  @param maxLength How many hits to return
182         *  @return the pages found
183         */
184        public List< Map< String, Object > > findPages( final String searchString, final int maxLength, final WikiContext wikiContext ) {
185            final StopWatch sw = new StopWatch();
186            sw.start();
187
188            final List< Map< String, Object > > list = new ArrayList<>( maxLength );
189            if( searchString.length() > 0 ) {
190                try {
191                    final Collection< SearchResult > c;
192                    if( m_searchProvider instanceof LuceneSearchProvider ) {
193                        c = ( ( LuceneSearchProvider )m_searchProvider ).findPages( searchString, 0, wikiContext );
194                    } else {
195                        c = m_searchProvider.findPages( searchString, wikiContext );
196                    }
197
198                    int count = 0;
199                    for( final Iterator< SearchResult > i = c.iterator(); i.hasNext() && count < maxLength; count++ ) {
200                        final SearchResult sr = i.next();
201                        final HashMap< String, Object > hm = new HashMap<>();
202                        hm.put( "page", sr.getPage().getName() );
203                        hm.put( "score", sr.getScore() );
204                        list.add( hm );
205                    }
206                } catch( final Exception e ) {
207                    log.info( "AJAX search failed; ", e );
208                }
209            }
210
211            sw.stop();
212            if( log.isDebugEnabled() ) {
213                log.debug( "AJAX search complete in " + sw );
214            }
215            return list;
216        }
217    }

220    /** {@inheritDoc} */
221    @Override
222    public void initialize( final Engine engine, final Properties properties ) throws FilterException {
223        m_engine = engine;
224        loadSearchProvider(properties);
225
226        try {
227            m_searchProvider.initialize( engine, properties );
228        } catch( final NoRequiredPropertyException | IOException e ) {
229            log.error( e.getMessage(), e );
230        }
231    }
232
233    private void loadSearchProvider( final Properties properties ) {
234        // See if we're using Lucene, and if so, ensure that its index directory is up to date.
235        final String providerClassName = properties.getProperty( PROP_SEARCHPROVIDER, DEFAULT_SEARCHPROVIDER );
236
237        try {
238            final Class<?> providerClass = ClassUtil.findClass( "org.apache.wiki.search", providerClassName );
239            m_searchProvider = ( SearchProvider )providerClass.newInstance();
240        } catch( final ClassNotFoundException | InstantiationException | IllegalAccessException e ) {
241            log.warn("Failed loading SearchProvider, will use BasicSearchProvider.", e);
242        }
243
244        if( null == m_searchProvider ) {
245            // FIXME: Make a static with the default search provider
246            m_searchProvider = new BasicSearchProvider();
247        }
248        log.debug("Loaded search provider " + m_searchProvider);
249    }
250
251    /** {@inheritDoc} */
252    @Override
253    public SearchProvider getSearchEngine()
254    {
255        return m_searchProvider;
256    }
257
258    /** {@inheritDoc} */
259    @Override
260    public void actionPerformed( final WikiEvent event ) {
261        if( event instanceof WikiPageEvent && event.getType() == WikiPageEvent.PAGE_DELETE_REQUEST ) {
262            final String pageName = ( ( WikiPageEvent ) event ).getPageName();
263
264            final WikiPage p = m_engine.getManager( PageManager.class ).getPage( pageName );
265            if( p != null ) {
266                pageRemoved( p );
267            }
268        }
269    }
