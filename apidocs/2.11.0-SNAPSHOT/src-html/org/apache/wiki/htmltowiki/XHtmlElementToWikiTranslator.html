042/**
043 * Converting XHtml to Wiki Markup.  This is the class which does all of the heavy loading.
044 *
045 */
046public class XHtmlElementToWikiTranslator
047{

059    /**
060     *  Create a new translator using the default config.
061     *
062     *  @param base The base element from which to start translating.
063     *  @throws IOException If reading of the DOM tree fails.
064     *  @throws JDOMException If the DOM tree is faulty.
065     */
066    public XHtmlElementToWikiTranslator(final Element base ) throws IOException, JDOMException
067    {
068        this( base, new XHtmlToWikiConfig() );
069    }

071    /**
072     *  Create a new translator using the specified config.
073     *
074     *  @param base The base element from which to start translating.
075     *  @param config The config to use.
076     *  @throws IOException If reading of the DOM tree fails.
077     *  @throws JDOMException If the DOM tree is faulty.
078     */
079    public XHtmlElementToWikiTranslator(final Element base, final XHtmlToWikiConfig config ) throws IOException, JDOMException
080    {
081        this.m_config = config;
082        m_outTimmer = new WhitespaceTrimWriter();
083        m_out = new PrintWriter( m_outTimmer );
084        print( base );
085    }

097    private void print( String s )
098    {
099        s = StringEscapeUtils.unescapeHtml4( s );
100        m_out.print( s );
101    }

103    private void print(final Object element ) throws IOException, JDOMException
104    {
105        if( element instanceof Text )
106        {
107            final Text t = (Text)element;
108            String s = t.getText();
109            if( m_preStack.isPreMode() )
110            {
111                m_out.print( s );
112            }
113            else
114            {
115                // remove all "line terminator" characters
116                s = s.replaceAll( "[\\r\\n\\f\\u0085\\u2028\\u2029]", "" );
117                m_out.print( s );
118            }
119        }
120        else if( element instanceof Element )
121        {
122            final Element base = (Element)element;
123            String n = base.getName().toLowerCase();
124            if( "imageplugin".equals( base.getAttributeValue( "class" ) ) )
125            {
126                printImage( base );
127            }
128            else if( "wikiform".equals( base.getAttributeValue( "class" ) ) )
129            {
130                // only print the children if the div's class="wikiform", but not the div itself.
131                printChildren( base );
132            }
133            else
134            {
135                boolean bold = false;
136                boolean italic = false;
137                boolean monospace = false;
138                String cssSpecial = null;
139                final String cssClass = base.getAttributeValue( "class" );
140
141                // accomodate a FCKeditor bug with Firefox: when a link is removed, it becomes <span class="wikipage">text</span>.
142                final boolean ignoredCssClass = cssClass != null && cssClass.matches( "wikipage|createpage|external|interwiki|attachment|inline-code" );
143
144                Map< Object, Object > styleProps = null;
145
146                // Only get the styles if it's not a link element. Styles for link elements are
147                // handled as an AugmentedWikiLink instead.
148                if( !n.equals( "a" ) )
149                {
150                    styleProps = getStylePropertiesLowerCase( base );
151                }
152
153                if("inline-code".equals(cssClass))
154                {
155                    monospace = true;
156                }
157
158                if( styleProps != null )
159                {
160                    final String weight = (String)styleProps.remove( "font-weight" );
161                    final String style = (String)styleProps.remove( "font-style" );
162
163                    if( n.equals( "p" ) )
164                    {
165                        // change it so we can print out the css styles for <p>
166                        n = "div";
167                    }
168
169                    italic = "oblique".equals( style ) || "italic".equals( style );
170                    bold = "bold".equals( weight ) || "bolder".equals( weight );
171                    if( !styleProps.isEmpty() )
172                    {
173                        cssSpecial = propsToStyleString( styleProps );
174                    }
175                }
176                if( cssClass != null && !ignoredCssClass )
177                {
178                    if( n.equals( "div" ) )
179                    {
180                        m_out.print( "\n%%" + cssClass + " \n" );
181                    }
182                    else if( n.equals( "span" ) )
183                    {
184                        m_out.print( "%%" + cssClass + " " );
185                    }
186                }
187                if( bold )
188                {
189                    m_out.print( "__" );
190                }
191                if( italic )
192                {
193                    m_out.print( "''" );
194                }
195                if( monospace )
196                {
197                    m_out.print( "{{{" );
198                    m_preStack.push();
199                }
200                if( cssSpecial != null )
201                {
202                    if( n.equals( "div" ) )
203                    {
204                        m_out.print( "\n%%(" + cssSpecial + " )\n" );
205                    }
206                    else
207                    {
208                        m_out.print( "%%(" + cssSpecial + " )" );
209                    }
210                }
211                printChildren( base );
212                if( cssSpecial != null )
213                {
214                    if( n.equals( "div" ) )
215                    {
216                        m_out.print( "\n/%\n" );
217                    }
218                    else
219                    {
220                        m_out.print( "/%" );
221                    }
222                }
223                if( monospace )
224                {
225                    m_preStack.pop();
226                    m_out.print( "}}}" );
227                }
228                if( italic )
229                {
230                    m_out.print( "''" );
231                }
232                if( bold )
233                {
234                    m_out.print( "__" );
235                }
236                if( cssClass != null && !ignoredCssClass )
237                {
238                    if( n.equals( "div" ) )
239                    {
240                        m_out.print( "\n/%\n" );
241                    }
242                    else if( n.equals( "span" ) )
243                    {
244                        m_out.print( "/%" );
245                    }
246                }
247            }
248        }
249    }

251    private void printChildren(final Element base ) throws IOException, JDOMException
252    {
253        for(final Iterator< Content > i = base.getContent().iterator(); i.hasNext(); )
254        {
255            final Content c = i.next();
256            if( c instanceof Element )
257            {
258                final Element e = (Element)c;
259                final String n = e.getName().toLowerCase();
260                switch ( n ) {
261                    case "h1":
262                        m_out.print( "\n!!! " );
263                        print( e );
264                        m_out.println();
265                        break;
266                    case "h2":
267                        m_out.print( "\n!!! " );
268                        print( e );
269                        m_out.println();
270                        break;
271                    case "h3":
272                        m_out.print( "\n!! " );
273                        print( e );
274                        m_out.println();
275                        break;
276                    case "h4":
277                        m_out.print( "\n! " );
278                        print( e );
279                        m_out.println();
280                        break;
281                    case "p":
282                        if ( e.getContentSize() != 0 ) // we don't want to print empty elements: <p></p>
283                        {
284                            m_out.println();
285                            print( e );
286                            m_out.println();
287                        }
288                        break;
289                    case "br":
290                        if ( m_preStack.isPreMode() ) {
291                            m_out.println();
292                        } else {
293                            final String parentElementName = base.getName().toLowerCase();
294
295                            //
296                            // To beautify the generated wiki markup, we print a newline character after a linebreak.
297                            // It's only safe to do this when the parent element is a <p> or <div>; when the parent
298                            // element is a table cell or list item, a newline character would break the markup.
299                            // We also check that this isn't being done inside a plugin body.
300                            //
301                            if ( parentElementName.matches( "p|div" )
302                                    && !base.getText().matches( "(?s).*\\[\\{.*\\}\\].*" ) ) {
303                                m_out.print( " \\\\\n" );
304                            } else {
305                                m_out.print( " \\\\" );
306                            }
307                        }
308                        print( e );
309                        break;
310                    case "hr":
311                        m_out.println();
312                        print( "----" );
313                        print( e );
314                        m_out.println();
315                        break;
316                    case "table":
317                        if ( !m_outTimmer.isCurrentlyOnLineBegin() ) {
318                            m_out.println();
319                        }
320                        print( e );
321                        break;
322                    case "tr":
323                        print( e );
324                        m_out.println();
325                        break;
326                    case "td":
327                        m_out.print( "| " );
328                        print( e );
329                        if ( !m_preStack.isPreMode() ) {
330                            print( " " );
331                        }
332                        break;
333                    case "th":
334                        m_out.print( "|| " );
335                        print( e );
336                        if ( !m_preStack.isPreMode() ) {
337                            print( " " );
338                        }
339                        break;
340                    case "a":
341                        if ( !isIgnorableWikiMarkupLink( e ) ) {
342                            if ( e.getChild( "IMG" ) != null ) {
343                                printImage( e );
344                            } else {
345                                String ref = e.getAttributeValue( "href" );
346                                if ( ref == null ) {
347                                    if ( isUndefinedPageLink( e ) ) {
348                                        m_out.print( "[" );
349                                        print( e );
350                                        m_out.print( "]" );
351                                    } else {
352                                        print( e );
353                                    }
354                                } else {
355                                    ref = trimLink( ref );
356                                    if ( ref != null ) {
357                                        if ( ref.startsWith( "#" ) ) // This is a link to a footnote.
358                                        {
359                                            // convert "#ref-PageName-1" to just "1"
360                                            final String href = ref.replaceFirst( "#ref-.+-(\\d+)", "$1" );
361
362                                            // remove the brackets around "[1]"
363                                            final String textValue = e.getValue().substring( 1, ( e.getValue().length() - 1 ) );
364
365                                            if ( href.equals( textValue ) ) { // handles the simplest case. Example: [1]
366                                                print( e );
367                                            } else { // handles the case where the link text is different from the href. Example: [something|1]
368                                                m_out.print( "[" + textValue + "|" + href + "]" );
369                                            }
370                                        } else {
371                                            final Map<String, String> augmentedWikiLinkAttributes = getAugmentedWikiLinkAttributes( e );
372
373                                            m_out.print( "[" );
374                                            print( e );
375                                            if ( !e.getTextTrim().equalsIgnoreCase( ref ) ) {
376                                                m_out.print( "|" );
377                                                print( ref );
378
379                                                if ( !augmentedWikiLinkAttributes.isEmpty() ) {
380                                                    m_out.print( "|" );
381
382                                                    final String augmentedWikiLink = augmentedWikiLinkMapToString( augmentedWikiLinkAttributes );
383                                                    m_out.print( augmentedWikiLink );
384                                                }
385                                            } else if ( !augmentedWikiLinkAttributes.isEmpty() ) {
386                                                // If the ref has the same value as the text and also if there
387                                                // are attributes, then just print: [ref|ref|attributes] .
388                                                m_out.print( "|" + ref + "|" );
389                                                final String augmentedWikiLink = augmentedWikiLinkMapToString( augmentedWikiLinkAttributes );
390                                                m_out.print( augmentedWikiLink );
391                                            }
392
393                                            m_out.print( "]" );
394                                        }
395                                    }
396                                }
397                            }
398                        }
399                        break;
400                    case "b":
401                    case "strong":
402                        m_out.print( "__" );
403                        print( e );
404                        m_out.print( "__" );
405                        break;
406                    case "i":
407                    case "em":
408                    case "address":
409                        m_out.print( "''" );
410                        print( e );
411                        m_out.print( "''" );
412                        break;
413                    case "u":
414                        m_out.print( "%%( text-decoration:underline; )" );
415                        print( e );
416                        m_out.print( "/%" );
417                        break;
418                    case "strike":
419                        m_out.print( "%%strike " );
420                        print( e );
421                        m_out.print( "/%" );
422                        // NOTE: don't print a space before or after the double percents because that can break words into two.
423                        // For example: %%(color:red)ABC%%%%(color:green)DEF%% is different from %%(color:red)ABC%% %%(color:green)DEF%%
424                        break;
425                    case "sup":
426                        m_out.print( "%%sup " );
427                        print( e );
428                        m_out.print( "/%" );
429                        break;
430                    case "sub":
431                        m_out.print( "%%sub " );
432                        print( e );
433                        m_out.print( "/%" );
434                        break;
435                    case "dl":
436                        m_out.print( "\n" );
437                        print( e );
438
439                        // print a newline after the definition list. If we don't,
440                        // it may cause problems for the subsequent element.
441                        m_out.print( "\n" );
442                        break;
443                    case "dt":
444                        m_out.print( ";" );
445                        print( e );
446                        break;
447                    case "dd":
448                        m_out.print( ":" );
449                        print( e );
450                        break;
451                    case "ul":
452                        m_out.println();
453                        m_liStack.push( "*" );
454                        print( e );
455                        m_liStack.pop();
456                        break;
457                    case "ol":
458                        m_out.println();
459                        m_liStack.push( "#" );
460                        print( e );
461                        m_liStack.pop();
462                        break;
463                    case "li":
464                        m_out.print( m_liStack + " " );
465                        print( e );
466
467                        // The following line assumes that the XHTML has been "pretty-printed"
468                        // (newlines separate child elements from their parents).
469                        final boolean lastListItem = base.indexOf( e ) == ( base.getContentSize() - 2 );
470                        final boolean sublistItem = m_liStack.toString().length() > 1;
471
472                        // only print a newline if this <li> element is not the last item within a sublist.
473                        if ( !sublistItem || !lastListItem ) {
474                            m_out.println();
475                        }
476                        break;
477                    case "pre":
478                        m_out.print( "\n{{{" ); // start JSPWiki "code blocks" on its own line
479
480                        m_preStack.push();
481                        print( e );
482                        m_preStack.pop();
483
484                        // print a newline after the closing braces
485                        // to avoid breaking any subsequent wiki markup that follows.
486                        m_out.print( "}}}\n" );
487                        break;
488                    case "code":
489                    case "tt":
490                        m_out.print( "{{" );
491                        m_preStack.push();
492                        print( e );
493                        m_preStack.pop();
494                        m_out.print( "}}" );
495                        // NOTE: don't print a newline after the closing brackets because if the Text is inside
496                        // a table or list, it would break it if there was a subsequent row or list item.
497                        break;
498                    case "img":
499                        if ( !isIgnorableWikiMarkupLink( e ) ) {
500                            m_out.print( "[" );
501                            print( trimLink( e.getAttributeValue( "src" ) ) );
502                            m_out.print( "]" );
503                        }
504                        break;
505                    case "form": {
506                        // remove the hidden input where name="formname" since a new one will be generated again when the xhtml is rendered.
507                        final Element formName = getXPathElement( e, "INPUT[@name='formname']" );
508                        if ( formName != null ) {
509                            formName.detach();
510                        }
511
512                        final String name = e.getAttributeValue( "name" );
513
514                        m_out.print( "\n[{FormOpen" );
515
516                        if ( name != null ) {
517                            m_out.print( " form='" + name + "'" );
518                        }
519
520                        m_out.print( "}]\n" );
521
522                        print( e );
523                        m_out.print( "\n[{FormClose}]\n" );
524                        break;
525                    }
526                    case "input": {
527                        final String type = e.getAttributeValue( "type" );
528                        String name = e.getAttributeValue( "name" );
529                        final String value = e.getAttributeValue( "value" );
530                        final String checked = e.getAttributeValue( "checked" );
531
532                        m_out.print( "[{FormInput" );
533
534                        if ( type != null ) {
535                            m_out.print( " type='" + type + "'" );
536                        }
537                        if ( name != null ) {
538                            // remove the "nbf_" that was prepended since new one will be generated again when the xhtml is rendered.
539                            if ( name.startsWith( "nbf_" ) ) {
540                                name = name.substring( 4, name.length() );
541                            }
542                            m_out.print( " name='" + name + "'" );
543                        }
544                        if ( value != null && !value.equals( "" ) ) {
545                            m_out.print( " value='" + value + "'" );
546                        }
547                        if ( checked != null ) {
548                            m_out.print( " checked='" + checked + "'" );
549                        }
550
551                        m_out.print( "}]" );
552
553                        print( e );
554                        break;
555                    }
556                    case "textarea": {
557                        String name = e.getAttributeValue( "name" );
558                        final String rows = e.getAttributeValue( "rows" );
559                        final String cols = e.getAttributeValue( "cols" );
560
561                        m_out.print( "[{FormTextarea" );
562
563                        if ( name != null ) {
564                            if ( name.startsWith( "nbf_" ) ) {
565                                name = name.substring( 4, name.length() );
566                            }
567                            m_out.print( " name='" + name + "'" );
568                        }
569                        if ( rows != null ) {
570                            m_out.print( " rows='" + rows + "'" );
571                        }
572                        if ( cols != null ) {
573                            m_out.print( " cols='" + cols + "'" );
574                        }
575
576                        m_out.print( "}]" );
577                        print( e );
578                        break;
579                    }
580                    case "select": {
581                        String name = e.getAttributeValue( "name" );
582
583                        m_out.print( "[{FormSelect" );
584
585                        if ( name != null ) {
586                            if ( name.startsWith( "nbf_" ) ) {
587                                name = name.substring( 4, name.length() );
588                            }
589                            m_out.print( " name='" + name + "'" );
590                        }
591
592                        m_out.print( " value='" );
593                        print( e );
594                        m_out.print( "'}]" );
595                        break;
596                    }
597                    case "option": {
598                        // If this <option> element isn't the second child element within the parent <select>
599                        // element, then we need to print a semicolon as a separator. (The first child element
600                        // is expected to be a newline character which is at index of 0).
601                        if ( base.indexOf( e ) != 1 ) {
602                            m_out.print( ";" );
603                        }
604
605                        final Attribute selected = e.getAttribute( "selected" );
606                        if ( selected != null ) {
607                            m_out.print( "*" );
608                        }
609
610                        final String value = e.getAttributeValue( "value" );
611                        if ( value != null ) {
612                            m_out.print( value );
613                        } else {
614                            print( e );
615                        }
616                        break;
617                    }
618                    default:
619                        print( e );
620                        break;
621                }
622            }
623            else
624            {
625                print( c );
626            }
627        }
628    }

630    private void printImage(final Element base )
631    {
632        Element child = getXPathElement( base, "TBODY/TR/TD/*" );
633        if( child == null )
634        {
635            child = base;
636        }
637        final Element img;
638        final String href;
639        final Map<Object,Object> map = new ForgetNullValuesLinkedHashMap<>();
640        if( child.getName().equals( "A" ) )
641        {
642            img = child.getChild( "IMG" );
643            href = child.getAttributeValue( "href" );
644        }
645        else
646        {
647            img = child;
648            href = null;
649        }
650        if( img == null )
651        {
652            return;
653        }
654        final String src = trimLink( img.getAttributeValue( "src" ) );
655        if( src == null )
656        {
657            return;
658        }
659        map.put( "align", base.getAttributeValue( "align" ) );
660        map.put( "height", img.getAttributeValue( "height" ) );
661        map.put( "width", img.getAttributeValue( "width" ) );
662        map.put( "alt", img.getAttributeValue( "alt" ) );
663        map.put( "caption", emptyToNull( ( Element )XPathFactory.instance().compile(  "CAPTION" ).evaluateFirst( base ) ) );
664        map.put( "link", href );
665        map.put( "border", img.getAttributeValue( "border" ) );
666        map.put( "style", base.getAttributeValue( "style" ) );
667        if( map.size() > 0 )
668        {
669            m_out.print( "[{Image src='" + src + "'" );
670            for(final Iterator i = map.entrySet().iterator(); i.hasNext(); )
671            {
672                final Map.Entry entry = (Map.Entry)i.next();
673                if( !entry.getValue().equals( "" ) )
674                {
675                    m_out.print( " " + entry.getKey() + "='" + entry.getValue() + "'" );
676                }
677            }
678            m_out.print( "}]" );
679        }
680        else
681        {
682            m_out.print( "[" + src + "]" );
683        }
684    }

686    Element getXPathElement( final Element base, final String expression ) {
687        final List< ? > nodes = XPathFactory.instance().compile( expression ).evaluate( base );
688        if( nodes == null || nodes.size() == 0 ) {
689            return null;
690        } else {
691            return ( Element )nodes.get( 0 );
692        }
693    }

695    private String emptyToNull( final Element e ) {
696        if( e == null ) {
697            return null;
698        }
699        final String s = e.getText();
700        return s == null ? null : ( s.replaceAll( "\\s", "" ).isEmpty() ? null : s );
701    }

703    private String propsToStyleString( final Map< Object, Object >  styleProps ) {
704        final StringBuilder style = new StringBuilder();
705        for(final Iterator< Map.Entry< Object, Object > > i = styleProps.entrySet().iterator(); i.hasNext(); ) {
706            final Map.Entry< Object, Object > entry = i.next();
707            style.append( " " ).append( entry.getKey() ).append( ": " ).append( entry.getValue() ).append( ";" );
708        }
709        return style.toString();
710    }

712    private boolean isIgnorableWikiMarkupLink( final Element a ) {
713        final String ref = a.getAttributeValue( "href" );
714        final String clazz = a.getAttributeValue( "class" );
715        return ( ref != null && ref.startsWith( m_config.getPageInfoJsp() ) )
716            || ( clazz != null && clazz.trim().equalsIgnoreCase( m_config.getOutlink() ) );
717    }

719    /**
720     * Checks if the link points to an undefined page.
721     */
722    private boolean isUndefinedPageLink(final Element a)
723    {
724        final String classVal = a.getAttributeValue( "class" );
725
726        return "createpage".equals( classVal );
727    }

824    /**
825     * Converts the entries in the map to a string for use in a wiki link.
826     */
827    private String augmentedWikiLinkMapToString(final Map attributesMap )
828    {
829        final StringBuilder sb = new StringBuilder();
830
831        for (final Iterator itr = attributesMap.entrySet().iterator(); itr.hasNext(); )
832        {
833            final Map.Entry entry = (Map.Entry)itr.next();
834            final String attributeName = (String)entry.getKey();
835            final String attributeValue = (String)entry.getValue();
836
837            sb.append( " " + attributeName + "='" + attributeValue + "'" );
838        }
839
840        return sb.toString().trim();
841    }

843    private Map< Object, Object > getStylePropertiesLowerCase(final Element base ) throws IOException
844    {
845        final String n = base.getName().toLowerCase();
846
847        //"font-weight: bold; font-style: italic;"
848        String style = base.getAttributeValue( "style" );
849        if( style == null )
850        {
851            style = "";
852        }
853
854        if( n.equals( "p" ) || n.equals( "div" ) )
855        {
856            final String align = base.getAttributeValue( "align" );
857            if( align != null )
858            {
859                // only add the value of the align attribute if the text-align style didn't already exist.
860                if( style.indexOf( "text-align" ) == -1 )
861                {
862                    style = style + ";text-align:" + align + ";";
863                }
864            }
865        }
866
867
868
869        if( n.equals( "font" ) )
870        {
871            final String color = base.getAttributeValue( "color" );
872            final String face = base.getAttributeValue( "face" );
873            final String size = base.getAttributeValue( "size" );
874            if( color != null )
875            {
876                style = style + "color:" + color + ";";
877            }
878            if( face != null )
879            {
880                style = style + "font-family:" + face + ";";
881            }
882            if( size != null )
883            {
884                switch ( size ) {
885                    case "1":
886                        style = style + "font-size:xx-small;";
887                        break;
888                    case "2":
889                        style = style + "font-size:x-small;";
890                        break;
891                    case "3":
892                        style = style + "font-size:small;";
893                        break;
894                    case "4":
895                        style = style + "font-size:medium;";
896                        break;
897                    case "5":
898                        style = style + "font-size:large;";
899                        break;
900                    case "6":
901                        style = style + "font-size:x-large;";
902                        break;
903                    case "7":
904                        style = style + "font-size:xx-large;";
905                        break;
906                }
907            }
908        }
909
910        if( style.equals( "" ) )
911        {
912            return null;
913        }
914
915        style = style.replace( ';', '\n' ).toLowerCase();
916        final LinkedHashMap< Object, Object > m = new LinkedHashMap<>();
917        new PersistentMapDecorator( m ).load( new ByteArrayInputStream( style.getBytes() ) );
918        return m;
919    }

921    private String trimLink( String ref )
922    {
923        if( ref == null )
924        {
925            return null;
926        }
927        try
928        {
929            ref = URLDecoder.decode( ref, StandardCharsets.UTF_8.name() );
930            ref = ref.trim();
931            if( ref.startsWith( m_config.getAttachPage() ) )
932            {
933                ref = ref.substring( m_config.getAttachPage().length() );
934            }
935            if( ref.startsWith( m_config.getWikiJspPage() ) )
936            {
937                ref = ref.substring( m_config.getWikiJspPage().length() );
938
939                // Handle links with section anchors.
940                // For example, we need to translate the html string "TargetPage#section-TargetPage-Heading2"
941                // to this wiki string "TargetPage#Heading2".
942                ref = ref.replaceFirst( ".+#section-(.+)-(.+)", "$1#$2" );
943            }
944            if( ref.startsWith( m_config.getEditJspPage() ) )
945            {
946                ref = ref.substring( m_config.getEditJspPage().length() );
947            }
948            if( m_config.getPageName() != null )
949            {
950                if( ref.startsWith( m_config.getPageName() ) )
951                {
952                    ref = ref.substring( m_config.getPageName().length() );
953                }
954            }
955        }
956        catch ( final UnsupportedEncodingException e )
957        {
958            // Shouldn't happen...
959        }
960        return ref;
961    }
