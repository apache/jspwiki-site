019package org.apache.wiki.parser;
020
021import org.apache.commons.text.StringEscapeUtils;
022import org.apache.wiki.WikiContext;
023import org.apache.wiki.api.exceptions.NoSuchVariableException;
024import org.apache.wiki.variables.VariableManager;
025import org.jdom2.Text;
026
027/**
028 *  Stores the contents of a WikiVariable in a WikiDocument DOM tree.
029 *  <p>
030 *  When the WikiDocument is rendered, if the {@link WikiContext#VAR_WYSIWYG_EDITOR_MODE} is set to {@link Boolean#TRUE}, the
031 *  variable declaration is rendered instead of the variable value.
032 *
033 *  @since  2.4
034 */
035public class VariableContent extends Text {
036
037    private static final long serialVersionUID = 1L;
038
039    private String m_varName;
040    
041    /**
042     *  Create a VariableContent for the given variable.
043     *  
044     *  @param varName The name of the variable.
045     */
046    public VariableContent( final String varName )
047    {
048        m_varName = varName;
049    }
050    
051    /**
052     *   Evaluates the variable and returns the contents. 
053     *   
054     *   @return The rendered value of the variable.
055     */
056    @Override public String getValue() {
057        String result;
058        final WikiDocument root = (WikiDocument) getDocument();
059
060        if( root == null ) {
061            // See similar note in PluginContent
062            return m_varName;
063        }
064        
065        final WikiContext context = root.getContext();
066        if( context == null ) {
067            return "No WikiContext available: INTERNAL ERROR";
068        }
069    
070        final Boolean wysiwygEditorMode = ( Boolean )context.getVariable( WikiContext.VAR_WYSIWYG_EDITOR_MODE );
071        if( wysiwygEditorMode != null && wysiwygEditorMode ) {
072            result = "[" + m_varName + "]";
073        } else {
074            try {
075                result = context.getEngine().getManager( VariableManager.class ).parseAndGetValue( context, m_varName );
076            } catch( final NoSuchVariableException e ) {
077                result = MarkupParser.makeError( "No such variable: " + e.getMessage() ).getText(); 
078            }
079        }
080
081        return StringEscapeUtils.escapeXml11( result );
082    }
083    
084    /**
085     *  Returns exactly getValue().
086     *  @return Whatever getValue() returns.
087     */
088    @Override public String getText() {
089        return getValue();
090    }
091
092    /**
093     *  Returns a debug-suitable string.
094     *  @return Debug string
095     */
096    @Override public String toString() {
097        return "VariableElement[\"" + m_varName + "\"]";
098    }
099
100}
