019package org.apache.wiki.htmltowiki.syntax;
020
021import org.apache.wiki.htmltowiki.XHtmlElementToWikiTranslator;
022import org.jdom2.JDOMException;
023
024import java.io.PrintWriter;
025import java.util.Stack;
026
027
028/**
029 * Translates to wiki syntax from a plain text handling css classes.
030 */
031public abstract class PlainTextCssDecorator {
032
033    final protected PrintWriter out;
034    final protected Stack< String > preStack;
035    final protected XHtmlElementToWikiTranslator chain;
036    final protected PlainTextBoldDecorator ptbd;
037
038    protected PlainTextCssDecorator( final PlainTextBoldDecorator ptbd, final PrintWriter out, final Stack< String > preStack, final XHtmlElementToWikiTranslator chain ) {
039        this.out = out;
040        this.preStack = preStack;
041        this.chain = chain;
042        this.ptbd = ptbd;
043    }
044
045    /**
046     * Translates the given XHTML element into wiki markup.
047     *
048     * @param dto XHTML element being translated.
049     */
050    public void decorate( final XHtmlElementToWikiTranslator.ElementDecoratorData dto ) throws JDOMException {
051        if( dto.cssClass != null && !dto.ignoredCssClass ) {
052            if( MarkupHelper.isHtmlBaseDiv( dto ) ) {
053                out.print( markupCssDivOpen( dto.cssClass ) );
054            } else if( MarkupHelper.isHtmlBaseSpan( dto ) ) {
055                out.print( markupCssSpanOpen( dto.cssClass ) );
056            }
057        }
058        ptbd.decorate( dto );
059        if( dto.cssClass != null && !dto.ignoredCssClass ) {
060            if( MarkupHelper.isHtmlBaseDiv( dto ) ) {
061                out.print( markupCssDivClose( dto.cssClass ) );
062            } else if( MarkupHelper.isHtmlBaseSpan( dto ) ) {
063                out.print( markupCssSpanClose( dto.cssClass ) );
064            }
065        }
066    }
067
068    /**
069     * Opening wiki markup for a css element.
070     *
071     * @return Opening wiki markup for a css element.
072     */
073    protected abstract String markupCssDivOpen( String cssClass );
074
075    /**
076     * Closing wiki markup for a css element.
077     *
078     * @return Closing wiki markup for a css element.
079     */
080    protected abstract String markupCssDivClose( String cssClass );
081
082    /**
083     * Opening wiki markup for a css element.
084     *
085     * @return Opening wiki markup for a css element.
086     */
087    protected abstract String markupCssSpanOpen( String cssClass );
088
089    /**
090     * Closing wiki markup for a css element.
091     *
092     * @return Closing wiki markup for a css element.
093     */
094    protected abstract String markupCssSpanClose( String cssClass );
095
096}
