034/**
035 *  Simple wrapper class for the Wiki page attributes.  The Wiki page content is moved around in Strings, though.
036 */
037
038// FIXME: We need to rethink how metadata is being used - probably the 
039//        author, date, etc. should also be part of the metadata.  We also
040//        need to figure out the metadata lifecycle.
041
042public class WikiPage implements Page {
043
044    private       String     m_name;
045    private       Engine     m_engine;
046    private       String     m_wiki;
047    private Date             m_lastModified;
048    private long             m_fileSize = -1;
049    private int              m_version = PageProvider.LATEST_VERSION;
050    private String           m_author = null;
051    private final Map<String,Object> m_attributes = new HashMap<>();
052    
053    private Acl m_accessList = null;
054    
055    /**
056     *  Create a new WikiPage using a given engine and name.
057     *  
058     *  @param engine The Engine that owns this page.
059     *  @param name   The name of the page.
060     */
061    public WikiPage( final Engine engine, final String name ) {
062        m_engine = engine;
063        m_name = name;
064        m_wiki = engine.getApplicationName();
065    }
066
067    /**
068     *  Returns the name of the page.
069     *  
070     *  @return The page name.
071     */
072    @Override
073    public String getName()
074    {
075        return m_name;
076    }
077
078    /**
079     *  A WikiPage may have a number of attributes, which might or might not be 
080     *  available.  Typically attributes are things that do not need to be stored
081     *  with the wiki page to the page repository, but are generated
082     *  on-the-fly.  A provider is not required to save them, but they
083     *  can do that if they really want.
084     *
085     *  @param key The key using which the attribute is fetched
086     *  @return The attribute.  If the attribute has not been set, returns null.
087     */
088    @Override
089    @SuppressWarnings( "unchecked" )
090    public < T > T getAttribute( final String key )
091    {
092        return ( T )m_attributes.get( key );
093    }
094
095    /**
096     *  Sets an metadata attribute.
097     *  
098     *  @see #getAttribute(String)
099     *  @param key The key for the attribute used to fetch the attribute later on.
100     *  @param attribute The attribute value
101     */
102    @Override
103    public void setAttribute( final String key, final Object attribute )
104    {
105        m_attributes.put( key, attribute );
106    }
107
108    /**
109     * Returns the full attributes Map, in case external code needs
110     * to iterate through the attributes.
111     * 
112     * @return The attribute Map.  Please note that this is a direct
113     *         reference, not a copy.
114     */
115    @Override
116    public Map< String, Object > getAttributes()
117    {
118        return m_attributes;
119    }
120
121    /**
122     *  Removes an attribute from the page, if it exists.
123     *  
124     *  @param  key The key for the attribute
125     *  @return If the attribute existed, returns the object.
126     *  @since 2.1.111
127     */
128    @Override
129    @SuppressWarnings( "unchecked" )
130    public < T > T removeAttribute( final String key )
131    {
132        return ( T )m_attributes.remove( key );
133    }

206     *  Returns the Acl for this page.  May return <code>null</code>, in case there is no Acl defined, or it has not yet been set by

216    /**
217     * Sets the Acl for this page. Note that method does <em>not</em> persist the Acl itself to back-end storage or in page markup;

225        m_accessList = acl;
226    }

229     *  Sets the author of the page.  Typically called only by the provider.

231     *  @param author The author name.

233    @Override

236        m_author = author;
237    }

240     *  Returns author name, or null, if no author has been defined.

244    @Override
245    public String getAuthor()
246    {
247        return m_author;
248    }

250    /**
251     *  Returns the wiki name for this page
252     *  
253     *  @return The name of the wiki.
254     */
255    @Override
256    public String getWiki()
257    {
258        return m_wiki;
259    }
260
261    /**
262     *  This method will remove all metadata from the page.
263     */
264    @Override
265    public void invalidateMetadata() {
266        m_hasMetadata = false;
267        setAcl( null );
268        m_attributes.clear();
269    }
270
271    private boolean m_hasMetadata = false;
272
273    /**
274     *  Returns <code>true</code> if the page has valid metadata; that is, it has been parsed. Note that this method is a kludge to
275     *  support our pre-3.0 metadata system, and as such will go away with the new API.
276     *  
277     *  @return true, if the page has metadata.
278     */
279    @Override
280    public boolean hasMetadata()
281    {
282        return m_hasMetadata;
283    }
284
285    /**
286     *  Sets the metadata flag to true.  Never call.
287     */
288    @Override
289    public void setHasMetadata()
290    {
291        m_hasMetadata = true;
292    }

294    /**
295     *  Returns a debug-suitable version of the page.
296     *  
297     *  @return A debug string.
298     */
299    @Override
300    public String toString()
301    {
302        return "WikiPage [" + m_wiki + ":" + m_name + ",ver=" + m_version + ",mod=" + m_lastModified + "]";
303    }
304
305    /**
306     *  Creates a deep clone of a WikiPage.  Strings are not cloned, since they're immutable.  Attributes are not cloned, only the internal
307     *  HashMap (so if you modify the contents of a value of an attribute, these will reflect back to everyone).
308     *  
309     *  @return A deep clone of the WikiPage
310     */
311    @Override
312    public WikiPage clone() {
313        final WikiPage p = new WikiPage( m_engine, m_name );
314       
315        p.m_wiki         = m_wiki;
316            
317        p.m_author       = m_author;
318        p.m_version      = m_version;
319        p.m_lastModified = m_lastModified != null ? (Date)m_lastModified.clone() : null;
320
321        p.m_fileSize     = m_fileSize;
322
323        for( final Map.Entry< String, Object > entry : m_attributes.entrySet() ) {
324            p.m_attributes.put( entry.getKey(), entry.getValue() );
325        }
326
327        if( m_accessList != null ) {
328            p.m_accessList = new AclImpl();
329            for( final Enumeration< AclEntry > entries = m_accessList.entries(); entries.hasMoreElements(); ) {
330                final AclEntry e = entries.nextElement();
331                p.m_accessList.addEntry( e );
332            }
333        }
334            
335        return p;
336    }
337    
338    /**
339     *  Compares a page with another by name using the defined PageNameComparator.  If the same name, compares their versions.
340     *  
341     *  @param page The page to compare against
342     *  @return -1, 0 or 1
343     */
344    @Override
345    public int compareTo( final Page page ) {
346        if( this == page ) {
347            return 0; // the same object
348        }
349
350        int res = m_engine.getManager( PageManager.class ).getPageSorter().compare( this.getName(), page.getName() );
351        if( res == 0 ) {
352            res = this.getVersion() - page.getVersion();
353        }
354        return res;
355    }
