040/**
041 *  Manages variables.  Variables are case-insensitive.  A list of all available variables is on a Wiki page called "WikiVariables".

052    static final String[] THE_BIG_NO_NO_LIST = {
053        "jspwiki.auth.masterpassword"
054    };

063    /**
064     *  Parses the link and finds a value.  This is essentially used once
065     *  {@link LinkParsingOperations#isVariableLink(String)} has found that
066     *  the link text actually contains a variable.  For example, you could
067     *  pass in "{$username}" and get back "JanneJalkanen".
068     *
069     *  @param  context The WikiContext
070     *  @param  link    The link text containing the variable name.
071     *  @return The variable value.
072     *  @throws IllegalArgumentException If the format is not valid (does not start with "{$", is zero length, etc.)
073     *  @throws NoSuchVariableException If a variable is not known.
074     */
075    public String parseAndGetValue( final WikiContext context, final String link ) throws IllegalArgumentException, NoSuchVariableException {
076        if( !link.startsWith( "{$" ) ) {
077            throw new IllegalArgumentException( "Link does not start with {$" );
078        }
079        if( !link.endsWith( "}" ) ) {
080            throw new IllegalArgumentException( "Link does not end with }" );
081        }
082        final String varName = link.substring( 2, link.length() - 1 );
083
084        return getValue( context, varName.trim() );
085    }

087    /**
088     *  This method does in-place expansion of any variables.  However, the expansion is not done twice, that is,
089     *  a variable containing text $variable will not be expanded.
090     *  <P>
091     *  The variables should be in the same format ({$variablename} as in the web pages.
092     *
093     *  @param context The WikiContext of the current page.
094     *  @param source  The source string.
095     *  @return The source string with variables expanded.
096     */
097    // FIXME: somewhat slow.
098    public String expandVariables( final WikiContext context, final String source ) {
099        final StringBuilder result = new StringBuilder();
100        for( int i = 0; i < source.length(); i++ ) {
101            if( source.charAt(i) == '{' ) {
102                if( i < source.length()-2 && source.charAt(i+1) == '$' ) {
103                    final int end = source.indexOf( '}', i );
104
105                    if( end != -1 ) {
106                        final String varname = source.substring( i+2, end );
107                        String value;
108
109                        try {
110                            value = getValue( context, varname );
111                        } catch( final NoSuchVariableException | IllegalArgumentException e ) {
112                            value = e.getMessage();
113                        }
114
115                        result.append( value );
116                        i = end;
117                    }
118                } else {
119                    result.append( '{' );
120                }
121            } else {
122                result.append( source.charAt(i) );
123            }
124        }
125
126        return result.toString();
127    }

129    /**
130     *  Returns the value of a named variable.  See {@link #getValue(WikiContext, String)}. The only difference is that
131     *  this method does not throw an exception, but it returns the given default value instead.
132     *
133     *  @param context WikiContext
134     *  @param varName The name of the variable
135     *  @param defValue A default value.
136     *  @return The variable value, or if not found, the default value.
137     */
138    public String getValue( final WikiContext context, final String varName, final String defValue ) {
139        try {
140            return getValue( context, varName );
141        } catch( final NoSuchVariableException e ) {
142            return defValue;
143        }
144    }

146    /**
147     *  Shortcut to getValue(). However, this method does not throw a NoSuchVariableException, but returns null
148     *  in case the variable does not exist.
149     *
150     *  @param context WikiContext to look the variable in
151     *  @param name Name of the variable to look for
152     *  @return Variable value, or null, if there is no such variable.
153     *  @since 2.2 on WikiEngine, moved to VariableManager on 2.11.0
154     */
155    public String getVariable( final WikiContext context, final String name ) {
156        return getValue( context, name, null );
157    }

159    /**
160     *  Returns a value of the named variable.  The resolving order is
161     *  <ol>
162     *    <li>Known "constant" name, such as "pagename", etc.  This is so
163     *        that pages could not override certain constants.
164     *    <li>WikiContext local variable.  This allows a programmer to
165     *        set a parameter which cannot be overridden by user.
166     *    <li>HTTP Session
167     *    <li>HTTP Request parameters
168     *    <li>WikiPage variable.  As set by the user with the SET directive.
169     *    <li>jspwiki.properties
170     *  </ol>
171     *
172     *  Use this method only whenever you really need to have a parameter that
173     *  can be overridden by anyone using the wiki.
174     *
175     *  @param context The WikiContext
176     *  @param varName Name of the variable.
177     *
178     *  @return The variable value.
179     *
180     *  @throws IllegalArgumentException If the name is somehow broken.
181     *  @throws NoSuchVariableException If a variable is not known.
182     */
183    public String getValue( final WikiContext context, final String varName ) throws IllegalArgumentException, NoSuchVariableException {
184        if( varName == null ) {
185            throw new IllegalArgumentException( "Null variable name." );
186        }
187        if( varName.length() == 0 ) {
188            throw new IllegalArgumentException( "Zero length variable name." );
189        }
190        // Faster than doing equalsIgnoreCase()
191        final String name = varName.toLowerCase();
192
193        for( final String value : THE_BIG_NO_NO_LIST ) {
194            if( name.equals( value ) ) {
195                return ""; // FIXME: Should this be something different?
196            }
197        }
198
199        try {
200            //
201            //  Using reflection to get system variables adding a new system variable
202            //  now only involves creating a new method in the SystemVariables class
203            //  with a name starting with get and the first character of the name of
204            //  the variable capitalized. Example:
205            //    public String getMysysvar(){
206            //      return "Hello World";
207            //    }
208            //
209            final SystemVariables sysvars = new SystemVariables( context );
210            final String methodName = "get" + Character.toUpperCase( name.charAt( 0 ) ) + name.substring( 1 );
211            final Method method = sysvars.getClass().getMethod( methodName );
212            return ( String )method.invoke( sysvars );
213        } catch( final NoSuchMethodException e1 ) {
214            //
215            //  It is not a system var. Time to handle the other cases.
216            //
217            //  Check if such a context variable exists, returning its string representation.
218            //
219            if( ( context.getVariable( varName ) ) != null ) {
220                return context.getVariable( varName ).toString();
221            }
222
223            //
224            //  Well, I guess it wasn't a final straw.  We also allow variables from the session and the request (in this order).
225            //
226            final HttpServletRequest req = context.getHttpRequest();
227            if( req != null && req.getSession() != null ) {
228                final HttpSession session = req.getSession();
229
230                try {
231                    String s = ( String )session.getAttribute( varName );
232
233                    if( s != null ) {
234                        return s;
235                    }
236
237                    s = context.getHttpParameter( varName );
238                    if( s != null ) {
239                        return s;
240                    }
241                } catch( final ClassCastException e ) {
242                    log.debug( "Not a String: " + varName );
243                }
244            }
245
246            //
247            // And the final straw: see if the current page has named metadata.
248            //
249            final WikiPage pg = context.getPage();
250            if( pg != null ) {
251                final Object metadata = pg.getAttribute( varName );
252                if( metadata != null ) {
253                    return metadata.toString();
254                }
255            }
256
257            //
258            // And the final straw part 2: see if the "real" current page has named metadata. This allows
259            // a parent page to control a inserted page through defining variables
260            //
261            final WikiPage rpg = context.getRealPage();
262            if( rpg != null ) {
263                final Object metadata = rpg.getAttribute( varName );
264                if( metadata != null ) {
265                    return metadata.toString();
266                }
267            }
268
269            //
270            // Next-to-final straw: attempt to fetch using property name. We don't allow fetching any other
271            // properties than those starting with "jspwiki.".  I know my own code, but I can't vouch for bugs
272            // in other people's code... :-)
273            //
274            if( varName.startsWith("jspwiki.") ) {
275                final Properties props = context.getEngine().getWikiProperties();
276                final String s = props.getProperty( varName );
277                if( s != null ) {
278                    return s;
279                }
280            }
281
282            //
283            //  Final defaults for some known quantities.
284            //
285            if( varName.equals( VAR_ERROR ) || varName.equals( VAR_MSG ) ) {
286                return "";
287            }
288
289            throw new NoSuchVariableException( "No variable " + varName + " defined." );
290        } catch( final Exception e ) {
291            log.info("Interesting exception: cannot fetch variable value", e );
292        }
293        return "";
294    }

296    /**
297     *  This class provides the implementation for the different system variables.
298     *  It is called via Reflection - any access to a variable called $xxx is mapped
299     *  to getXxx() on this class.
300     *  <p>
301     *  This is a lot neater than using a huge if-else if branching structure
302     *  that we used to have before.
303     *  <p>
304     *  Note that since we are case insensitive for variables, and VariableManager
305     *  calls var.toLowerCase(), the getters for the variables do not have
306     *  capitalization anywhere.  This may look a bit odd, but then again, this
307     *  is not meant to be a public class.
308     *
309     *  @since 2.7.0
310     */
311    @SuppressWarnings( "unused" )
312    private static class SystemVariables
313    {
314        private final WikiContext m_context;
315
316        public SystemVariables( final WikiContext context )
317        {
318            m_context=context;
319        }
320
321        public String getPagename()
322        {
323            return m_context.getPage().getName();
324        }
325
326        public String getApplicationname()
327        {
328            return m_context.getEngine().getApplicationName();
329        }
330
331        public String getJspwikiversion()
332        {
333            return Release.getVersionString();
334        }
335
336        public String getEncoding() {
337            return m_context.getEngine().getContentEncoding().displayName();
338        }
339
340        public String getTotalpages() {
341            return Integer.toString( m_context.getEngine().getPageCount() );
342        }
343
344        public String getPageprovider()
345        {
346            return m_context.getEngine().getCurrentProvider();
347        }
348
349        public String getPageproviderdescription()
350        {
351            return m_context.getEngine().getCurrentProviderInfo();
352        }
353
354        public String getAttachmentprovider() {
355            final WikiProvider p = m_context.getEngine().getAttachmentManager().getCurrentProvider();
356            return (p != null) ? p.getClass().getName() : "-";
357        }
358
359        public String getAttachmentproviderdescription() {
360            final WikiProvider p = m_context.getEngine().getAttachmentManager().getCurrentProvider();
361            return (p != null) ? p.getProviderInfo() : "-";
362        }
363
364        public String getInterwikilinks() {
365            final StringBuilder res = new StringBuilder();
366
367            for( final String link : m_context.getEngine().getAllInterWikiLinks() ) {
368                if( res.length() > 0 ) {
369                    res.append( ", " );
370                }
371                res.append( link );
372                res.append( " --> " );
373                res.append( m_context.getEngine().getInterWikiURL( link ) );
374            }
375            return res.toString();
376        }
377
378        public String getInlinedimages() {
379            final StringBuilder res = new StringBuilder();
380            for( final String ptrn : m_context.getEngine().getAllInlinedImagePatterns() ) {
381                if( res.length() > 0 ) {
382                    res.append( ", " );
383                }
384
385                res.append( ptrn );
386            }
387
388            return res.toString();
389        }
390
391        public String getPluginpath() {
392            final String s = m_context.getEngine().getPluginManager().getPluginSearchPath();
393
394            return ( s == null ) ? "-" : s;
395        }
396
397        public String getBaseurl()
398        {
399            return m_context.getEngine().getBaseURL();
400        }
401
402        public String getUptime() {
403            final Date now = new Date();
404            long secondsRunning = ( now.getTime() - m_context.getEngine().getStartTime().getTime() ) / 1_000L;
405
406            final long seconds = secondsRunning % 60;
407            final long minutes = (secondsRunning /= 60) % 60;
408            final long hours = (secondsRunning /= 60) % 24;
409            final long days = secondsRunning /= 24;
410
411            return days + "d, " + hours + "h " + minutes + "m " + seconds + "s";
412        }
413
414        public String getLoginstatus() {
415            final WikiSession session = m_context.getWikiSession();
416            return Preferences.getBundle( m_context, InternationalizationManager.CORE_BUNDLE ).getString( "varmgr." + session.getStatus() );
417        }
418
419        public String getUsername() {
420            final Principal wup = m_context.getCurrentUser();
421            final ResourceBundle rb = Preferences.getBundle( m_context, InternationalizationManager.CORE_BUNDLE );
422            return wup != null ? wup.getName() : rb.getString( "varmgr.not.logged.in" );
423        }
424
425        public String getRequestcontext()
426        {
427            return m_context.getRequestContext();
428        }
429
430        public String getPagefilters() {
431            final FilterManager fm = m_context.getEngine().getFilterManager();
432            final List< PageFilter > filters = fm.getFilterList();
433            final StringBuilder sb = new StringBuilder();
434            for( final PageFilter pf : filters ) {
435                final String f = pf.getClass().getName();
436                if( pf instanceof InternalModule ) {
437                    continue;
438                }
439
440                if( sb.length() > 0 ) {
441                    sb.append( ", " );
442                }
443                sb.append( f );
444            }
445            return sb.toString();
446        }
447    }
