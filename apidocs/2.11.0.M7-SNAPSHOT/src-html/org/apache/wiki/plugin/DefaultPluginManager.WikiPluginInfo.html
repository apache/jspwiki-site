068/**
069 *  Manages plugin classes.  There exists a single instance of PluginManager
070 *  per each instance of Engine, that is, each JSPWiki instance.
071 *  <P>
072 *  A plugin is defined to have three parts:
073 *  <OL>
074 *    <li>The plugin class
075 *    <li>The plugin parameters
076 *    <li>The plugin body
077 *  </ol>
078 *
079 *  For example, in the following line of code:
080 *  <pre>
081 *  [{INSERT org.apache.wiki.plugin.FunnyPlugin  foo='bar'
082 *  blob='goo'
083 *
084 *  abcdefghijklmnopqrstuvw
085 *  01234567890}]
086 *  </pre>
087 *
088 *  The plugin class is "org.apache.wiki.plugin.FunnyPlugin", the
089 *  parameters are "foo" and "blob" (having values "bar" and "goo",
090 *  respectively), and the plugin body is then
091 *  "abcdefghijklmnopqrstuvw\n01234567890".   The plugin body is
092 *  accessible via a special parameter called "_body".
093 *  <p>
094 *  If the parameter "debug" is set to "true" for the plugin,
095 *  JSPWiki will output debugging information directly to the page if there
096 *  is an exception.
097 *  <P>
098 *  The class name can be shortened, and marked without the package.
099 *  For example, "FunnyPlugin" would be expanded to
100 *  "org.apache.wiki.plugin.FunnyPlugin" automatically.  It is also
101 *  possible to define other packages, by setting the
102 *  "jspwiki.plugin.searchPath" property.  See the included
103 *  jspwiki.properties file for examples.
104 *  <P>
105 *  Even though the nominal way of writing the plugin is
106 *  <pre>
107 *  [{INSERT pluginclass WHERE param1=value1...}],
108 *  </pre>
109 *  it is possible to shorten this quite a lot, by skipping the
110 *  INSERT, and WHERE words, and dropping the package name.  For
111 *  example:
112 *
113 *  <pre>
114 *  [{INSERT org.apache.wiki.plugin.Counter WHERE name='foo'}]
115 *  </pre>
116 *
117 *  is the same as
118 *  <pre>
119 *  [{Counter name='foo'}]
120 *  </pre>
121 *  <h3>Plugin property files</h3>
122 *  <p>
123 *  Since 2.3.25 you can also define a generic plugin XML properties file per
124 *  each JAR file.
125 *  <pre>
126 *  <modules>
127 *   <plugin class="org.apache.wiki.foo.TestPlugin">
128 *       <author>Janne Jalkanen</author>
129 *       <script>foo.js</script>
130 *       <stylesheet>foo.css</stylesheet>
131 *       <alias>code</alias>
132 *   </plugin>
133 *   <plugin class="org.apache.wiki.foo.TestPlugin2">
134 *       <author>Janne Jalkanen</author>
135 *   </plugin>
136 *   </modules>
137 *  </pre>
138 *  <h3>Plugin lifecycle</h3>
139 *
140 *  <p>Plugin can implement multiple interfaces to let JSPWiki know at which stages they should
141 *  be invoked:
142 *  <ul>
143 *  <li>InitializablePlugin: If your plugin implements this interface, the initialize()-method is
144 *      called once for this class
145 *      before any actual execute() methods are called.  You should use the initialize() for e.g.
146 *      precalculating things.  But notice that this method is really called only once during the
147 *      entire Engine lifetime.  The InitializablePlugin is available from 2.5.30 onwards.</li>
148 *  <li>ParserStagePlugin: If you implement this interface, the executeParse() method is called
149 *      when JSPWiki is forming the DOM tree.  You will receive an incomplete DOM tree, as well
150 *      as the regular parameters.  However, since JSPWiki caches the DOM tree to speed up later
151 *      places, which means that whatever this method returns would be irrelevant.  You can do some DOM
152 *      tree manipulation, though.  The ParserStagePlugin is available from 2.5.30 onwards.</li>
153 *  <li>Plugin: The regular kind of plugin which is executed at every rendering stage.  Each
154 *      new page load is guaranteed to invoke the plugin, unlike with the ParserStagePlugins.</li>
155 *  </ul>
156 *
157 *  @since 1.6.1
158 */

173    /**
174     *  Create a new PluginManager.
175     *
176     *  @param engine Engine which owns this manager.
177     *  @param props Contents of a "jspwiki.properties" file.
178     */
179    public DefaultPluginManager( final Engine engine, final Properties props ) {
180        super( engine );
181        final String packageNames = props.getProperty( PROP_SEARCHPATH );
182        if ( packageNames != null ) {
183            final StringTokenizer tok = new StringTokenizer( packageNames, "," );
184            while( tok.hasMoreTokens() ) {
185                m_searchPath.add( tok.nextToken().trim() );
186            }
187        }
188
189        final String externalJars = props.getProperty( PROP_EXTERNALJARS );
190        if( externalJars != null ) {
191            final StringTokenizer tok = new StringTokenizer( externalJars, "," );
192            while( tok.hasMoreTokens() ) {
193                m_externalJars.add( tok.nextToken().trim() );
194            }
195        }
196
197        registerPlugins();
198
199        //  The default packages are always added.
200        m_searchPath.add( DEFAULT_PACKAGE );
201        m_searchPath.add( DEFAULT_FORMS_PACKAGE );
202
203        final PatternCompiler compiler = new Perl5Compiler();
204        try {
205            m_pluginPattern = compiler.compile( PLUGIN_INSERT_PATTERN );
206        } catch( final MalformedPatternException e ) {
207            log.fatal( "Internal error: someone messed with pluginmanager patterns.", e );
208            throw new InternalWikiException( "PluginManager patterns are broken" , e );
209        }
210    }

247    /** Outputs a HTML-formatted version of a stack trace. */
248    private String stackTrace( final Map<String,String> params, final Throwable t ) {
249        final Element div = XhtmlUtil.element( XHTML.div, "Plugin execution failed, stack trace follows:" );
250        div.setAttribute( XHTML.ATTR_class, "debug" );
251
252        final StringWriter out = new StringWriter();
253        t.printStackTrace( new PrintWriter( out ) );
254        div.addContent( XhtmlUtil.element( XHTML.pre, out.toString() ) );
255        div.addContent( XhtmlUtil.element( XHTML.b, "Parameters to the plugin" ) );
256
257        final Element list = XhtmlUtil.element( XHTML.ul );
258        for( final Map.Entry< String, String > e : params.entrySet() ) {
259            final String key = e.getKey();
260            list.addContent( XhtmlUtil.element( XHTML.li, key + "'='" + e.getValue() ) );
261        }
262        div.addContent( list );
263        return XhtmlUtil.serialize( div );
264    }

266    /** {@inheritDoc} */
267    @Override
268    public String execute( final Context context, final String classname, final Map< String, String > params ) throws PluginException {
269        if( !m_pluginsEnabled ) {
270            return "";
271        }
272
273        final ResourceBundle rb = Preferences.getBundle( context, Plugin.CORE_PLUGINS_RESOURCEBUNDLE );
274        final boolean debug = TextUtil.isPositive( params.get( PARAM_DEBUG ) );
275        try {
276            //   Create...
277            final Plugin plugin = newWikiPlugin( classname, rb );
278            if( plugin == null ) {
279                return "Plugin '" + classname + "' not compatible with this version of JSPWiki";
280            }
281
282            //  ...and launch.
283            try {
284                return plugin.execute( context, params );
285            } catch( final PluginException e ) {
286                if( debug ) {
287                    return stackTrace( params, e );
288                }
289
290                // Just pass this exception onward.
291                throw ( PluginException )e.fillInStackTrace();
292            } catch( final Throwable t ) {
293                // But all others get captured here.
294                log.info( "Plugin failed while executing:", t );
295                if( debug ) {
296                    return stackTrace( params, t );
297                }
298
299                throw new PluginException( rb.getString( "plugin.error.failed" ), t );
300            }
301
302        } catch( final ClassCastException e ) {
303            throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.notawikiplugin" ), classname ), e );
304        }
305    }

307    /** {@inheritDoc} */
308    @Override
309    public Map< String, String > parseArgs( final String argstring ) throws IOException {
310        final Map< String, String > arglist = new HashMap<>();
311        //  Protection against funny users.
312        if( argstring == null ) {
313            return arglist;
314        }
315
316        arglist.put( PARAM_CMDLINE, argstring );
317        final StringReader in = new StringReader( argstring );
318        final StreamTokenizer tok = new StreamTokenizer( in );
319        tok.eolIsSignificant( true );
320
321        String param = null;
322        String value;
323        boolean potentialEmptyLine = false;
324        boolean quit = false;
325        while( !quit ) {
326            final String s;
327            final int type = tok.nextToken();
328
329            switch( type ) {
330            case StreamTokenizer.TT_EOF:
331                quit = true;
332                s = null;
333                break;
334
335            case StreamTokenizer.TT_WORD:
336                s = tok.sval;
337                potentialEmptyLine = false;
338                break;
339
340            case StreamTokenizer.TT_EOL:
341                quit = potentialEmptyLine;
342                potentialEmptyLine = true;
343                s = null;
344                break;
345
346            case StreamTokenizer.TT_NUMBER:
347                s = Integer.toString( ( int )tok.nval );
348                potentialEmptyLine = false;
349                break;
350
351            case '\'':
352                s = tok.sval;
353                break;
354
355            default:
356                s = null;
357            }
358
359            //  Assume that alternate words on the line are parameter and value, respectively.
360            if( s != null ) {
361                if( param == null ) {
362                    param = s;
363                } else {
364                    value = s;
365                    arglist.put( param, value );
366                    param = null;
367                }
368            }
369        }
370
371        //  Now, we'll check the body.
372        if( potentialEmptyLine ) {
373            final StringWriter out = new StringWriter();
374            FileUtil.copyContents( in, out );
375            final String bodyContent = out.toString();
376            if( bodyContent != null ) {
377                arglist.put( PARAM_BODY, bodyContent );
378            }
379        }
380
381        return arglist;
382    }

384    /** {@inheritDoc} */
385    @Override
386    public String execute( final Context context, final String commandline ) throws PluginException {
387        if( !m_pluginsEnabled ) {
388            return "";
389        }
390
391        final ResourceBundle rb = Preferences.getBundle( context, Plugin.CORE_PLUGINS_RESOURCEBUNDLE );
392        final PatternMatcher matcher = new Perl5Matcher();
393
394        try {
395            if( matcher.contains( commandline, m_pluginPattern ) ) {
396                final MatchResult res = matcher.getMatch();
397                final String plugin = res.group( 2 );
398                final int endIndex = commandline.length() - ( commandline.charAt( commandline.length() - 1 ) == '}' ? 1 : 0 );
399                final String args = commandline.substring( res.endOffset( 0 ), endIndex );
400                final Map< String, String > arglist = parseArgs( args );
401                return execute( context, plugin, arglist );
402            }
403        } catch( final NoSuchElementException e ) {
404            final String msg =  "Missing parameter in plugin definition: " + commandline;
405            log.warn( msg, e );
406            throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.missingparameter" ), commandline ) );
407        } catch( final IOException e ) {
408            final String msg = "Zyrf.  Problems with parsing arguments: " + commandline;
409            log.warn( msg, e );
410            throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.parsingarguments" ), commandline ) );
411        }
412
413        // FIXME: We could either return an empty string "", or the original line.  If we want unsuccessful requests
414        // to be invisible, then we should return an empty string.
415        return commandline;
416    }

418    /** Register a plugin. */
419    private void registerPlugin( final WikiPluginInfo pluginClass ) {
420        String name;
421
422        // Register the plugin with the className without the package-part
423        name = pluginClass.getName();
424        if( name != null ) {
425            log.debug( "Registering plugin [name]: " + name );
426            m_pluginClassMap.put( name, pluginClass );
427        }
428
429        // Register the plugin with a short convenient name.
430        name = pluginClass.getAlias();
431        if( name != null ) {
432            log.debug( "Registering plugin [shortName]: " + name );
433            m_pluginClassMap.put( name, pluginClass );
434        }
435
436        // Register the plugin with the className with the package-part
437        name = pluginClass.getClassName();
438        if( name != null ) {
439            log.debug( "Registering plugin [className]: " + name );
440            m_pluginClassMap.put( name, pluginClass );
441        }
442
443        pluginClass.initializePlugin( pluginClass, m_engine, m_searchPath, m_externalJars );
444    }
445
446    private void registerPlugins() {
447        // Register all plugins which have created a resource containing its properties.
448        log.info( "Registering plugins" );
449        final List< Element > plugins = XmlUtil.parse( PLUGIN_RESOURCE_LOCATION, "/modules/plugin" );
450
451        // Get all resources of all plugins.
452        for( final Element pluginEl : plugins ) {
453            final String className = pluginEl.getAttributeValue( "class" );
454            final WikiPluginInfo pluginInfo = WikiPluginInfo.newInstance( className, pluginEl ,m_searchPath, m_externalJars );
455            if( pluginInfo != null ) {
456                registerPlugin( pluginInfo );
457            }
458        }
459    }

468        private String    m_className;
469        private String    m_alias;
470        private String    m_ajaxAlias;
471        private Class<?>  m_clazz;

475        /**
476         *  Creates a new plugin info object which can be used to access a plugin.
477         *
478         *  @param className Either a fully qualified class name, or a "short" name which is then checked against the internal list of plugin packages.
479         *  @param el A JDOM Element containing the information about this class.
480         *  @param searchPath A List of Strings, containing different package names.
481         *  @param externalJars the list of external jars to search
482         *  @return A WikiPluginInfo object.
483         */
484        protected static WikiPluginInfo newInstance( final String className, final Element el, final List<String> searchPath, final List<String> externalJars ) {
485            if( className == null || className.length() == 0 ) {
486                return null;
487            }
488
489            final WikiPluginInfo info = new WikiPluginInfo( className );
490            info.initializeFromXML( el );
491            return info;
492        }

494        /**
495         *  Initializes a plugin, if it has not yet been initialized. If the plugin extends {@link HttpServlet} it will automatically
496         *  register it as AJAX using {@link WikiAjaxDispatcherServlet#registerServlet(String, WikiAjaxServlet)}.
497         *
498         *  @param engine The Engine
499         *  @param searchPath A List of Strings, containing different package names.
500         *  @param externalJars the list of external jars to search
501         */
502        protected void initializePlugin( final WikiPluginInfo info, final Engine engine , final List<String> searchPath, final List<String> externalJars) {
503            if( !m_initialized ) {
504                // This makes sure we only try once per class, even if init fails.
505                m_initialized = true;
506
507                try {
508                    final Plugin p = newPluginInstance(searchPath, externalJars);
509                    if( p instanceof InitializablePlugin ) {
510                        ( ( InitializablePlugin )p ).initialize( engine );
511                    }
512                    if( p instanceof WikiAjaxServlet ) {
513                        WikiAjaxDispatcherServlet.registerServlet( (WikiAjaxServlet) p );
514                        final String ajaxAlias = info.getAjaxAlias();
515                        if (StringUtils.isNotBlank(ajaxAlias)) {
516                            WikiAjaxDispatcherServlet.registerServlet( info.getAjaxAlias(), (WikiAjaxServlet) p );
517                        }
518                    }
519                } catch( final Exception e ) {
520                    log.info( "Cannot initialize plugin " + m_className, e );
521                }
522            }
523        }

584         *  @return A new plugin.

598        /**
599         *  Returns a text for IncludeResources.
600         *
601         *  @param type Either "script" or "stylesheet"
602         *  @return Text, or an empty string, if there is nothing to be included.
603         */
604        public String getIncludeText( final String type ) {
605            try {
606                if( "script".equals( type ) ) {
607                    return getScriptText();
608                } else if( "stylesheet".equals( type ) ) {
609                    return getStylesheetText();
610                }
611            } catch( final Exception ex ) {
612                // We want to fail gracefully here
613                return ex.getMessage();
614            }
615
616            return null;
617        }
618
619        private String getScriptText() throws IOException {
620            if( m_scriptText != null ) {
621                return m_scriptText;
622            }
623
624            if( m_scriptLocation == null ) {
625                return "";
626            }
627
628            try {
629                m_scriptText = getTextResource(m_scriptLocation);
630            } catch( final IOException ex ) {
631                // Only throw this exception once!
632                m_scriptText = "";
633                throw ex;
634            }
635
636            return m_scriptText;
637        }
638
639        private String getStylesheetText() throws IOException {
640            if( m_stylesheetText != null ) {
641                return m_stylesheetText;
642            }
643
644            if( m_stylesheetLocation == null ) {
645                return "";
646            }
647
648            try {
649                m_stylesheetText = getTextResource(m_stylesheetLocation);
650            } catch( final IOException ex ) {
651                // Only throw this exception once!
652                m_stylesheetText = "";
653                throw ex;
654            }
655
656            return m_stylesheetText;
657        }

693     * @throws PluginException if there is a problem building the {@link Plugin}.
694     */
695    @Override
696    public Plugin newWikiPlugin( final String pluginName, final ResourceBundle rb ) throws PluginException {
697        Plugin plugin = null;
698        WikiPluginInfo pluginInfo = m_pluginClassMap.get( pluginName );
699        try {
700            if( pluginInfo == null ) {
701                pluginInfo = WikiPluginInfo.newInstance( findPluginClass( pluginName ) );
702                registerPlugin( pluginInfo );
703            }
704
705            if( !checkCompatibility( pluginInfo ) ) {
706                final String msg = "Plugin '" + pluginInfo.getName() + "' not compatible with this version of JSPWiki";
707                log.info( msg );
708            } else {
709                plugin = pluginInfo.newPluginInstance(m_searchPath, m_externalJars);
710            }
711        } catch( final ClassNotFoundException e ) {
712            throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.couldnotfind" ), pluginName ), e );
713        } catch( final InstantiationException e ) {
714            throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.cannotinstantiate" ), pluginName ), e );
715        } catch( final IllegalAccessException e ) {
716            throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.notallowed" ), pluginName ), e );
717        } catch( final Exception e ) {
718            throw new PluginException( MessageFormat.format( rb.getString( "plugin.error.instantationfailed" ), pluginName ), e );
719        }
720        return plugin;
721    }
