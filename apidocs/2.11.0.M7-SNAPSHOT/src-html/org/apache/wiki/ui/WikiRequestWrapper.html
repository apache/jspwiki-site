019package org.apache.wiki.ui;
020
021import org.apache.wiki.WikiEngine;
022import org.apache.wiki.WikiSession;
023import org.apache.wiki.auth.SessionMonitor;
024import org.apache.wiki.auth.authorize.Role;
025
026import javax.servlet.http.HttpServletRequest;
027import javax.servlet.http.HttpServletRequestWrapper;
028import java.security.Principal;
029
030/**
031 * Servlet request wrapper that encapsulates an incoming HTTP request and overrides its security methods so that the request returns
032 * JSPWiki-specific values.
033 * 
034 * @since 2.8
035 */
036public class WikiRequestWrapper extends HttpServletRequestWrapper {
037
038    private final WikiSession m_session;
039
040    /**
041     * Constructs a new wrapped request.
042     * 
043     * @param engine the wiki engine
044     * @param request the request to wrap
045     */
046    public WikiRequestWrapper( final WikiEngine engine, final HttpServletRequest request ) {
047        super( request );
048
049        // Get and stash a reference to the current WikiSession
050        m_session = SessionMonitor.getInstance( engine ).find( request.getSession() );
051    }
052
053    /**
054     * Returns the remote user for the HTTP request, taking into account both container and JSPWiki custom authentication status.
055     * Specifically, if the wrapped request contains a remote user, this method returns that remote user. Otherwise, if the user's
056     * WikiSession is an authenticated session (that is, {@link WikiSession#isAuthenticated()} returns <code>true</code>,
057     * this method returns the name of the principal returned by {@link WikiSession#getLoginPrincipal()}.
058     */
059    public String getRemoteUser() {
060        if( super.getRemoteUser() != null ) {
061            return super.getRemoteUser();
062        }
063
064        if( m_session.isAuthenticated() ) {
065            return m_session.getLoginPrincipal().getName();
066        }
067        return null;
068    }
069
070    /**
071     * Returns the user principal for the HTTP request, taking into account both container and JSPWiki custom authentication status.
072     * Specifically, if the wrapped request contains a user principal, this method returns that principal. Otherwise, if the user's
073     * WikiSession is an authenticated session (that is, {@link WikiSession#isAuthenticated()} returns
074     * <code>true</code>, this method returns the value of {@link WikiSession#getLoginPrincipal()}.
075     */
076    public Principal getUserPrincipal() {
077        if( super.getUserPrincipal() != null ) {
078            return super.getUserPrincipal();
079        }
080
081        if( m_session.isAuthenticated() ) {
082            return m_session.getLoginPrincipal();
083        }
084        return null;
085    }
086
087    /**
088     * Determines whether the current user possesses a supplied role, taking into account both container and JSPWIki custom authentication
089     * status. Specifically, if the wrapped request shows that the user possesses the role, this method returns <code>true</code>. If not,
090     * this method iterates through the built-in Role objects (<em>e.g.</em>, ANONYMOUS, ASSERTED, AUTHENTICATED) returned by
091     * {@link WikiSession#getRoles()} and checks to see if any of these principals' names match the supplied role.
092     */
093    public boolean isUserInRole( final String role ) {
094        final boolean hasContainerRole = super.isUserInRole(role);
095        if( hasContainerRole ) {
096            return true;
097        }
098
099        // Iterate through all of the built-in roles and look for a match
100        final Principal[] principals = m_session.getRoles();
101        for( final Principal value : principals ) {
102            if( value instanceof Role ) {
103                final Role principal = ( Role )value;
104                if( Role.isBuiltInRole( principal ) && principal.getName().equals( role ) ) {
105                    return true;
106                }
107            }
108        }
109
110        // None of the built-in roles match, so no luck
111        return false;
112    }
113
114}
