019package org.apache.wiki.htmltowiki.syntax;
020
021import org.apache.wiki.htmltowiki.XHtmlElementToWikiTranslator;
022import org.jdom2.Element;
023import org.jdom2.JDOMException;
024
025import java.io.PrintWriter;
026
027
028/**
029 * Translates to wiki syntax from a {@code TR} element.
030 */
031public class TrDecorator {
032
033    final protected PrintWriter out;
034    final protected XHtmlElementToWikiTranslator chain;
035
036    public TrDecorator( final PrintWriter out, final XHtmlElementToWikiTranslator chain ) {
037        this.out = out;
038        this.chain = chain;
039    }
040
041    /**
042     * Translates the given XHTML element into wiki markup.
043     *
044     * @param element XHTML element being translated.
045     */
046    public void decorate( final Element element ) throws JDOMException {
047        chain.translate( element );
048        out.println();
049    }
050
051}
