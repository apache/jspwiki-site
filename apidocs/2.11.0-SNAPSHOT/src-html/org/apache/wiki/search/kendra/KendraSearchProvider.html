089/**
090 * Search provider that implements {link SearchProvider} using AWS Kendra for
091 * indexing. Note that we are using a Custom DataSource which limits the
092 * attributes that can be uploaded / searched for each page (as per
093 * https://docs.aws.amazon.com/kendra/latest/dg/custom-attributes.html). This
094 * could be overcome by using an S3 bucket where any custom attributes can be
095 * added.
096 *
097 * @since 2.11.0
098 */
099public class KendraSearchProvider implements SearchProvider {

121  /**
122   * {@inheritDoc}
123   */
124  @Override
125  public void initialize(Engine engine, Properties properties) throws NoRequiredPropertyException, IOException {
126    this.engine = engine;
127    this.properties = properties;
128    this.contentTypes = getContentTypes();
129
130    setKendra(buildClient());
131
132    this.indexName = TextUtil.getRequiredProperty(this.properties, PROP_KENDRA_INDEX_NAME);
133    this.dataSourceName = TextUtil.getRequiredProperty(this.properties, PROP_KENDRA_DATA_SOURCE_NAME);
134    int initialDelay = TextUtil.getIntegerProperty(this.properties, PROP_KENDRA_INITIALDELAY,
135        KendraUpdater.INITIAL_DELAY);
136    int indexDelay = TextUtil.getIntegerProperty(this.properties, PROP_KENDRA_INDEXDELAY, KendraUpdater.INDEX_DELAY);
137
138    // Start the Kendra update thread, which waits first for a little while
139    // before starting to go through the "pages that need updating".
140    if (initialDelay >= 0) {
141      KendraUpdater updater = new KendraUpdater(engine, this, initialDelay, indexDelay);
142      updater.start();
143    }
144  }

146  private Map<String, Object> getContentTypes() {
147    Gson gson = new GsonBuilder().create();
148    try (InputStream in = KendraSearchProvider.class.getResourceAsStream("content_types.json")) {
149      if (in != null) {
150        Type collectionType = new TypeToken<HashMap<String, Object>>(){}.getType();
151        return gson.fromJson(new InputStreamReader(in), collectionType);
152      }
153    } catch (IOException e) {
154      LOG.error(format("Unable to load default propertyfile 'content_types.json': %s", e.getMessage()), e);
155    }
156    return null;
157  }

170  @Override
171  public void pageRemoved(Page page) {
172    String pageName = page.getName();
173    BatchDeleteDocumentRequest request = new BatchDeleteDocumentRequest().withIndexId(indexId)
174        .withDocumentIdList(pageName);
175    BatchDeleteDocumentResult result = getKendra().batchDeleteDocument(request);
176    if (result.getFailedDocuments().size() == 0) {
177      LOG.debug(format("Page '%s' was removed from index", pageName));
178    } else {
179      LOG.error(format("Failed to remove Page '%s' from index", pageName));
180    }
181  }

191    }

194  /**
195   * {@inheritDoc}
196   */
197  @Override
198  public Collection<SearchResult> findPages(String query, Context wikiContext) throws ProviderException, IOException {
199    QueryRequest request = new QueryRequest().withIndexId(indexId).withQueryText(query);
200    List<QueryResultItem> items = null;
201    try {
202      items = getKendra().query(request).getResultItems();
203    } catch (ThrottlingException e) {
204      LOG.error(format("ThrottlingException. Skipping..."));
205      return new ArrayList<>();
206    }
207    List<SearchResult> searchResults = new ArrayList<>(items.size());
208    AuthorizationManager mgr = engine.getManager(AuthorizationManager.class);
209
210    for (QueryResultItem item : items) {
211      switch (QueryResultType.fromValue(item.getType())) {
212      case DOCUMENT:
213        String documentId = item.getDocumentId();
214        String documentExcerpt = item.getDocumentExcerpt().getText();
215        String scoreConfidence = item.getScoreAttributes().getScoreConfidence();
216        Page page = this.engine.getManager(PageManager.class).getPage(documentId, PageProvider.LATEST_VERSION);
217        if (page != null) {
218          PagePermission pp = new PagePermission(page, PagePermission.VIEW_ACTION);
219          if (mgr.checkPermission(wikiContext.getWikiSession(), pp)) {
220            SearchResult searchResult = new SearchResultImpl(page, confidence2score(scoreConfidence),
221                new String[] { documentExcerpt });
222            searchResults.add(searchResult);
223          } else {
224            LOG.error(format("Page '%s' is not accessible", documentId));
225          }
226        } else {
227          LOG.error(
228              format("Kendra found a result page '%s' that could not be loaded, removing from index", documentId));
229          pageRemoved(Wiki.contents().page(this.engine, documentId));
230        }
231        break;
232      default:
233        LOG.error(format("Unknown query result type: %s", item.getType()));
234      }
235    }
236    return searchResults;
237  }

239  /**
240   * This method initialize the AWS Kendra Index and Datasources to be used.
241   * 
242   * @throws InterruptedException
243   */
244  public void initializeIndexAndDataSource() throws InterruptedException {
245    this.indexId = getIndexId(indexName);
246    if (this.indexId == null) {
247      String message = format("Index '%s' does not exists", indexName);
248      LOG.error(message);
249      throw new IllegalArgumentException(message);
250    }
251    this.dataSourceId = getDatasourceId(this.indexId, dataSourceName);
252    if (this.dataSourceId == null) {
253      String message = format("Datasource '%s' does not exists in index %s", dataSourceName, indexName);
254      LOG.error(message);
255      throw new IllegalArgumentException(message);
256    }
257  }

259  /**
260   * Given an Kendra's Index name, returns the corresponding Index Id, or
261   * {@code null} if it does not exists
262   * 
263   * @param indexName the name of the index to look up
264   * @return the index id or {@code null}
265   */
266  private String getIndexId(String indexName) {
267    ListIndicesRequest request = new ListIndicesRequest();
268    ListIndicesResult result = getKendra().listIndices(request);
269    String nextToken = "";
270    while (nextToken != null) {
271      List<IndexConfigurationSummary> items = result.getIndexConfigurationSummaryItems();
272      if (items == null || items.isEmpty()) {
273        return null;
274      }
275      for (IndexConfigurationSummary item : items) {
276        if (StringUtils.equals(item.getName(), indexName)) {
277          return item.getId();
278        }
279      }
280      nextToken = result.getNextToken();
281      request = new ListIndicesRequest().withNextToken(result.getNextToken());
282      result = getKendra().listIndices(request);
283    }
284    return null;
285  }
286
287  /**
288   * Given an Kendra's Datasource name, returns the corresponding Datasource Id,
289   * or {@code null} if it does not exists
290   * 
291   * @param dataSourceName the name of the datasource to look up
292   * @return the datasource id or {@code null}
293   */
294  private String getDatasourceId(String indexId, String dataSourceName) {
295    ListDataSourcesRequest request = new ListDataSourcesRequest().withIndexId(indexId);
296    ListDataSourcesResult result = getKendra().listDataSources(request);
297    String nextToken = "";
298    while (nextToken != null) {
299      List<DataSourceSummary> items = result.getSummaryItems();
300      if (items == null || items.isEmpty()) {
301        return null;
302      }
303      
304      for (DataSourceSummary item : items) {
305        if (StringUtils.equals(item.getName(), dataSourceName)) {
306          return item.getId();
307        }
308      }
309      nextToken = result.getNextToken();
310      request = new ListDataSourcesRequest().withNextToken(result.getNextToken());
311      result = getKendra().listDataSources(request);
312    }
313    return null;
314  }

316  /*
317   * Converts a SCORE Confidence from Kendra to an "equivalent" integer score
318   */
319  private int confidence2score(String scoreConfidence) {
320    switch (ScoreConfidence.fromValue(scoreConfidence)) {
321    case VERY_HIGH:
322      return 100;
323    case HIGH:
324      return 75;
325    case MEDIUM:
326      return 50;
327    case LOW:
328      return 25;
329    default:
330      return 0;
331    }
332  }

334  /**
335   * This method re-index all the pages found in the Wiki. It is mainly used at
336   * startup.
337   * 
338   * @throws IOException in case some page can not be read
339   */
340  private void doFullReindex() throws IOException {
341    try {
342      Collection<Page> pages = engine.getManager(PageManager.class).getAllPages();
343      if (pages.isEmpty()) {
344        return;
345      }
346      LOG.debug(format("Indexing all %d pages. Please wait ...", pages.size()));
347      String executionId = startExecution();
348      for (Page page : pages) {
349        // Since I do not want to handle the size limit
350        // (https://docs.aws.amazon.com/goto/WebAPI/kendra-2019-02-03/BatchPutDocument)
351        // uploading documents one at a time
352        indexOnePage(page, executionId);
353      }
354    } catch (ProviderException e) {
355      LOG.error(e.getMessage());
356      throw new IOException(e);
357    } finally {
358      stopExecution();
359    }
360  }
361
362  /**
363   * This method re-index all pages marked as updated. It is used to periodically
364   * index pages that have been modified
365   */
366  private void doPartialReindex() {
367    if (updates.isEmpty()) {
368      return;
369    }
370    LOG.debug(format("Indexing updated pages. Please wait ..."));
371    String executionId = startExecution();
372    synchronized (updates) {
373      try {
374        while (updates.size() > 0) {
375          indexOnePage(updates.remove(0), executionId);
376        }
377      } finally {
378        stopExecution();
379      }
380    }
381  }
382
383  /**
384   * Returns an ExecutiuonId that is required to keep track of the modifed 
385   * documents
386   * @return The execution id
387   */
388  private String startExecution() {
389    StartDataSourceSyncJobRequest request = new StartDataSourceSyncJobRequest().withIndexId(indexId)
390        .withId(dataSourceId);
391    StartDataSourceSyncJobResult result = getKendra().startDataSourceSyncJob(request);
392    return result.getExecutionId();
393  }
394
395  /**
396   * Stop the execution for the given index Id and DataSource Id.
397   */
398  private void stopExecution() {
399    StopDataSourceSyncJobRequest request = new StopDataSourceSyncJobRequest().withIndexId(indexId).withId(dataSourceId);
400    getKendra().stopDataSourceSyncJob(request);
401  }

403  /**
404   * Index on single {@link Page} into the Kendra Index
405   * @param page the {@link Page} to index
406   * @param executionId The Execution Id
407   */
408  private void indexOnePage(Page page, String executionId) {
409    String pageName = page.getName();
410    try {
411      Document document = newDocument(page, executionId);
412      BatchPutDocumentRequest request = new BatchPutDocumentRequest().withIndexId(indexId)
413          .withDocuments(document);
414      BatchPutDocumentResult result = getKendra().batchPutDocument(request);
415      if (result.getFailedDocuments().size() == 0) {
416        LOG.info(format("Successfully indexed Page '%s' as %s", page.getName(), document.getContentType()));
417      } else {
418        for (BatchPutDocumentResponseFailedDocument failedDocument : result.getFailedDocuments()) {
419          LOG.error(format("Failed to index Page '%s': %s", failedDocument.getId(), failedDocument.getErrorMessage()));
420        }
421      }
422    } catch (IOException e) {
423      LOG.error(format("Failed to index Page '%s': %s", pageName, e.getMessage()));
424    }
425  }

428  /**
429   * Given a {@link Page}, returns the corresponding Kendra {@link Document}.
430   * 
431   * @param page        the {@link Page} to be indexed
432   * @param executionId an execution id to identify when the {@link Page} was
433   *                    indexed for the last time.
434   * @return a {@link Document} containing the searchable attributes.
435   * @throws IOException if the {@link Page}'s {@link Attachment} can not be read.
436   */
437  private Document newDocument(Page page, String executionId) throws IOException {
438    String pageName = page.getName();
439    List<DocumentAttribute> attrs = new ArrayList<>();
440    // These 2 are required as per
441    // https://docs.aws.amazon.com/kendra/latest/dg/data-source-custom.html#custom-required-attributes
442    attrs.add(newAttribute("_data_source_id", dataSourceId));
443    attrs.add(newAttribute("_data_source_sync_job_execution_id", executionId));
444
445    String title = TextUtil.beautifyString(pageName);
446    ByteBuffer blob = null;
447    ContentType contentType = ContentType.PLAIN_TEXT;
448    if (page instanceof Attachment) {
449      Attachment attachment = (Attachment) page;
450      InputStream is = null;
451      try {
452        String filename = attachment.getFileName();
453        contentType = getContentType(filename);
454        is = engine.getManager(AttachmentManager.class).getAttachmentStream(attachment);
455        blob = ByteBuffer.wrap(IOUtils.toByteArray(is));
456      } catch (ProviderException e) {
457        throw new IOException(e);
458      } finally {
459        IOUtils.closeQuietly(is, null);
460      }
461      // contentType should be set to its real value
462    } else {
463      String text = engine.getManager(PageManager.class).getPureText(page);
464      blob = ByteBuffer.wrap(text.getBytes(StandardCharsets.UTF_8));
465    }
466    return new Document().withId(pageName).withTitle(title).withAttributes(attrs).withBlob(blob)
467        .withContentType(contentType);
468  }
469
470  private DocumentAttribute newAttribute(String key, String value) {
471    return new DocumentAttribute().withKey(key).withValue(new DocumentAttributeValue().withStringValue(value));
472  }

474  @SuppressWarnings("unchecked")
475  private ContentType getContentType(String filename) {
476    String extention = FilenameUtils.getExtension(filename);
477    Map<String, String> ct = (Map<String, String>) this.contentTypes.get("ContentTypes");
478    return ContentType.fromValue(ct.getOrDefault(extention, ContentType.PLAIN_TEXT.name()));
479  }

481  /**
482   * Updater thread that updates Kendra indexes.
483   */
484  private static final class KendraUpdater extends WikiBackgroundThread {
485    protected static final int INDEX_DELAY = 5;
486    protected static final int INITIAL_DELAY = 10;
487    private KendraSearchProvider provider;
488
489    private int initialDelay;
490
491    private WatchDog watchdog;
492
493    private KendraUpdater(Engine engine, KendraSearchProvider provider, int initialDelay, int indexDelay) {
494      super(engine, indexDelay);
495      this.provider = provider;
496      this.initialDelay = initialDelay;
497      setName("JSPWiki Kendra Indexer");
498    }
499
500    @Override
501    public void startupTask() throws Exception {
502      watchdog = WatchDog.getCurrentWatchDog(getEngine());
503      try {
504        Thread.sleep(initialDelay * 1000L);
505      } catch (InterruptedException e) {
506        throw new InternalWikiException("Interrupted while waiting to start.", e);
507      }
508      watchdog.enterState("Full reindex");
509      provider.initializeIndexAndDataSource();
510      provider.doFullReindex();
511      watchdog.exitState();
512    }
513
514    @Override
515    public void backgroundTask() {
516      watchdog.enterState("Reindexing ...", 60);
517      provider.doPartialReindex();
518      watchdog.exitState();
519    }
520  }
