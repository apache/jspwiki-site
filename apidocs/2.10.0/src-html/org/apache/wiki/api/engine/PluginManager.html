019    package org.apache.wiki.api.engine;
020    
021    import java.io.IOException;
022    import java.util.Collection;
023    import java.util.Map;
024    import java.util.ResourceBundle;
025    
026    import org.apache.oro.text.regex.Pattern;
027    import org.apache.wiki.WikiContext;
028    import org.apache.wiki.api.exceptions.PluginException;
029    import org.apache.wiki.api.plugin.WikiPlugin;
030    
031    
032    public interface PluginManager {
033        
034        /** The property name defining which packages will be searched for properties. */
035        String PROP_SEARCHPATH = "jspwiki.plugin.searchPath";
036        
037        /** This is the default package to try in case the instantiation fails. */
038        String DEFAULT_PACKAGE = "org.apache.wiki.plugin";
039    
040        /**
041         *  The name of the body content.  Current value is "_body".
042         */
043        String PARAM_BODY      = "_body";
044    
045        /** The name of the command line content parameter. The value is "_cmdline". */
046        String PARAM_CMDLINE   = "_cmdline";
047    
048        /**
049         *  The name of the parameter containing the start and end positions in the
050         *  read stream of the plugin text (stored as a two-element int[], start
051         *  and end resp.).
052         */
053        String PARAM_BOUNDS    = "_bounds";
054    
055        /** A special name to be used in case you want to see debug output */
056        String PARAM_DEBUG     = "debug";
057    
058        /**
059         * Enables or disables plugin execution.
060         * 
061         * @param enabled True, if plugins should be globally enabled; false, if disabled.
062         */
063        void enablePlugins( boolean enabled );
064    
065        /**
066         * Returns plugin execution status. If false, plugins are not
067         * executed when they are encountered on a WikiPage, and an
068         * empty string is returned in their place.
069         * 
070         * @return True, if plugins are enabled; false otherwise.
071         */
072        boolean pluginsEnabled();
073        
074        /**
075         * Returns plugin insert pattern.
076         * 
077         * @return plugin insert pattern.
078         */
079        Pattern getPluginPattern();
080        
081        /**
082         * Returns plugins' search path.
083         * 
084         * @return plugins' search path.
085         */
086        String getPluginSearchPath();
087    
088        /**
089         *  Executes a plugin class in the given context.
090         *  <P>Used to be private, but is public since 1.9.21.
091         *
092         *  @param context The current WikiContext.
093         *  @param classname The name of the class.  Can also be a
094         *  shortened version without the package name, since the class name is searched from the
095         *  package search path.
096         *
097         *  @param params A parsed map of key-value pairs.
098         *
099         *  @return Whatever the plugin returns.
100         *
101         *  @throws PluginException If the plugin execution failed for
102         *  some reason.
103         *
104         *  @since 2.0
105         */
106        String execute( WikiContext context, String classname, Map< String, String > params ) throws PluginException;
107    
108        /**
109         *  Parses plugin arguments.  Handles quotes and all other kewl stuff.
110         *
111         *  <h3>Special parameters</h3>
112         *  The plugin body is put into a special parameter defined by {@link #PARAM_BODY};
113         *  the plugin's command line into a parameter defined by {@link #PARAM_CMDLINE};
114         *  and the bounds of the plugin within the wiki page text by a parameter defined
115         *  by {@link #PARAM_BOUNDS}, whose value is stored as a two-element int[] array,
116         *  i.e., <tt>[start,end]</tt>.
117         *
118         * @param argstring The argument string to the plugin.  This is
119         *  typically a list of key-value pairs, using "'" to escape
120         *  spaces in strings, followed by an empty line and then the
121         *  plugin body.  In case the parameter is null, will return an
122         *  empty parameter list.
123         *
124         * @return A parsed list of parameters.
125         *
126         * @throws IOException If the parsing fails.
127         */
128        Map< String, String > parseArgs( String argstring ) throws IOException;
129    
130        /**
131         *  Parses a plugin.  Plugin commands are of the form:
132         *  [{INSERT myplugin WHERE param1=value1, param2=value2}]
133         *  myplugin may either be a class name or a plugin alias.
134         *  <P>
135         *  This is the main entry point that is used.
136         *
137         *  @param context The current WikiContext.
138         *  @param commandline The full command line, including plugin
139         *  name, parameters and body.
140         *
141         *  @return HTML as returned by the plugin, or possibly an error
142         *  message.
143         *  
144         *  @throws PluginException From the plugin itself, it propagates, waah!
145         */
146        String execute( WikiContext context, String commandline ) throws PluginException;
147    
148        /**
149         * Returns a collection of modules currently managed by this ModuleManager.  Each
150         * entry is an instance of the WikiModuleInfo class.  This method should return something
151         * which is safe to iterate over, even if the underlying collection changes.
152         * 
153         * @return A Collection of WikiModuleInfo instances.
154         */
155        Collection modules();
156        
157        /**
158         * Creates a {@link WikiPlugin}.
159         * 
160         * @param pluginName plugin's classname
161         * @param rb {@link ResourceBundle} with i18ned text for exceptions.
162         * @return a {@link WikiPlugin}.
163         * @throws PluginException if there is a problem building the {@link WikiPlugin}.
164         */
165        WikiPlugin newWikiPlugin( String pluginName, ResourceBundle rb ) throws PluginException;
166        
167    }
