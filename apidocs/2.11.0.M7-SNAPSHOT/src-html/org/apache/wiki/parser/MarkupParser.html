128    /**
129     *  Constructs a MarkupParser.  The subclass must call this constructor to set up the necessary bits and pieces.
130     *
131     *  @param context The WikiContext.
132     *  @param in The reader from which we are reading the bytes from.
133     */
134    protected MarkupParser( final WikiContext context, final Reader in ) {
135        m_engine = context.getEngine();
136        m_context = context;
137        m_linkParsingOperations = new LinkParsingOperations( m_context );
138        setInputReader( in );
139    }
140
141    /**
142     *  Replaces the current input character stream with a new one.
143     *
144     *  @param in New source for input.  If null, this method does nothing.
145     *  @return the old stream
146     */
147    public Reader setInputReader( final Reader in ) {
148        final Reader old = m_in;
149        if( in != null ) {
150            m_in = new PushbackReader( new BufferedReader( in ), PUSHBACK_BUFFER_SIZE );
151        }
152
153        return old;
154    }

156    /**
157     *  Adds a hook for processing link texts.  This hook is called when the link text is written into the output stream, and
158     *  you may use it to modify the text.  It does not affect the actual link, only the user-visible text.
159     *
160     *  @param mutator The hook to call.  Null is safe.
161     */
162    public void addLinkTransmutator( final StringTransmutator mutator ) {
163        addLinkHook( m_linkMutators, mutator );
164    }

166    /**
167     *  Adds a hook for processing local links.  The engine transforms both non-existing and existing page links.
168     *
169     *  @param mutator The hook to call.  Null is safe.

171    public void addLocalLinkHook( final StringTransmutator mutator ) {
172        addLinkHook( m_localLinkMutatorChain, mutator );
173    }

175    /**
176     *  Adds a hook for processing external links.  This includes all http:// ftp://, etc. links, including inlined images.
177     *
178     *  @param mutator The hook to call.  Null is safe.
179     */
180    public void addExternalLinkHook( final StringTransmutator mutator ) {
181        addLinkHook( m_externalLinkMutatorChain, mutator );
182    }
183
184    /**
185     *  Adds a hook for processing attachment links.
186     *
187     *  @param mutator The hook to call.  Null is safe.
188     */
189    public void addAttachmentLinkHook( final StringTransmutator mutator ) {
190        addLinkHook( m_attachmentLinkMutatorChain, mutator );
191    }

193    void addLinkHook( final List< StringTransmutator > mutatorChain, final StringTransmutator mutator ) {
194        if( mutator != null ) {
195            mutatorChain.add( mutator );
196        }

240    protected final void initInlineImagePatterns() {
241        final PatternCompiler compiler = new GlobCompiler();
242
243        //  We cache compiled patterns in the engine, since their creation is really expensive
244        List< Pattern > compiledpatterns = m_engine.getAttribute( INLINE_IMAGE_PATTERNS );
245
246        if( compiledpatterns == null ) {
247            compiledpatterns = new ArrayList< >( 20 );
248            final Collection< String > ptrns = m_engine.getAllInlinedImagePatterns();
249
250            //  Make them into Regexp Patterns.  Unknown patterns are ignored.
251            for( final String pattern : ptrns ) {
252                try {
253                    compiledpatterns.add( compiler.compile( pattern, GlobCompiler.DEFAULT_MASK | GlobCompiler.READ_ONLY_MASK ) );
254                } catch( final MalformedPatternException e ) {
255                    log.error( "Malformed pattern [" + pattern + "] in properties: ", e );
256                }
257            }
258
259            m_engine.setAttribute( INLINE_IMAGE_PATTERNS, compiledpatterns );
260        }
261
262        m_inlineImagePatterns = Collections.unmodifiableList( compiledpatterns );
263    }
264
265    public List< Pattern > getInlineImagePatterns() {
266        if( m_inlineImagePatterns == null ) {
267            initInlineImagePatterns();
268        }
269        return m_inlineImagePatterns;
270    }

272    /**
273     *  Parses the document.
274     *
275     *  @return the parsed document, as a WikiDocument
276     *  @throws IOException If something goes wrong.
277     */
278    public abstract WikiDocument parse() throws IOException;
279
280    /**
281     *  Return the current position in the reader stream. The value will be -1 prior to reading.
282     *
283     * @return the reader position as an int.
284     */
285    public int getPosition()
286    {
287        return m_pos;
288    }
289
290    /**
291     * Returns the next token in the stream.  This is the most called method in the entire parser, so it needs to be lean and mean.
292     *
293     * @return The next token in the stream; or, if the stream is ended, -1.
294     * @throws IOException If something bad happens
295     * @throws NullPointerException If you have not yet created an input document.
296     */
297    protected final int nextToken() throws IOException, NullPointerException {
298        // if( m_in == null ) return -1;
299        m_pos++;
300        return m_in.read();
301    }
302
303    /**
304     *  Push back any character to the current input.  Does not push back a read EOF, though.
305     *
306     *  @param c Character to push back.
307     *  @throws IOException In case the character cannot be pushed back.
308     */
309    protected void pushBack( final int c ) throws IOException {
310        if( c != -1 && m_in != null ) {
311            m_pos--;
312            m_in.unread( c );
313        }
314    }

316    /**
317     *  Writes HTML for error message.  Does not add it to the document, you have to do it yourself.
318     *
319     *  @param error The error string.
320     *  @return An Element containing the error.
321     */
322    public static Element makeError( final String error ) {
323        return new Element( "span" ).setAttribute( "class", "error" ).addContent( error );
324    }

326    /**
327     *  Cleans a Wiki name.  The functionality of this method was changed in 2.6 so that the list of allowed characters is much larger.
328     *  Use {@link #wikifyLink(String)} to get the legacy behaviour.
329     *  <P>
330     *  [ This is a link ] -&gt; This is a link
331     *
332     *  @param link Link to be cleared. Null is safe, and causes this to return null.
333     *  @return A cleaned link.
334     *
335     *  @since 2.0
336     */
337    public static String cleanLink( final String link ) {
338        return TextUtil.cleanString( link, TextUtil.PUNCTUATION_CHARS_ALLOWED );
339    }
340
341    /**
342     *  Cleans away extra legacy characters.  This method functions exactly like pre-2.6 cleanLink()
343     *  <P>
344     *  [ This is a link ] -&gt; ThisIsALink
345     *
346     *  @param link Link to be cleared. Null is safe, and causes this to return null.
347     *  @return A cleaned link.
348     *  @since 2.6
349     */
350    public static String wikifyLink( final String link ) {
351        return TextUtil.cleanString( link, TextUtil.LEGACY_CHARS_ALLOWED );
352    }
