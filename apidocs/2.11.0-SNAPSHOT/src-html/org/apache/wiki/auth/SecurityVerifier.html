148    /**
149     * Constructs a new SecurityVerifier for a supplied Engine and WikiSession.
150     *
151     * @param engine the wiki engine
152     * @param session the wiki session (typically, that of an administrator)
153     */
154    public SecurityVerifier( final Engine engine, final Session session ) {
155        m_engine = engine;
156        m_session = session;
157        m_session.clearMessages();
158        verifyJaas();
159        verifyPolicy();
160        try {
161            verifyPolicyAndContainerRoles();
162        } catch( final WikiException e ) {
163            m_session.addMessage( ERROR_ROLES, e.getMessage() );
164        }
165        verifyGroupDatabase();
166        verifyUserDatabase();
167    }

181    /**
182     * Formats and returns an HTML table containing sample permissions and what
183     * roles are allowed to have them. This method will throw an
184     * {@link IllegalStateException} if the authorizer is not of type
185     * {@link org.apache.wiki.auth.authorize.WebContainerAuthorizer}
186     * @return the formatted HTML table containing the result of the tests
187     */
188    public String policyRoleTable()
189    {
190        final Principal[] roles = m_policyPrincipals;
191        final String wiki = m_engine.getApplicationName();
192
193        final String[] pages = new String[]
194        { "Main", "Index", "GroupTest", "GroupAdmin" };
195        final String[] pageActions = new String[]
196        { "view", "edit", "modify", "rename", "delete" };
197
198        final String[] groups = new String[]
199        { "Admin", "TestGroup", "Foo" };
200        final String[] groupActions = new String[]
201        { "view", "edit", null, null, "delete" };
202
203
204        final int rolesLength = roles.length;
205        final int pageActionsLength = pageActions.length;
206        // Calculate column widths
207        final String colWidth;
208        if( pageActionsLength > 0 && rolesLength > 0 ) {
209            colWidth = ( 67f / ( pageActionsLength * rolesLength ) ) + "%";
210        } else {
211            colWidth = "67%";
212        }
213
214        final StringBuilder s = new StringBuilder();
215
216        // Write the table header
217        s.append( "<table class=\"wikitable\" border=\"1\">\n" );
218        s.append( "  <colgroup span=\"1\" width=\"33%\"/>\n" );
219        s.append( "  <colgroup span=\"" + pageActionsLength * rolesLength + "\" width=\"" + colWidth
220                + "\" align=\"center\"/>\n" );
221        s.append( "  <tr>\n" );
222        s.append( "    <th rowspan=\"2\" valign=\"bottom\">Permission</th>\n" );
223        for( int i = 0; i < rolesLength; i++ )
224        {
225            s.append( "    <th colspan=\"" + pageActionsLength + "\" title=\"" + roles[i].getClass().getName() + "\">"
226                    + roles[i].getName() + "</th>\n" );
227        }
228        s.append( "  </tr>\n" );
229
230        // Print a column for each role
231        s.append( "  <tr>\n" );
232        for( int i = 0; i < rolesLength; i++ )
233        {
234            for( final String pageAction : pageActions )
235            {
236                final String action = pageAction.substring( 0, 1 );
237                s.append( "    <th title=\"" + pageAction + "\">" + action + "</th>\n" );
238            }
239        }
240        s.append( "  </tr>\n" );
241
242        // Write page permission tests first
243        for( final String page : pages ) {
244            s.append( "  <tr>\n" );
245            s.append( "    <td>PagePermission \"" + wiki + ":" + page + "\"</td>\n" );
246            for( final Principal role : roles ) {
247                for( final String pageAction : pageActions ) {
248                    final Permission permission = PermissionFactory.getPagePermission( wiki + ":" + page, pageAction );
249                    s.append( printPermissionTest( permission, role, 1 ) );
250                }
251            }
252            s.append( "  </tr>\n" );
253        }
254
255        // Now do the group tests
256        for( final String group : groups ) {
257            s.append( "  <tr>\n" );
258            s.append( "    <td>GroupPermission \"" + wiki + ":" + group + "\"</td>\n" );
259            for( final Principal role : roles ) {
260                for( final String groupAction : groupActions ) {
261                    Permission permission = null;
262                    if( groupAction != null ) {
263                        permission = new GroupPermission( wiki + ":" + group, groupAction );
264                    }
265                    s.append( printPermissionTest( permission, role, 1 ) );
266                }
267            }
268            s.append( "  </tr>\n" );
269        }
270
271
272        // Now check the wiki-wide permissions
273        final String[] wikiPerms = new String[] { "createGroups", "createPages", "login", "editPreferences", "editProfile" };
274        for( final String wikiPerm : wikiPerms ) {
275            s.append( "  <tr>\n" );
276            s.append( "    <td>WikiPermission \"" + wiki + "\",\"" + wikiPerm + "\"</td>\n" );
277            for( final Principal role : roles ) {
278                final Permission permission = new WikiPermission( wiki, wikiPerm );
279                s.append( printPermissionTest( permission, role, pageActionsLength ) );
280            }
281            s.append( "  </tr>\n" );
282        }
283
284        // Lastly, check for AllPermission
285        s.append( "  <tr>\n" );
286        s.append( "    <td>AllPermission \"" + wiki + "\"</td>\n" );
287        for( final Principal role : roles )
288        {
289            final Permission permission = new AllPermission( wiki );
290            s.append( printPermissionTest( permission, role, pageActionsLength ) );
291        }
292        s.append( "  </tr>\n" );
293
294        // We're done!
295        s.append( "</table>" );
296        return s.toString();
297    }

299    /**
300     * Prints a &lt;td&gt; HTML element with the results of a permission test.
301     * @param permission the permission to format
302     * @param principal
303     * @param cols
304     */
305    private String printPermissionTest( final Permission permission, final Principal principal, final int cols ) {
306        final StringBuilder s = new StringBuilder();
307        if( permission == null ) {
308            s.append( "    <td colspan=\"" + cols + "\" align=\"center\" title=\"N/A\">" );
309            s.append( "&nbsp;</td>\n" );
310        } else {
311            final boolean allowed = verifyStaticPermission( principal, permission );
312            s.append( "    <td colspan=\"" + cols + "\" align=\"center\" title=\"" );
313            s.append( allowed ? "ALLOW: " : "DENY: " );
314            s.append( permission.getClass().getName() );
315            s.append( " &quot;" );
316            s.append( permission.getName() );
317            s.append( "&quot;" );
318            if ( permission.getName() != null )
319            {
320                s.append( ",&quot;" );
321                s.append( permission.getActions() );
322                s.append( "&quot;" );
323            }
324            s.append( " " );
325            s.append( principal.getClass().getName() );
326            s.append( " &quot;" );
327            s.append( principal.getName() );
328            s.append( "&quot;" );
329            s.append( "\"" );
330            s.append( allowed ? BG_GREEN + ">" : BG_RED + ">" );
331            s.append( "&nbsp;</td>\n" );
332        }
333        return s.toString();
334    }

336    /**
337     * Formats and returns an HTML table containing the roles the web container
338     * is aware of, and whether each role maps to particular JSPs. This method
339     * throws an {@link IllegalStateException} if the authorizer is not of type
340     * {@link org.apache.wiki.auth.authorize.WebContainerAuthorizer}
341     * @return the formatted HTML table containing the result of the tests
342     * @throws WikiException if tests fail for unexpected reasons
343     */
344    public String containerRoleTable() throws WikiException {
345        final AuthorizationManager authorizationManager = m_engine.getManager( AuthorizationManager.class );
346        final Authorizer authorizer = authorizationManager.getAuthorizer();
347
348        // If authorizer not WebContainerAuthorizer, print error message
349        if ( !( authorizer instanceof WebContainerAuthorizer ) ) {
350            throw new IllegalStateException( "Authorizer should be WebContainerAuthorizer" );
351        }
352
353        // Now, print a table with JSP pages listed on the left, and
354        // an evaluation of each pages' constraints for each role
355        // we discovered
356        final StringBuilder s = new StringBuilder();
357        final Principal[] roles = authorizer.getRoles();
358        s.append( "<table class=\"wikitable\" border=\"1\">\n" );
359        s.append( "<thead>\n" );
360        s.append( "  <tr>\n" );
361        s.append( "    <th rowspan=\"2\">Action</th>\n" );
362        s.append( "    <th rowspan=\"2\">Page</th>\n" );
363        s.append( "    <th colspan=\"" + roles.length + 1 + "\">Roles</th>\n" );
364        s.append( "  </tr>\n" );
365        s.append( "  <tr>\n" );
366        s.append( "    <th>Anonymous</th>\n" );
367        for( final Principal role : roles ) {
368            s.append( "    <th>" + role.getName() + "</th>\n" );
369        }
370        s.append( "</tr>\n" );
371        s.append( "</thead>\n" );
372        s.append( "<tbody>\n" );
373
374        final WebContainerAuthorizer wca = (WebContainerAuthorizer) authorizer;
375        for( int i = 0; i < CONTAINER_ACTIONS.length; i++ ) {
376            final String action = CONTAINER_ACTIONS[i];
377            final String jsp = CONTAINER_JSPS[i];
378
379            // Print whether the page is constrained for each role
380            final boolean allowsAnonymous = !wca.isConstrained( jsp, Role.ALL );
381            s.append( "  <tr>\n" );
382            s.append( "    <td>" + action + "</td>\n" );
383            s.append( "    <td>" + jsp + "</td>\n" );
384            s.append( "    <td title=\"" );
385            s.append( allowsAnonymous ? "ALLOW: " : "DENY: " );
386            s.append( jsp );
387            s.append( " Anonymous" );
388            s.append( "\"" );
389            s.append( allowsAnonymous ? BG_GREEN + ">" : BG_RED + ">" );
390            s.append( "&nbsp;</td>\n" );
391            for( final Principal role : roles )
392            {
393                final boolean allowed = allowsAnonymous || wca.isConstrained( jsp, (Role)role );
394                s.append( "    <td title=\"" );
395                s.append( allowed ? "ALLOW: " : "DENY: " );
396                s.append( jsp );
397                s.append( " " );
398                s.append( role.getClass().getName() );
399                s.append( " &quot;" );
400                s.append( role.getName() );
401                s.append( "&quot;" );
402                s.append( "\"" );
403                s.append( allowed ? BG_GREEN + ">" : BG_RED + ">" );
404                s.append( "&nbsp;</td>\n" );
405            }
406            s.append( "  </tr>\n" );
407        }
408
409        s.append( "</tbody>\n" );
410        s.append( "</table>\n" );
411        return s.toString();
412    }

438    /**
439     * Verifies that the roles given in the security policy are reflected by the
440     * container <code>web.xml</code> file.
441     * @throws WikiException if the web authorizer cannot verify the roles
442     */
443    protected void verifyPolicyAndContainerRoles() throws WikiException
444    {
445        final Authorizer authorizer = m_engine.getManager( AuthorizationManager.class ).getAuthorizer();
446        final Principal[] containerRoles = authorizer.getRoles();
447        boolean missing = false;
448        for( final Principal principal : m_policyPrincipals )
449        {
450            if ( principal instanceof Role )
451            {
452                final Role role = (Role) principal;
453                final boolean isContainerRole = ArrayUtils.contains( containerRoles, role );
454                if ( !Role.isBuiltInRole( role ) && !isContainerRole )
455                {
456                    m_session.addMessage( ERROR_ROLES, "Role '" + role.getName() + "' is defined in security policy but not in web.xml." );
457                    missing = true;
458                }
459            }
460        }
461        if ( !missing )
462        {
463            m_session.addMessage( INFO_ROLES, "Every non-standard role defined in the security policy was also found in web.xml." );
464        }
465    }

467    /**
468     * Verifies that the group datbase was initialized properly, and that
469     * user add and delete operations work as they should.
470     */
471    protected void verifyGroupDatabase()
472    {
473        final GroupManager mgr = m_engine.getManager( GroupManager.class );
474        GroupDatabase db = null;
475        try {
476            db = m_engine.getManager( GroupManager.class ).getGroupDatabase();
477        } catch ( final WikiSecurityException e ) {
478            m_session.addMessage( ERROR_GROUPS, "Could not retrieve GroupManager: " + e.getMessage() );
479        }
480
481        // Check for obvious error conditions
482        if ( mgr == null || db == null ) {
483            if ( mgr == null ) {
484                m_session.addMessage( ERROR_GROUPS, "GroupManager is null; JSPWiki could not initialize it. Check the error logs." );
485            }
486            if ( db == null ) {
487                m_session.addMessage( ERROR_GROUPS, "GroupDatabase is null; JSPWiki could not initialize it. Check the error logs." );
488            }
489            return;
490        }
491
492        // Everything initialized OK...
493
494        // Tell user what class of database this is.
495        m_session.addMessage( INFO_GROUPS, "GroupDatabase is of type '" + db.getClass().getName() + "'. It appears to be initialized properly." );
496
497        // Now, see how many groups we have.
498        final int oldGroupCount;
499        try {
500            final Group[] groups = db.groups();
501            oldGroupCount = groups.length;
502            m_session.addMessage( INFO_GROUPS, "The group database contains " + oldGroupCount + " groups." );
503        } catch( final WikiSecurityException e ) {
504            m_session.addMessage( ERROR_GROUPS, "Could not obtain a list of current groups: " + e.getMessage() );
505            return;
506        }
507
508        // Try adding a bogus group with random name
509        final String name = "TestGroup" + System.currentTimeMillis();
510        final Group group;
511        try {
512            // Create dummy test group
513            group = mgr.parseGroup( name, "", true );
514            final Principal user = new WikiPrincipal( "TestUser" );
515            group.add( user );
516            db.save( group, new WikiPrincipal( "SecurityVerifier" ) );
517
518            // Make sure the group saved successfully
519            if( db.groups().length == oldGroupCount ) {
520                m_session.addMessage( ERROR_GROUPS, "Could not add a test group to the database." );
521                return;
522            }
523            m_session.addMessage( INFO_GROUPS, "The group database allows new groups to be created, as it should." );
524        } catch( final WikiSecurityException e ) {
525            m_session.addMessage( ERROR_GROUPS, "Could not add a group to the database: " + e.getMessage() );
526            return;
527        }
528
529        // Now delete the group; should be back to old count
530        try {
531            db.delete( group );
532            if( db.groups().length != oldGroupCount ) {
533                m_session.addMessage( ERROR_GROUPS, "Could not delete a test group from the database." );
534                return;
535            }
536            m_session.addMessage( INFO_GROUPS, "The group database allows groups to be deleted, as it should." );
537        } catch( final WikiSecurityException e ) {
538            m_session.addMessage( ERROR_GROUPS, "Could not delete a test group from the database: " + e.getMessage() );
539            return;
540        }
541
542        m_session.addMessage( INFO_GROUPS, "The group database configuration looks fine." );
543    }

545    /**
546     * Verfies the JAAS configuration. The configuration is valid if value of the
547     * <code>jspwiki.properties<code> property
548     * {@value org.apache.wiki.auth.AuthenticationManager#PROP_LOGIN_MODULE}
549     * resolves to a valid class on the classpath.
550     */
551    protected void verifyJaas() {
552        // Verify that the specified JAAS moduie corresponds to a class we can load successfully.
553        final String jaasClass = m_engine.getWikiProperties().getProperty( AuthenticationManager.PROP_LOGIN_MODULE );
554        if( jaasClass == null || jaasClass.isEmpty() ) {
555            m_session.addMessage( ERROR_JAAS, "The value of the '" + AuthenticationManager.PROP_LOGIN_MODULE
556                    + "' property was null or blank. This is a fatal error. This value should be set to a valid LoginModule implementation "
557                    + "on the classpath." );
558            return;
559        }
560
561        // See if we can find the LoginModule on the classpath
562        Class< ? > c = null;
563        try {
564            m_session.addMessage( INFO_JAAS,
565                    "The property '" + AuthenticationManager.PROP_LOGIN_MODULE + "' specified the class '" + jaasClass + ".'" );
566            c = Class.forName( jaasClass );
567        } catch( final ClassNotFoundException e ) {
568            m_session.addMessage( ERROR_JAAS, "We could not find the the class '" + jaasClass + "' on the " + "classpath. This is fatal error." );
569        }
570
571        // Is the specified class actually a LoginModule?
572        if( LoginModule.class.isAssignableFrom( c ) ) {
573            m_session.addMessage( INFO_JAAS, "We found the the class '" + jaasClass + "' on the classpath, and it is a LoginModule implementation. Good!" );
574        } else {
575            m_session.addMessage( ERROR_JAAS, "We found the the class '" + jaasClass + "' on the classpath, but it does not seem to be LoginModule implementation! This is fatal error." );
576        }
577    }

579    /**
580     * Looks up a file name based on a JRE system property and returns the associated
581     * File object if it exists. This method adds messages with the topic prefix 
582     * {@link #ERROR} and {@link #INFO} as appropriate, with the suffix matching the 
583     * supplied property.
584     * @param property the system property to look up
585     * @return the file object, or <code>null</code> if not found
586     */
587    protected File getFileFromProperty( final String property )
588    {
589        String propertyValue = null;
590        try
591        {
592            propertyValue = System.getProperty( property );
593            if ( propertyValue == null )
594            {
595                m_session.addMessage( "Error." + property, "The system property '" + property + "' is null." );
596                return null;
597            }
598
599            //
600            //  It's also possible to use "==" to mark a property.  We remove that
601            //  here so that we can actually find the property file, then.
602            //
603            if( propertyValue.startsWith("=") )
604            {
605                propertyValue = propertyValue.substring(1);
606            }
607
608            try
609            {
610                m_session.addMessage( "Info." + property, "The system property '" + property + "' is set to: "
611                        + propertyValue + "." );
612
613                // Prepend a file: prefix if not there already
614                if ( !propertyValue.startsWith( "file:" ) )
615                {
616                  propertyValue = "file:" + propertyValue;
617                }
618                final URL url = new URL( propertyValue );
619                final File file = new File( url.getPath() );
620                if ( file.exists() )
621                {
622                    m_session.addMessage( "Info." + property, "File '" + propertyValue + "' exists in the filesystem." );
623                    return file;
624                }
625            }
626            catch( final MalformedURLException e )
627            {
628                // Swallow exception because we can't find it anyway
629            }
630            m_session.addMessage( "Error." + property, "File '" + propertyValue
631                    + "' doesn't seem to exist. This might be a problem." );
632            return null;
633        }
634        catch( final SecurityException e )
635        {
636            m_session.addMessage( "Error." + property, "We could not read system property '" + property
637                    + "'. This is probably because you are running with a security manager." );
638            return null;
639        }
640    }

642    /**
643     * Verfies the Java security policy configuration. The configuration is
644     * valid if value of the local policy (at <code>WEB-INF/jspwiki.policy</code>
645     * resolves to an existing file, and the policy file contained therein
646     * represents a valid policy.
647     */
648    @SuppressWarnings("unchecked")
649    protected void verifyPolicy() {
650        // Look up the policy file and set the status text.
651        final URL policyURL = m_engine.findConfigFile( AuthorizationManager.DEFAULT_POLICY );
652        String path = policyURL.getPath();
653        if ( path.startsWith("file:") ) {
654            path = path.substring( 5 );
655        }
656        final File policyFile = new File( path );
657
658        // Next, verify the policy
659        try {
660            // Get the file
661            final PolicyReader policy = new PolicyReader( policyFile );
662            m_session.addMessage( INFO_POLICY, "The security policy '" + policy.getFile() + "' exists." );
663
664            // See if there is a keystore that's valid
665            final KeyStore ks = policy.getKeyStore();
666            if ( ks == null ) {
667                m_session.addMessage( WARNING_POLICY,
668                    "Policy file does not have a keystore... at least not one that we can locate. If your policy file " +
669                    "does not contain any 'signedBy' blocks, this is probably ok." );
670            } else {
671                m_session.addMessage( INFO_POLICY,
672                    "The security policy specifies a keystore, and we were able to locate it in the filesystem." );
673            }
674
675            // Verify the file
676            policy.read();
677            final List<Exception> errors = policy.getMessages();
678            if ( errors.size() > 0 ) {
679                for( final Exception e : errors ) {
680                    m_session.addMessage( ERROR_POLICY, e.getMessage() );
681                }
682            } else {
683                m_session.addMessage( INFO_POLICY, "The security policy looks fine." );
684                m_isSecurityPolicyConfigured = true;
685            }
686
687            // Stash the unique principals mentioned in the file,
688            // plus our standard roles.
689            final Set<Principal> principals = new LinkedHashSet<>();
690            principals.add( Role.ALL );
691            principals.add( Role.ANONYMOUS );
692            principals.add( Role.ASSERTED );
693            principals.add( Role.AUTHENTICATED );
694            final ProtectionDomain[] domains = policy.getProtectionDomains();
695            for ( final ProtectionDomain domain : domains ) {
696                principals.addAll(Arrays.asList(domain.getPrincipals()));
697            }
698            m_policyPrincipals = principals.toArray( new Principal[0] );
699        } catch( final IOException e ) {
700            m_session.addMessage( ERROR_POLICY, e.getMessage() );
701        }
702    }

704    /**
705     * Verifies that a particular Principal possesses a Permission, as defined
706     * in the security policy file.
707     * @param principal the principal
708     * @param permission the permission
709     * @return the result, based on consultation with the active Java security
710     *         policy
711     */
712    protected boolean verifyStaticPermission( final Principal principal, final Permission permission )
713    {
714        final Subject subject = new Subject();
715        subject.getPrincipals().add( principal );
716        final boolean allowedByGlobalPolicy = (Boolean)
717            Subject.doAsPrivileged( subject, ( PrivilegedAction< Object > )() -> {
718                try {
719                    AccessController.checkPermission( permission );
720                    return Boolean.TRUE;
721                } catch( final AccessControlException e ) {
722                    return Boolean.FALSE;
723                }
724            }, null );
725
726        if ( allowedByGlobalPolicy )
727        {
728            return true;
729        }
730
731        // Check local policy
732        final Principal[] principals = new Principal[]{ principal };
733        return m_engine.getManager( AuthorizationManager.class ).allowedByLocalPolicy( principals, permission );
734    }

736    /**
737     * Verifies that the user datbase was initialized properly, and that
738     * user add and delete operations work as they should.
739     */
740    protected void verifyUserDatabase()
741    {
742        final UserDatabase db = m_engine.getManager( UserManager.class ).getUserDatabase();
743
744        // Check for obvious error conditions
745        if ( db == null )
746        {
747            m_session.addMessage( ERROR_DB, "UserDatabase is null; JSPWiki could not " +
748                    "initialize it. Check the error logs." );
749            return;
750        }
751
752        if ( db instanceof DummyUserDatabase )
753        {
754            m_session.addMessage( ERROR_DB, "UserDatabase is DummyUserDatabase; JSPWiki " +
755                    "may not have been able to initialize the database you supplied in " +
756                    "jspwiki.properties, or you left the 'jspwiki.userdatabase' property " +
757                    "blank. Check the error logs." );
758        }
759
760        // Tell user what class of database this is.
761        m_session.addMessage( INFO_DB, "UserDatabase is of type '" + db.getClass().getName() +
762                                       "'. It appears to be initialized properly." );
763
764        // Now, see how many users we have.
765        final int oldUserCount;
766        try {
767            final Principal[] users = db.getWikiNames();
768            oldUserCount = users.length;
769            m_session.addMessage( INFO_DB, "The user database contains " + oldUserCount + " users." );
770        } catch( final WikiSecurityException e ) {
771            m_session.addMessage( ERROR_DB, "Could not obtain a list of current users: " + e.getMessage() );
772            return;
773        }
774
775        // Try adding a bogus user with random name
776        final String loginName = "TestUser" + System.currentTimeMillis();
777        try {
778            final UserProfile profile = db.newProfile();
779            profile.setEmail( "jspwiki.tests@mailinator.com" );
780            profile.setLoginName( loginName );
781            profile.setFullname( "FullName" + loginName );
782            profile.setPassword( "password" );
783            db.save( profile );
784
785            // Make sure the profile saved successfully
786            if( db.getWikiNames().length == oldUserCount ) {
787                m_session.addMessage( ERROR_DB, "Could not add a test user to the database." );
788                return;
789            }
790            m_session.addMessage( INFO_DB, "The user database allows new users to be created, as it should." );
791        } catch( final WikiSecurityException e ) {
792            m_session.addMessage( ERROR_DB, "Could not add a test user to the database: " + e.getMessage() );
793            return;
794        }
795
796        // Now delete the profile; should be back to old count
797        try {
798            db.deleteByLoginName( loginName );
799            if( db.getWikiNames().length != oldUserCount ) {
800                m_session.addMessage( ERROR_DB, "Could not delete a test user from the database." );
801                return;
802            }
803            m_session.addMessage( INFO_DB, "The user database allows users to be deleted, as it should." );
804        } catch( final WikiSecurityException e ) {
805            m_session.addMessage( ERROR_DB, "Could not delete a test user to the database: " + e.getMessage() );
806            return;
807        }
808
809        m_session.addMessage( INFO_DB, "The user database configuration looks fine." );
810    }
