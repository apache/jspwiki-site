088    /**
089     *  Reloads the preferences from the PageContext into the WikiContext.
090     *
091     *  @param pageContext The page context.
092     */
093    // FIXME: The way that date preferences are chosen is currently a bit wacky: it all
094    //        gets saved to the cookie based on the browser state with which the user
095    //        happened to first arrive to the site with.  This, unfortunately, means that
096    //        even if the user changes e.g. language preferences (like in a web cafe),
097    //        the old preferences still remain in a site cookie.
098    public static void reloadPreferences( PageContext pageContext )
099    {
100        Preferences prefs = new Preferences();
101        Properties props = PropertyReader.loadWebAppProps( pageContext.getServletContext() );
102        WikiContext ctx = WikiContext.findContext( pageContext );
103        String dateFormat = ctx.getEngine().getInternationalizationManager().get( InternationalizationManager.CORE_BUNDLE,
104                                                                                  getLocale( ctx ),
105                                                                                  "common.datetimeformat" );
106        
107        prefs.put("SkinName", TextUtil.getStringProperty( props, "jspwiki.defaultprefs.template.skinname", "PlainVanilla" ) );
108        prefs.put("DateFormat", TextUtil.getStringProperty( props, "jspwiki.defaultprefs.template.dateformat", dateFormat ) );
109        prefs.put("TimeZone", TextUtil.getStringProperty( props, "jspwiki.defaultprefs.template.timezone", java.util.TimeZone.getDefault().getID() ) );
110        prefs.put("Orientation", TextUtil.getStringProperty( props, "jspwiki.defaultprefs.template.orientation", "fav-left" ) );
111        prefs.put("Sidebar", TextUtil.getStringProperty( props, "jspwiki.defaultprefs.template.sidebar", "active" ) );
112        prefs.put("Layout", TextUtil.getStringProperty( props, "jspwiki.defaultprefs.template.layout", "fluid" ) );
113        prefs.put("Language", TextUtil.getStringProperty( props, "jspwiki.defaultprefs.template.language", getLocale( ctx ).toString() ) );
114        prefs.put("SectionEditing", TextUtil.getStringProperty( props, "jspwiki.defaultprefs.template.sectionediting", "true" ) );
115
116        // FIXME: editormanager reads jspwiki.editor -- which of both properties should continue
117        prefs.put("editor", TextUtil.getStringProperty( props, "jspwiki.defaultprefs.template.editor", "plain" ) );
118
119        parseJSONPreferences( (HttpServletRequest) pageContext.getRequest(), prefs );
120
121        pageContext.getSession().setAttribute( SESSIONPREFS, prefs );
122    }

125    /**
126     *  Parses new-style preferences stored as JSON objects and stores them
127     *  in the session.  Everything in the cookie is stored.
128     *
129     *  @param request
130     *  @param prefs The default hashmap of preferences
131     *
132     */
133    private static void parseJSONPreferences( HttpServletRequest request, Preferences prefs ) {
134        String prefVal = TextUtil.urlDecodeUTF8( HttpUtil.retrieveCookieValue( request, "JSPWikiUserPrefs" ) );
135
136        if( prefVal != null ) {
137            // Convert prefVal JSON to a generic hashmap
138            @SuppressWarnings("unchecked")
139            Map<String,String> map = new Gson().fromJson(prefVal, Map.class );
140
141            for (String key : map.keySet()) {
142                key = TextUtil.replaceEntities( key );
143                // Sometimes this is not a String as it comes from the Cookie set by Javascript
144                Object value = map.get(key);
145                if (value != null) {
146                    prefs.put( key, value.toString() );
147                }
148            }
149        }
150    }

153     *  Returns a preference value programmatically.
154     *  FIXME
155     *
156     *  @param wikiContext
157     *  @param name
158     *  @return the preference value
159     */
160    public static String getPreference( WikiContext wikiContext, String name ) {
161        HttpServletRequest request = wikiContext.getHttpRequest();
162        if ( request == null ) return null;
163
164        Preferences prefs = (Preferences)request.getSession().getAttribute( SESSIONPREFS );
165
166        if( prefs != null ) {
167            return prefs.get( name );
168        }
169
170        return null;
171    }

192    /**
193     * Get Locale according to user-preference settings or the user browser locale
194     *
195     * @param context The context to examine.
196     * @return a Locale object.
197     * @since 2.8
198     */
199    public static Locale getLocale( WikiContext context ) {
200        Locale loc = null;
201
202        String langSetting = getPreference( context, "Language" );
203
204        // parse language and construct valid Locale object
205        if( langSetting != null ) {
206            String language = "";
207            String country  = "";
208            String variant  = "";
209
210            String[] res = StringUtils.split( langSetting, "-_" );
211
212            if( res.length > 2 ) variant = res[2];
213            if( res.length > 1 ) country = res[1];
214
215            if( res.length > 0 ) {
216                language = res[0];
217
218                loc = new Locale( language, country, variant );
219            }
220        }
221        
222        // see if default locale is set server side
223        if( loc == null ) {
224            String locale = context.getEngine().getWikiProperties().getProperty( "jspwiki.preferences.default-locale" );
225            try {
226                loc = LocaleUtils.toLocale( locale );
227            } catch( IllegalArgumentException iae ) {
228                log.error( iae.getMessage() );
229            }
230        }
231
232        // otherwise try to find out the browser's preferred language setting, or use the JVM's default
233        if( loc == null ) {
234            HttpServletRequest request = context.getHttpRequest();
235            loc = ( request != null ) ? request.getLocale() : Locale.getDefault();
236        }
237
238        log.debug( "using locale "+loc.toString() );
239        return loc;
240    }

242    /**
243     *  Locates the i18n ResourceBundle given.  This method interprets
244     *  the request locale, and uses that to figure out which language the
245     *  user wants.
246     *  @see org.apache.wiki.i18n.InternationalizationManager
247     *  @param context {@link WikiContext} holding the user's locale
248     *  @param bundle  The name of the bundle you are looking for.
249     *  @return A localized string (or from the default language, if not found)
250     *  @throws MissingResourceException If the bundle cannot be found
251     */
252    public static ResourceBundle getBundle( WikiContext context, String bundle )
253        throws MissingResourceException
254    {
255        Locale loc = getLocale( context );
256        InternationalizationManager i18n = context.getEngine().getInternationalizationManager();
257        return i18n.getBundle( bundle, loc );
258    }

260    /**
261     *  Get SimpleTimeFormat according to user browser locale and preferred time
262     *  formats. If not found, it will revert to whichever format is set for the
263     *  default
264     *
265     *  @param context WikiContext to use for rendering.
266     *  @param tf Which version of the dateformat you are looking for?
267     *  @return A SimpleTimeFormat object which you can use to render
268     *  @since 2.8
269     */
270    public static SimpleDateFormat getDateFormat( WikiContext context, TimeFormat tf )
271    {
272        InternationalizationManager imgr = context.getEngine().getInternationalizationManager();
273        Locale clientLocale = getLocale( context );
274        String prefTimeZone = getPreference( context, "TimeZone" );
275        String prefDateFormat;
276
277        log.debug("Checking for preferences...");
278
279        switch( tf )
280        {
281            case DATETIME:
282                prefDateFormat = getPreference( context, "DateFormat" );
283                log.debug("Preferences fmt = "+prefDateFormat);
284                if( prefDateFormat == null )
285                {
286                    prefDateFormat = imgr.get( InternationalizationManager.CORE_BUNDLE,
287                                               clientLocale,
288                                               "common.datetimeformat" );
289                    log.debug("Using locale-format = "+prefDateFormat);
290                }
291                break;
292
293            case TIME:
294                prefDateFormat = imgr.get( "common.timeformat" );
295                break;
296
297            case DATE:
298                prefDateFormat = imgr.get( "common.dateformat" );
299                break;
300
301            default:
302                throw new InternalWikiException( "Got a TimeFormat for which we have no value!" );
303        }
304
305        try
306        {
307            SimpleDateFormat fmt = new SimpleDateFormat( prefDateFormat, clientLocale );
308
309            if( prefTimeZone != null )
310            {
311                TimeZone tz = TimeZone.getTimeZone( prefTimeZone );
312                // TimeZone tz = TimeZone.getDefault();
313                // tz.setRawOffset(Integer.parseInt(prefTimeZone));
314
315                fmt.setTimeZone( tz );
316            }
317
318            return fmt;
319        }
320        catch( Exception e )
321        {
322            return null;
323        }
324    }

326    /**
327     *  A simple helper function to render a date based on the user preferences.
328     *  This is useful for example for all plugins.
329     *
330     *  @param context  The context which is used to get the preferences
331     *  @param date     The date to render.
332     *  @param tf       In which format the date should be rendered.
333     *  @return A ready-rendered date.
334     *  @since 2.8
335     */
336    public static String renderDate( WikiContext context, Date date, TimeFormat tf )
337    {
338        DateFormat df = getDateFormat( context, tf );
339
340        return df.format( date );
341    }

343    /**
344     *  Is used to choose between the different date formats that JSPWiki supports.
345     *  <ul>
346     *   <li>TIME: A time format, without  date</li>
347     *   <li>DATE: A date format, without a time</li>
348     *   <li>DATETIME: A date format, with a time</li>
349     *  </ul>
350     *
351     *  @since 2.8
352     */
353    public enum TimeFormat
354    {
355        /** A time format, no date. */
356        TIME,
357
358        /** A date format, no time. */
359        DATE,
360
361        /** A date+time format. */
362        DATETIME
363    }
