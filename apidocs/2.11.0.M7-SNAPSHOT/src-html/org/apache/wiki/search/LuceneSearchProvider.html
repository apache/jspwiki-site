134    @Override
135    public void initialize( final Engine engine, final Properties props ) throws NoRequiredPropertyException, IOException  {
136        m_engine = engine;
137        searchExecutor = Executors.newCachedThreadPool();
138
139        m_luceneDirectory = engine.getWorkDir()+File.separator+LUCENE_DIR;
140
141        final int initialDelay = TextUtil.getIntegerProperty( props, PROP_LUCENE_INITIALDELAY, LuceneUpdater.INITIAL_DELAY );
142        final int indexDelay   = TextUtil.getIntegerProperty( props, PROP_LUCENE_INDEXDELAY, LuceneUpdater.INDEX_DELAY );
143
144        m_analyzerClass = TextUtil.getStringProperty( props, PROP_LUCENE_ANALYZER, m_analyzerClass );
145        // FIXME: Just to be simple for now, we will do full reindex
146        // only if no files are in lucene directory.
147
148        final File dir = new File(m_luceneDirectory);
149        log.info("Lucene enabled, cache will be in: "+dir.getAbsolutePath());
150        try {
151            if( !dir.exists() ) {
152                dir.mkdirs();
153            }
154
155            if( !dir.exists() || !dir.canWrite() || !dir.canRead() ) {
156                log.error("Cannot write to Lucene directory, disabling Lucene: "+dir.getAbsolutePath());
157                throw new IOException( "Invalid Lucene directory." );
158            }
159
160            final String[] filelist = dir.list();
161            if( filelist == null ) {
162                throw new IOException( "Invalid Lucene directory: cannot produce listing: "+dir.getAbsolutePath());
163            }
164        } catch( final IOException e ) {
165            log.error("Problem while creating Lucene index - not using Lucene.", e);
166        }
167
168        // Start the Lucene update thread, which waits first
169        // for a little while before starting to go through
170        // the Lucene "pages that need updating".
171        final LuceneUpdater updater = new LuceneUpdater( m_engine, this, initialDelay, indexDelay );
172        updater.start();
173    }

185    /**
186     *  Performs a full Lucene reindex, if necessary.
187     *
188     *  @throws IOException If there's a problem during indexing
189     */
190    protected void doFullLuceneReindex() throws IOException {
191        final File dir = new File(m_luceneDirectory);
192        final String[] filelist = dir.list();
193        if( filelist == null ) {
194            throw new IOException( "Invalid Lucene directory: cannot produce listing: "+dir.getAbsolutePath());
195        }
196
197        try {
198            if( filelist.length == 0 ) {
199                //
200                //  No files? Reindex!
201                //
202                final Date start = new Date();
203
204                log.info("Starting Lucene reindexing, this can take a couple of minutes...");
205
206                final Directory luceneDir = new SimpleFSDirectory( dir.toPath() );
207                try( final IndexWriter writer = getIndexWriter( luceneDir ) ) {
208                    final Collection< WikiPage > allPages = m_engine.getManager( PageManager.class ).getAllPages();
209                    for( final WikiPage page : allPages ) {
210                        try {
211                            final String text = m_engine.getManager( PageManager.class ).getPageText( page.getName(), WikiProvider.LATEST_VERSION );
212                            luceneIndexPage( page, text, writer );
213                        } catch( final IOException e ) {
214                            log.warn( "Unable to index page " + page.getName() + ", continuing to next ", e );
215                        }
216                    }
217
218                    final Collection< Attachment > allAttachments = m_engine.getManager( AttachmentManager.class ).getAllAttachments();
219                    for( final Attachment att : allAttachments ) {
220                        try {
221                            final String text = getAttachmentContent( att.getName(), WikiProvider.LATEST_VERSION );
222                            luceneIndexPage( att, text, writer );
223                        } catch( final IOException e ) {
224                            log.warn( "Unable to index attachment " + att.getName() + ", continuing to next", e );
225                        }
226                    }
227
228                }
229
230                final Date end = new Date();
231                log.info( "Full Lucene index finished in " + (end.getTime() - start.getTime()) + " milliseconds." );
232            } else {
233                log.info("Files found in Lucene directory, not reindexing.");
234            }
235        } catch ( final IOException e ) {
236            log.error("Problem while creating Lucene index - not using Lucene.", e);
237        } catch ( final ProviderException e ) {
238            log.error("Problem reading pages while creating Lucene index (JSPWiki won't start.)", e);
239            throw new IllegalArgumentException("unable to create Lucene index");
240        } catch( final Exception e ) {
241            log.error("Unable to start lucene",e);
242        }
243
244    }

246    /**
247     *  Fetches the attachment content from the repository.
248     *  Content is flat text that can be used for indexing/searching or display
249     *  
250     *  @param attachmentName Name of the attachment.
251     *  @param version The version of the attachment.
252     *  
253     *  @return the content of the Attachment as a String.
254     */
255    protected String getAttachmentContent( final String attachmentName, final int version ) {
256        final AttachmentManager mgr = m_engine.getManager( AttachmentManager.class );
257        try {
258            final Attachment att = mgr.getAttachmentInfo( attachmentName, version );
259            //FIXME: Find out why sometimes att is null
260            if( att != null ) {
261                return getAttachmentContent( att );
262            }
263        } catch( final ProviderException e ) {
264            log.error("Attachment cannot be loaded", e);
265        }
266        return null;
267    }

269    /**
270     * @param att Attachment to get content for. Filename extension is used to determine the type of the attachment.
271     * @return String representing the content of the file.
272     * FIXME This is a very simple implementation of some text-based attachment, mainly used for testing.
273     * This should be replaced /moved to Attachment search providers or some other 'pluggable' way to search attachments
274     */
275    protected String getAttachmentContent( final Attachment att ) {
276        final AttachmentManager mgr = m_engine.getManager( AttachmentManager.class );
277        //FIXME: Add attachment plugin structure
278
279        final String filename = att.getFileName();
280
281        boolean searchSuffix = false;
282        for( final String suffix : SEARCHABLE_FILE_SUFFIXES ) {
283            if( filename.endsWith( suffix ) ) {
284                searchSuffix = true;
285                break;
286            }
287        }
288
289        String out = filename;
290        if( searchSuffix ) {
291            try( final InputStream attStream = mgr.getAttachmentStream( att ); final StringWriter sout = new StringWriter() ) {
292                FileUtil.copyContents( new InputStreamReader( attStream ), sout );
293                out = out + " " + sout.toString();
294            } catch( final ProviderException | IOException e ) {
295                log.error("Attachment cannot be loaded", e);
296            }
297        }
298
299        return out;
300    }

302    /**
303     *  Updates the lucene index for a single page.
304     *
305     *  @param page The WikiPage to check
306     *  @param text The page text to index.
307     */
308    protected synchronized void updateLuceneIndex( final WikiPage page, final String text ) {
309        log.debug("Updating Lucene index for page '" + page.getName() + "'...");
310        pageRemoved( page );
311
312        // Now add back the new version.
313        try( final Directory luceneDir = new SimpleFSDirectory( new File( m_luceneDirectory ).toPath() );
314             final IndexWriter writer = getIndexWriter( luceneDir ) ) {
315            luceneIndexPage( page, text, writer );
316        } catch( final IOException e ) {
317            log.error("Unable to update page '" + page.getName() + "' from Lucene index", e);
318            // reindexPage( page );
319        } catch( final Exception e ) {
320            log.error("Unexpected Lucene exception - please check configuration!",e);
321            // reindexPage( page );
322        }
323
324        log.debug("Done updating Lucene index for page '" + page.getName() + "'.");
325    }

327    private Analyzer getLuceneAnalyzer() throws ProviderException {
328        try {
329            final Class< ? > clazz = ClassUtil.findClass( "", m_analyzerClass );
330            final Constructor< ? > constructor = clazz.getConstructor();
331            return ( Analyzer )constructor.newInstance();
332        } catch( final Exception e ) {
333            final String msg = "Could not get LuceneAnalyzer class " + m_analyzerClass + ", reason: ";
334            log.error( msg, e );
335            throw new ProviderException( msg + e );
336        }

339    /**
340     *  Indexes page using the given IndexWriter.
341     *
342     *  @param page WikiPage
343     *  @param text Page text to index
344     *  @param writer The Lucene IndexWriter to use for indexing
345     *  @return the created index Document
346     *  @throws IOException If there's an indexing problem
347     */
348    protected Document luceneIndexPage( final WikiPage page, final String text, final IndexWriter writer ) throws IOException {
349        if( log.isDebugEnabled() ) {
350            log.debug( "Indexing " + page.getName() + "..." );
351        }
352        
353        // make a new, empty document
354        final Document doc = new Document();
355
356        if( text == null ) {
357            return doc;
358        }
359        final String indexedText = text.replace( "__", " " ); // be nice to Language Analyzers - cfr. JSPWIKI-893
360
361        // Raw name is the keyword we'll use to refer to this document for updates.
362        Field field = new Field( LUCENE_ID, page.getName(), StringField.TYPE_STORED );
363        doc.add( field );
364
365        // Body text.  It is stored in the doc for search contexts.
366        field = new Field( LUCENE_PAGE_CONTENTS, indexedText, TextField.TYPE_STORED );
367        doc.add( field );
368
369        // Allow searching by page name. Both beautified and raw
370        final String unTokenizedTitle = StringUtils.replaceChars( page.getName(), TextUtil.PUNCTUATION_CHARS_ALLOWED, c_punctuationSpaces );
371        field = new Field( LUCENE_PAGE_NAME, TextUtil.beautifyString( page.getName() ) + " " + unTokenizedTitle, TextField.TYPE_STORED );
372        doc.add( field );
373
374        // Allow searching by authorname
375        if( page.getAuthor() != null ) {
376            field = new Field( LUCENE_AUTHOR, page.getAuthor(), TextField.TYPE_STORED );
377            doc.add( field );
378        }
379
380        // Now add the names of the attachments of this page
381        try {
382            final List< Attachment > attachments = m_engine.getManager( AttachmentManager.class ).listAttachments( page );
383            String attachmentNames = "";
384
385            for( final Attachment att : attachments ) {
386                attachmentNames += att.getName() + ";";
387            }
388            field = new Field( LUCENE_ATTACHMENTS, attachmentNames, TextField.TYPE_STORED );
389            doc.add( field );
390
391        } catch( final ProviderException e ) {
392            // Unable to read attachments
393            log.error( "Failed to get attachments for page", e );
394        }
395
396        // also index page keywords, if available
397        if( page.getAttribute( "keywords" ) != null ) {
398            field = new Field( LUCENE_PAGE_KEYWORDS, page.getAttribute( "keywords" ).toString(), TextField.TYPE_STORED );
399            doc.add( field );
400        }
401        writer.addDocument(doc);
402
403        return doc;
404    }

406    /**
407     *  {@inheritDoc}
408     */
409    @Override
410    public void pageRemoved( final WikiPage page ) {
411        try( final Directory luceneDir = new SimpleFSDirectory( new File( m_luceneDirectory ).toPath() );
412             final IndexWriter writer = getIndexWriter( luceneDir ) ) {
413            final Query query = new TermQuery( new Term( LUCENE_ID, page.getName() ) );
414            writer.deleteDocuments( query );
415        } catch ( final Exception e ) {
416            log.error("Unable to remove page '" + page.getName() + "' from Lucene index", e);
417        }
418    }

420    IndexWriter getIndexWriter(final  Directory luceneDir ) throws IOException, ProviderException {
421        final IndexWriterConfig writerConfig = new IndexWriterConfig( getLuceneAnalyzer() );
422        writerConfig.setOpenMode( OpenMode.CREATE_OR_APPEND );
423        return new IndexWriter( luceneDir, writerConfig );
424    }

426    /**
427     *  Adds a page-text pair to the lucene update queue.  Safe to call always
428     *
429     *  @param page WikiPage to add to the update queue.
430     */
431    @Override
432    public void reindexPage( final WikiPage page ) {
433        if( page != null ) {
434            final String text;
435
436            // TODO: Think if this was better done in the thread itself?
437            if( page instanceof Attachment ) {
438                text = getAttachmentContent( ( Attachment )page );
439            } else {
440                text = m_engine.getManager( PageManager.class ).getPureText( page );
441            }
442
443            if( text != null ) {
444                // Add work item to m_updates queue.
445                final Object[] pair = new Object[2];
446                pair[0] = page;
447                pair[1] = text;
448                m_updates.add( pair );
449                log.debug("Scheduling page " + page.getName() + " for index update");
450            }
451        }
452    }

462    /**
463     *  Create contexts also.  Generating contexts can be expensive,
464     *  so they're not on by default.
465     */
466    public static final int FLAG_CONTEXTS = 0x01;
467
468    /**
469     *  Searches pages using a particular combination of flags.
470     *
471     *  @param query The query to perform in Lucene query language
472     *  @param flags A set of flags
473     *  @return A Collection of SearchResult instances
474     *  @throws ProviderException if there is a problem with the backend
475     */
476    public Collection< SearchResult > findPages( final String query, final int flags, final WikiContext wikiContext ) throws ProviderException {
477        ArrayList<SearchResult> list = null;
478        Highlighter highlighter = null;
479
480        try( final Directory luceneDir = new SimpleFSDirectory( new File( m_luceneDirectory ).toPath() );
481             final IndexReader reader = DirectoryReader.open( luceneDir ) ) {
482            final String[] queryfields = { LUCENE_PAGE_CONTENTS, LUCENE_PAGE_NAME, LUCENE_AUTHOR, LUCENE_ATTACHMENTS, LUCENE_PAGE_KEYWORDS };
483            final QueryParser qp = new MultiFieldQueryParser( queryfields, getLuceneAnalyzer() );
484            final Query luceneQuery = qp.parse( query );
485            final IndexSearcher searcher = new IndexSearcher( reader, searchExecutor );
486
487            if( (flags & FLAG_CONTEXTS) != 0 ) {
488                highlighter = new Highlighter(new SimpleHTMLFormatter("<span class=\"searchmatch\">", "</span>"),
489                                              new SimpleHTMLEncoder(),
490                                              new QueryScorer( luceneQuery ) );
491            }
492
493            final ScoreDoc[] hits = searcher.search(luceneQuery, MAX_SEARCH_HITS).scoreDocs;
494            final AuthorizationManager mgr = m_engine.getManager( AuthorizationManager.class );
495
496            list = new ArrayList<>(hits.length);
497            for( final ScoreDoc hit : hits ) {
498                final int docID = hit.doc;
499                final Document doc = searcher.doc( docID );
500                final String pageName = doc.get( LUCENE_ID );
501                final WikiPage page = m_engine.getManager( PageManager.class ).getPage( pageName, WikiPageProvider.LATEST_VERSION );
502
503                if( page != null ) {
504                    final PagePermission pp = new PagePermission( page, PagePermission.VIEW_ACTION );
505                    if( mgr.checkPermission( wikiContext.getWikiSession(), pp ) ) {
506                        final int score = ( int )( hit.score * 100 );
507
508                        // Get highlighted search contexts
509                        final String text = doc.get( LUCENE_PAGE_CONTENTS );
510
511                        String[] fragments = new String[ 0 ];
512                        if( text != null && highlighter != null ) {
513                            final TokenStream tokenStream = getLuceneAnalyzer()
514                                    .tokenStream( LUCENE_PAGE_CONTENTS, new StringReader( text ) );
515                            fragments = highlighter.getBestFragments( tokenStream, text, MAX_FRAGMENTS );
516                        }
517
518                        final SearchResult result = new SearchResultImpl( page, score, fragments );
519                        list.add( result );
520                    }
521                } else {
522                    log.error( "Lucene found a result page '" + pageName + "' that could not be loaded, removing from Lucene cache" );
523                    pageRemoved( new WikiPage( m_engine, pageName ) );
524                }
525            }
526        } catch( final IOException e ) {
527            log.error("Failed during lucene search",e);
528        } catch( final ParseException e ) {
529            log.info("Broken query; cannot parse query: " + query, e);
530            throw new ProviderException( "You have entered a query Lucene cannot process [" + query + "]: " + e.getMessage() );
531        } catch( final InvalidTokenOffsetsException e ) {
532            log.error("Tokens are incompatible with provided text ",e);
533        }
534
535        return list;
536    }

547    /**
548     * Updater thread that updates Lucene indexes.
549     */
550    private static final class LuceneUpdater extends WikiBackgroundThread {
551        protected static final int INDEX_DELAY    = 5;
552        protected static final int INITIAL_DELAY = 60;
553        private final LuceneSearchProvider m_provider;
554
555        private int m_initialDelay;
556
557        private WatchDog m_watchdog;
558
559        private LuceneUpdater( final Engine engine, final LuceneSearchProvider provider, final int initialDelay, final int indexDelay ) {
560            super( engine, indexDelay );
561            m_provider = provider;
562            m_initialDelay = initialDelay;
563            setName("JSPWiki Lucene Indexer");
564        }
565
566        @Override
567        public void startupTask() throws Exception {
568            m_watchdog = WatchDog.getCurrentWatchDog( getEngine() );
569
570            // Sleep initially...
571            try {
572                Thread.sleep( m_initialDelay * 1000L );
573            } catch( final InterruptedException e ) {
574                throw new InternalWikiException("Interrupted while waiting to start.", e);
575            }
576
577            m_watchdog.enterState( "Full reindex" );
578            // Reindex everything
579            m_provider.doFullLuceneReindex();
580            m_watchdog.exitState();
581        }
582
583        @Override
584        public void backgroundTask() {
585            m_watchdog.enterState("Emptying index queue", 60);
586
587            synchronized ( m_provider.m_updates ) {
588                while( m_provider.m_updates.size() > 0 ) {
589                    final Object[] pair = m_provider.m_updates.remove(0);
590                    final WikiPage page = ( WikiPage ) pair[0];
591                    final String text = ( String ) pair[1];
592                    m_provider.updateLuceneIndex(page, text);
593                }
594            }
595
596            m_watchdog.exitState();
597        }
598
599    }
