019package org.apache.wiki.api.filters;
020
021import org.apache.wiki.api.core.Engine;
022import org.apache.wiki.api.exceptions.FilterException;
023
024import java.lang.reflect.Method;
025import java.util.Properties;
026
027import static org.apache.wiki.api.filters.FilterSupportOperations.executePageFilterPhase;
028import static org.apache.wiki.api.filters.FilterSupportOperations.methodOfNonPublicAPI;
029
030
031/**
032 * Provides a base implementation of a PageFilter.  None of the callbacks do anything, so it is a good idea for you to extend from this
033 * class and implement only methods that you need.
034 */
035public class BasePageFilter implements PageFilter {
036
037    protected Engine m_engine;
038
039    /**
040     * If you override this, you should call super.initialize() first.
041     *
042     * {@inheritDoc}
043     */
044    @Override
045    public void initialize( final Engine engine, final Properties properties ) throws FilterException {
046        m_engine = engine;
047        final Method m = methodOfNonPublicAPI( this, "initialize", "org.apache.wiki.WikiEngine", "java.util.Properties" );
048        executePageFilterPhase( () -> null, m, this, engine, properties );
049    }
050
051}
