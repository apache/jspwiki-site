043/**
044 * Stores the contents of a plugin in a WikiDocument DOM tree.
045 * <p/>
046 * If the WikiContext.VAR_WYSIWYG_EDITOR_MODE is set to Boolean.TRUE in the context, then the plugin is rendered as WikiMarkup.
047 * This allows an HTML editor to work without rendering the plugin each time as well.
048 * <p/>
049 * If WikiContext.VAR_EXECUTE_PLUGINS is set to Boolean.FALSE, then the plugin is not executed.
050 *
051 * @since 2.4
052 */
053public class PluginContent extends Text {
054
055    private static final String BLANK = "";
056    private static final String CMDLINE = "_cmdline";
057    private static final String ELEMENT_BR = "<br/>";
058    private static final String EMITTABLE_PLUGINS = "Image|FormOpen|FormClose|FormInput|FormTextarea|FormSelect";
059    private static final String LINEBREAK = "\n";
060    private static final String PLUGIN_START = "[{";
061    private static final String PLUGIN_END = "}]";
062    private static final String SPACE = " ";
063
064    private static final long serialVersionUID = 1L;
065
066    private static Logger log = Logger.getLogger(PluginContent.class);
067
068    private String m_pluginName;
069    private Map<String, String> m_params;
070
071    /**
072     * Creates a new DOM element with the given plugin name and a map of parameters.
073     *
074     * @param pluginName The FQN of a plugin.
075     * @param parameters A Map of parameters.
076     */
077    public PluginContent( final String pluginName, final Map< String, String > parameters) {
078        m_pluginName = pluginName;
079        m_params = parameters;
080    }
081
082    /**
083     * Returns the name of the plugin invoked by the DOM element.
084     *
085     * @return Name of the plugin
086     * @since 2.5.7
087     */
088    public String getPluginName() {
089        return m_pluginName;
090    }
091
092    /**
093     * Returns a parameter value from the parameter map.
094     *
095     * @param name the name of the parameter.
096     * @return The value from the map, or null, if no such parameter exists.
097     */
098    public String getParameter( final String name) {
099        return m_params.get(name);
100    }
101
102    /**
103     * Returns the parameter map given in the constructor.
104     *
105     * @return The parameter map.
106     */
107    public Map< String, String > getParameters() {
108        return m_params;
109    }
110
111    /**
112     * Returns the rendered plugin.  Only calls getText().
113     *
114     * @return HTML
115     */
116    public String getValue() {
117        return getText();
118    }
119
120    /**
121     * The main invocation for the plugin.  When the getText() is called, it
122     * invokes the plugin and returns its contents.  If there is no Document
123     * yet, only returns the plugin name itself.
124     *
125     * @return The plugin rendered according to the options set in the WikiContext.
126     */
127    public String getText() {
128        final WikiDocument doc = ( WikiDocument )getDocument();
129        if( doc == null ) {
130            //
131            // This element has not yet been attached anywhere, so we simply assume there is no rendering and return the plugin name.
132            // This is required e.g. when the paragraphify() checks whether the element is empty or not.  We can't of course know
133            // whether the rendering would result in an empty string or not, but let us assume it does not.
134            //
135            return getPluginName();
136        }
137
138        final WikiContext context = doc.getContext();
139        if( context == null ) {
140            log.info( "WikiContext garbage-collected, cannot proceed" );
141            return getPluginName();
142        }
143
144        return invoke( context );
145    }
146
147    /**
148     * Performs plugin invocation and return its contents.
149     * 
150     * @param context WikiContext in which the plugin is executed. Must NOT be null.
151     * @return plugin contents.
152     */
153    public String invoke( final WikiContext context ) {
154        String result;
155        final Boolean wysiwygVariable = ( Boolean )context.getVariable( WikiContext.VAR_WYSIWYG_EDITOR_MODE );
156        boolean wysiwygEditorMode = false;
157        if( wysiwygVariable != null ) {
158            wysiwygEditorMode = wysiwygVariable;
159        }
160
161        try {
162            //
163            //  Determine whether we should emit the actual code for this plugin or
164            //  whether we should execute it.  For some plugins we always execute it,
165            //  since they can be edited visually.
166            //
167            // FIXME: The plugin name matching should not be done here, but in a per-editor resource
168            if (wysiwygEditorMode && !m_pluginName.matches(EMITTABLE_PLUGINS)) {
169                result = PLUGIN_START + m_pluginName + SPACE;
170
171                // convert newlines to <br> in case the plugin has a body.
172                final String cmdLine = m_params.get( CMDLINE ).replaceAll( LINEBREAK, ELEMENT_BR );
173                result = result + cmdLine + PLUGIN_END;
174            } else {
175                final Boolean b = ( Boolean )context.getVariable( WikiContext.VAR_EXECUTE_PLUGINS );
176                if (b != null && !b ) {
177                    return BLANK;
178                }
179
180                final WikiEngine engine = context.getEngine();
181                final Map< String, String > parsedParams = new HashMap<>();
182
183                //  Parse any variable instances from the string
184                for( final Map.Entry< String, String > e : m_params.entrySet() ) {
185                    String val = e.getValue();
186                    val = engine.getVariableManager().expandVariables( context, val );
187                    parsedParams.put( e.getKey(), val );
188                }
189                final PluginManager pm = engine.getPluginManager();
190                result = pm.execute( context, m_pluginName, parsedParams );
191            }
192        } catch( final Exception e ) {
193            if( wysiwygEditorMode ) {
194                result = "";
195            } else {
196                // log.info("Failed to execute plugin",e);
197                final ResourceBundle rb = Preferences.getBundle( context, WikiPlugin.CORE_PLUGINS_RESOURCEBUNDLE );
198                result = MarkupParser.makeError( MessageFormat.format( rb.getString( "plugin.error.insertionfailed" ), 
199                                                                       context.getRealPage().getWiki(), 
200                                                                       context.getRealPage().getName(), 
201                                                                       e.getMessage() ) ).getText();
202            }
203        }
204
205        return result;
206    }
207
208    /**
209     * Executes the executeParse() method.
210     *
211     * @param context The WikiContext
212     * @throws PluginException If something goes wrong.
213     */
214    public void executeParse( final WikiContext context ) throws PluginException {
215        final PluginManager pm = context.getEngine().getPluginManager();
216        if( pm.pluginsEnabled() ) {
217            final ResourceBundle rb = Preferences.getBundle(context, WikiPlugin.CORE_PLUGINS_RESOURCEBUNDLE);
218            final Map< String, String > params = getParameters();
219            final WikiPlugin plugin = pm.newWikiPlugin( getPluginName(), rb );
220            try {
221                if( plugin instanceof ParserStagePlugin ) {
222                    ( ( ParserStagePlugin )plugin ).executeParser(this, context, params );
223                }
224            } catch( final ClassCastException e ) {
225                throw new PluginException( MessageFormat.format( rb.getString("plugin.error.notawikiplugin"), getPluginName() ), e );
226            }
227        }
228    }

230    /**
231     * Parses a plugin invocation and returns a DOM element.
232     *
233     * @param context     The WikiContext
234     * @param commandline The line to parse
235     * @param pos         The position in the stream parsing.
236     * @return A DOM element
237     * @throws PluginException If plugin invocation is faulty
238     * @since 2.10.0
239     */
240    public static PluginContent parsePluginLine( final WikiContext context, final String commandline, final int pos ) throws PluginException {
241        final PatternMatcher matcher = new Perl5Matcher();
242
243        try {
244            final PluginManager pm = context.getEngine().getPluginManager();
245            if( matcher.contains( commandline, pm.getPluginPattern() ) ) {
246                final MatchResult res = matcher.getMatch();
247                final String plugin = res.group( 2 );
248                final String args = commandline.substring( res.endOffset( 0 ),
249                                                           commandline.length() - ( commandline.charAt( commandline.length() - 1 ) == '}' ? 1 : 0 ) );
250                final Map< String, String > arglist = pm.parseArgs( args );
251
252                // set wikitext bounds of plugin as '_bounds' parameter, e.g., [345,396]
253                if( pos != -1 ) {
254                    final int end = pos + commandline.length() + 2;
255                    final String bounds = pos + "|" + end;
256                    arglist.put( PluginManager.PARAM_BOUNDS, bounds );
257                }
258
259                return new PluginContent( plugin, arglist );
260            }
261        } catch( final ClassCastException e ) {
262            log.error( "Invalid type offered in parsing plugin arguments.", e );
263            throw new InternalWikiException( "Oops, someone offered !String!", e );
264        } catch( final NoSuchElementException e ) {
265            final String msg = "Missing parameter in plugin definition: " + commandline;
266            log.warn( msg, e );
267            throw new PluginException( msg );
268        } catch( final IOException e ) {
269            final String msg = "Zyrf.  Problems with parsing arguments: " + commandline;
270            log.warn( msg, e );
271            throw new PluginException( msg );
272        }
273
274        return null;
275    }
