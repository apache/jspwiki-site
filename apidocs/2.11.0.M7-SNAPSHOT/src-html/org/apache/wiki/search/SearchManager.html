090    /**
091     *  Provides a JSON RPC API to the JSPWiki Search Engine.
092     */
093    public class JSONSearch implements WikiAjaxServlet {
094
095        public static final String AJAX_ACTION_SUGGESTIONS = "suggestions";
096        public static final String AJAX_ACTION_PAGES = "pages";
097        public static final int DEFAULT_MAX_RESULTS = 20;
098        public int maxResults = DEFAULT_MAX_RESULTS;
099
100        @Override
101        public String getServletMapping() {
102            return JSON_SEARCH;
103        }
104
105        @Override
106        public void service( final HttpServletRequest req,
107                             final HttpServletResponse resp,
108                             final String actionName,
109                             final List< String > params ) throws ServletException, IOException {
110            String result = "";
111            if( StringUtils.isNotBlank( actionName ) ) {
112                if( params.size() < 1 ) {
113                    return;
114                }
115                final String itemId = params.get( 0 );
116                log.debug( "itemId=" + itemId );
117                if( params.size() > 1 ) {
118                    final String maxResultsParam = params.get( 1 );
119                    log.debug( "maxResultsParam=" + maxResultsParam );
120                    if( StringUtils.isNotBlank( maxResultsParam ) && StringUtils.isNumeric( maxResultsParam ) ) {
121                        maxResults = Integer.parseInt( maxResultsParam );
122                    }
123                }
124
125                if( actionName.equals( AJAX_ACTION_SUGGESTIONS ) ) {
126                    log.debug( "Calling getSuggestions() START" );
127                    final List< String > callResults = getSuggestions( itemId, maxResults );
128                    log.debug( "Calling getSuggestions() DONE. " + callResults.size() );
129                    result = AjaxUtil.toJson( callResults );
130                } else if( actionName.equals( AJAX_ACTION_PAGES ) ) {
131                    log.debug("Calling findPages() START");
132                    final WikiContext wikiContext = new WikiContext( m_engine, req, WikiContext.VIEW );
133                    final List< Map< String, Object > > callResults = findPages( itemId, maxResults, wikiContext );
134                    log.debug( "Calling findPages() DONE. " + callResults.size() );
135                    result = AjaxUtil.toJson( callResults );
136                }
137            }
138            log.debug( "result=" + result );
139            resp.getWriter().write( result );
140        }
141
142        /**
143         *  Provides a list of suggestions to use for a page name. Currently the algorithm just looks into the value parameter,
144         *  and returns all page names from that.
145         *
146         *  @param wikiName the page name
147         *  @param maxLength maximum number of suggestions
148         *  @return the suggestions
149         */
150        public List< String > getSuggestions( String wikiName, final int maxLength ) {
151            final StopWatch sw = new StopWatch();
152            sw.start();
153            final List< String > list = new ArrayList<>( maxLength );
154
155            if( wikiName.length() > 0 ) {
156                // split pagename and attachment filename
157                String filename = "";
158                final int pos = wikiName.indexOf("/");
159                if( pos >= 0 ) {
160                    filename = wikiName.substring( pos ).toLowerCase();
161                    wikiName = wikiName.substring( 0, pos );
162                }
163
164                final String cleanWikiName = MarkupParser.cleanLink(wikiName).toLowerCase() + filename;
165                final String oldStyleName = MarkupParser.wikifyLink(wikiName).toLowerCase() + filename;
166                final Set< String > allPages = m_engine.getReferenceManager().findCreated();
167
168                int counter = 0;
169                for( final Iterator< String > i = allPages.iterator(); i.hasNext() && counter < maxLength; ) {
170                    final String p = i.next();
171                    final String pp = p.toLowerCase();
172                    if( pp.startsWith( cleanWikiName) || pp.startsWith( oldStyleName ) ) {
173                        list.add( p );
174                        counter++;
175                    }
176                }
177            }
178
179            sw.stop();
180            if( log.isDebugEnabled() ) {
181                log.debug( "Suggestion request for " + wikiName + " done in " + sw );
182            }
183            return list;
184        }
185
186        /**
187         *  Performs a full search of pages.
188         *
189         *  @param searchString The query string
190         *  @param maxLength How many hits to return
191         *  @return the pages found
192         */
193        public List< Map< String, Object > > findPages( final String searchString, final int maxLength, final WikiContext wikiContext ) {
194            final StopWatch sw = new StopWatch();
195            sw.start();
196
197            final List< Map< String, Object > > list = new ArrayList<>( maxLength );
198            if( searchString.length() > 0 ) {
199                try {
200                    final Collection< SearchResult > c;
201
202                    if( m_searchProvider instanceof LuceneSearchProvider ) {
203                        c = ( ( LuceneSearchProvider )m_searchProvider ).findPages( searchString, 0, wikiContext );
204                    } else {
205                        c = m_searchProvider.findPages( searchString, wikiContext );
206                    }
207
208                    int count = 0;
209                    for( final Iterator< SearchResult > i = c.iterator(); i.hasNext() && count < maxLength; count++ ) {
210                        final SearchResult sr = i.next();
211                        final HashMap< String, Object > hm = new HashMap<>();
212                        hm.put( "page", sr.getPage().getName() );
213                        hm.put( "score", sr.getScore() );
214                        list.add( hm );
215                    }
216                } catch( final Exception e ) {
217                    log.info( "AJAX search failed; ", e );
218                }
219            }
220
221            sw.stop();
222            if( log.isDebugEnabled() ) {
223                log.debug( "AJAX search complete in " + sw );
224            }
225            return list;
226        }
227    }

230    /**
231     *  This particular method starts off indexing and all sorts of various activities,
232     *  so you need to run this last, after things are done.
233     *
234     * @param engine the wiki engine
235     * @param properties the properties used to initialize the wiki engine
236     * @throws FilterException if the search provider failed to initialize
237     */
238    @Override
239    public void initialize( final WikiEngine engine, final Properties properties ) throws FilterException {
240        m_engine = engine;
241        loadSearchProvider(properties);
242
243        try {
244            m_searchProvider.initialize( engine, properties );
245        } catch( final NoRequiredPropertyException | IOException e ) {
246            log.error( e.getMessage(), e );
247        }
248    }
249
250    private void loadSearchProvider( final Properties properties ) {
251        //
252        // See if we're using Lucene, and if so, ensure that its index directory is up to date.
253        //
254        final String providerClassName = properties.getProperty( PROP_SEARCHPROVIDER, DEFAULT_SEARCHPROVIDER );
255
256        try {
257            final Class<?> providerClass = ClassUtil.findClass( "org.apache.wiki.search", providerClassName );
258            m_searchProvider = (SearchProvider)providerClass.newInstance();
259        } catch( final ClassNotFoundException | InstantiationException | IllegalAccessException e ) {
260            log.warn("Failed loading SearchProvider, will use BasicSearchProvider.", e);
261        }
262
263        if( null == m_searchProvider ) {
264            // FIXME: Make a static with the default search provider
265            m_searchProvider = new BasicSearchProvider();
266        }
267        log.debug("Loaded search provider " + m_searchProvider);
268    }

270    /**
271     *  Returns the SearchProvider used.
272     *
273     *  @return The current SearchProvider.
274     */
275    public SearchProvider getSearchEngine()
276    {
277        return m_searchProvider;
278    }
279
280    /**
281     *  Sends a search to the current search provider. The query is is whatever native format
282     *  the query engine wants to use.
283     *
284     * @param query The query.  Null is safe, and is interpreted as an empty query.
285     * @param wikiContext the context within which to run the search
286     * @return A collection of WikiPages that matched.
287     * @throws ProviderException If the provider fails and a search cannot be completed.
288     * @throws IOException If something else goes wrong.
289     */
290    public Collection< SearchResult > findPages( String query, final WikiContext wikiContext ) throws ProviderException, IOException {
291        if( query == null ) {
292            query = "";
293        }
294        return m_searchProvider.findPages( query, wikiContext );
295    }
296
297    /**
298     *  Removes the page from the search cache (if any).
299     *  @param page  The page to remove
300     */
301    public void pageRemoved( final WikiPage page )
302    {
303        m_searchProvider.pageRemoved(page);
304    }
305
306    /**
307     *  Reindexes the page.
308     *
309     *  @param wikiContext {@inheritDoc}
310     *  @param content {@inheritDoc}
311     */
312    @Override
313    public void postSave( final WikiContext wikiContext, final String content ) {
314        //  Makes sure that we're indexing the latest version of this page.
315        final WikiPage p = m_engine.getManager( PageManager.class ).getPage( wikiContext.getPage().getName() );
316        reindexPage( p );
317    }
318
319    /**
320     *   Forces the reindex of the given page.
321     *
322     *   @param page The page.
323     */
324    public void reindexPage( final WikiPage page )
325    {
326        m_searchProvider.reindexPage( page );
327    }
328
329    /**
330     *  If the page has been deleted, removes it from the index.
331     *
332     *  @param event {@inheritDoc}
333     */
334    @Override
335    public void actionPerformed( final WikiEvent event ) {
336        if( event instanceof WikiPageEvent && event.getType() == WikiPageEvent.PAGE_DELETE_REQUEST ) {
337            final String pageName = ( ( WikiPageEvent ) event ).getPageName();
338
339            final WikiPage p = m_engine.getManager( PageManager.class ).getPage( pageName );
340            if( p != null ) {
341                pageRemoved( p );
342            }
343        }
344    }
