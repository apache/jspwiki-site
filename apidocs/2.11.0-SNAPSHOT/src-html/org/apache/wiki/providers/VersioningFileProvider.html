047/**
048 *  Provides a simple directory based repository for Wiki pages.
049 *  Pages are held in a directory structure:
050 *  <PRE>
051 *    Main.txt
052 *    Foobar.txt
053 *    OLD/
054 *       Main/
055 *          1.txt
056 *          2.txt
057 *          page.properties
058 *       Foobar/
059 *          page.properties
060 *  </PRE>
061 *
062 *  In this case, "Main" has three versions, and "Foobar" just one version.
063 *  <P>
064 *  The properties file contains the necessary metainformation (such as author)
065 *  information of the page.  DO NOT MESS WITH IT!
066 *
067 *  <P>
068 *  All files have ".txt" appended to make life easier for those
069 *  who insist on using Windows or other software which makes assumptions
070 *  on the files contents based on its name.
071 *
072 */
073public class VersioningFileProvider extends AbstractFileProvider {
074
075    private static final Logger log = LogManager.getLogger( VersioningFileProvider.class );
076
077    /** Name of the directory where the old versions are stored. */
078    public static final String PAGEDIR = "OLD";
079
080    /** Name of the property file which stores the metadata. */
081    public static final String PROPERTYFILE = "page.properties";

085    /**
086     *  {@inheritDoc}
087     */
088    @Override
089    public void initialize( final Engine engine, final Properties properties ) throws NoRequiredPropertyException, IOException {
090        super.initialize( engine, properties );
091        // some additional sanity checks :
092        final File oldpages = new File( getPageDirectory(), PAGEDIR );
093        if( !oldpages.exists() ) {
094            if( !oldpages.mkdirs() ) {
095                throw new IOException( "Failed to create page version directory " + oldpages.getAbsolutePath() );
096            }
097        } else {
098            if( !oldpages.isDirectory() ) {
099                throw new IOException( "Page version directory is not a directory: " + oldpages.getAbsolutePath() );
100            }
101            if( !oldpages.canWrite() ) {
102                throw new IOException( "Page version directory is not writable: " + oldpages.getAbsolutePath() );
103            }
104        }
105        log.info( "Using directory " + oldpages.getAbsolutePath() + " for storing old versions of pages" );
106    }

108    /**
109     *  Returns the directory where the old versions of the pages
110     *  are being kept.
111     */
112    private File findOldPageDir( final String page ) {
113        if( page == null ) {
114            throw new InternalWikiException( "Page may NOT be null in the provider!" );
115        }
116        final File oldpages = new File( getPageDirectory(), PAGEDIR );
117        return new File( oldpages, mangleName( page ) );
118    }

164    private int findLatestVersion( final String page ) {
165        int version = -1;
166
167        try {
168            final Properties props = getPageProperties( page );
169
170            for( final Object o : props.keySet() ) {
171                final String key = ( String )o;
172                if( key.endsWith( ".author" ) ) {
173                    final int cutpoint = key.indexOf( '.' );
174                    if( cutpoint > 0 ) {
175                        final String pageNum = key.substring( 0, cutpoint );
176
177                        try {
178                            final int res = Integer.parseInt( pageNum );
179
180                            if( res > version ) {
181                                version = res;
182                            }
183                        } catch( final NumberFormatException e ) {
184                        } // It's okay to skip these.
185                    }
186                }
187            }
188        } catch( final IOException e ) {
189            log.error( "Unable to figure out latest version - dying...", e );
190        }
191
192        return version;
193    }

195    /**
196     *  Reads page properties from the file system.
197     */
198    private Properties getPageProperties( final String page ) throws IOException {
199        final File propertyFile = new File( findOldPageDir(page), PROPERTYFILE );
200        if( propertyFile.exists() ) {
201            final long lastModified = propertyFile.lastModified();
202
203            //
204            //   The profiler showed that when calling the history of a page the propertyfile
205            //   was read just as much times as there were versions of that file. The loading
206            //   of a propertyfile is a cpu-intensive job. So now hold on to the last propertyfile
207            //   read because the next method will with a high probability ask for the same propertyfile.
208            //   The time it took to show a historypage with 267 versions dropped with 300%.
209            //
210
211            CachedProperties cp = m_cachedProperties;
212
213            if( cp != null && cp.m_page.equals( page ) && cp.m_lastModified == lastModified ) {
214                return cp.m_props;
215            }
216
217            try( final InputStream in = new BufferedInputStream(new FileInputStream( propertyFile ) ) ) {
218                final Properties props = new Properties();
219                props.load( in );
220                cp = new CachedProperties( page, props, lastModified );
221                m_cachedProperties = cp; // Atomic
222
223                return props;
224            }
225        }
226
227        return new Properties(); // Returns an empty object
228    }

230    /**
231     *  Writes the page properties back to the file system.
232     *  Note that it WILL overwrite any previous properties.
233     */
234    private void putPageProperties( final String page, final Properties properties ) throws IOException {
235        final File propertyFile = new File( findOldPageDir(page), PROPERTYFILE );
236        try( final OutputStream out = new FileOutputStream( propertyFile ) ) {
237            properties.store( out, " JSPWiki page properties for "+page+". DO NOT MODIFY!" );
238        }
239
240        // The profiler showed the probability was very high that when  calling for the history of
241        // a page the propertyfile would be read as much times as there were versions of that file.
242        // It is statistically likely the propertyfile will be examined many times before it is updated.
243        final CachedProperties cp = new CachedProperties( page, properties, propertyFile.lastModified() );
244        m_cachedProperties = cp; // Atomic
245    }

247    /**
248     *  Figures out the real version number of the page and also checks for its existence.
249     *
250     *  @throws NoSuchVersionException if there is no such version.
251     */
252    private int realVersion( final String page, final int requestedVersion ) throws NoSuchVersionException {
253        //  Quickly check for the most common case.
254        if( requestedVersion == WikiProvider.LATEST_VERSION ) {
255            return -1;
256        }
257
258        final int latest = findLatestVersion(page);
259
260        if( requestedVersion == latest || (requestedVersion == 1 && latest == -1 ) ) {
261            return -1;
262        } else if( requestedVersion <= 0 || requestedVersion > latest ) {
263            throw new NoSuchVersionException( "Requested version " + requestedVersion + ", but latest is " + latest );
264        }
265
266        return requestedVersion;
267    }

269    /**
270     *  {@inheritDoc}
271     */
272    @Override
273    public synchronized String getPageText( final String page, int version ) throws ProviderException {
274        final File dir = findOldPageDir( page );
275
276        version = realVersion( page, version );
277        if( version == -1 ) {
278            // We can let the FileSystemProvider take care of these requests.
279            return super.getPageText( page, PageProvider.LATEST_VERSION );
280        }
281
282        final File pageFile = new File( dir, ""+version+FILE_EXT );
283        if( !pageFile.exists() ) {
284            throw new NoSuchVersionException("Version "+version+"does not exist.");
285        }
286
287        return readFile( pageFile );
288    }

291    // FIXME: Should this really be here?
292    private String readFile( final File pagedata ) throws ProviderException {
293        String result = null;
294        if( pagedata.exists() ) {
295            if( pagedata.canRead() ) {
296                try( final InputStream in = new FileInputStream( pagedata ) ) {
297                    result = FileUtil.readContents( in, m_encoding );
298                } catch( final IOException e ) {
299                    log.error("Failed to read", e);
300                    throw new ProviderException("I/O error: "+e.getMessage());
301                }
302            } else {
303                log.warn("Failed to read page from '"+pagedata.getAbsolutePath()+"', possibly a permissions problem");
304                throw new ProviderException("I cannot read the requested page.");
305            }
306        } else {
307            // This is okay.
308            // FIXME: is it?
309            log.info("New page");
310        }
311
312        return result;
313    }

317    /*
318      This is how the page directory should look like:
319
320         version    pagedir       olddir
321          none       empty         empty
322           1         Main.txt (1)  empty
323           2         Main.txt (2)  1.txt
324           3         Main.txt (3)  1.txt, 2.txt
325    */
326    /**
327     *  {@inheritDoc}
328     */
329    @Override
330    public synchronized void putPageText( final Page page, final String text ) throws ProviderException {
331        // This is a bit complicated.  We'll first need to copy the old file to be the newest file.
332        final int  latest  = findLatestVersion( page.getName() );
333        final File pageDir = findOldPageDir( page.getName() );
334        if( !pageDir.exists() ) {
335            pageDir.mkdirs();
336        }
337
338        try {
339            // Copy old data to safety, if one exists.
340            final File oldFile = findPage( page.getName() );
341
342            // Figure out which version should the old page be? Numbers should always start at 1.
343            // "most recent" = -1 ==> 1
344            // "first"       = 1  ==> 2
345            int versionNumber = (latest > 0) ? latest : 1;
346            final boolean firstUpdate = (versionNumber == 1);
347
348            if( oldFile != null && oldFile.exists() ) {
349                final File pageFile = new File( pageDir, versionNumber + FILE_EXT );
350                try( final InputStream in = new BufferedInputStream( new FileInputStream( oldFile ) );
351                     final OutputStream out = new BufferedOutputStream( new FileOutputStream( pageFile ) ) ) {
352                    FileUtil.copyContents( in, out );
353
354                    // We need also to set the date, since we rely on this.
355                    pageFile.setLastModified( oldFile.lastModified() );
356
357                    // Kludge to make the property code to work properly.
358                    versionNumber++;
359                }
360            }
361
362            //  Let superclass handler writing data to a new version.
363            super.putPageText( page, text );
364
365            //  Finally, write page version data.
366            // FIXME: No rollback available.
367            final Properties props = getPageProperties( page.getName() );
368
369            String authorFirst = null;
370            // if the following file exists, we are NOT migrating from FileSystemProvider
371            final File pagePropFile = new File(getPageDirectory() + File.separator + PAGEDIR + File.separator + mangleName(page.getName()) + File.separator + "page" + FileSystemProvider.PROP_EXT);
372            if( firstUpdate && ! pagePropFile.exists() ) {
373                // we might not yet have a versioned author because the old page was last maintained by FileSystemProvider
374                final Properties props2 = getHeritagePageProperties( page.getName() );
375
376                // remember the simulated original author (or something) in the new properties
377                authorFirst = props2.getProperty( "1.author", "unknown" );
378                props.setProperty( "1.author", authorFirst );
379            }
380
381            String newAuthor = page.getAuthor();
382            if ( newAuthor == null ) {
383                newAuthor = ( authorFirst != null ) ? authorFirst : "unknown";
384            }
385            page.setAuthor(newAuthor);
386            props.setProperty( versionNumber + ".author", newAuthor );
387
388            final String changeNote = page.getAttribute( Page.CHANGENOTE );
389            if( changeNote != null ) {
390                props.setProperty( versionNumber + ".changenote", changeNote );
391            }
392
393            // Get additional custom properties from page and add to props
394            getCustomProperties( page, props );
395            putPageProperties( page.getName(), props );
396        } catch( final IOException e ) {
397            log.error( "Saving failed", e );
398            throw new ProviderException("Could not save page text: "+e.getMessage());
399        }
400    }

402    /**
403     *  {@inheritDoc}
404     */
405    @Override
406    public Page getPageInfo( final String page, final int version ) throws ProviderException {
407        final int latest = findLatestVersion( page );
408        final int realVersion;
409
410        Page p = null;
411
412        if( version == PageProvider.LATEST_VERSION || version == latest || (version == 1 && latest == -1) ) {
413            //
414            // Yes, we need to talk to the top level directory to get this version.
415            //
416            // I am listening to Press Play On Tape's guitar version of the good old C64 "Wizardry" -tune at this moment.
417            // Oh, the memories...
418            //
419            realVersion = (latest >= 0) ? latest : 1;
420
421            p = super.getPageInfo( page, PageProvider.LATEST_VERSION );
422
423            if( p != null ) {
424                p.setVersion( realVersion );
425            }
426        } else {
427            // The file is not the most recent, so we'll need to find it from the deep trenches of the "OLD" directory structure.
428            realVersion = version;
429            final File dir = findOldPageDir( page );
430            if( !dir.exists() || !dir.isDirectory() ) {
431                return null;
432            }
433
434            final File file = new File( dir, version + FILE_EXT );
435            if( file.exists() ) {
436                p = Wiki.contents().page( m_engine, page );
437
438                p.setLastModified( new Date( file.lastModified() ) );
439                p.setVersion( version );
440            }
441        }
442
443        //  Get author and other metadata information (Modification date has already been set.)
444        if( p != null ) {
445            try {
446                final Properties props = getPageProperties( page );
447                String author = props.getProperty( realVersion + ".author" );
448                if( author == null ) {
449                    // we might not have a versioned author because the old page was last maintained by FileSystemProvider
450                    final Properties props2 = getHeritagePageProperties( page );
451                    author = props2.getProperty( Page.AUTHOR );
452                }
453                if( author != null ) {
454                    p.setAuthor( author );
455                }
456
457                final String changenote = props.getProperty( realVersion + ".changenote" );
458                if( changenote != null ) {
459                    p.setAttribute( Page.CHANGENOTE, changenote );
460                }
461
462                // Set the props values to the page attributes
463                setCustomProperties( p, props );
464            } catch( final IOException e ) {
465                log.error( "Cannot get author for page" + page + ": ", e );
466            }
467        }
468
469        return p;
470    }

475    @Override
476    public boolean pageExists( final String pageName, final int version ) {
477        if (version == PageProvider.LATEST_VERSION || version == findLatestVersion( pageName ) ) {
478            return pageExists(pageName);
479        }
480
481        final File dir = findOldPageDir( pageName );
482        if( !dir.exists() || !dir.isDirectory() ) {
483            return false;
484        }
485
486        return new File( dir, version + FILE_EXT ).exists();
487    }

507    /*
508     * Support for migration of simple properties created by the FileSystemProvider when coming under Versioning management.
509     * Simulate an initial version.
510     */
511    private Properties getHeritagePageProperties( final String page ) throws IOException {
512        final File propertyFile = new File( getPageDirectory(), mangleName( page ) + FileSystemProvider.PROP_EXT );
513        if ( propertyFile.exists() ) {
514            final long lastModified = propertyFile.lastModified();
515
516            CachedProperties cp = m_cachedProperties;
517            if ( cp != null && cp.m_page.equals(page) && cp.m_lastModified == lastModified ) {
518                return cp.m_props;
519            }
520
521            try( final InputStream in = new BufferedInputStream( new FileInputStream( propertyFile ) ) ) {
522                final Properties props = new Properties();
523                props.load( in );
524
525                final String originalAuthor = props.getProperty( Page.AUTHOR );
526                if ( !originalAuthor.isEmpty() ) {
527                    // simulate original author as if already versioned but put non-versioned property in special cache too
528                    props.setProperty( "1.author", originalAuthor );
529
530                    // The profiler showed the probability was very high that when calling for the history of a page the
531                    // propertyfile would be read as much times as there were versions of that file. It is statistically
532                    // likely the propertyfile will be examined many times before it is updated.
533                    cp = new CachedProperties( page, props, propertyFile.lastModified() );
534                    m_cachedProperties = cp; // Atomic
535                }
536
537                return props;
538            }
539        }
540
541        return new Properties(); // Returns an empty object
542    }

544    /**
545     *  Removes the relevant page directory under "OLD" -directory as well, but does not remove any extra subdirectories from it.
546     *  It will only touch those files that it thinks to be WikiPages.
547     *
548     *  @param page {@inheritDoc}
549     *  @throws {@inheritDoc}
550     */
551    // FIXME: Should log errors.
552    @Override
553    public void deletePage( final String page ) throws ProviderException {
554        super.deletePage( page );
555        final File dir = findOldPageDir( page );
556        if( dir.exists() && dir.isDirectory() ) {
557            final File[] files = dir.listFiles( new WikiFileFilter() );
558            for( final File file : files ) {
559                file.delete();
560            }
561
562            final File propfile = new File( dir, PROPERTYFILE );
563            if( propfile.exists() ) {
564                propfile.delete();
565            }
566
567            dir.delete();
568        }
569    }

571    /**
572     *  {@inheritDoc}
573     *
574     *  Deleting versions has never really worked, JSPWiki assumes that version histories are "not gappy". Using deleteVersion() is
575     *  definitely not recommended.
576     */
577    @Override
578    public void deleteVersion( final String page, final int version ) throws ProviderException {
579        final File dir = findOldPageDir( page );
580        int latest = findLatestVersion( page );
581        if( version == PageProvider.LATEST_VERSION ||
582            version == latest ||
583            (version == 1 && latest == -1) ) {
584            //  Delete the properties
585            try {
586                final Properties props = getPageProperties( page );
587                props.remove( ((latest > 0) ? latest : 1)+".author" );
588                putPageProperties( page, props );
589            } catch( final IOException e ) {
590                log.error("Unable to modify page properties",e);
591                throw new ProviderException("Could not modify page properties: " + e.getMessage());
592            }
593
594            // We can let the FileSystemProvider take care of the actual deletion
595            super.deleteVersion( page, PageProvider.LATEST_VERSION );
596
597            //  Copy the old file to the new location
598            latest = findLatestVersion( page );
599
600            final File pageDir = findOldPageDir( page );
601            final File previousFile = new File( pageDir, latest + FILE_EXT );
602            final File pageFile = findPage(page);
603            try( final InputStream in = new BufferedInputStream( new FileInputStream( previousFile ) );
604                 final OutputStream out = new BufferedOutputStream( new FileOutputStream( pageFile ) ) ) {
605                if( previousFile.exists() ) {
606                    FileUtil.copyContents( in, out );
607                    // We need also to set the date, since we rely on this.
608                    pageFile.setLastModified( previousFile.lastModified() );
609                }
610            } catch( final IOException e ) {
611                log.fatal("Something wrong with the page directory - you may have just lost data!",e);
612            }
613
614            return;
615        }
616
617        final File pageFile = new File( dir, ""+version+FILE_EXT );
618        if( pageFile.exists() ) {
619            if( !pageFile.delete() ) {
620                log.error("Unable to delete page." + pageFile.getPath() );
621            }
622        } else {
623            throw new NoSuchVersionException("Page "+page+", version="+version);
624        }
625    }

655    @Override
656    public void movePage( final String from, final String to ) {
657        // Move the file itself
658        final File fromFile = findPage( from );
659        final File toFile = findPage( to );
660        fromFile.renameTo( toFile );
661
662        // Move any old versions
663        final File fromOldDir = findOldPageDir( from );
664        final File toOldDir = findOldPageDir( to );
665        fromOldDir.renameTo( toOldDir );
666    }

668    /*
669     * The profiler showed that when calling the history of a page, the propertyfile was read just as many
670     * times as there were versions of that file. The loading of a propertyfile is a cpu-intensive job.
671     * This Class holds onto the last propertyfile read, because the probability is high that the next call
672     * will with ask for the same propertyfile. The time it took to show a historypage with 267 versions dropped
673     * by 300%. Although each propertyfile in a history could be cached, there is likely to be little performance
674     * gain over simply keeping the last one requested.
675     */
676    private static class CachedProperties {
677        String m_page;
678        Properties m_props;
679        long m_lastModified;
680
681        /**
682         * Because a Constructor is inherently synchronised, there is no need to synchronise the arguments.
683         *
684         * @param pageName page name
685         * @param props  Properties to use for initialization
686         * @param lastModified last modified date
687         */
688        public CachedProperties( final String pageName, final Properties props, final long lastModified ) {
689            if ( pageName == null ) {
690                throw new NullPointerException ( "pageName must not be null!" );
691            }
692            this.m_page = pageName;
693            if ( props == null ) {
694                throw new NullPointerException ( "properties must not be null!" );
695            }
696            m_props = props;
697            this.m_lastModified = lastModified;
698        }
699    }
