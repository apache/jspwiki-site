349    // FIXME: Potential make-things-easier thingy here: no need to fetch the wikiengine anymore
350    //        Wiki.jsp.jspInit() [really old code]; it's probably even faster to fetch it
351    //        using this method every time than go to pageContext.getAttribute().

353    public static synchronized WikiEngine getInstance( ServletContext context,
354                                                       Properties props )
355        throws InternalWikiException
356    {
357        WikiEngine engine = (WikiEngine) context.getAttribute( ATTR_WIKIENGINE );
358
359        if( engine == null )
360        {
361            String appid = Integer.toString(context.hashCode()); //FIXME: Kludge, use real type.
362
363            context.log(" Assigning new engine to "+appid);
364            try
365            {
366                if( props == null )
367                {
368                    props = PropertyReader.loadWebAppProps( context );
369                }
370
371                engine = new WikiEngine( context, appid, props );
372                context.setAttribute( ATTR_WIKIENGINE, engine );
373            }
374            catch( Exception e )
375            {
376                context.log( "ERROR: Failed to create a Wiki engine: "+e.getMessage() );
377                log.error( "ERROR: Failed to create a Wiki engine, stacktrace follows " , e);
378                throw new InternalWikiException( "No wiki engine, check logs." , e);
379            }
380
381        }
382
383        return engine;
384    }

420        if ( context != null )

422            context.setAttribute( ATTR_WIKIENGINE,  this );
423            m_rootPath = context.getRealPath("/");
424        }
425
426        try
427        {
428            //

435        {
436            String msg = Release.APPNAME+": Unable to load and setup properties from jspwiki.properties. "+e.getMessage();
437            if ( context != null )
438            {
439                context.log( msg );
440            }
441            throw new WikiException( msg, e );
442        }

445    /**
446     *  Does all the real initialization.
447     */
448    private void initialize( Properties props )
449        throws WikiException
450    {
451        m_startTime  = new Date();
452        m_properties = props;
453
454        //
455        //  Initialize log4j.  However, make sure that we don't initialize it multiple times.
456        //  By default we load the log4j config statements from jspwiki.properties, unless
457        //  the property jspwiki.use.external.logconfig=true, in that case we let log4j figure out the
458        //  logging configuration.
459        //
460        if( !c_configured )
461        {
462            String useExternalLogConfig = TextUtil.getStringProperty(props,"jspwiki.use.external.logconfig","false");
463            if( useExternalLogConfig == null || useExternalLogConfig.equals("false"))
464            {
465                PropertyConfigurator.configure( props );
466            }
467            c_configured = true;
468        }
469
470        log.info("*******************************************");
471        log.info(Release.APPNAME+" "+Release.getVersionString()+" starting. Whee!");
472
473        fireEvent( WikiEngineEvent.INITIALIZING ); // begin initialization
474
475        log.debug("Java version: "+System.getProperty("java.runtime.version"));
476        log.debug("Java vendor: "+System.getProperty("java.vm.vendor"));
477        log.debug("OS: "+System.getProperty("os.name")+" "+System.getProperty("os.version")+" "+System.getProperty("os.arch"));
478        log.debug("Default server locale: "+Locale.getDefault());
479        log.debug("Default server timezone: "+TimeZone.getDefault().getDisplayName(true, TimeZone.LONG));
480
481        if( m_servletContext != null )
482        {
483            log.info("Servlet container: "+m_servletContext.getServerInfo() );
484            if( m_servletContext.getMajorVersion() < 2 ||
485                (m_servletContext.getMajorVersion() == 2 && m_servletContext.getMinorVersion() < 4) )
486            {
487                throw new InternalWikiException("I require a container which supports at least version 2.4 of Servlet specification");
488            }
489        }
490
491        log.debug("Configuring WikiEngine...");
492
493        //  Initializes the CommandResolver
494        m_commandResolver  = new CommandResolver( this, props );
495
496        //
497        //  Create and find the default working directory.
498        //
499        m_workDir = TextUtil.getStringProperty( props, PROP_WORKDIR, null );
500
501        if( m_workDir == null ) {
502            m_workDir = System.getProperty("java.io.tmpdir", ".");
503            m_workDir += File.separator+Release.APPNAME+"-"+m_appid;
504        }
505
506        try {
507            File f = new File( m_workDir );
508            f.mkdirs();
509
510            //
511            //  A bunch of sanity checks
512            //
513            if( !f.exists() ) throw new WikiException("Work directory does not exist: "+m_workDir);
514            if( !f.canRead() ) throw new WikiException("No permission to read work directory: "+m_workDir);
515            if( !f.canWrite() ) throw new WikiException("No permission to write to work directory: "+m_workDir);
516            if( !f.isDirectory() ) throw new WikiException("jspwiki.workDir does not point to a directory: "+m_workDir);
517        } catch( SecurityException e ) {
518            log.fatal( "Unable to find or create the working directory: "+m_workDir, e );
519            throw new IllegalArgumentException( "Unable to find or create the working dir: " + m_workDir, e );
520        }
521
522        log.info("JSPWiki working directory is '"+m_workDir+"'");
523
524        m_saveUserInfo   = TextUtil.getBooleanProperty( props, PROP_STOREUSERNAME, m_saveUserInfo );
525        m_useUTF8        = StandardCharsets.UTF_8.name().equals( TextUtil.getStringProperty( props, PROP_ENCODING, StandardCharsets.ISO_8859_1.name() ) );
526        m_beautifyTitle  = TextUtil.getBooleanProperty( props, PROP_BEAUTIFYTITLE, m_beautifyTitle );
527        m_templateDir    = TextUtil.getStringProperty( props, PROP_TEMPLATEDIR, "default" );
528        enforceValidTemplateDirectory();
529        m_frontPage      = TextUtil.getStringProperty( props, PROP_FRONTPAGE,   "Main" );
530
531        //
532        //  Initialize the important modules.  Any exception thrown by the
533        //  managers means that we will not start up.
534        //
535
536        // FIXME: This part of the code is getting unwieldy.  We must think
537        //        of a better way to do the startup-sequence.
538        try
539        {
540            Class< ? > urlclass = ClassUtil.findClass( "org.apache.wiki.url",
541                                                       TextUtil.getStringProperty( props, PROP_URLCONSTRUCTOR, "DefaultURLConstructor" ) );
542            m_urlConstructor = (URLConstructor) urlclass.getDeclaredConstructor().newInstance();
543            m_urlConstructor.initialize( this, props );
544
545            m_pageManager           = ClassUtil.getMappedObject( PageManager.class.getName(), this, props );
546            m_pluginManager         = ClassUtil.getMappedObject( PluginManager.class.getName(), this, props );
547            m_differenceManager     = ClassUtil.getMappedObject( DifferenceManager.class.getName(), this, props );
548            m_attachmentManager     = ClassUtil.getMappedObject( AttachmentManager.class.getName(), this, props );
549            m_variableManager       = ClassUtil.getMappedObject( VariableManager.class.getName(), props );
550            m_renderingManager      = ClassUtil.getMappedObject( RenderingManager.class.getName() );
551            m_searchManager         = ClassUtil.getMappedObject( SearchManager.class.getName(), this, props );
552            m_authenticationManager = ClassUtil.getMappedObject( AuthenticationManager.class.getName() );
553            m_authorizationManager  = ClassUtil.getMappedObject( AuthorizationManager.class.getName() );
554            m_userManager           = ClassUtil.getMappedObject( UserManager.class.getName() );
555            m_groupManager          = ClassUtil.getMappedObject( GroupManager.class.getName() );
556            m_editorManager         = ClassUtil.getMappedObject( EditorManager.class.getName(), this );
557            m_editorManager.initialize( props );
558
559            m_progressManager   = new ProgressManager();
560
561            // Initialize the authentication, authorization, user and acl managers
562            m_authenticationManager.initialize( this, props );
563            m_authorizationManager.initialize( this, props );
564            m_userManager.initialize( this, props );
565            m_groupManager.initialize( this, props );
566            m_aclManager = getAclManager();
567
568            // Start the Workflow manager
569            m_workflowMgr = ClassUtil.getMappedObject(WorkflowManager.class.getName());
570            m_workflowMgr.initialize(this, props);
571            m_tasksManager = ClassUtil.getMappedObject(TasksManager.class.getName());
572
573            m_internationalizationManager = ClassUtil.getMappedObject(InternationalizationManager.class.getName(),this);
574            m_templateManager = ClassUtil.getMappedObject(TemplateManager.class.getName(), this, props );
575
576            // Since we want to use a page filters initilize() method
577            // as a engine startup listener where we can initialize global event listeners,
578            // it must be called lastly, so that all object references in the engine
579            // are availabe to the initialize() method
580            m_filterManager = ClassUtil.getMappedObject(FilterManager.class.getName(), this, props );
581
582            m_adminBeanManager = ClassUtil.getMappedObject(AdminBeanManager.class.getName(),this);
583
584            // RenderingManager depends on FilterManager events.
585            m_renderingManager.initialize( this, props );
586
587            //
588            //  ReferenceManager has the side effect of loading all
589            //  pages.  Therefore after this point, all page attributes
590            //  are available.
591            //
592            //  initReferenceManager is indirectly using m_filterManager, therefore
593            //  it has to be called after it was initialized.
594            //
595            initReferenceManager();
596
597            //
598            //  Hook the different manager routines into the system.
599            //
600            m_filterManager.addPageFilter(m_referenceManager, -1001 );
601            m_filterManager.addPageFilter(m_searchManager, -1002 );
602        }
603
604        catch( RuntimeException e )
605        {
606            // RuntimeExceptions may occur here, even if they shouldn't.
607            log.fatal( "Failed to start managers.", e );
608            throw new WikiException( "Failed to start managers: " + e.getMessage(), e );
609        }
610        catch (ClassNotFoundException e)
611        {
612            log.fatal( "JSPWiki could not start, URLConstructor was not found: " + e.getMessage(), e );
613            throw new WikiException(e.getMessage(), e );
614        }
615        catch (InstantiationException e)
616        {
617            log.fatal( "JSPWiki could not start, URLConstructor could not be instantiated: " + e.getMessage(), e );
618            throw new WikiException(e.getMessage(), e );
619        }
620        catch (IllegalAccessException e)
621        {
622            log.fatal( "JSPWiki could not start, URLConstructor cannot be accessed: " + e.getMessage(), e );
623            throw new WikiException(e.getMessage(), e );
624        }
625        catch( Exception e )
626        {
627            // Final catch-all for everything
628            log.fatal( "JSPWiki could not start, due to an unknown exception when starting.",e );
629            throw new WikiException( "Failed to start. Caused by: " + e.getMessage() +
630                                     "; please check log files for better information.", e );
631        }
632
633        //
634        //  Initialize the good-to-have-but-not-fatal modules.
635        //
636        try {
637            if( TextUtil.getBooleanProperty( props, RSSGenerator.PROP_GENERATE_RSS,false ) ) {
638                m_rssGenerator = ClassUtil.getMappedObject( RSSGenerator.class.getName(), this, props );
639            }
640
641            m_pageRenamer = ClassUtil.getMappedObject( PageRenamer.class.getName(), this, props );
642        } catch( final Exception e ) {
643            log.error( "Unable to start RSS generator - JSPWiki will still work, but there will be no RSS feed.", e );
644        }
645
646        // Start the RSS generator & generator thread
647        if( m_rssGenerator != null )
648        {
649            m_rssFile = TextUtil.getStringProperty( props,
650                    RSSGenerator.PROP_RSSFILE, "rss.rdf" );
651            File rssFile=null;
652            if (m_rssFile.startsWith(File.separator))
653            {
654                // honor absolute pathnames:
655                rssFile = new File(m_rssFile );
656            }
657            else
658            {
659                // relative path names are anchored from the webapp root path:
660                rssFile = new File( getRootPath(), m_rssFile );
661            }
662            int rssInterval = TextUtil.getIntegerProperty( props,
663                    RSSGenerator.PROP_INTERVAL, 3600 );
664            RSSThread rssThread = new RSSThread( this, rssFile, rssInterval );
665            rssThread.start();
666        }
667
668        fireEvent( WikiEngineEvent.INITIALIZED ); // initialization complete
669
670        log.info("WikiEngine configured.");
671        m_isConfigured = true;
672    }

1461    /**
1462     *  Helper method for doing the HTML translation.
1463     *
1464     *  @param context The WikiContext in which to do the conversion
1465     *  @param pagedata The data to render
1466     *  @param localLinkHook Is called whenever a wiki link is found
1467     *  @param extLinkHook   Is called whenever an external link is found
1468     *  @param parseAccessRules Parse the access rules if we encounter them
1469     *  @param justParse Just parses the pagedata, does not actually render.  In this case,
1470     *                   this methods an empty string.
1471     *  @return HTML-rendered page text.
1472
1473     */
1474    private String textToHTML( WikiContext context,
1475                               String pagedata,
1476                               StringTransmutator localLinkHook,
1477                               StringTransmutator extLinkHook,
1478                               StringTransmutator attLinkHook,
1479                               boolean            parseAccessRules,
1480                               boolean            justParse )
1481    {
1482        String result = "";
1483
1484        if( pagedata == null )
1485        {
1486            log.error("NULL pagedata to textToHTML()");
1487            return null;
1488        }
1489
1490        boolean runFilters = "true".equals(m_variableManager.getValue(context,PROP_RUNFILTERS,"true"));
1491
1492        try
1493        {
1494            StopWatch sw = new StopWatch();
1495            sw.start();
1496
1497            if( runFilters && m_filterManager != null )
1498                pagedata = m_filterManager.doPreTranslateFiltering( context, pagedata );
1499
1500            MarkupParser mp = m_renderingManager.getParser( context, pagedata );
1501            mp.addLocalLinkHook( localLinkHook );
1502            mp.addExternalLinkHook( extLinkHook );
1503            mp.addAttachmentLinkHook( attLinkHook );
1504
1505            if( !parseAccessRules ) mp.disableAccessRules();
1506
1507            WikiDocument doc = mp.parse();
1508
1509            //
1510            //  In some cases it's better just to parse, not to render
1511            //
1512            if( !justParse )
1513            {
1514                result = m_renderingManager.getHTML( context, doc );
1515
1516                if( runFilters && m_filterManager != null )
1517                    result = m_filterManager.doPostTranslateFiltering( context, result );
1518            }
1519
1520            sw.stop();
1521
1522            if( log.isDebugEnabled() )
1523                log.debug("Page "+context.getRealPage().getName()+" rendered, took "+sw );
1524        }
1525        catch( IOException e )
1526        {
1527            log.error( "Failed to scan page data: ", e );
1528        }
1529        catch( FilterException e )
1530        {
1531            log.error( "page filter threw exception: ", e );
1532            // FIXME: Don't yet know what to do
1533        }
1534
1535        return result;
1536    }

1552    /**
1553     *  Writes the WikiText of a page into the page repository. If the <code>jspwiki.properties</code> file contains
1554     *  the property <code>jspwiki.approver.workflow.saveWikiPage</code> and its value resolves to a valid user,
1555     *  {@link org.apache.wiki.auth.authorize.Group} or {@link org.apache.wiki.auth.authorize.Role}, this method will
1556     *  place a {@link org.apache.wiki.workflow.Decision} in the approver's workflow inbox and throw a
1557     *  {@link org.apache.wiki.workflow.DecisionRequiredException}. If the submitting user is authenticated and the
1558     *  page save is rejected, a notification will be placed in the user's decision queue.
1559     *
1560     *  @since 2.1.28
1561     *  @param context The current WikiContext
1562     *  @param text    The Wiki markup for the page.
1563     *  @throws WikiException if the save operation encounters an error during the save operation. If the page-save
1564     *  operation requires approval, the exception will be of type {@link org.apache.wiki.workflow.DecisionRequiredException}.
1565     *  Individual PageFilters, such as the {@link org.apache.wiki.filters.SpamFilter} may also throw a
1566     *  {@link org.apache.wiki.api.exceptions.RedirectException}.
1567     */
1568    public void saveText( WikiContext context, String text ) throws WikiException {
1569        // Check if page data actually changed; bail if not
1570        WikiPage page = context.getPage();
1571        String oldText = getPureText( page );
1572        String proposedText = TextUtil.normalizePostData( text );
1573        if ( oldText != null && oldText.equals( proposedText ) ) {
1574            return;
1575        }
1576
1577        // Check if creation of empty pages is allowed; bail if not
1578        boolean allowEmpty = TextUtil.getBooleanProperty( m_properties, PROP_ALLOW_CREATION_OF_EMPTY_PAGES, false );
1579        if ( !allowEmpty && !pageExists( page ) && text.trim().equals( "" ) ) {
1580            return;
1581        }
1582
1583        // Create approval workflow for page save; add the diffed, proposed and old text versions as
1584        // Facts for the approver (if approval is required). If submitter is authenticated, any reject
1585        // messages will appear in his/her workflow inbox.
1586        WorkflowBuilder builder = WorkflowBuilder.getBuilder( this );
1587        Principal submitter = context.getCurrentUser();
1588        Step prepTask = m_tasksManager.buildPreSaveWikiPageTask( context, proposedText );
1589        Step completionTask = m_tasksManager.buildSaveWikiPageTask();
1590        String diffText = m_differenceManager.makeDiff( context, oldText, proposedText );
1591        boolean isAuthenticated = context.getWikiSession().isAuthenticated();
1592        Fact[] facts = new Fact[ 5 ];
1593        facts[ 0 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_PAGE_NAME, page.getName() );
1594        facts[ 1 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_DIFF_TEXT, diffText );
1595        facts[ 2 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_PROPOSED_TEXT, proposedText );
1596        facts[ 3 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_CURRENT_TEXT, oldText);
1597        facts[ 4 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_IS_AUTHENTICATED, Boolean.valueOf( isAuthenticated ) );
1598        String rejectKey = isAuthenticated ? WorkflowManager.WF_WP_SAVE_REJECT_MESSAGE_KEY : null;
1599        Workflow workflow = builder.buildApprovalWorkflow( submitter,
1600                                                           WorkflowManager.WF_WP_SAVE_APPROVER,
1601                                                           prepTask,
1602                                                           WorkflowManager.WF_WP_SAVE_DECISION_MESSAGE_KEY,
1603                                                           facts,
1604                                                           completionTask,
1605                                                           rejectKey );
1606        m_workflowMgr.start( workflow );
1607
1608        // Let callers know if the page-save requires approval
1609        if ( workflow.getCurrentStep() instanceof Decision ) {
1610            throw new DecisionRequiredException( "The page contents must be approved before they become active." );
1611        }
1612    }

1917            } else {

1922                    final List< Attachment > attachments = m_attachmentManager.listAttachments( p );
1923                    for( final Attachment attachment : attachments ) {
1924                        if( refTo != null ) {
1925                            refTo.remove( attachment.getName() );
1926                        }
1927
1928                        m_attachmentManager.deleteAttachment( attachment );
1929                    }
1930                }
1931                m_pageManager.deletePage( p );
1932                firePageEvent( WikiPageEvent.PAGE_DELETED, pageName );
1933            }
1934        }
1935    }
