019package org.apache.wiki.workflow;
020
021import org.apache.wiki.api.core.Engine;
022import org.apache.wiki.api.core.Session;
023import org.apache.wiki.api.exceptions.WikiException;
024import org.apache.wiki.auth.AuthorizationManager;
025import org.apache.wiki.auth.acl.UnresolvedPrincipal;
026import org.apache.wiki.event.WikiEvent;
027import org.apache.wiki.event.WorkflowEvent;
028
029import java.security.Principal;
030import java.util.ArrayList;
031import java.util.List;
032import java.util.Map;
033import java.util.Properties;
034import java.util.Set;
035import java.util.concurrent.ConcurrentHashMap;
036import java.util.concurrent.CopyOnWriteArrayList;
037
038
039/**
040 * <p>
041 * Monitor class that tracks running Workflows. The WorkflowManager also keeps track of the names of
042 * users or groups expected to approve particular Workflows.
043 * </p>
044 */
045public class DefaultWorkflowManager implements WorkflowManager {
046
047    private final DecisionQueue m_queue = new DecisionQueue();
048    private final Set< Workflow > m_workflows;
049    private final Map< String, Principal > m_approvers;
050    private final List< Workflow > m_completed;
051    private Engine m_engine = null;
052
053    /**
054     * Constructs a new WorkflowManager, with an empty workflow cache. New Workflows are automatically assigned unique identifiers,
055     * starting with 1.
056     */
057    public DefaultWorkflowManager() {
058        m_next = 1;
059        m_workflows = ConcurrentHashMap.newKeySet();
060        m_approvers = new ConcurrentHashMap<>();
061        m_completed = new CopyOnWriteArrayList<>();
062    }
063
064    /**
065     * {@inheritDoc}
066     */
067    @Override
068    public void start( final Workflow workflow ) throws WikiException {
069        m_workflows.add( workflow );
070        workflow.setWorkflowManager( this );
071        workflow.setId( nextId() );
072        workflow.start();
073    }
074
075    /**
076     * {@inheritDoc}
077     */
078    @Override
079    public Set< Workflow > getWorkflows() {
080        final Set< Workflow > workflows = ConcurrentHashMap.newKeySet();
081        workflows.addAll( m_workflows );
082        return workflows;
083    }
084
085    /**
086     * {@inheritDoc}
087     */
088    @Override
089    public List< Workflow > getCompletedWorkflows() {
090        return new CopyOnWriteArrayList< >( m_completed );
091    }
092
093    /**
094     * {@inheritDoc}
095     */
096    @Override
097    public void initialize( final Engine engine, final Properties props ) {
098        m_engine = engine;
099
100        // Identify the workflows requiring approvals
101        for( final Object o : props.keySet() ) {
102            final String prop = ( String )o;
103            if( prop.startsWith( PROPERTY_APPROVER_PREFIX ) ) {
104                // For the key, everything after the prefix is the workflow name
105                final String key = prop.substring( PROPERTY_APPROVER_PREFIX.length() );
106                if( key.length() > 0 ) {
107                    // Only use non-null/non-blank approvers
108                    final String approver = props.getProperty( prop );
109                    if( approver != null && approver.length() > 0 ) {
110                        m_approvers.put( key, new UnresolvedPrincipal( approver ) );
111                    }
112                }
113            }
114        }
115    }
116
117    /**
118     * {@inheritDoc}
119     */
120    @Override
121    public boolean requiresApproval( final String messageKey ) {
122        return  m_approvers.containsKey( messageKey );
123    }
124
125    /**
126     * {@inheritDoc}
127     */
128    @Override
129    public Principal getApprover( final String messageKey ) throws WikiException {
130        Principal approver = m_approvers.get( messageKey );
131        if ( approver == null ) {
132            throw new WikiException( "Workflow '" + messageKey + "' does not require approval." );
133        }
134
135        // Try to resolve UnresolvedPrincipals
136        if ( approver instanceof UnresolvedPrincipal ) {
137            final String name = approver.getName();
138            approver = m_engine.getManager( AuthorizationManager.class ).resolvePrincipal( name );
139
140            // If still unresolved, throw exception; otherwise, freshen our cache
141            if ( approver instanceof UnresolvedPrincipal ) {
142                throw new WikiException( "Workflow approver '" + name + "' cannot not be resolved." );
143            }
144
145            m_approvers.put( messageKey, approver );
146        }
147        return approver;
148    }
149
150    /**
151     * Protected helper method that returns the associated Engine
152     *
153     * @return the wiki engine
154     */
155    protected Engine getEngine() {
156        if ( m_engine == null ) {
157            throw new IllegalStateException( "Engine cannot be null; please initialize WorkflowManager first." );
158        }
159        return m_engine;
160    }
161
162    /**
163     * Returns the DecisionQueue associated with this WorkflowManager
164     *
165     * @return the decision queue
166     */
167    @Override public DecisionQueue getDecisionQueue()
168    {
169        return m_queue;
170    }
171
172    private volatile int m_next;
173
174    /**
175     * Returns the next available unique identifier, which is subsequently
176     * incremented.
177     *
178     * @return the id
179     */
180    private synchronized int nextId() {
181        final int current = m_next;
182        m_next++;
183        return current;
184    }
185
186    /**
187     * {@inheritDoc}
188     */
189    @Override
190    public List< Workflow > getOwnerWorkflows( final Session session ) {
191        final List< Workflow > workflows = new ArrayList<>();
192        if ( session.isAuthenticated() ) {
193            final Principal[] sessionPrincipals = session.getPrincipals();
194            for( final Workflow w : m_workflows ) {
195                final Principal owner = w.getOwner();
196                for ( final Principal sessionPrincipal : sessionPrincipals ) {
197                    if ( sessionPrincipal.equals( owner ) ) {
198                        workflows.add( w );
199                        break;
200                    }
201                }
202            }
203        }
204        return workflows;
205    }
206
207    /**
208     * Listens for {@link WorkflowEvent} objects emitted by Workflows. In particular, this
209     * method listens for {@link WorkflowEvent#CREATED},
210     * {@link WorkflowEvent#ABORTED} and {@link WorkflowEvent#COMPLETED}
211     * events. If a workflow is created, it is automatically added to the cache. If one is aborted or completed, it 
212     * is automatically removed.
213     * 
214     * @param event the event passed to this listener
215     */
216    @Override
217    public void actionPerformed( final WikiEvent event ) {
218        if( event instanceof WorkflowEvent ) {
219            final Workflow workflow = event.getSrc();
220            switch ( event.getType() ) {
221                 // Remove from manager
222                 case WorkflowEvent.ABORTED   :
223                 case WorkflowEvent.COMPLETED : remove( workflow ); break;
224                // Add to manager
225                case WorkflowEvent.CREATED    : add( workflow ); break;
226                default: break;
227            }
228        }
229    }
230
231    /**
232     * Protected helper method that adds a newly created Workflow to the cache, and sets its {@code workflowManager} and
233     * {@code Id} properties if not set.
234     *
235     * @param workflow the workflow to add
236     */
237    protected void add( final Workflow workflow ) {
238        if ( workflow.getWorkflowManager() == null ) {
239            workflow.setWorkflowManager( this );
240        }
241        if ( workflow.getId() == Workflow.ID_NOT_SET ) {
242            workflow.setId( nextId() );
243        }
244        m_workflows.add( workflow );
245    }
246
247    /**
248     * Protected helper method that removes a specified Workflow from the cache, and moves it to the workflow history list. This method
249     * defensively checks to see if the workflow has not yet been removed.
250     *
251     * @param workflow the workflow to remove
252     */
253    protected void remove( final Workflow workflow ) {
254        if( m_workflows.contains( workflow ) ) {
255            m_workflows.remove( workflow );
256            m_completed.add( workflow );
257        }
258    }
259
