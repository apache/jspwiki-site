073/**
074 * Manages the WikiPages.  This class functions as an unified interface towards the page providers.  It handles initialization
075 * and management of the providers, and provides utility methods for accessing the contents.
076 * <p/>
077 * Saving a page is a two-stage Task; first the pre-save operations and then the actual save.  See the descriptions of the tasks
078 * for further information.
079 *
080 * @since 2.0
081 */

096    /**
097     * Creates a new PageManager.
098     *
099     * @param engine WikiEngine instance
100     * @param props  Properties to use for initialization
101     * @throws NoSuchElementException {@value #PROP_PAGEPROVIDER} property not found on WikiEngine properties
102     * @throws WikiException If anything goes wrong, you get this.
103     */
104    public DefaultPageManager(final WikiEngine engine, final Properties props) throws NoSuchElementException, WikiException {
105        super(engine);
106        final String classname;
107        m_engine = engine;
108        final boolean useCache = "true".equals(props.getProperty(PROP_USECACHE));
109
110        m_expiryTime = TextUtil.parseIntParameter(props.getProperty(PROP_LOCKEXPIRY), 60);
111
112        //
113        //  If user wants to use a cache, then we'll use the CachingProvider.
114        //
115        if (useCache) {
116            classname = "org.apache.wiki.providers.CachingProvider";
117        } else {
118            classname = TextUtil.getRequiredProperty(props, PROP_PAGEPROVIDER);
119        }
120
121        pageSorter.initialize( props );
122
123        try {
124            LOG.debug("Page provider class: '" + classname + "'");
125            final Class<?> providerclass = ClassUtil.findClass("org.apache.wiki.providers", classname);
126            m_provider = (WikiPageProvider) providerclass.newInstance();
127
128            LOG.debug("Initializing page provider class " + m_provider);
129            m_provider.initialize(m_engine, props);
130        } catch (final ClassNotFoundException e) {
131            LOG.error("Unable to locate provider class '" + classname + "' (" + e.getMessage() + ")", e);
132            throw new WikiException("No provider class. (" + e.getMessage() + ")", e);
133        } catch (final InstantiationException e) {
134            LOG.error("Unable to create provider class '" + classname + "' (" + e.getMessage() + ")", e);
135            throw new WikiException("Faulty provider class. (" + e.getMessage() + ")", e);
136        } catch (final IllegalAccessException e) {
137            LOG.error("Illegal access to provider class '" + classname + "' (" + e.getMessage() + ")", e);
138            throw new WikiException("Illegal provider class. (" + e.getMessage() + ")", e);
139        } catch (final NoRequiredPropertyException e) {
140            LOG.error("Provider did not found a property it was looking for: " + e.getMessage(), e);
141            throw e;  // Same exception works.
142        } catch (final IOException e) {
143            LOG.error("An I/O exception occurred while trying to create a new page provider: " + classname, e);
144            throw new WikiException("Unable to start page provider: " + e.getMessage(), e);
145        }
146
147    }

167    /**
168     * {@inheritDoc}
169     * @see org.apache.wiki.pages.PageManager#getPageText(java.lang.String, int)
170     */
171    @Override
172    public String getPageText( final String pageName, final int version ) throws ProviderException {
173        if (pageName == null || pageName.length() == 0) {
174            throw new ProviderException( "Illegal page name" );
175        }
176        String text;
177
178        try {
179            text = m_provider.getPageText( pageName, version );
180        } catch ( final RepositoryModifiedException e ) {
181            //  This only occurs with the latest version.
182            LOG.info( "Repository has been modified externally while fetching page " + pageName );
183
184            //  Empty the references and yay, it shall be recalculated
185            final WikiPage p = m_provider.getPageInfo( pageName, version );
186
187            m_engine.getReferenceManager().updateReferences( p );
188            m_engine.getSearchManager().reindexPage( p );
189            text = m_provider.getPageText( pageName, version );
190        }
191
192        return text;
193    }

222    public void saveText( final WikiContext context, final String text ) throws WikiException {
223        // Check if page data actually changed; bail if not
224        final WikiPage page = context.getPage();
225        final String oldText = getPureText( page );
226        final String proposedText = TextUtil.normalizePostData( text );
227        if ( oldText != null && oldText.equals( proposedText ) ) {
228            return;
229        }
230
231        // Check if creation of empty pages is allowed; bail if not
232        final boolean allowEmpty = TextUtil.getBooleanProperty( m_engine.getWikiProperties(),
233                                                                WikiEngine.PROP_ALLOW_CREATION_OF_EMPTY_PAGES,
234                                                         false );
235        if ( !allowEmpty && !wikiPageExists( page ) && text.trim().equals( "" ) ) {
236            return;
237        }
238
239        // Create approval workflow for page save; add the diffed, proposed and old text versions as
240        // Facts for the approver (if approval is required). If submitter is authenticated, any reject
241        // messages will appear in his/her workflow inbox.
242        final WorkflowBuilder builder = WorkflowBuilder.getBuilder( m_engine );
243        final Principal submitter = context.getCurrentUser();
244        final Step prepTask = m_engine.getTasksManager().buildPreSaveWikiPageTask( context, proposedText );
245        final Step completionTask = m_engine.getTasksManager().buildSaveWikiPageTask();
246        final String diffText = m_engine.getDifferenceManager().makeDiff( context, oldText, proposedText );
247        final boolean isAuthenticated = context.getWikiSession().isAuthenticated();
248        final Fact[] facts = new Fact[ 5 ];
249        facts[ 0 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_PAGE_NAME, page.getName() );
250        facts[ 1 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_DIFF_TEXT, diffText );
251        facts[ 2 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_PROPOSED_TEXT, proposedText );
252        facts[ 3 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_CURRENT_TEXT, oldText);
253        facts[ 4 ] = new Fact( WorkflowManager.WF_WP_SAVE_FACT_IS_AUTHENTICATED, isAuthenticated );
254        final String rejectKey = isAuthenticated ? WorkflowManager.WF_WP_SAVE_REJECT_MESSAGE_KEY : null;
255        final Workflow workflow = builder.buildApprovalWorkflow( submitter,
256                                                                 WorkflowManager.WF_WP_SAVE_APPROVER,
257                                                                 prepTask,
258                                                                 WorkflowManager.WF_WP_SAVE_DECISION_MESSAGE_KEY,
259                                                                 facts,
260                                                                 completionTask,
261                                                                 rejectKey );
262        m_engine.getWorkflowManager().start( workflow );
263
264        // Let callers know if the page-save requires approval
265        if ( workflow.getCurrentStep() instanceof Decision ) {
266            throw new DecisionRequiredException( "The page contents must be approved before they become active." );
267        }
268    }

292    /**
293     * {@inheritDoc}
294     * @see org.apache.wiki.pages.PageManager#lockPage(org.apache.wiki.WikiPage, java.lang.String)
295     */
296    @Override
297    public PageLock lockPage( final WikiPage page, final String user) {
298        if (m_reaper == null) {
299            //
300            //  Start the lock reaper lazily.  We don't want to start it in
301            //  the constructor, because starting threads in constructors
302            //  is a bad idea when it comes to inheritance.  Besides,
303            //  laziness is a virtue.
304            //
305            m_reaper = new LockReaper(m_engine);
306            m_reaper.start();
307        }
308
309        fireEvent(WikiPageEvent.PAGE_LOCK, page.getName()); // prior to or after actual lock?
310        PageLock lock = m_pageLocks.get(page.getName());
311
312        if (lock == null) {
313            //
314            //  Lock is available, so make a lock.
315            //
316            final Date d = new Date();
317            lock = new PageLock(page, user, d, new Date(d.getTime() + m_expiryTime * 60 * 1000L));
318            m_pageLocks.put(page.getName(), lock);
319            LOG.debug("Locked page " + page.getName() + " for " + user);
320        } else {
321            LOG.debug("Page " + page.getName() + " already locked by " + lock.getLocker());
322            lock = null; // Nothing to return
323        }
324
325        return lock;
326    }

393    public WikiPage getPageInfo( final String pageName, final int version) throws ProviderException {
394        if (pageName == null || pageName.length() == 0) {
395            throw new ProviderException("Illegal page name '" + pageName + "'");
396        }
397
398        WikiPage page;
399
400        try {
401            page = m_provider.getPageInfo(pageName, version);
402        } catch ( final RepositoryModifiedException e) {
403            //  This only occurs with the latest version.
404            LOG.info("Repository has been modified externally while fetching info for " + pageName);
405            page = m_provider.getPageInfo(pageName, version);
406            if (page != null) {
407                m_engine.getReferenceManager().updateReferences(page);
408            } else {
409                m_engine.getReferenceManager().pageRemoved(new WikiPage(m_engine, pageName));
410            }
411        }
412
413        return page;
414    }

518    /**
519     * {@inheritDoc}
520     * @see org.apache.wiki.pages.PageManager#wikiPageExists(java.lang.String)
521     */
522    public boolean wikiPageExists( final String page ) {
523        if( m_engine.getCommandResolver().getSpecialPageReference( page ) != null ) {
524            return true;
525        }
526
527        Attachment att = null;
528        try {
529            if( m_engine.getFinalPageName( page ) != null ) {
530                return true;
531            }
532
533            att = m_engine.getAttachmentManager().getAttachmentInfo( null, page );
534        } catch( final ProviderException e ) {
535            LOG.debug( "pageExists() failed to find attachments", e );
536        }
537
538        return att != null;
539    }

541    /**
542     * {@inheritDoc}
543     * @see org.apache.wiki.pages.PageManager#wikiPageExists(java.lang.String, int)
544     */
545    public boolean wikiPageExists( final String page, final int version ) throws ProviderException {
546        if( m_engine.getCommandResolver().getSpecialPageReference( page ) != null ) {
547            return true;
548        }
549
550        boolean isThere = false;
551        final String finalName = m_engine.getFinalPageName( page );
552        if( finalName != null ) {
553            isThere = pageExists( finalName, version );
554        }
555
556        if( !isThere ) {
557            //  Go check if such an attachment exists.
558            try {
559                isThere = m_engine.getAttachmentManager().getAttachmentInfo( null, page, version ) != null;
560            } catch( final ProviderException e ) {
561                LOG.debug( "wikiPageExists() failed to find attachments", e );
562            }
563        }
564
565        return isThere;
566    }

582    /**
583     * {@inheritDoc}
584     * @see org.apache.wiki.pages.PageManager#deletePage(java.lang.String)
585     */
586    public void deletePage( final String pageName ) throws ProviderException {
587        final WikiPage p = getPage( pageName );
588        if( p != null ) {
589            if( p instanceof Attachment ) {
590                m_engine.getAttachmentManager().deleteAttachment( ( Attachment )p );
591            } else {
592                final Collection< String > refTo = m_engine.getReferenceManager().findRefersTo( pageName );
593                // May return null, if the page does not exist or has not been indexed yet.
594
595                if( m_engine.getAttachmentManager().hasAttachments( p ) ) {
596                    final List< Attachment > attachments = m_engine.getAttachmentManager().listAttachments( p );
597                    for( final Attachment attachment : attachments ) {
598                        if( refTo != null ) {
599                            refTo.remove( attachment.getName() );
600                        }
601
602                        m_engine.getAttachmentManager().deleteAttachment( attachment );
603                    }
604                }
605                deletePage( p );
606                fireEvent( WikiPageEvent.PAGE_DELETED, pageName );
607            }
608        }
609    }

622    /**
623     * This is a simple reaper thread that runs roughly every minute
624     * or so (it's not really that important, as long as it runs),
625     * and removes all locks that have expired.
626     */
627    private class LockReaper extends WikiBackgroundThread {
628        /**
629         * Create a LockReaper for a given engine.
630         *
631         * @param engine WikiEngine to own this thread.
632         */
633        public LockReaper( final WikiEngine engine) {
634            super(engine, 60);
635            setName("JSPWiki Lock Reaper");
636        }
637
638        @Override
639        public void backgroundTask() {
640            final Collection< PageLock > entries = m_pageLocks.values();
641            for( final Iterator<PageLock> i = entries.iterator(); i.hasNext(); ) {
642                final PageLock p = i.next();
643
644                if ( p.isExpired() ) {
645                    i.remove();
646
647                    LOG.debug( "Reaped lock: " + p.getPage() +
648                               " by " + p.getLocker() +
649                               ", acquired " + p.getAcquisitionTime() +
650                               ", and expired " + p.getExpiryTime() );
651                }
652            }
653        }
654    }

689    /**
690     * Listens for {@link org.apache.wiki.event.WikiSecurityEvent#PROFILE_NAME_CHANGED}
691     * events. If a user profile's name changes, each page ACL is inspected. If an entry contains
692     * a name that has changed, it is replaced with the new one. No events are emitted
693     * as a consequence of this method, because the page contents are still the same; it is
694     * only the representations of the names within the ACL that are changing.
695     *
696     * @param event The event
697     */
698    @Override
699    public void actionPerformed( final WikiEvent event ) {
700        if( !( event instanceof WikiSecurityEvent ) ) {
701            return;
702        }
703
704        final WikiSecurityEvent se = ( WikiSecurityEvent ) event;
705        if( se.getType() == WikiSecurityEvent.PROFILE_NAME_CHANGED ) {
706            final UserProfile[] profiles = (UserProfile[]) se.getTarget();
707            final Principal[] oldPrincipals = new Principal[] { new WikiPrincipal( profiles[ 0 ].getLoginName() ),
708                                                                new WikiPrincipal( profiles[ 0 ].getFullname()),
709                                                                new WikiPrincipal( profiles[ 0 ].getWikiName() ) };
710            final Principal newPrincipal = new WikiPrincipal( profiles[ 1 ].getFullname() );
711
712            // Examine each page ACL
713            try {
714                int pagesChanged = 0;
715                final Collection< WikiPage > pages = getAllPages();
716                for( final WikiPage page : pages ) {
717                    final boolean aclChanged = changeAcl( page, oldPrincipals, newPrincipal );
718                    if( aclChanged ) {
719                        // If the Acl needed changing, change it now
720                        try {
721                            m_engine.getAclManager().setPermissions( page, page.getAcl() );
722                        } catch( final WikiSecurityException e ) {
723                            LOG.error("Could not change page ACL for page " + page.getName() + ": " + e.getMessage(), e);
724                        }
725                        pagesChanged++;
726                    }
727                }
728                LOG.info( "Profile name change for '" + newPrincipal.toString() + "' caused " + pagesChanged + " page ACLs to change also." );
729            } catch( final ProviderException e ) {
730                // Oooo! This is really bad...
731                LOG.error( "Could not change user name in Page ACLs because of Provider error:" + e.getMessage(), e );
732            }
733        }
734    }

736    /**
737     * For a single wiki page, replaces all Acl entries matching a supplied array of Principals with a new Principal.
738     *
739     * @param page the wiki page whose Acl is to be modified
740     * @param oldPrincipals an array of Principals to replace; all AclEntry objects whose {@link AclEntry#getPrincipal()} method returns
741     *                      one of these Principals will be replaced
742     * @param newPrincipal the Principal that should receive the old Principals' permissions
743     * @return <code>true</code> if the Acl was actually changed; <code>false</code> otherwise
744     */
745    protected boolean changeAcl( final WikiPage page, final Principal[] oldPrincipals, final Principal newPrincipal ) {
746        final Acl acl = page.getAcl();
747        boolean pageChanged = false;
748        if( acl != null ) {
749            final Enumeration< AclEntry > entries = acl.entries();
750            final Collection< AclEntry > entriesToAdd = new ArrayList<>();
751            final Collection< AclEntry > entriesToRemove = new ArrayList<>();
752            while( entries.hasMoreElements() ) {
753                final AclEntry entry = entries.nextElement();
754                if( ArrayUtils.contains(oldPrincipals, entry.getPrincipal() ) ) {
755                    // Create new entry
756                    final AclEntry newEntry = new AclEntryImpl();
757                    newEntry.setPrincipal( newPrincipal );
758                    final Enumeration<Permission> permissions = entry.permissions();
759                    while( permissions.hasMoreElements() ) {
760                        final Permission permission = permissions.nextElement();
761                        newEntry.addPermission(permission);
762                    }
763                    pageChanged = true;
764                    entriesToRemove.add(entry);
765                    entriesToAdd.add(newEntry);
766                }
767            }
768            for( final AclEntry entry : entriesToRemove ) {
769                acl.removeEntry( entry );
770            }
771            for( final AclEntry entry : entriesToAdd ) {
772                acl.addEntry( entry );
773            }
774        }
775        return pageChanged;
776    }
